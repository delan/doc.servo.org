Helper trait to convert a tuple of atoms into Atoms.

use egui::{Atoms, Image, IntoAtoms, RichText};
let atoms: Atoms = (
    "Some text",
    RichText::new("Some RichText"),
    Image::new("some_image_url"),
).into_atoms();
