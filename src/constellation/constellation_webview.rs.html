18/// The `Constellation`'s view of a `WebView` in the embedding layer. This tracks all of the
19/// `Constellation` state for this `WebView`.
20pub(crate) struct ConstellationWebView {
21    /// The [`WebViewId`] of this [`ConstellationWebView`].
22    webview_id: WebViewId,
23
24    /// The currently focused browsing context in this webview for key events.
25    /// The focused pipeline is the current entry of the focused browsing
26    /// context.
27    pub focused_browsing_context_id: BrowsingContextId,
28
29    /// The [`BrowsingContextId`] of the currently hovered browsing context, to use for
30    /// knowing which frame is currently receiving cursor events.
31    pub hovered_browsing_context_id: Option<BrowsingContextId>,
32
33    /// The last mouse move point in the coordinate space of the Pipeline that it
34    /// happened int.
35    pub last_mouse_move_point: Point2D<f32, CSSPixel>,
36
37    /// The joint session history for this webview.
38    pub session_history: JointSessionHistory,
39
40    /// The [`UserContentManagerId`] for all pipelines in this `WebView`. This is `Some`
41    /// if the embedder has set a `UserContentManager` using the WebViewBuilder API and
42    /// it is `None` otherwise.
43    pub user_content_manager_id: Option<UserContentManagerId>,
44
45    /// The [`Theme`] that this [`ConstellationWebView`] uses. This is communicated to all
46    /// `ScriptThread`s so that they know how to render the contents of a particular `WebView.
47    theme: Theme,
48}
49
50impl ConstellationWebView {
51    pub(crate) fn new(
52        webview_id: WebViewId,
53        focused_browsing_context_id: BrowsingContextId,
54        user_content_manager_id: Option<UserContentManagerId>,
55    ) -> Self {
56        Self {
57            webview_id,
58            user_content_manager_id,
59            focused_browsing_context_id,
60            hovered_browsing_context_id: None,
61            last_mouse_move_point: Default::default(),
62            session_history: JointSessionHistory::new(),
63            theme: Theme::Light,
64        }
65    }
66
67    /// Set the [`Theme`] on this [`ConstellationWebView`] returning true if the theme changed.
68    pub(crate) fn set_theme(&mut self, new_theme: Theme) -> bool {
69        let old_theme = std::mem::replace(&mut self.theme, new_theme);
70        old_theme != self.theme
71    }
72
73    /// Get the [`Theme`] of this [`ConstellationWebView`].
74    pub(crate) fn theme(&self) -> Theme {
75        self.theme
76    }
77
78    fn target_pipeline_id_for_input_event(
79        &self,
80        event: &ConstellationInputEvent,
81        browsing_contexts: &FxHashMap<BrowsingContextId, BrowsingContext>,
82    ) -> Option<PipelineId> {
83        if let Some(hit_test_result) = &event.hit_test_result {
84            return Some(hit_test_result.pipeline_id);
85        }
86
87        // If there's no hit test, send the event to either the hovered or focused browsing context,
88        // depending on the event type.
89        let browsing_context_id = if matches!(event.event.event, InputEvent::MouseLeftViewport(_)) {
90            self.hovered_browsing_context_id
91                .unwrap_or(self.focused_browsing_context_id)
92        } else {
93            self.focused_browsing_context_id
94        };
95
96        Some(browsing_contexts.get(&browsing_context_id)?.pipeline_id)
97    }
98
99    /// Forward the [`InputEvent`] to this [`ConstellationWebView`]. Returns false if
100    /// the event could not be forwarded or true otherwise.
101    pub(crate) fn forward_input_event(
102        &mut self,
103        event: ConstellationInputEvent,
104        pipelines: &FxHashMap<PipelineId, Pipeline>,
105        browsing_contexts: &FxHashMap<BrowsingContextId, BrowsingContext>,
106    ) -> bool {
107        let Some(pipeline_id) = self.target_pipeline_id_for_input_event(&event, browsing_contexts)
108        else {
109            warn!("Unknown pipeline for input event. Ignoring.");
110            return false;
111        };
112        let Some(pipeline) = pipelines.get(&pipeline_id) else {
113            warn!("Unknown pipeline id {pipeline_id:?} for input event. Ignoring.");
114            return false;
115        };
116
117        let mut update_hovered_browsing_context =
118            |newly_hovered_browsing_context_id, focus_moving_to_another_iframe: bool| {
119                let old_hovered_context_id = std::mem::replace(
120                    &mut self.hovered_browsing_context_id,
121                    newly_hovered_browsing_context_id,
122                );
123                if old_hovered_context_id == newly_hovered_browsing_context_id {
124                    return;
125                }
126                let Some(old_hovered_context_id) = old_hovered_context_id else {
127                    return;
128                };
129                let Some(pipeline) = browsing_contexts
130                    .get(&old_hovered_context_id)
131                    .and_then(|browsing_context| pipelines.get(&browsing_context.pipeline_id))
132                else {
133                    return;
134                };
135
136                let mut synthetic_mouse_leave_event = event.clone();
137                synthetic_mouse_leave_event.event.event =
138                    InputEvent::MouseLeftViewport(MouseLeftViewportEvent {
139                        focus_moving_to_another_iframe,
140                    });
141
142                let _ = pipeline
143                    .event_loop
144                    .send(ScriptThreadMessage::SendInputEvent(
145                        self.webview_id,
146                        pipeline.id,
147                        synthetic_mouse_leave_event,
148                    ));
149            };
150
151        if let InputEvent::MouseLeftViewport(_) = &event.event.event {
152            update_hovered_browsing_context(None, false);
153            return true;
154        }
155
156        if let InputEvent::MouseMove(_) = &event.event.event {
157            update_hovered_browsing_context(Some(pipeline.browsing_context_id), true);
158            self.last_mouse_move_point = event
159                .hit_test_result
160                .as_ref()
161                .expect("MouseMove events should always have hit tests.")
162                .point_in_viewport;
163        }
164
165        let _ = pipeline
166            .event_loop
167            .send(ScriptThreadMessage::SendInputEvent(
168                self.webview_id,
169                pipeline.id,
170                event,
171            ));
172        true
173    }
