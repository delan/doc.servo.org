100impl Actor for WalkerActor {
101    fn name(&self) -> String {
102        self.name.clone()
103    }
104
105    /// The walker actor can handle the following messages:
106    ///
107    /// - `children`: Returns a list of children nodes of the specified node
108    ///
109    /// - `clearPseudoClassLocks`: Placeholder
110    ///
111    /// - `documentElement`: Returns the base document element node
112    ///
113    /// - `getLayoutInspector`: Returns the Layout inspector actor, placeholder
114    ///
115    /// - `getMutations`: Returns the list of attribute changes since it was last called
116    ///
117    /// - `getOffsetParent`: Placeholder
118    ///
119    /// - `querySelector`: Recursively looks for the specified selector in the tree, reutrning the
120    ///   node and its ascendents
121    fn handle_message(
122        &self,
123        mut request: ClientRequest,
124        registry: &ActorRegistry,
125        msg_type: &str,
126        msg: &Map<String, Value>,
127        _id: StreamId,
128    ) -> Result<(), ActorError> {
129        match msg_type {
130            "children" => {
131                let target = msg
132                    .get("node")
133                    .ok_or(ActorError::MissingParameter)?
134                    .as_str()
135                    .ok_or(ActorError::BadParameterType)?;
136                let Some((tx, rx)) = generic_channel::channel() else {
137                    return Err(ActorError::Internal);
138                };
139                self.script_chan
140                    .send(GetChildren(
141                        self.pipeline,
142                        registry.actor_to_script(target.into()),
143                        tx,
144                    ))
145                    .map_err(|_| ActorError::Internal)?;
146                let children = rx
147                    .recv()
148                    .map_err(|_| ActorError::Internal)?
149                    .ok_or(ActorError::Internal)?;
150
151                let msg = ChildrenReply {
152                    has_first: true,
153                    has_last: true,
154                    nodes: children
155                        .into_iter()
156                        .map(|child| {
157                            child.encode(
158                                registry,
159                                self.script_chan.clone(),
160                                self.pipeline,
161                                self.name(),
162                            )
163                        })
164                        .collect(),
165                    from: self.name(),
166                };
167                request.reply_final(&msg)?
168            },
169            "clearPseudoClassLocks" => {
170                let msg = EmptyReplyMsg { from: self.name() };
171                request.reply_final(&msg)?
172            },
173            "documentElement" => {
174                let Some((tx, rx)) = generic_channel::channel() else {
175                    return Err(ActorError::Internal);
176                };
177                self.script_chan
178                    .send(GetDocumentElement(self.pipeline, tx))
179                    .map_err(|_| ActorError::Internal)?;
180                let doc_elem_info = rx
181                    .recv()
182                    .map_err(|_| ActorError::Internal)?
183                    .ok_or(ActorError::Internal)?;
184                let node = doc_elem_info.encode(
185                    registry,
186                    self.script_chan.clone(),
187                    self.pipeline,
188                    self.name(),
189                );
190
191                let msg = DocumentElementReply {
192                    from: self.name(),
193                    node,
194                };
195                request.reply_final(&msg)?
196            },
197            "getLayoutInspector" => {
198                // TODO: Create actual layout inspector actor
199                let layout = LayoutInspectorActor::new(registry.new_name::<LayoutInspectorActor>());
200                let actor = layout.encode(registry);
201                registry.register(layout);
202
203                let msg = GetLayoutInspectorReply {
204                    from: self.name(),
205                    actor,
206                };
207                request.reply_final(&msg)?
208            },
209            "getMutations" => {
210                let msg = GetMutationsReply {
211                    from: self.name(),
212                    mutations: self
213                        .mutations
214                        .borrow_mut()
215                        .drain(..)
216                        .map(|(mutation, target)| MutationMsg {
217                            attribute_name: mutation.attribute_name,
218                            new_value: mutation.new_value,
219                            target,
220                            type_: "attributes".into(),
221                        })
222                        .collect(),
223                };
224                request.reply_final(&msg)?
225            },
226            "getOffsetParent" => {
227                let msg = GetOffsetParentReply {
228                    from: self.name(),
229                    node: None,
230                };
231                request.reply_final(&msg)?
232            },
233            "querySelector" => {
234                let selector = msg
235                    .get("selector")
236                    .ok_or(ActorError::MissingParameter)?
237                    .as_str()
238                    .ok_or(ActorError::BadParameterType)?;
239                let node = msg
240                    .get("node")
241                    .ok_or(ActorError::MissingParameter)?
242                    .as_str()
243                    .ok_or(ActorError::BadParameterType)?;
244                let mut hierarchy = find_child(
245                    &self.script_chan,
246                    self.pipeline,
247                    &self.name,
248                    registry,
249                    node,
250                    vec![],
251                    |msg| msg.display_name == selector,
252                )
253                .map_err(|_| ActorError::Internal)?;
254                hierarchy.reverse();
255                let node = hierarchy.pop().ok_or(ActorError::Internal)?;
256
257                let msg = QuerySelectorReply {
258                    from: self.name(),
259                    node,
260                    new_parents: hierarchy,
261                };
262                request.reply_final(&msg)?
263            },
264            "watchRootNode" => {
265                let msg = WatchRootNodeNotification {
266                    type_: "root-available".into(),
267                    from: self.name(),
268                    node: self.root(registry)?,
269                };
270                let _ = request.write_json_packet(&msg);
271
272                let msg = EmptyReplyMsg { from: self.name() };
273                request.reply_final(&msg)?
274            },
275            _ => return Err(ActorError::UnrecognizedPacketType),
276        };
277        Ok(())
278    }
279}

281impl WalkerActor {
282    pub(crate) fn new_mutations(
283        &self,
284        request: &mut ClientRequest,
285        target: &str,
286        modifications: &[AttrModification],
287    ) {
288        {
289            let mut mutations = self.mutations.borrow_mut();
290            mutations.extend(modifications.iter().cloned().map(|m| (m, target.into())));
291        }
292        let _ = request.write_json_packet(&NewMutationsNotification {
293            from: self.name(),
294            type_: "newMutations".into(),
295        });
296    }
297
298    pub(crate) fn root(&self, registry: &ActorRegistry) -> Result<NodeActorMsg, ActorError> {
299        let (tx, rx) = generic_channel::channel().ok_or(ActorError::Internal)?;
300        self.script_chan
301            .send(GetRootNode(self.pipeline, tx))
302            .map_err(|_| ActorError::Internal)?;
303        let root_node = rx
304            .recv()
305            .map_err(|_| ActorError::Internal)?
306            .ok_or(ActorError::Internal)?;
307        Ok(root_node.encode(
308            registry,
309            self.script_chan.clone(),
310            self.pipeline,
311            self.name(),
312        ))
313    }
314}

316/// Recursively searches for a child with the specified selector
317/// If it is found, returns a list with the child and all of its ancestors.
318/// TODO: Investigate how to cache this to some extent.
319pub fn find_child(
320    script_chan: &GenericSender<DevtoolScriptControlMsg>,
321    pipeline: PipelineId,
322    name: &str,
323    registry: &ActorRegistry,
324    node: &str,
325    mut hierarchy: Vec<NodeActorMsg>,
326    compare_fn: impl Fn(&NodeActorMsg) -> bool + Clone,
327) -> Result<Vec<NodeActorMsg>, Vec<NodeActorMsg>> {
328    let (tx, rx) = generic_channel::channel().unwrap();
329    script_chan
330        .send(GetChildren(
331            pipeline,
332            registry.actor_to_script(node.into()),
333            tx,
334        ))
335        .unwrap();
336    let children = rx.recv().unwrap().ok_or(vec![])?;
337
338    for child in children {
339        let msg = child.encode(registry, script_chan.clone(), pipeline, name.into());
340        if compare_fn(&msg) {
341            hierarchy.push(msg);
342            return Ok(hierarchy);
343        };
344
345        if msg.num_children == 0 {
346            continue;
347        }
348
349        match find_child(
350            script_chan,
351            pipeline,
352            name,
353            registry,
354            &msg.actor,
355            hierarchy,
356            compare_fn.clone(),
357        ) {
358            Ok(mut hierarchy) => {
359                hierarchy.push(msg);
360                return Ok(hierarchy);
361            },
362            Err(e) => {
363                hierarchy = e;
364            },
365        }
366    }
367    Err(hierarchy)
368}
