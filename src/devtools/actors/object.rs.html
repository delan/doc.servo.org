44impl ObjectActor {
45    pub fn register(registry: &ActorRegistry, uuid: String) -> String {
46        if !registry.script_actor_registered(uuid.clone()) {
47            let name = registry.new_name::<Self>();
48            let actor = ObjectActor {
49                name: name.clone(),
50                _uuid: uuid.clone(),
51            };
52
53            registry.register_script_actor(uuid, name.clone());
54            registry.register(actor);
55
56            name
57        } else {
58            registry.script_to_actor(uuid)
59        }
60    }
61}
