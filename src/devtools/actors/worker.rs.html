22#[derive(Clone, Copy)]
23#[expect(dead_code)]
24pub enum WorkerType {
25    Dedicated = 0,
26    Shared = 1,
27    Service = 2,
28}

30pub(crate) struct WorkerActor {
31    pub name: String,
32    pub console: String,
33    pub thread: String,
34    pub worker_id: WorkerId,
35    pub url: ServoUrl,
36    pub type_: WorkerType,
37    pub script_chan: GenericSender<DevtoolScriptControlMsg>,
38    pub streams: AtomicRefCell<HashMap<StreamId, TcpStream>>,
39}
40
41impl ResourceAvailable for WorkerActor {
42    fn actor_name(&self) -> String {
43        self.name.clone()
44    }
45}
46
47impl Actor for WorkerActor {
48    fn name(&self) -> String {
49        self.name.clone()
50    }
51    fn handle_message(
52        &self,
53        mut request: ClientRequest,
54        _registry: &ActorRegistry,
55        msg_type: &str,
56        _msg: &Map<String, Value>,
57        stream_id: StreamId,
58    ) -> Result<(), ActorError> {
59        match msg_type {
60            "attach" => {
61                let msg = AttachedReply {
62                    from: self.name(),
63                    type_: "attached".to_owned(),
64                    url: self.url.as_str().to_owned(),
65                };
66                // FIXME: we don’t send an actual reply (message without type), which seems to be a bug?
67                request.write_json_packet(&msg)?;
68                self.streams
69                    .borrow_mut()
70                    .insert(stream_id, request.try_clone_stream().unwrap());
71                // FIXME: fix messages to not require forging a pipeline for worker messages
72                self.script_chan
73                    .send(WantsLiveNotifications(TEST_PIPELINE_ID, true))
74                    .unwrap();
75            },
76
77            "connect" => {
78                let msg = ConnectReply {
79                    from: self.name(),
80                    type_: "connected".to_owned(),
81                    thread_actor: self.thread.clone(),
82                    console_actor: self.console.clone(),
83                };
84                // FIXME: we don’t send an actual reply (message without type), which seems to be a bug?
85                request.write_json_packet(&msg)?;
86            },
87
88            "detach" => {
89                let msg = DetachedReply {
90                    from: self.name(),
91                    type_: "detached".to_string(),
92                };
93                self.cleanup(stream_id);
94                // FIXME: we don’t send an actual reply (message without type), which seems to be a bug?
95                request.write_json_packet(&msg)?;
96            },
97
98            _ => return Err(ActorError::UnrecognizedPacketType),
99        };
100        Ok(())
101    }
102
103    fn cleanup(&self, stream_id: StreamId) {
104        self.streams.borrow_mut().remove(&stream_id);
105        if self.streams.borrow().is_empty() {
106            self.script_chan
107                .send(WantsLiveNotifications(TEST_PIPELINE_ID, false))
108                .unwrap();
109        }
110    }
111}

113#[derive(Serialize)]
114struct DetachedReply {
115    from: String,
116    #[serde(rename = "type")]
117    type_: String,
118}
119
120#[derive(Serialize)]
121struct AttachedReply {
122    from: String,
123    #[serde(rename = "type")]
124    type_: String,
125    url: String,
126}
127
128#[derive(Serialize)]
129#[serde(rename_all = "camelCase")]
130struct ConnectReply {
131    from: String,
132    #[serde(rename = "type")]
133    type_: String,
134    thread_actor: String,
135    console_actor: String,
136}

138#[derive(Serialize)]
139#[serde(rename_all = "camelCase")]
140struct WorkerTraits {
141    is_parent_intercept_enabled: bool,
142    supports_top_level_target_flag: bool,
143}
144
145#[derive(Serialize)]
146#[serde(rename_all = "camelCase")]
147pub(crate) struct WorkerActorMsg {
148    actor: String,
149    console_actor: String,
150    thread_actor: String,
151    id: String,
152    url: String,
153    traits: WorkerTraits,
154    #[serde(rename = "type")]
155    type_: u32,
156    #[serde(rename = "targetType")]
157    target_type: String,
158}
159
160impl ActorEncode<WorkerActorMsg> for WorkerActor {
161    fn encode(&self, _: &ActorRegistry) -> WorkerActorMsg {
162        WorkerActorMsg {
163            actor: self.name(),
164            console_actor: self.console.clone(),
165            thread_actor: self.thread.clone(),
166            id: self.worker_id.0.to_string(),
167            url: self.url.to_string(),
168            traits: WorkerTraits {
169                is_parent_intercept_enabled: false,
170                supports_top_level_target_flag: false,
171            },
172            type_: self.type_ as u32,
173            target_type: "worker".to_string(),
174        }
175    }
