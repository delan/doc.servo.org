60// e.g. quartz, FreeType. It provides access to metrics and tables
61// needed by the text shaper as well as access to the underlying font

64pub trait PlatformFontMethods: Sized {
65    #[servo_tracing::instrument(name = "PlatformFontMethods::new_from_template", skip_all)]
66    fn new_from_template(
67        template: FontTemplateRef,
68        pt_size: Option<Au>,
69        variations: &[FontVariation],
70        data: &Option<FontData>,
71        synthetic_bold: bool,
72    ) -> Result<PlatformFont, &'static str> {
73        let template = template.borrow();
74        let font_identifier = template.identifier.clone();
75
76        match font_identifier {
77            FontIdentifier::Local(font_identifier) => Self::new_from_local_font_identifier(
78                font_identifier,
79                pt_size,
80                variations,
81                synthetic_bold,
82            ),
83            FontIdentifier::Web(_) => Self::new_from_data(
84                font_identifier,
85                data.as_ref()
86                    .expect("Should never create a web font without data."),
87                pt_size,
88                variations,
89                synthetic_bold,
90            ),
91        }
92    }
93
94    fn new_from_local_font_identifier(
95        font_identifier: LocalFontIdentifier,
96        pt_size: Option<Au>,
97        variations: &[FontVariation],
98        synthetic_bold: bool,
99    ) -> Result<PlatformFont, &'static str>;
100
101    fn new_from_data(
102        font_identifier: FontIdentifier,
103        data: &FontData,
104        pt_size: Option<Au>,
105        variations: &[FontVariation],
106        synthetic_bold: bool,
107    ) -> Result<PlatformFont, &'static str>;
108
109    /// Get a [`FontTemplateDescriptor`] from a [`PlatformFont`]. This is used to get
110    /// descriptors for web fonts.
111    fn descriptor(&self) -> FontTemplateDescriptor;
112
113    fn glyph_index(&self, codepoint: char) -> Option<GlyphId>;
114    fn glyph_h_advance(&self, _: GlyphId) -> Option<FractionalPixel>;
115    fn glyph_h_kerning(&self, glyph0: GlyphId, glyph1: GlyphId) -> FractionalPixel;
116
117    fn metrics(&self) -> FontMetrics;
118    fn table_for_tag(&self, _: Tag) -> Option<FontTable>;
119    fn typographic_bounds(&self, _: GlyphId) -> Rect<f32>;
120
121    /// Get the necessary [`FontInstanceFlags`]` for this font.
122    fn webrender_font_instance_flags(&self) -> FontInstanceFlags;
123
124    /// Return all the variation values that the font was instantiated with.
125    fn variations(&self) -> &[FontVariation];
126
127    fn descriptor_from_os2_table(os2: &Os2) -> FontTemplateDescriptor {
128        let mut style = FontStyle::NORMAL;
129        if os2.fs_selection().contains(SelectionFlags::ITALIC) {
130            style = FontStyle::ITALIC;
131        }
132
133        let weight = FontWeight::from_float(os2.us_weight_class() as f32);
134        let stretch = match os2.us_width_class() {
135            1 => FontStretch::ULTRA_CONDENSED,
136            2 => FontStretch::EXTRA_CONDENSED,
137            3 => FontStretch::CONDENSED,
138            4 => FontStretch::SEMI_CONDENSED,
139            5 => FontStretch::NORMAL,
140            6 => FontStretch::SEMI_EXPANDED,
141            7 => FontStretch::EXPANDED,
142            8 => FontStretch::EXTRA_EXPANDED,
143            9 => FontStretch::ULTRA_EXPANDED,
144            _ => FontStretch::NORMAL,
145        };
146
147        FontTemplateDescriptor::new(weight, stretch, style)
148    }
149}

188#[derive(Debug, Default)]
189struct CachedShapeData {
190    glyph_advances: HashMap<GlyphId, FractionalPixel>,
191    glyph_indices: HashMap<char, Option<GlyphId>>,
192    shaped_text: HashMap<ShapeCacheEntry, Arc<GlyphStore>>,
193}
194
195impl malloc_size_of::MallocSizeOf for CachedShapeData {
196    fn size_of(&self, ops: &mut malloc_size_of::MallocSizeOfOps) -> usize {
197        // Estimate the size of the shaped text cache. This will be smaller, because
198        // HashMap has some overhead, but we are mainly interested in the actual data.
199        let shaped_text_size = self
200            .shaped_text
201            .iter()
202            .map(|(key, value)| key.size_of(ops) + (*value).size_of(ops))
203            .sum::<usize>();
204        self.glyph_advances.size_of(ops) + self.glyph_indices.size_of(ops) + shaped_text_size
205    }
206}

386/// An entry in the shape cache.
387#[derive(Clone, Debug, Eq, Hash, PartialEq)]
388struct ShapeCacheEntry {
389    text: String,
390    options: ShapingOptions,
391}
392
393impl Font {
394    pub fn shape_text(&self, text: &str, options: &ShapingOptions) -> Arc<GlyphStore> {
395        let lookup_key = ShapeCacheEntry {
396            text: text.to_owned(),
397            options: *options,
398        };
399        {
400            let cache = self.cached_shape_data.read();
401            if let Some(shaped_text) = cache.shaped_text.get(&lookup_key) {
402                return shaped_text.clone();
403            }
404        }
405
406        let is_single_preserved_newline = text.len() == 1 && text.starts_with('\n');
407        let start_time = Instant::now();
408        let mut glyphs = GlyphStore::new(
409            text.len(),
410            options
411                .flags
412                .contains(ShapingFlags::IS_WHITESPACE_SHAPING_FLAG),
413            options
414                .flags
415                .contains(ShapingFlags::ENDS_WITH_WHITESPACE_SHAPING_FLAG),
416            is_single_preserved_newline,
417            options.flags.contains(ShapingFlags::RTL_FLAG),
418        );
419
420        if self.can_do_fast_shaping(text, options) {
421            debug!("shape_text: Using ASCII fast path.");
422            self.shape_text_fast(text, options, &mut glyphs);
423        } else {
424            debug!("shape_text: Using Harfbuzz.");
425            self.shape_text_harfbuzz(text, options, &mut glyphs);
426        }
427
428        let shaped_text = Arc::new(glyphs);
429        let mut cache = self.cached_shape_data.write();
430        cache.shaped_text.insert(lookup_key, shaped_text.clone());
431
432        let end_time = Instant::now();
433        TEXT_SHAPING_PERFORMANCE_COUNTER.fetch_add(
434            (end_time.duration_since(start_time).as_nanos()) as usize,
435            Ordering::Relaxed,
436        );
437
438        shaped_text
439    }
440
441    fn shape_text_harfbuzz(&self, text: &str, options: &ShapingOptions, glyphs: &mut GlyphStore) {
442        self.shaper
443            .get_or_init(|| Shaper::new(self))
444            .shape_text(text, options, glyphs);
445    }
446
447    /// Whether not a particular text and [`ShapingOptions`] combination can use
448    /// "fast shaping" ie shaping without Harfbuzz.
449    ///
450    /// Note: This will eventually be removed.
451    pub fn can_do_fast_shaping(&self, text: &str, options: &ShapingOptions) -> bool {
452        options.script == Script::Latin &&
453            !options.flags.contains(ShapingFlags::RTL_FLAG) &&
454            *self.can_do_fast_shaping.get_or_init(|| {
455                self.table_for_tag(KERN).is_some() &&
456                    self.table_for_tag(GPOS).is_none() &&
457                    self.table_for_tag(GSUB).is_none()
458            }) &&
459            text.is_ascii()
460    }
461
462    /// Fast path for ASCII text that only needs simple horizontal LTR kerning.
463    fn shape_text_fast(&self, text: &str, options: &ShapingOptions, glyphs: &mut GlyphStore) {
464        let mut prev_glyph_id = None;
465        for (i, byte) in text.bytes().enumerate() {
466            let character = byte as char;
467            let glyph_id = match self.glyph_index(character) {
468                Some(id) => id,
469                None => continue,
470            };
471
472            let mut advance = advance_for_shaped_glyph(
473                Au::from_f64_px(self.glyph_h_advance(glyph_id)),
474                character,
475                options,
476            );
477            let offset = prev_glyph_id.map(|prev| {
478                let h_kerning = Au::from_f64_px(self.glyph_h_kerning(prev, glyph_id));
479                advance += h_kerning;
480                Point2D::new(h_kerning, Au::zero())
481            });
482
483            let glyph = GlyphData::new(glyph_id, advance, offset, true, true);
484            glyphs.add_glyph_for_byte_index(ByteIndex(i as isize), character, &glyph);
485            prev_glyph_id = Some(glyph_id);
486        }
487        glyphs.finalize_changes();
488    }

507    #[inline]
508    pub fn glyph_index(&self, codepoint: char) -> Option<GlyphId> {
509        {
510            let cache = self.cached_shape_data.read();
511            if let Some(glyph) = cache.glyph_indices.get(&codepoint) {
512                return *glyph;
513            }
514        }
515        let codepoint = match self.descriptor.variant {
516            font_variant_caps::T::SmallCaps => codepoint.to_ascii_uppercase(),
517            font_variant_caps::T::Normal => codepoint,
518        };
519        let glyph_index = self.handle.glyph_index(codepoint);
520
521        let mut cache = self.cached_shape_data.write();
522        cache.glyph_indices.insert(codepoint, glyph_index);
523        glyph_index
524    }

538    pub fn glyph_h_advance(&self, glyph_id: GlyphId) -> FractionalPixel {
539        {
540            let cache = self.cached_shape_data.read();
541            if let Some(width) = cache.glyph_advances.get(&glyph_id) {
542                return *width;
543            }
544        }
545
546        let new_width = self
547            .handle
548            .glyph_h_advance(glyph_id)
549            .unwrap_or(LAST_RESORT_GLYPH_ADVANCE as FractionalPixel);
550        let mut cache = self.cached_shape_data.write();
551        cache.glyph_advances.insert(glyph_id, new_width);
552        new_width
553    }

575/// A `FontGroup` is a prioritised list of fonts for a given set of font styles. It is used by
576/// `TextRun` to decide which font to render a character with. If none of the fonts listed in the
577/// styles are suitable, a fallback font may be used.
578#[derive(MallocSizeOf)]
579pub struct FontGroup {
580    descriptor: FontDescriptor,
581    families: SmallVec<[FontGroupFamily; 8]>,
582}
583
584impl FontGroup {
585    pub(crate) fn new(style: &FontStyleStruct, descriptor: FontDescriptor) -> FontGroup {
586        let families: SmallVec<[FontGroupFamily; 8]> = style
587            .font_family
588            .families
589            .iter()
590            .map(FontGroupFamily::local_or_web)
591            .collect();
592
593        FontGroup {
594            descriptor,
595            families,
596        }
597    }
598
599    /// Finds the first font, or else the first fallback font, which contains a glyph for
600    /// `codepoint`. If no such font is found, returns the first available font or fallback font
601    /// (which will cause a "glyph not found" character to be rendered). If no font at all can be
602    /// found, returns None.
603    pub fn find_by_codepoint(
604        &self,
605        font_context: &FontContext,
606        codepoint: char,
607        next_codepoint: Option<char>,
608        first_fallback: Option<FontRef>,
609        lang: Option<String>,
610    ) -> Option<FontRef> {
611        // Tab characters are converted into spaces when rendering.
612        // TODO: We should not render a tab character. Instead they should be converted into tab stops
613        // based upon the width of a space character in inline formatting contexts.
614        let codepoint = match codepoint {
615            '\t' => ' ',
616            _ => codepoint,
617        };
618
619        let options = FallbackFontSelectionOptions::new(codepoint, next_codepoint, lang);
620
621        let should_look_for_small_caps = self.descriptor.variant == font_variant_caps::T::SmallCaps &&
622            options.character.is_ascii_lowercase();
623        let font_or_synthesized_small_caps = |font: FontRef| {
624            if should_look_for_small_caps && font.synthesized_small_caps.is_some() {
625                return font.synthesized_small_caps.clone();
626            }
627            Some(font)
628        };
629
630        let font_has_glyph_and_presentation = |font: &FontRef| {
631            // Do not select this font if it goes against our emoji preference.
632            match options.presentation_preference {
633                EmojiPresentationPreference::Text if font.has_color_bitmap_or_colr_table() => {
634                    return false;
635                },
636                EmojiPresentationPreference::Emoji if !font.has_color_bitmap_or_colr_table() => {
637                    return false;
638                },
639                _ => {},
640            }
641            font.has_glyph_for(options.character)
642        };
643
644        let char_in_template =
645            |template: FontTemplateRef| template.char_in_unicode_range(options.character);
646
647        if let Some(font) = self.find(
648            font_context,
649            &char_in_template,
650            &font_has_glyph_and_presentation,
651        ) {
652            return font_or_synthesized_small_caps(font);
653        }
654
655        if let Some(ref first_fallback) = first_fallback {
656            if char_in_template(first_fallback.template.clone()) &&
657                font_has_glyph_and_presentation(first_fallback)
658            {
659                return font_or_synthesized_small_caps(first_fallback.clone());
660            }
661        }
662
663        if let Some(font) = self.find_fallback(
664            font_context,
665            options.clone(),
666            &char_in_template,
667            &font_has_glyph_and_presentation,
668        ) {
669            return font_or_synthesized_small_caps(font);
670        }
671
672        self.first(font_context)
673    }
674
675    /// Find the first available font in the group, or the first available fallback font.
676    pub fn first(&self, font_context: &FontContext) -> Option<FontRef> {
677        // From https://drafts.csswg.org/css-fonts/#first-available-font:
678        // > The first available font, used for example in the definition of font-relative lengths
679        // > such as ex or in the definition of the line-height property, is defined to be the first
680        // > font for which the character U+0020 (space) is not excluded by a unicode-range, given the
681        // > font families in the font-family list (or a user agents default font if none are
682        // > available).
683        // > Note: it does not matter whether that font actually has a glyph for the space character.
684        let space_in_template = |template: FontTemplateRef| template.char_in_unicode_range(' ');
685        let font_predicate = |_: &FontRef| true;
686        self.find(font_context, &space_in_template, &font_predicate)
687            .or_else(|| {
688                self.find_fallback(
689                    font_context,
690                    FallbackFontSelectionOptions::default(),
691                    &space_in_template,
692                    &font_predicate,
693                )
694            })
695    }
696
697    /// Attempts to find a font which matches the given `template_predicate` and `font_predicate`.
698    /// This method mutates because we may need to load new font data in the process of finding
699    /// a suitable font.
700    fn find(
701        &self,
702        font_context: &FontContext,
703        template_predicate: &impl Fn(FontTemplateRef) -> bool,
704        font_predicate: &impl Fn(&FontRef) -> bool,
705    ) -> Option<FontRef> {
706        self.families
707            .iter()
708            .flat_map(|family| family.templates(font_context, &self.descriptor))
709            .find_map(|template| {
710                template.font_if_matches(
711                    font_context,
712                    &self.descriptor,
713                    template_predicate,
714                    font_predicate,
715                )
716            })
717    }
718
719    /// Attempts to find a suitable fallback font which matches the given `template_predicate` and
720    /// `font_predicate`. The default family (i.e. "serif") will be tried first, followed by
721    /// platform-specific family names. If a `codepoint` is provided, then its Unicode block may be
722    /// used to refine the list of family names which will be tried.
723    fn find_fallback(
724        &self,
725        font_context: &FontContext,
726        options: FallbackFontSelectionOptions,
727        template_predicate: &impl Fn(FontTemplateRef) -> bool,
728        font_predicate: &impl Fn(&FontRef) -> bool,
729    ) -> Option<FontRef> {
730        iter::once(FontFamilyDescriptor::default())
731            .chain(
732                fallback_font_families(options)
733                    .into_iter()
734                    .map(|family_name| {
735                        let family = SingleFontFamily::FamilyName(FamilyName {
736                            name: family_name.into(),
737                            syntax: FontFamilyNameSyntax::Quoted,
738                        });
739                        FontFamilyDescriptor::new(family, FontSearchScope::Local)
740                    }),
741            )
742            .find_map(|family_descriptor| {
743                FontGroupFamily::from(family_descriptor)
744                    .templates(font_context, &self.descriptor)
745                    .find_map(|template| {
746                        template.font_if_matches(
747                            font_context,
748                            &self.descriptor,
749                            template_predicate,
750                            font_predicate,
751                        )
752                    })
753            })
754    }
755}

757/// A [`FontGroupFamily`] can have multiple associated `FontTemplate`s if it is a
758/// "composite face", meaning that it is defined by multiple `@font-face`
759/// declarations which vary only by their `unicode-range` descriptors. In this case,
760/// font selection will select a single member that contains the necessary unicode
761/// character. Unicode ranges are specified by the [`FontGroupFamilyTemplate::template`]
762/// member.
763#[derive(MallocSizeOf)]
764struct FontGroupFamilyTemplate {
765    #[ignore_malloc_size_of = "This measured in the FontContext template cache."]
766    template: FontTemplateRef,
767    #[ignore_malloc_size_of = "This measured in the FontContext font cache."]
768    font: OnceLock<Option<FontRef>>,
769}
770
771impl From<FontTemplateRef> for FontGroupFamilyTemplate {
772    fn from(template: FontTemplateRef) -> Self {
773        Self {
774            template,
775            font: Default::default(),
776        }
777    }
778}
779
780impl FontGroupFamilyTemplate {
781    fn font(
782        &self,
783        font_context: &FontContext,
784        font_descriptor: &FontDescriptor,
785    ) -> Option<FontRef> {
786        self.font
787            .get_or_init(|| font_context.font(self.template.clone(), font_descriptor))
788            .clone()
789    }
790
791    fn font_if_matches(
792        &self,
793        font_context: &FontContext,
794        font_descriptor: &FontDescriptor,
795        template_predicate: &impl Fn(FontTemplateRef) -> bool,
796        font_predicate: &impl Fn(&FontRef) -> bool,
797    ) -> Option<FontRef> {
798        if !template_predicate(self.template.clone()) {
799            return None;
800        }
801        self.font(font_context, font_descriptor)
802            .filter(font_predicate)
803    }
804}

806/// A `FontGroupFamily` is a single font family in a `FontGroup`. It corresponds to one of the
807/// families listed in the `font-family` CSS property. The corresponding font data is lazy-loaded,
808/// only if actually needed. A single `FontGroupFamily` can have multiple fonts, in the case that
809/// individual fonts only cover part of the Unicode range.
810#[derive(MallocSizeOf)]
811struct FontGroupFamily {
812    family_descriptor: FontFamilyDescriptor,
813    members: OnceLock<Vec<FontGroupFamilyTemplate>>,
814}
815
816impl From<FontFamilyDescriptor> for FontGroupFamily {
817    fn from(family_descriptor: FontFamilyDescriptor) -> Self {
818        Self {
819            family_descriptor,
820            members: Default::default(),
821        }
822    }
823}
824
825impl FontGroupFamily {
826    fn local_or_web(family: &SingleFontFamily) -> FontGroupFamily {
827        FontFamilyDescriptor::new(family.clone(), FontSearchScope::Any).into()
828    }
829
830    fn templates(
831        &self,
832        font_context: &FontContext,
833        font_descriptor: &FontDescriptor,
834    ) -> impl Iterator<Item = &FontGroupFamilyTemplate> {
835        self.members
836            .get_or_init(|| {
837                font_context
838                    .matching_templates(font_descriptor, &self.family_descriptor)
839                    .into_iter()
840                    .map(Into::into)
841                    .collect()
842            })
843            .iter()
844    }
845}

883/// Given a mapping array `mapping` and a value, map that value onto
884/// the value specified by the array. For instance, for FontConfig
885/// values of weights, we would map these onto the CSS [0..1000] range
886/// by creating an array as below. Values that fall between two mapped
887/// values, will be adjusted by the weighted mean.
888///
889/// ```rust
890/// let mapping = [
891///     (0., 0.),
892///     (FC_WEIGHT_REGULAR as f64, 400 as f64),
893///     (FC_WEIGHT_BOLD as f64, 700 as f64),
894///     (FC_WEIGHT_EXTRABLACK as f64, 1000 as f64),
895/// ];
896/// let mapped_weight = apply_font_config_to_style_mapping(&mapping, weight as f64);
897/// ```
898#[cfg(all(
899    any(target_os = "linux", target_os = "macos"),
900    not(target_env = "ohos")
901))]
902pub(crate) fn map_platform_values_to_style_values(mapping: &[(f64, f64)], value: f64) -> f64 {
903    if value < mapping[0].0 {
904        return mapping[0].1;
905    }
906
907    for window in mapping.windows(2) {
908        let (font_config_value_a, css_value_a) = window[0];
909        let (font_config_value_b, css_value_b) = window[1];
910
911        if value >= font_config_value_a && value <= font_config_value_b {
912            let ratio = (value - font_config_value_a) / (font_config_value_b - font_config_value_a);
913            return css_value_a + ((css_value_b - css_value_a) * ratio);
914        }
915    }
916
917    mapping[mapping.len() - 1].1
918}

920/// Computes the total advance for a glyph, taking `letter-spacing` and `word-spacing` into account.
921pub(super) fn advance_for_shaped_glyph(
922    mut advance: Au,
923    character: char,
924    options: &ShapingOptions,
925) -> Au {
926    if let Some(letter_spacing) = options.letter_spacing {
927        advance += letter_spacing;
928    };
929
930    // CSS 2.1  16.4 states that "word spacing affects each space (U+0020) and non-breaking
931    // space (U+00A0) left in the text after the white space processing rules have been
932    // applied. The effect of the property on other word-separator characters is undefined."
933    // We elect to only space the two required code points.
934    if character == ' ' || character == '\u{a0}' {
935        // https://drafts.csswg.org/css-text-3/#word-spacing-property
936        advance += options.word_spacing;
937    }
938
939    advance
940}
