20/// GlyphEntry is a port of Gecko's CompressedGlyph scheme for storing glyph data compactly.
21///
22/// In the common case (reasonable glyph advances, no offsets from the font em-box, and one glyph
23/// per character), we pack glyph advance, glyph id, and some flags into a single u32.
24///
25/// In the uncommon case (multiple glyphs per unicode character, large glyph index/advance, or
26/// glyph offsets), we pack the glyph count into GlyphEntry, and store the other glyph information
27/// in DetailedGlyphStore.
28#[derive(Clone, Copy, Debug, Deserialize, MallocSizeOf, PartialEq, Serialize)]
29pub(crate) struct GlyphEntry {
30    value: u32,
31}
32
33impl GlyphEntry {
34    fn new(value: u32) -> GlyphEntry {
35        GlyphEntry { value }
36    }
37
38    fn initial() -> GlyphEntry {
39        GlyphEntry::new(0)
40    }
41
42    // Creates a GlyphEntry for the common case
43    fn simple(id: GlyphId, advance: Au) -> GlyphEntry {
44        assert!(is_simple_glyph_id(id));
45        assert!(is_simple_advance(advance));
46
47        let id_mask = id;
48        let Au(advance) = advance;
49        let advance_mask = (advance as u32) << GLYPH_ADVANCE_SHIFT;
50
51        GlyphEntry::new(id_mask | advance_mask | FLAG_IS_SIMPLE_GLYPH)
52    }
53
54    // Create a GlyphEntry for uncommon case; should be accompanied by
55    // initialization of the actual DetailedGlyph data in DetailedGlyphStore
56    fn complex(starts_cluster: bool, starts_ligature: bool, glyph_count: usize) -> GlyphEntry {
57        assert!(glyph_count <= u16::MAX as usize);
58
59        debug!(
60            "creating complex glyph entry: starts_cluster={}, starts_ligature={}, \
61             glyph_count={}",
62            starts_cluster, starts_ligature, glyph_count
63        );
64
65        GlyphEntry::new(glyph_count as u32)
66    }
67
68    fn is_initial(&self) -> bool {
69        *self == GlyphEntry::initial()
70    }
71}

73/// The id of a particular glyph within a font
74pub(crate) type GlyphId = u32;
75
76// TODO: make this more type-safe.
77
78const FLAG_CHAR_IS_WORD_SEPARATOR: u32 = 0x40000000;
79const FLAG_IS_SIMPLE_GLYPH: u32 = 0x80000000;
80
81// glyph advance; in Au's.
82const GLYPH_ADVANCE_MASK: u32 = 0x3FFF0000;
83const GLYPH_ADVANCE_SHIFT: u32 = 16;
84const GLYPH_ID_MASK: u32 = 0x0000FFFF;
85
86// Non-simple glyphs (more than one glyph per char; missing glyph,
87// newline, tab, large advance, or nonzero x/y offsets) may have one
88// or more detailed glyphs associated with them. They are stored in a
89// side array so that there is a 1:1 mapping of GlyphEntry to
90// unicode char.
91
92// The number of detailed glyphs for this char.
93const GLYPH_COUNT_MASK: u32 = 0x0000FFFF;
94
95fn is_simple_glyph_id(id: GlyphId) -> bool {
96    (id & GLYPH_ID_MASK) == id
97}
98
99fn is_simple_advance(advance: Au) -> bool {
100    advance >= Au::zero() && {
101        let unsigned_au = advance.0 as u32;
102        (unsigned_au & (GLYPH_ADVANCE_MASK >> GLYPH_ADVANCE_SHIFT)) == unsigned_au
103    }
104}

106// Getters and setters for GlyphEntry. Setter methods are functional,
107// because GlyphEntry is immutable and only a u32 in size.
108impl GlyphEntry {
109    #[inline(always)]
110    fn advance(&self) -> Au {
111        Au::new(((self.value & GLYPH_ADVANCE_MASK) >> GLYPH_ADVANCE_SHIFT) as i32)
112    }
113
114    #[inline]
115    fn id(&self) -> GlyphId {
116        self.value & GLYPH_ID_MASK
117    }
118
119    /// True if the original character was a word separator. These include spaces
120    /// (U+0020), non-breaking spaces (U+00A0), and a few other characters
121    /// non-exhaustively listed in the specification. Other characters may map to the same
122    /// glyphs, but this function does not take mapping into account.
123    ///
124    /// See <https://drafts.csswg.org/css-text/#word-separator>.
125    fn char_is_word_separator(&self) -> bool {
126        self.has_flag(FLAG_CHAR_IS_WORD_SEPARATOR)
127    }
128
129    #[inline(always)]
130    fn set_char_is_word_separator(&mut self) {
131        self.value |= FLAG_CHAR_IS_WORD_SEPARATOR;
132    }
133
134    fn glyph_count(&self) -> u16 {
135        assert!(!self.is_simple());
136        (self.value & GLYPH_COUNT_MASK) as u16
137    }
138
139    #[inline(always)]
140    fn is_simple(&self) -> bool {
141        self.has_flag(FLAG_IS_SIMPLE_GLYPH)
142    }
143
144    #[inline(always)]
145    fn has_flag(&self, flag: u32) -> bool {
146        (self.value & flag) != 0
147    }
148}

150// Stores data for a detailed glyph, in the case that several glyphs
151// correspond to one character, or the glyph's data couldn't be packed.
152#[derive(Clone, Copy, Debug, Deserialize, MallocSizeOf, Serialize)]
153struct DetailedGlyph {
154    id: GlyphId,
155    // glyph's advance, in the text's direction (LTR or RTL)
156    advance: Au,
157    // glyph's offset from the font's em-box (from top-left)
158    offset: Point2D<Au>,
159}
160
161impl DetailedGlyph {
162    fn new(id: GlyphId, advance: Au, offset: Point2D<Au>) -> DetailedGlyph {
163        DetailedGlyph {
164            id,
165            advance,
166            offset,
167        }
168    }
169}

171#[derive(Clone, Copy, Debug, Deserialize, Eq, MallocSizeOf, PartialEq, Serialize)]
172struct DetailedGlyphRecord {
173    // source string offset/GlyphEntry offset in the TextRun
174    entry_offset: ByteIndex,
175    // offset into the detailed glyphs buffer
176    detail_offset: usize,
177}
178
179impl Ord for DetailedGlyphRecord {
180    fn cmp(&self, other: &DetailedGlyphRecord) -> Ordering {
181        self.entry_offset.cmp(&other.entry_offset)
182    }
183}
184
185impl PartialOrd for DetailedGlyphRecord {
186    fn partial_cmp(&self, other: &DetailedGlyphRecord) -> Option<Ordering> {
187        Some(self.cmp(other))
188    }
189}

191// Manages the lookup table for detailed glyphs. Sorting is deferred
192// until a lookup is actually performed; this matches the expected
193// usage pattern of setting/appending all the detailed glyphs, and
194// then querying without setting.
195#[derive(Clone, Deserialize, MallocSizeOf, Serialize)]
196struct DetailedGlyphStore {
197    // TODO(pcwalton): Allocation of this buffer is expensive. Consider a small-vector
198    // optimization.
199    detail_buffer: Vec<DetailedGlyph>,
200    // TODO(pcwalton): Allocation of this buffer is expensive. Consider a small-vector
201    // optimization.
202    detail_lookup: Vec<DetailedGlyphRecord>,
203    lookup_is_sorted: bool,
204}
205
206impl<'a> DetailedGlyphStore {
207    fn new() -> DetailedGlyphStore {
208        DetailedGlyphStore {
209            detail_buffer: vec![], // TODO: default size?
210            detail_lookup: vec![],
211            lookup_is_sorted: false,
212        }
213    }
214
215    fn add_detailed_glyphs_for_entry(&mut self, entry_offset: ByteIndex, glyphs: &[DetailedGlyph]) {
216        let entry = DetailedGlyphRecord {
217            entry_offset,
218            detail_offset: self.detail_buffer.len(),
219        };
220
221        debug!(
222            "Adding entry[off={:?}] for detailed glyphs: {:?}",
223            entry_offset, glyphs
224        );
225
226        debug_assert!(!self.detail_lookup.contains(&entry));
227        self.detail_lookup.push(entry);
228        self.detail_buffer.extend_from_slice(glyphs);
229        self.lookup_is_sorted = false;
230    }
231
232    fn detailed_glyphs_for_entry(
233        &'a self,
234        entry_offset: ByteIndex,
235        count: u16,
236    ) -> &'a [DetailedGlyph] {
237        debug!(
238            "Requesting detailed glyphs[n={}] for entry[off={:?}]",
239            count, entry_offset
240        );
241
242        // FIXME: Is this right? --pcwalton
243        // TODO: should fix this somewhere else
244        if count == 0 {
245            return &self.detail_buffer[0..0];
246        }
247
248        assert!((count as usize) <= self.detail_buffer.len());
249        assert!(self.lookup_is_sorted);
250
251        let key = DetailedGlyphRecord {
252            entry_offset,
253            detail_offset: 0, // unused
254        };
255
256        let i = self
257            .detail_lookup
258            .binary_search(&key)
259            .expect("Invalid index not found in detailed glyph lookup table!");
260        let main_detail_offset = self.detail_lookup[i].detail_offset;
261        assert!(main_detail_offset + (count as usize) <= self.detail_buffer.len());
262        // return a slice into the buffer
263        &self.detail_buffer[main_detail_offset..main_detail_offset + count as usize]
264    }
265
266    fn detailed_glyph_with_index(
267        &'a self,
268        entry_offset: ByteIndex,
269        detail_offset: u16,
270    ) -> &'a DetailedGlyph {
271        assert!((detail_offset as usize) <= self.detail_buffer.len());
272        assert!(self.lookup_is_sorted);
273
274        let key = DetailedGlyphRecord {
275            entry_offset,
276            detail_offset: 0, // unused
277        };
278
279        let i = self
280            .detail_lookup
281            .binary_search(&key)
282            .expect("Invalid index not found in detailed glyph lookup table!");
283        let main_detail_offset = self.detail_lookup[i].detail_offset;
284        assert!(main_detail_offset + (detail_offset as usize) < self.detail_buffer.len());
285        &self.detail_buffer[main_detail_offset + (detail_offset as usize)]
286    }
287
288    fn ensure_sorted(&mut self) {
289        if self.lookup_is_sorted {
290            return;
291        }
292
293        // Sorting a unique vector is surprisingly hard. The following
294        // code is a good argument for using DVecs, but they require
295        // immutable locations thus don't play well with freezing.
296
297        // Thar be dragons here. You have been warned. (Tips accepted.)
298        let mut unsorted_records: Vec<DetailedGlyphRecord> = vec![];
299        mem::swap(&mut self.detail_lookup, &mut unsorted_records);
300        let mut mut_records: Vec<DetailedGlyphRecord> = unsorted_records;
301        mut_records.sort_by(|a, b| {
302            if a < b {
303                Ordering::Less
304            } else {
305                Ordering::Greater
306            }
307        });
308        let mut sorted_records = mut_records;
309        mem::swap(&mut self.detail_lookup, &mut sorted_records);
310
311        self.lookup_is_sorted = true;
312    }
313}

345// This enum is a proxy that's provided to GlyphStore clients when iterating
346// through glyphs (either for a particular TextRun offset, or all glyphs).
347// Rather than eagerly assembling and copying glyph data, it only retrieves
348// values as they are needed from the GlyphStore, using provided offsets.
349#[derive(Clone, Copy)]
350pub enum GlyphInfo<'a> {
351    Simple(&'a GlyphStore, ByteIndex),
352    Detail(&'a GlyphStore, ByteIndex, u16),
353}
354
355impl GlyphInfo<'_> {
356    pub fn id(self) -> GlyphId {
357        match self {
358            GlyphInfo::Simple(store, entry_i) => store.entry_buffer[usize::from(entry_i)].id(),
359            GlyphInfo::Detail(store, entry_i, detail_j) => {
360                store
361                    .detail_store
362                    .detailed_glyph_with_index(entry_i, detail_j)
363                    .id
364            },
365        }
366    }
367
368    #[inline(always)]
369    pub fn advance(self) -> Au {
370        match self {
371            GlyphInfo::Simple(store, entry_i) => store.entry_buffer[usize::from(entry_i)].advance(),
372            GlyphInfo::Detail(store, entry_i, detail_j) => {
373                store
374                    .detail_store
375                    .detailed_glyph_with_index(entry_i, detail_j)
376                    .advance
377            },
378        }
379    }
380
381    #[inline]
382    pub fn offset(self) -> Option<Point2D<Au>> {
383        match self {
384            GlyphInfo::Simple(_, _) => None,
385            GlyphInfo::Detail(store, entry_i, detail_j) => Some(
386                store
387                    .detail_store
388                    .detailed_glyph_with_index(entry_i, detail_j)
389                    .offset,
390            ),
391        }
392    }
393
394    pub fn char_is_word_separator(self) -> bool {
395        let (store, entry_i) = match self {
396            GlyphInfo::Simple(store, entry_i) => (store, entry_i),
397            GlyphInfo::Detail(store, entry_i, _) => (store, entry_i),
398        };
399
400        store.char_is_word_separator(entry_i)
401    }
402}

404/// Stores the glyph data belonging to a text run.
405///
406/// Simple glyphs are stored inline in the `entry_buffer`, detailed glyphs are
407/// stored as pointers into the `detail_store`.
408///
409/// ~~~ascii
410/// +- GlyphStore --------------------------------+
411/// |               +---+---+---+---+---+---+---+ |
412/// | entry_buffer: |   | s |   | s |   | s | s | |  d = detailed
413/// |               +-|-+---+-|-+---+-|-+---+---+ |  s = simple
414/// |                 |       |       |           |
415/// |                 |   +---+-------+           |
416/// |                 |   |                       |
417/// |               +-V-+-V-+                     |
418/// | detail_store: | d | d |                     |
419/// |               +---+---+                     |
420/// +---------------------------------------------+
421/// ~~~
422#[derive(Clone, Deserialize, MallocSizeOf, Serialize)]
423pub struct GlyphStore {
424    // TODO(pcwalton): Allocation of this buffer is expensive. Consider a small-vector
425    // optimization.
426    /// A buffer of glyphs within the text run, in the order in which they
427    /// appear in the input text.
428    /// Any changes will also need to be reflected in
429    /// transmute_entry_buffer_to_u32_buffer().
430    entry_buffer: Vec<GlyphEntry>,
431    /// A store of the detailed glyph data. Detailed glyphs contained in the
432    /// `entry_buffer` point to locations in this data structure.
433    detail_store: DetailedGlyphStore,
434
435    /// A cache of the advance of the entire glyph store.
436    total_advance: Au,
437
438    /// A cache of the number of word separators in the entire glyph store.
439    /// See <https://drafts.csswg.org/css-text/#word-separator>.
440    total_word_separators: usize,
441
442    /// Used to check if fast path should be used in glyph iteration.
443    has_detailed_glyphs: bool,
444
445    /// Whether or not this glyph store contains only glyphs for whitespace.
446    is_whitespace: bool,
447
448    /// Whether or not this glyph store ends with whitespace glyphs.
449    /// Typically whitespace glyphs are placed in a separate store,
450    /// but that may not be the case with `white-space: break-spaces`.
451    ends_with_whitespace: bool,
452
453    /// Whether or not this glyph store contains only a single glyph for a single
454    /// preserved newline.
455    is_single_preserved_newline: bool,
456
457    is_rtl: bool,
458}

516    #[inline(never)]
517    fn cache_total_advance_and_word_separators(&mut self) {
518        let mut total_advance = Au::zero();
519        let mut total_word_separators = 0;
520        for glyph in self.iter_glyphs_for_byte_range(TextByteRange::new(ByteIndex(0), self.len())) {
521            total_advance += glyph.advance();
522            if glyph.char_is_word_separator() {
523                total_word_separators += 1;
524            }
525        }
526        self.total_advance = total_advance;
527        self.total_word_separators = total_word_separators;
528    }

530    /// Adds a single glyph.
531    pub(crate) fn add_glyph_for_byte_index(
532        &mut self,
533        i: ByteIndex,
534        character: char,
535        data: &GlyphData,
536    ) {
537        let glyph_is_compressible = is_simple_glyph_id(data.id) &&
538            is_simple_advance(data.advance) &&
539            data.offset == Point2D::zero() &&
540            data.cluster_start; // others are stored in detail buffer
541
542        debug_assert!(data.ligature_start); // can't compress ligature continuation glyphs.
543        debug_assert!(i < self.len());
544
545        let mut entry = if glyph_is_compressible {
546            GlyphEntry::simple(data.id, data.advance)
547        } else {
548            let glyph = &[DetailedGlyph::new(data.id, data.advance, data.offset)];
549            self.has_detailed_glyphs = true;
550            self.detail_store.add_detailed_glyphs_for_entry(i, glyph);
551            GlyphEntry::complex(data.cluster_start, data.ligature_start, 1)
552        };
553
554        // This list is taken from the non-exhaustive list of word separator characters in
555        // the CSS Text Module Level 3 Spec:
556        // See https://drafts.csswg.org/css-text/#word-separator
557        if matches!(
558            character,
559            ' ' |
560            '\u{00A0}' | // non-breaking space
561            '\u{1361}' | // Ethiopic word space
562            '\u{10100}' | // Aegean word separator
563            '\u{10101}' | // Aegean word separator
564            '\u{1039F}' | // Ugartic word divider
565            '\u{1091F}' // Phoenician word separator
566        ) {
567            entry.set_char_is_word_separator();
568        }
569
570        self.entry_buffer[usize::from(i)] = entry;
571    }
572
573    pub(crate) fn add_glyphs_for_byte_index(
574        &mut self,
575        i: ByteIndex,
576        data_for_glyphs: &[GlyphData],
577    ) {
578        assert!(i < self.len());
579        assert!(!data_for_glyphs.is_empty());
580
581        let glyph_count = data_for_glyphs.len();
582
583        let first_glyph_data = data_for_glyphs[0];
584        let glyphs_vec: Vec<DetailedGlyph> = (0..glyph_count)
585            .map(|i| {
586                DetailedGlyph::new(
587                    data_for_glyphs[i].id,
588                    data_for_glyphs[i].advance,
589                    data_for_glyphs[i].offset,
590                )
591            })
592            .collect();
593
594        self.has_detailed_glyphs = true;
595        self.detail_store
596            .add_detailed_glyphs_for_entry(i, &glyphs_vec);
597
598        let entry = GlyphEntry::complex(
599            first_glyph_data.cluster_start,
600            first_glyph_data.ligature_start,
601            glyph_count,
602        );
603
604        debug!(
605            "Adding multiple glyphs[idx={:?}, count={}]: {:?}",
606            i, glyph_count, entry
607        );
608
609        self.entry_buffer[usize::from(i)] = entry;
610    }

612    #[inline]
613    pub fn iter_glyphs_for_byte_range(
614        &self,
615        range: TextByteRange,
616    ) -> impl Iterator<Item = GlyphInfo<'_>> + use<'_> {
617        if range.start >= self.len() {
618            panic!("iter_glyphs_for_range: range.begin beyond length!");
619        }
620        if range.end > self.len() {
621            panic!("iter_glyphs_for_range: range.end beyond length!");
622        }
623
624        let range_it = if self.is_rtl {
625            Either::Left(range.rev())
626        } else {
627            Either::Right(range)
628        };
629
630        range_it.into_iter().flat_map(move |range_idx| {
631            let entry = self.entry_buffer[usize::from(range_idx)];
632            let result = if entry.is_simple() {
633                Either::Left(once(GlyphInfo::Simple(self, range_idx)))
634            } else {
635                // Slow path for complex glyphs
636                let glyphs = self
637                    .detail_store
638                    .detailed_glyphs_for_entry(range_idx, entry.glyph_count());
639
640                let complex_glyph_range =
641                    TextByteRange::new(ByteIndex::zero(), ByteIndex(glyphs.len() as isize));
642                Either::Right(complex_glyph_range.map(move |i| {
643                    GlyphInfo::Detail(self, range_idx, usize::from(i) as u16 /* ??? */)
644                }))
645            };
646
647            result.into_iter()
648        })
649    }

651    #[inline]
652    pub fn advance_for_byte_range(&self, range: TextByteRange, extra_word_spacing: Au) -> Au {
653        if range.start == ByteIndex(0) && range.end == self.len() {
654            self.total_advance + extra_word_spacing * (self.total_word_separators as i32)
655        } else {
656            self.advance_for_byte_range_simple_glyphs(range, extra_word_spacing)
657        }
658    }
659
660    #[inline]
661    pub(crate) fn advance_for_byte_range_simple_glyphs(
662        &self,
663        range: TextByteRange,
664        extra_word_spacing: Au,
665    ) -> Au {
666        self.iter_glyphs_for_byte_range(range)
667            .map(|glyph| {
668                if glyph.char_is_word_separator() {
669                    glyph.advance() + extra_word_spacing
670                } else {
671                    glyph.advance()
672                }
673            })
674            .sum()
675    }

677    pub(crate) fn char_is_word_separator(&self, i: ByteIndex) -> bool {
678        assert!(i < self.len());
679        self.entry_buffer[usize::from(i)].char_is_word_separator()
680    }
681}
682
683impl fmt::Debug for GlyphStore {
684    fn fmt(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
685        writeln!(formatter, "GlyphStore:")?;
686        let mut detailed_buffer = self.detail_store.detail_buffer.iter();
687        for entry in self.entry_buffer.iter() {
688            if entry.is_simple() {
689                writeln!(
690                    formatter,
691                    "  simple id={:?} advance={:?}",
692                    entry.id(),
693                    entry.advance()
694                )?;
695                continue;
696            }
697            if entry.is_initial() {
698                continue;
699            }
700            write!(formatter, "  complex...")?;
701            if detailed_buffer.next().is_none() {
702                continue;
703            }
704            writeln!(
705                formatter,
706                "  detailed id={:?} advance={:?}",
707                entry.id(),
708                entry.advance()
709            )?;
710        }
711        Ok(())
712    }
713}

723}
