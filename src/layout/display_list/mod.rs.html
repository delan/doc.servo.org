169    pub(crate) fn build(
170        stacking_context_tree: &mut StackingContextTree,
171        fragment_tree: &FragmentTree,
172        image_resolver: Arc<ImageResolver>,
173        device_pixel_ratio: Scale<f32, StyloCSSPixel, StyloDevicePixel>,
174        highlighted_dom_node: Option<OpaqueNode>,
175        debug: &DiagnosticsLogging,
176        lcp_candidate_collector: Option<&mut LargestContentfulPaintCandidateCollector>,
177    ) -> BuiltDisplayList {
178        // Build the rest of the display list which inclues all of the WebRender primitives.
179        let paint_info = &mut stacking_context_tree.paint_info;
180        let pipeline_id = paint_info.pipeline_id;
181        let mut webrender_display_list_builder =
182            webrender_api::DisplayListBuilder::new(pipeline_id);
183        webrender_display_list_builder.begin();
184
185        // `dump_serialized_display_list` doesn't actually print anything. It sets up
186        // the display list for printing the serialized version when `finalize()` is called.
187        // We need to call this before adding any display items so that they are printed
188        // during `finalize()`.
189        if debug.display_list {
190            webrender_display_list_builder.dump_serialized_display_list();
191        }
192
193        let _span = profile_traits::trace_span!("DisplayListBuilder::build").entered();
194        let mut builder = DisplayListBuilder {
195            current_scroll_node_id: paint_info.root_reference_frame_id,
196            current_reference_frame_scroll_node_id: paint_info.root_reference_frame_id,
197            current_clip_id: ClipId::INVALID,
198            webrender_display_list_builder: &mut webrender_display_list_builder,
199            paint_info,
200            inspector_highlight: highlighted_dom_node.map(InspectorHighlight::for_node),
201            paint_body_background: true,
202            clip_map: Default::default(),
203            image_resolver,
204            device_pixel_ratio,
205            lcp_candidate_collector,
206        };
207
208        builder.add_all_spatial_nodes();
209
210        for clip in stacking_context_tree.clip_store.0.iter() {
211            builder.add_clip_to_display_list(clip);
212        }
213
214        // Add a single hit test that covers the entire viewport, so that WebRender knows
215        // which pipeline it hits when doing hit testing.
216        let pipeline_id = builder.paint_info.pipeline_id;
217        let viewport_size = builder.paint_info.viewport_details.size;
218        let viewport_rect = LayoutRect::from_size(viewport_size.cast_unit());
219        builder.wr().push_hit_test(
220            viewport_rect,
221            ClipChainId::INVALID,
222            SpatialId::root_reference_frame(pipeline_id),
223            PrimitiveFlags::default(),
224            (0, 0), /* tag */
225        );
226
227        // Paint the canvasâ€™ background (if any) before/under everything else
228        stacking_context_tree
229            .root_stacking_context
230            .build_canvas_background_display_list(&mut builder, fragment_tree);
231        stacking_context_tree
232            .root_stacking_context
233            .build_display_list(&mut builder);
234        builder.paint_dom_inspector_highlight();
235
236        webrender_display_list_builder.end().1
237    }

265    pub(crate) fn add_all_spatial_nodes(&mut self) {
266        // A count of the number of SpatialTree nodes pushed to the WebRender display
267        // list. This is merely to ensure that the currently-unused SpatialTreeItemKey
268        // produced for every SpatialTree node is unique.
269        let mut spatial_tree_count = 0;
270        let mut scroll_tree = std::mem::take(&mut self.paint_info.scroll_tree);
271        let mut mapping = Vec::with_capacity(scroll_tree.nodes.len());
272
273        mapping.push(SpatialId::root_reference_frame(self.pipeline_id()));
274        mapping.push(SpatialId::root_scroll_node(self.pipeline_id()));
275
276        let pipeline_id = self.pipeline_id();
277        let pipeline_tag = ((pipeline_id.0 as u64) << 32) | pipeline_id.1 as u64;
278
279        for node in scroll_tree.nodes.iter().skip(2) {
280            let parent_scroll_node_id = node
281                .parent
282                .expect("Should have already added root reference frame");
283            let parent_spatial_node_id = mapping
284                .get(parent_scroll_node_id.index)
285                .expect("Should add spatial nodes to display list in order");
286
287            // Produce a new SpatialTreeItemKey. This is currently unused by WebRender,
288            // but has to be unique to the entire scene.
289            spatial_tree_count += 1;
290            let spatial_tree_item_key = SpatialTreeItemKey::new(pipeline_tag, spatial_tree_count);
291
292            mapping.push(match &node.info {
293                SpatialTreeNodeInfo::ReferenceFrame(info) => {
294                    let spatial_id = self.wr().push_reference_frame(
295                        info.origin,
296                        *parent_spatial_node_id,
297                        info.transform_style,
298                        PropertyBinding::Value(*info.transform.to_transform()),
299                        info.kind,
300                        spatial_tree_item_key,
301                    );
302                    self.wr().pop_reference_frame();
303                    spatial_id
304                },
305                SpatialTreeNodeInfo::Scroll(info) => {
306                    self.wr().define_scroll_frame(
307                        *parent_spatial_node_id,
308                        info.external_id,
309                        info.content_rect,
310                        info.clip_rect,
311                        LayoutVector2D::zero(), /* external_scroll_offset */
312                        0,                      /* scroll_offset_generation */
313                        wr::HasScrollLinkedEffect::No,
314                        spatial_tree_item_key,
315                    )
316                },
317                SpatialTreeNodeInfo::Sticky(info) => {
318                    self.wr().define_sticky_frame(
319                        *parent_spatial_node_id,
320                        info.frame_rect,
321                        info.margins,
322                        info.vertical_offset_bounds,
323                        info.horizontal_offset_bounds,
324                        LayoutVector2D::zero(), /* previously_applied_offset */
325                        spatial_tree_item_key,
326                        None, /* transform */
327                    )
328                },
329            });
330        }
331
332        scroll_tree.update_mapping(mapping);
333        self.paint_info.scroll_tree = scroll_tree;
334    }

336    /// Add the given [`Clip`] to the WebRender display list and create a mapping from
337    /// its [`ClipId`] to a WebRender [`ClipChainId`]. This happens:
338    ///  - When WebRender display list construction starts: All clips created during the
339    ///    `StackingContextTree` construction are added in one batch. These clips are used
340    ///    for things such as `overflow: scroll` elements.
341    ///  - When a clip is added during WebRender display list construction for individual
342    ///    items. In that case, this is called by [`Self::maybe_create_clip`].
343    pub(crate) fn add_clip_to_display_list(&mut self, clip: &Clip) -> ClipChainId {
344        assert_eq!(
345            clip.id.0,
346            self.clip_map.len(),
347            "Clips should be added in order"
348        );
349
350        let spatial_id = self.spatial_id(clip.parent_scroll_node_id);
351        let new_clip_id = if clip.radii.is_zero() {
352            self.wr().define_clip_rect(spatial_id, clip.rect)
353        } else {
354            self.wr().define_clip_rounded_rect(
355                spatial_id,
356                ComplexClipRegion {
357                    rect: clip.rect,
358                    radii: clip.radii,
359                    mode: ClipMode::Clip,
360                },
361            )
362        };
363
364        // WebRender has two different ways of expressing "no clip." ClipChainId::INVALID should be
365        // used for primitives, but `None` is used for stacking contexts and clip chains. We convert
366        // to the `Option<ClipChainId>` representation here. Just passing Some(ClipChainId::INVALID)
367        // leads to a crash.
368        let parent_clip_chain_id = match self.clip_chain_id(clip.parent_clip_id) {
369            ClipChainId::INVALID => None,
370            parent => Some(parent),
371        };
372        let clip_chain_id = self
373            .wr()
374            .define_clip_chain(parent_clip_chain_id, [new_clip_id]);
375        self.clip_map.push(clip_chain_id);
376        clip_chain_id
377    }

418    fn paint_dom_inspector_highlight(&mut self) {
419        let Some(highlight) = self
420            .inspector_highlight
421            .take()
422            .and_then(|highlight| highlight.state)
423        else {
424            return;
425        };
426
427        const CONTENT_BOX_HIGHLIGHT_COLOR: webrender_api::ColorF = webrender_api::ColorF {
428            r: 0.23,
429            g: 0.7,
430            b: 0.87,
431            a: 0.5,
432        };
433
434        const PADDING_BOX_HIGHLIGHT_COLOR: webrender_api::ColorF = webrender_api::ColorF {
435            r: 0.49,
436            g: 0.3,
437            b: 0.7,
438            a: 0.5,
439        };
440
441        const BORDER_BOX_HIGHLIGHT_COLOR: webrender_api::ColorF = webrender_api::ColorF {
442            r: 0.2,
443            g: 0.2,
444            b: 0.2,
445            a: 0.5,
446        };
447
448        const MARGIN_BOX_HIGHLIGHT_COLOR: webrender_api::ColorF = webrender_api::ColorF {
449            r: 1.,
450            g: 0.93,
451            b: 0.,
452            a: 0.5,
453        };
454
455        // Highlight content box
456        let content_box = highlight.content_box.to_webrender();
457        let properties = wr::CommonItemProperties {
458            clip_rect: content_box,
459            spatial_id: highlight.spatial_id,

461            flags: wr::PrimitiveFlags::default(),
462        };

464        self.wr()
465            .push_rect(&properties, content_box, CONTENT_BOX_HIGHLIGHT_COLOR);

467        // Highlight margin, border and padding
468        if let Some(box_fragment) = highlight.maybe_box_fragment {
469            let mut paint_highlight =
470                |color: webrender_api::ColorF,
471                 fragment_relative_bounds: PhysicalRect<Au>,
472                 widths: webrender_api::units::LayoutSideOffsets| {
473                    if widths.is_zero() {
474                        return;
475                    }
476
477                    let bounds = box_fragment
478                        .borrow()
479                        .offset_by_containing_block(&fragment_relative_bounds)
480                        .to_webrender();
481
482                    // We paint each highlighted area as if it was a border for simplicity
483                    let border_style = wr::BorderSide {
484                        color,
485                        style: wr::BorderStyle::Solid,
486                    };
487
488                    let details = wr::BorderDetails::Normal(wr::NormalBorder {
489                        top: border_style,
490                        right: border_style,
491                        bottom: border_style,
492                        left: border_style,
493                        radius: webrender_api::BorderRadius::default(),
494                        do_aa: true,
495                    });
496
497                    let common = wr::CommonItemProperties {
498                        clip_rect: bounds,
499                        spatial_id: highlight.spatial_id,
500                        clip_chain_id: highlight.clip_chain_id,
501                        flags: wr::PrimitiveFlags::default(),
502                    };
503                    self.wr().push_border(&common, bounds, widths, details)
504                };
505
506            let box_fragment = box_fragment.borrow();
507            paint_highlight(
508                PADDING_BOX_HIGHLIGHT_COLOR,
509                box_fragment.padding_rect(),
510                box_fragment.padding.to_webrender(),
511            );
512            paint_highlight(
513                BORDER_BOX_HIGHLIGHT_COLOR,
514                box_fragment.border_rect(),
515                box_fragment.border.to_webrender(),
516            );
517            paint_highlight(
518                MARGIN_BOX_HIGHLIGHT_COLOR,
519                box_fragment.margin_rect(),
520                box_fragment.margin.to_webrender(),
521            );
522        }
523    }

549impl InspectorHighlight {
550    fn register_fragment_of_highlighted_dom_node(
551        &mut self,
552        fragment: &Fragment,
553        spatial_id: SpatialId,
554        clip_chain_id: ClipChainId,
555        containing_block: &PhysicalRect<Au>,
556    ) {
557        let state = self.state.get_or_insert(HighlightTraversalState {
558            content_box: euclid::Rect::zero(),
559            spatial_id,
560            clip_chain_id,
561            maybe_box_fragment: None,
562        });
563
564        // We expect all fragments generated by one node to be in the same scroll tree node and clip node
565        debug_assert_eq!(spatial_id, state.spatial_id);
566        if clip_chain_id != ClipChainId::INVALID && state.clip_chain_id != ClipChainId::INVALID {
567            debug_assert_eq!(
568                clip_chain_id, state.clip_chain_id,
569                "Fragments of the same node must either have no clip chain or the same one"
570            );
571        }
572
573        let Some(fragment_relative_rect) = fragment.base().map(|base| base.rect) else {
574            return;
575        };
576        state.maybe_box_fragment = match fragment {
577            Fragment::Box(fragment) | Fragment::Float(fragment) => Some(fragment.clone()),
578            _ => None,
579        };
580
581        state.content_box = state
582            .content_box
583            .union(&fragment_relative_rect.translate(containing_block.origin.to_vector()));
584    }

587impl Fragment {
588    pub(crate) fn build_display_list(
589        &self,
590        builder: &mut DisplayListBuilder,
591        containing_block: &PhysicalRect<Au>,
592        section: StackingContextSection,
593        is_hit_test_for_scrollable_overflow: bool,
594        is_collapsed_table_borders: bool,
595        text_decorations: &Arc<Vec<FragmentTextDecoration>>,
596    ) {
597        let spatial_id = builder.spatial_id(builder.current_scroll_node_id);
598        let clip_chain_id = builder.clip_chain_id(builder.current_clip_id);
599        if let Some(inspector_highlight) = &mut builder.inspector_highlight {
600            if self.tag() == Some(inspector_highlight.tag) {
601                inspector_highlight.register_fragment_of_highlighted_dom_node(
602                    self,
603                    spatial_id,
604                    clip_chain_id,
605                    containing_block,
606                );
607            }
608        }
609
610        match self {
611            Fragment::Box(box_fragment) | Fragment::Float(box_fragment) => {
612                let box_fragment = &*box_fragment.borrow();
613                match box_fragment.style().get_inherited_box().visibility {
614                    Visibility::Visible => BuilderForBoxFragment::new(
615                        box_fragment,
616                        containing_block,
617                        is_hit_test_for_scrollable_overflow,
618                        is_collapsed_table_borders,
619                    )
620                    .build(builder, section),
621                    Visibility::Hidden => (),
622                    Visibility::Collapse => (),
623                }
624            },
625            Fragment::AbsoluteOrFixedPositioned(_) | Fragment::Positioning(_) => {},
626            Fragment::Image(image) => {
627                let image = image.borrow();
628                let style = image.base.style();
629                match style.get_inherited_box().visibility {
630                    Visibility::Visible => {
631                        builder.mark_is_contentful();
632
633                        let image_rendering =
634                            style.get_inherited_box().image_rendering.to_webrender();
635                        let rect = image
636                            .base
637                            .rect
638                            .translate(containing_block.origin.to_vector())
639                            .to_webrender();
640                        let clip = image
641                            .clip
642                            .translate(containing_block.origin.to_vector())
643                            .to_webrender();
644                        let common = builder.common_properties(clip, &style);
645
646                        if let Some(image_key) = image.image_key {
647                            builder.wr().push_image(
648                                &common,
649                                rect,
650                                image_rendering,
651                                wr::AlphaType::PremultipliedAlpha,
652                                image_key,
653                                wr::ColorF::WHITE,
654                            );
655                        }

658                            self.build_display_list_for_broken_image_border(
659                                builder,
660                                containing_block,
661                                &common,
662                            );
663                        }
664
665                        let lcp_candidate_id = image
666                            .base
667                            .tag
668                            .map(|tag| LCPCandidateID(tag.node.id()))
669                            .unwrap_or(LCPCandidateID(0));

671                            LargestContentfulPaintType::Image,
672                            lcp_candidate_id,
673                            common.clip_rect,
674                            rect,
675                        );
676                    },
677                    Visibility::Hidden => (),
678                    Visibility::Collapse => (),
679                }
680            },
681            Fragment::IFrame(iframe) => {

683                let style = iframe.base.style();

685                    Visibility::Visible => {

688                            .base
689                            .rect
690                            .translate(containing_block.origin.to_vector());
691
692                        let common = builder.common_properties(rect.to_webrender(), &style);
693                        builder.wr().push_iframe(
694                            rect.to_webrender(),
695                            common.clip_rect,
696                            &wr::SpaceAndClipInfo {
697                                spatial_id: common.spatial_id,
698                                clip_chain_id: common.clip_chain_id,
699                            },
700                            iframe.pipeline_id.into(),
701                            true,
702                        );
703                    },
704                    Visibility::Hidden => (),
705                    Visibility::Collapse => (),
706                }
707            },
708            Fragment::Text(text) => {
709                let text = &*text.borrow();
710                match text.base.style().get_inherited_box().visibility {
711                    Visibility::Visible => self.build_display_list_for_text_fragment(
712                        text,
713                        builder,
714                        containing_block,
715                        text_decorations,
716                    ),
717                    Visibility::Hidden => (),
718                    Visibility::Collapse => (),
719                }
720            },
721        }

724    fn build_display_list_for_text_fragment(
725        &self,
726        fragment: &TextFragment,
727        builder: &mut DisplayListBuilder,
728        containing_block: &PhysicalRect<Au>,
729        text_decorations: &Arc<Vec<FragmentTextDecoration>>,
730    ) {
731        // NB: The order of painting text components (CSS Text Decoration Module Level 3) is:
732        // shadows, underline, overline, text, text-emphasis, and then line-through.
733
734        builder.mark_is_contentful();
735
736        let rect = fragment
737            .base
738            .rect
739            .translate(containing_block.origin.to_vector());
740        let mut baseline_origin = rect.origin;
741        baseline_origin.y += fragment.font_metrics.ascent;
742        let include_whitespace =
743            fragment.has_selection() || text_decorations.iter().any(|item| !item.line.is_empty());
744
745        let glyphs = glyphs(
746            &fragment.glyphs,
747            baseline_origin,
748            fragment.justification_adjustment,
749            include_whitespace,
750        );
751        if glyphs.is_empty() {
752            return;
753        }
754
755        let parent_style = fragment.base.style();
756        let color = parent_style.clone_color();
757        let font_metrics = &fragment.font_metrics;
758        let dppx = builder.device_pixel_ratio.get();
759        let common = builder.common_properties(rect.to_webrender(), &parent_style);
760
761        // Shadows. According to CSS-BACKGROUNDS, text shadows render in *reverse* order (front to
762        // back).
763        let shadows = &parent_style.get_inherited_text().text_shadow;
764        for shadow in shadows.0.iter().rev() {
765            builder.wr().push_shadow(
766                &wr::SpaceAndClipInfo {
767                    spatial_id: common.spatial_id,
768                    clip_chain_id: common.clip_chain_id,
769                },
770                wr::Shadow {
771                    offset: LayoutVector2D::new(shadow.horizontal.px(), shadow.vertical.px()),
772                    color: rgba(shadow.color.resolve_to_absolute(&color)),
773                    blur_radius: shadow.blur.px(),
774                },
775                true, /* should_inflate */
776            );
777        }
778
779        for text_decoration in text_decorations.iter() {
780            if text_decoration.line.contains(TextDecorationLine::UNDERLINE) {
781                let mut rect = rect;
782                rect.origin.y += font_metrics.ascent - font_metrics.underline_offset;
783                rect.size.height =
784                    Au::from_f32_px(font_metrics.underline_size.to_nearest_pixel(dppx));
785
786                self.build_display_list_for_text_decoration(
787                    &parent_style,
788                    builder,
789                    &rect,
790                    text_decoration,
791                    TextDecorationLine::UNDERLINE,
792                );
793            }
794        }
795
796        for text_decoration in text_decorations.iter() {
797            if text_decoration.line.contains(TextDecorationLine::OVERLINE) {
798                let mut rect = rect;
799                rect.size.height =
800                    Au::from_f32_px(font_metrics.underline_size.to_nearest_pixel(dppx));
801                self.build_display_list_for_text_decoration(
802                    &parent_style,
803                    builder,
804                    &rect,
805                    text_decoration,
806                    TextDecorationLine::OVERLINE,
807                );
808            }
809        }
810
811        // TODO: This caret/text selection implementation currently does not account for vertical text
812        // and RTL text properly.
813        if let Some(range) = fragment.selection_range.clone() {
814            let baseline_origin = rect.origin;
815            if !range.is_empty() {
816                let start = glyphs_advance_by_index(
817                    &fragment.glyphs,
818                    range.start,
819                    baseline_origin,
820                    fragment.justification_adjustment,
821                );
822
823                let end = glyphs_advance_by_index(
824                    &fragment.glyphs,
825                    range.end,
826                    baseline_origin,
827                    fragment.justification_adjustment,
828                );
829
830                let selection_rect = LayoutRect::new(
831                    Point2D::new(start.x.to_f32_px(), containing_block.min_y().to_f32_px()),
832                    Point2D::new(end.x.to_f32_px(), containing_block.max_y().to_f32_px()),
833                );
834                if let Some(selection_color) = fragment
835                    .selected_style
836                    .borrow()
837                    .clone_background_color()
838                    .as_absolute()
839                {
840                    let selection_common = builder.common_properties(selection_rect, &parent_style);
841                    builder.wr().push_rect(
842                        &selection_common,
843                        selection_rect,
844                        rgba(*selection_color),
845                    );
846                }
847            } else {
848                let insertion_point = glyphs_advance_by_index(
849                    &fragment.glyphs,
850                    range.start,
851                    baseline_origin,
852                    fragment.justification_adjustment,
853                );
854
855                let insertion_point_rect = LayoutRect::new(
856                    Point2D::new(
857                        insertion_point.x.to_f32_px(),
858                        containing_block.min_y().to_f32_px(),
859                    ),
860                    Point2D::new(
861                        insertion_point.x.to_f32_px() + INSERTION_POINT_LOGICAL_WIDTH.to_f32_px(),
862                        containing_block.max_y().to_f32_px(),
863                    ),
864                );
865                let insertion_point_common =
866                    builder.common_properties(insertion_point_rect, &parent_style);
867                // TODO: The color of the caret is currently hardcoded to the text color.
868                // We should be retrieving the caret color from the style properly.
869                builder
870                    .wr()
871                    .push_rect(&insertion_point_common, insertion_point_rect, rgba(color));
872            }
873        }
874
875        builder.wr().push_text(
876            &common,
877            rect.to_webrender(),
878            &glyphs,
879            fragment.font_key,
880            rgba(color),
881            None,
882        );
883
884        for text_decoration in text_decorations.iter() {
885            if text_decoration
886                .line
887                .contains(TextDecorationLine::LINE_THROUGH)
888            {
889                let mut rect = rect;
890                rect.origin.y += font_metrics.ascent - font_metrics.strikeout_offset;
891                rect.size.height =
892                    Au::from_f32_px(font_metrics.strikeout_size.to_nearest_pixel(dppx));
893                self.build_display_list_for_text_decoration(
894                    &parent_style,
895                    builder,
896                    &rect,
897                    text_decoration,
898                    TextDecorationLine::LINE_THROUGH,
899                );
900            }
901        }
902
903        if !shadows.0.is_empty() {
904            builder.wr().pop_all_shadows();
905        }
906    }

908    fn build_display_list_for_text_decoration(
909        &self,
910        parent_style: &ServoArc<ComputedValues>,
911        builder: &mut DisplayListBuilder,
912        rect: &PhysicalRect<Au>,
913        text_decoration: &FragmentTextDecoration,
914        line: TextDecorationLine,
915    ) {
916        if text_decoration.style == ComputedTextDecorationStyle::MozNone {
917            return;
918        }
919
920        let mut rect = rect.to_webrender();
921        let line_thickness = rect.height().ceil();
922
923        if text_decoration.style == ComputedTextDecorationStyle::Wavy {
924            rect = rect.inflate(0.0, line_thickness * 1.0);
925        }
926
927        let common_properties = builder.common_properties(rect, parent_style);
928        builder.wr().push_line(
929            &common_properties,
930            &rect,
931            line_thickness,
932            wr::LineOrientation::Horizontal,
933            &rgba(text_decoration.color),
934            text_decoration.style.to_webrender(),
935        );
936
937        if text_decoration.style == TextDecorationStyle::Double {
938            let half_height = (rect.height() / 2.0).floor().max(1.0);
939            let y_offset = match line {
940                TextDecorationLine::OVERLINE => -rect.height() - half_height,
941                _ => rect.height() + half_height,
942            };
943            let rect = rect.translate(Vector2D::new(0.0, y_offset));
944            let common_properties = builder.common_properties(rect, parent_style);
945            builder.wr().push_line(
946                &common_properties,
947                &rect,
948                line_thickness,
949                wr::LineOrientation::Horizontal,
950                &rgba(text_decoration.color),
951                text_decoration.style.to_webrender(),
952            );
953        }
954    }

1155    fn build_background_for_painter(

1177    fn build_background(&mut self, builder: &mut DisplayListBuilder) {
1178        let flags = self.fragment.base.flags;
1179
1180        // The root element's background is painted separately as it might inherit the `<body>`'s
1181        // background.
1182        if flags.intersects(FragmentFlags::IS_ROOT_ELEMENT) {
1183            return;
1184        }
1185        // If the `<body>` background was inherited by the root element, don't paint it again here.
1186        if !builder.paint_body_background &&
1187            flags.intersects(FragmentFlags::IS_BODY_ELEMENT_OF_HTML_ELEMENT_ROOT)
1188        {
1189            return;
1190        }
1191
1192        // If this BoxFragment does not paint a background, do nothing.
1193        if let BackgroundMode::None = self.fragment.background_mode {
1194            return;
1195        }
1196
1197        // Paint all extra backgrounds for this BoxFragment. These are painted first, as that's
1198        // the order that they are expected to be painted for table cells (where this feature
1199        // is used).
1200        if let BackgroundMode::Extra(ref extra_backgrounds) = self.fragment.background_mode {
1201            for extra_background in extra_backgrounds {
1202                let positioning_area = extra_background.rect;
1203                let painter = BackgroundPainter {
1204                    style: &extra_background.style.borrow_mut(),
1205                    painting_area_override: None,
1206                    positioning_area_override: Some(
1207                        positioning_area
1208                            .translate(self.containing_block.origin.to_vector())
1209                            .to_webrender(),
1210                    ),
1211                };
1212                self.build_background_for_painter(builder, &painter);
1213            }
1214        }
1215
1216        let painter = BackgroundPainter {
1217            style: &self.fragment.style(),
1218            painting_area_override: None,
1219            positioning_area_override: None,
1220        };
1221        self.build_background_for_painter(builder, &painter);
1222    }

1224    fn build_background_image(
1225        &mut self,
1226        builder: &mut DisplayListBuilder,
1227        painter: &BackgroundPainter,
1228    ) {
1229        let style = painter.style;
1230        let b = style.get_background();
1231        let node = self.fragment.base.tag.map(|tag| tag.node);
1232        // Reverse because the property is top layer first, we want to paint bottom layer first.
1233        for (index, image) in b.background_image.0.iter().enumerate().rev() {
1234            match builder.image_resolver.resolve_image(node, image) {
1235                Err(_) => {},
1236                Ok(ResolvedImage::Gradient(gradient)) => {
1237                    let intrinsic = NaturalSizes::empty();
1238                    let Some(layer) =
1239                        &background::layout_layer(self, painter, builder, index, intrinsic)
1240                    else {
1241                        continue;
1242                    };
1243
1244                    match gradient::build(style, gradient, layer.tile_size, builder) {
1245                        WebRenderGradient::Linear(linear_gradient) => builder.wr().push_gradient(
1246                            &layer.common,
1247                            layer.bounds,
1248                            linear_gradient,
1249                            layer.tile_size,
1250                            layer.tile_spacing,
1251                        ),
1252                        WebRenderGradient::Radial(radial_gradient) => {
1253                            builder.wr().push_radial_gradient(
1254                                &layer.common,
1255                                layer.bounds,
1256                                radial_gradient,
1257                                layer.tile_size,
1258                                layer.tile_spacing,
1259                            )
1260                        },
1261                        WebRenderGradient::Conic(conic_gradient) => {
1262                            builder.wr().push_conic_gradient(
1263                                &layer.common,
1264                                layer.bounds,
1265                                conic_gradient,
1266                                layer.tile_size,
1267                                layer.tile_spacing,
1268                            )
1269                        },
1270                    }
1271                },
1272                Ok(ResolvedImage::Image { image, size }) => {
1273                    // FIXME: https://drafts.csswg.org/css-images-4/#the-image-resolution
1274                    let dppx = 1.0;
1275                    let intrinsic =
1276                        NaturalSizes::from_width_and_height(size.width / dppx, size.height / dppx);
1277                    let layer = background::layout_layer(self, painter, builder, index, intrinsic);
1278                    let image_wr_key = match image {
1279                        CachedImage::Raster(raster_image) => raster_image.id,
1280                        CachedImage::Vector(vector_image) => {
1281                            let scale = builder.device_pixel_ratio.get();
1282                            let default_size: DeviceIntSize =
1283                                Size2D::new(size.width * scale, size.height * scale).to_i32();
1284                            let layer_size = layer.as_ref().map(|layer| {
1285                                Size2D::new(
1286                                    layer.tile_size.width * scale,
1287                                    layer.tile_size.height * scale,
1288                                )
1289                                .to_i32()
1290                            });
1291
1292                            node.and_then(|node| {
1293                                let size = layer_size.unwrap_or(default_size);
1294                                builder.image_resolver.rasterize_vector_image(
1295                                    vector_image.id,
1296                                    size,
1297                                    node,
1298                                )
1299                            })
1300                            .and_then(|rasterized_image| rasterized_image.id)
1301                        },
1302                    };
1303
1304                    let Some(image_key) = image_wr_key else {
1305                        continue;
1306                    };
1307

1555    fn build_outline(&mut self, builder: &mut DisplayListBuilder) {
1556        let style = self.fragment.style();
1557        let outline = style.get_outline();
1558        if outline.outline_style.none_or_hidden() {
1559            return;
1560        }
1561        let width = outline.outline_width.0.to_f32_px();
1562        if width == 0.0 {
1563            return;
1564        }
1565        // <https://drafts.csswg.org/css-ui-3/#outline-offset>
1566        // > Negative values must cause the outline to shrink into the border box. Both
1567        // > the height and the width of outside of the shape drawn by the outline should
1568        // > not become smaller than twice the computed value of the outline-width
1569        // > property, to make sure that an outline can be rendered even with large
1570        // > negative values. User agents should apply this constraint independently in
1571        // > each dimension. If the outline is drawn as multiple disconnected shapes, this
1572        // > constraint applies to each shape separately.
1573        let offset = outline.outline_offset.to_f32_px() + width;
1574        let outline_rect = self.border_rect.inflate(
1575            offset.max(-self.border_rect.width() / 2.0 + width),
1576            offset.max(-self.border_rect.height() / 2.0 + width),
1577        );
1578        let common = builder.common_properties(outline_rect, &style);
1579        let widths = SideOffsets2D::new_all_same(width);
1580        let border_style = match outline.outline_style {
1581            // TODO: treating 'auto' as 'solid' is allowed by the spec,
1582            // but we should do something better.
1583            OutlineStyle::Auto => BorderStyle::Solid,
1584            OutlineStyle::BorderStyle(s) => s,
1585        };
1586        let side = self.build_border_side(BorderStyleColor {
1587            style: border_style,
1588            color: style.resolve_color(&outline.outline_color),
1589        });
1590        let details = wr::BorderDetails::Normal(wr::NormalBorder {
1591            top: side,
1592            right: side,
1593            bottom: side,
1594            left: side,
1595            radius: offset_radii(self.border_radius, offset),
1596            do_aa: true,
1597        });
1598        builder
1599            .wr()
1600            .push_border(&common, outline_rect, widths, details)
1601    }

1646fn glyphs(
1647    glyph_runs: &[Arc<GlyphStore>],
1648    mut baseline_origin: PhysicalPoint<Au>,
1649    justification_adjustment: Au,
1650    include_whitespace: bool,
1651) -> Vec<wr::GlyphInstance> {
1652    use fonts_traits::ByteIndex;
1653
1654    let mut glyphs = vec![];
1655    for run in glyph_runs {
1656        for glyph in run.iter_glyphs_for_byte_range(TextByteRange::new(ByteIndex(0), run.len())) {
1657            if !run.is_whitespace() || include_whitespace {
1658                let glyph_offset = glyph.offset().unwrap_or(Point2D::zero());
1659                let point = units::LayoutPoint::new(
1660                    baseline_origin.x.to_f32_px() + glyph_offset.x.to_f32_px(),
1661                    baseline_origin.y.to_f32_px() + glyph_offset.y.to_f32_px(),
1662                );
1663                let glyph = wr::GlyphInstance {
1664                    index: glyph.id(),
1665                    point,
1666                };
1667                glyphs.push(glyph);
1668            }
1669
1670            if glyph.char_is_word_separator() {
1671                baseline_origin.x += justification_adjustment;
1672            }
1673            baseline_origin.x += glyph.advance();
1674        }
1675    }
1676    glyphs
1677}

1696/// Radii for the padding edge or content edge
1697fn inner_radii(mut radii: wr::BorderRadius, insets: units::LayoutSideOffsets) -> wr::BorderRadius {
1698    assert!(insets.left >= 0.0, "left inset must not be negative");
1699    radii.top_left.width = (radii.top_left.width - insets.left).max(0.0);
1700    radii.bottom_left.width = (radii.bottom_left.width - insets.left).max(0.0);
1701
1702    assert!(insets.right >= 0.0, "left inset must not be negative");
1703    radii.top_right.width = (radii.top_right.width - insets.right).max(0.0);
1704    radii.bottom_right.width = (radii.bottom_right.width - insets.right).max(0.0);
1705
1706    assert!(insets.top >= 0.0, "top inset must not be negative");
1707    radii.top_left.height = (radii.top_left.height - insets.top).max(0.0);
1708    radii.top_right.height = (radii.top_right.height - insets.top).max(0.0);
1709
1710    assert!(insets.bottom >= 0.0, "bottom inset must not be negative");
1711    radii.bottom_left.height = (radii.bottom_left.height - insets.bottom).max(0.0);
1712    radii.bottom_right.height = (radii.bottom_right.height - insets.bottom).max(0.0);
1713    radii
1714}

1716fn offset_radii(mut radii: wr::BorderRadius, offset: f32) -> wr::BorderRadius {
1717    if offset == 0.0 {
1718        return radii;
1719    }
1720    if offset < 0.0 {
1721        return inner_radii(radii, units::LayoutSideOffsets::new_all_same(-offset));
1722    }
1723    let expand = |radius: &mut f32| {
1724        // Expand the radius by the specified amount, but keeping sharp corners.
1725        // TODO: this behavior is not continuous, it's being discussed in the CSSWG:
1726        // https://github.com/w3c/csswg-drafts/issues/7103
1727        if *radius > 0.0 {
1728            *radius += offset;
1729        }
1730    };
1731    expand(&mut radii.top_left.width);
1732    expand(&mut radii.top_left.height);
1733    expand(&mut radii.top_right.width);
1734    expand(&mut radii.top_right.height);
1735    expand(&mut radii.bottom_right.width);
1736    expand(&mut radii.bottom_right.height);
1737    expand(&mut radii.bottom_left.width);
1738    expand(&mut radii.bottom_left.height);
1739    radii
1740}

1810pub(super) fn normalize_radii(rect: &units::LayoutRect, radius: &mut wr::BorderRadius) {
1811    // Normalize radii that add up to > 100%.
1812    // https://www.w3.org/TR/css-backgrounds-3/#corner-overlap
1813    // > Let f = min(L_i/S_i), where i âˆˆ {top, right, bottom, left},
1814    // > S_i is the sum of the two corresponding radii of the corners on side i,
1815    // > and L_top = L_bottom = the width of the box,
1816    // > and L_left = L_right = the height of the box.
1817    // > If f < 1, then all corner radii are reduced by multiplying them by f.
1818    let f = (rect.width() / (radius.top_left.width + radius.top_right.width))
1819        .min(rect.width() / (radius.bottom_left.width + radius.bottom_right.width))
1820        .min(rect.height() / (radius.top_left.height + radius.bottom_left.height))
1821        .min(rect.height() / (radius.top_right.height + radius.bottom_right.height));
1822    if f < 1.0 {
1823        radius.top_left *= f;
1824        radius.top_right *= f;
1825        radius.bottom_right *= f;
1826        radius.bottom_left *= f;
1827    }
1828}

1830/// <https://drafts.csswg.org/css-shapes-1/#valdef-shape-box-margin-box>
1831/// > The corner radii of this shape are determined by the corresponding
1832/// > border-radius and margin values. If the ratio of border-radius/margin is 1 or more,
1833/// > or margin is negative or zero, then the margin box corner radius is
1834/// > max(border-radius + margin, 0). If the ratio of border-radius/margin is less than 1,
1835/// > and margin is positive, then the margin box corner radius is
1836/// > border-radius + margin * (1 + (ratio-1)^3).
1837pub(super) fn compute_margin_box_radius(
1838    radius: wr::BorderRadius,
1839    layout_rect: LayoutSize,
1840    fragment: &BoxFragment,
1841) -> wr::BorderRadius {
1842    let style = fragment.style();
1843    let margin = style.physical_margin();
1844    let adjust_radius = |radius: f32, margin: f32| -> f32 {
1845        if margin <= 0. || (radius / margin) >= 1. {
1846            (radius + margin).max(0.)
1847        } else {
1848            radius + (margin * (1. + (radius / margin - 1.).powf(3.)))
1849        }
1850    };
1851    let compute_margin_radius = |radius: LayoutSize,
1852                                 layout_rect: LayoutSize,
1853                                 margin: Size2D<LengthPercentageOrAuto, UnknownUnit>|
1854     -> LayoutSize {
1855        let zero = LengthPercentage::zero();
1856        let width = margin
1857            .width
1858            .auto_is(|| &zero)
1859            .to_used_value(Au::from_f32_px(layout_rect.width));
1860        let height = margin
