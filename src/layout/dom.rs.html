41/// The data that is stored in each DOM node that is used by layout.
42#[derive(Default, MallocSizeOf)]
43pub struct InnerDOMLayoutData {
44    pub(super) self_box: ArcRefCell<Option<LayoutBox>>,
45    pub(super) pseudo_boxes: SmallVec<[PseudoLayoutData; 2]>,
46}
47
48impl InnerDOMLayoutData {
49    fn pseudo_layout_data(
50        &self,
51        pseudo_element: PseudoElement,
52    ) -> Option<ArcRefCell<InnerDOMLayoutData>> {
53        for pseudo_layout_data in self.pseudo_boxes.iter() {
54            if pseudo_element == pseudo_layout_data.pseudo {
55                return Some(pseudo_layout_data.data.clone());
56            }
57        }
58        None
59    }
60
61    fn create_pseudo_layout_data(
62        &mut self,
63        pseudo_element: PseudoElement,
64    ) -> ArcRefCell<InnerDOMLayoutData> {
65        let data: ArcRefCell<InnerDOMLayoutData> = Default::default();
66        self.pseudo_boxes.push(PseudoLayoutData {
67            pseudo: pseudo_element,
68            data: data.clone(),
69        });
70        data
71    }
72
73    fn fragments(&self) -> Vec<Fragment> {
74        self.self_box
75            .borrow()
76            .as_ref()
77            .and_then(|layout_box| layout_box.with_base(LayoutBoxBase::fragments))
78            .unwrap_or_default()
79    }
80
81    fn repair_style(&self, node: &ServoThreadSafeLayoutNode, context: &SharedStyleContext) {
82        if let Some(layout_object) = &*self.self_box.borrow() {
83            layout_object.repair_style(context, node, &node.style(context));
84        }
85
86        for pseudo_layout_data in self.pseudo_boxes.iter() {
87            let Some(node_with_pseudo) = node.with_pseudo(pseudo_layout_data.pseudo) else {
88                continue;
89            };
90            pseudo_layout_data
91                .data
92                .borrow()
93                .repair_style(&node_with_pseudo, context);
94        }
95    }
96
97    fn with_layout_box_base(&self, callback: impl Fn(&LayoutBoxBase)) {
98        if let Some(data) = self.self_box.borrow().as_ref() {
99            data.with_base(callback);
100        }
101    }
102
103    fn with_layout_box_base_including_pseudos(&self, callback: impl Fn(&LayoutBoxBase)) {
104        self.with_layout_box_base(&callback);
105        for pseudo_layout_data in self.pseudo_boxes.iter() {
106            pseudo_layout_data
107                .data
108                .borrow()
109                .with_layout_box_base(&callback);
110        }
111    }
112}

114/// A box that is stored in one of the `DOMLayoutData` slots.
115#[derive(MallocSizeOf)]
116pub(super) enum LayoutBox {
117    DisplayContents(SharedInlineStyles),
118    BlockLevel(ArcRefCell<BlockLevelBox>),
119    InlineLevel(InlineItem),
120    FlexLevel(ArcRefCell<FlexLevelBox>),
121    TableLevelBox(TableLevelBox),
122    TaffyItemBox(ArcRefCell<TaffyItemBox>),
123}
124
125impl LayoutBox {
126    pub(crate) fn with_base<T>(&self, callback: impl FnOnce(&LayoutBoxBase) -> T) -> Option<T> {
127        Some(match self {
128            LayoutBox::DisplayContents(..) => return None,
129            LayoutBox::BlockLevel(block_level_box) => block_level_box.borrow().with_base(callback),
130            LayoutBox::InlineLevel(inline_item) => inline_item.with_base(callback),
131            LayoutBox::FlexLevel(flex_level_box) => flex_level_box.borrow().with_base(callback),
132            LayoutBox::TaffyItemBox(taffy_item_box) => taffy_item_box.borrow().with_base(callback),
133            LayoutBox::TableLevelBox(table_box) => table_box.with_base(callback),
134        })
135    }
136
137    pub(crate) fn with_base_mut<T>(
138        &mut self,
139        callback: impl FnOnce(&mut LayoutBoxBase) -> T,
140    ) -> Option<T> {
141        Some(match self {
142            LayoutBox::DisplayContents(..) => return None,
143            LayoutBox::BlockLevel(block_level_box) => {
144                block_level_box.borrow_mut().with_base_mut(callback)
145            },
146            LayoutBox::InlineLevel(inline_item) => inline_item.with_base_mut(callback),
147            LayoutBox::FlexLevel(flex_level_box) => {
148                flex_level_box.borrow_mut().with_base_mut(callback)
149            },
150            LayoutBox::TaffyItemBox(taffy_item_box) => {
151                taffy_item_box.borrow_mut().with_base_mut(callback)
152            },
153            LayoutBox::TableLevelBox(table_box) => table_box.with_base_mut(callback),
154        })
155    }
156
157    fn repair_style(
158        &self,
159        context: &SharedStyleContext,
160        node: &ServoThreadSafeLayoutNode,
161        new_style: &ServoArc<ComputedValues>,
162    ) {
163        match self {
164            LayoutBox::DisplayContents(inline_shared_styles) => {
165                *inline_shared_styles.style.borrow_mut() = new_style.clone();
166                *inline_shared_styles.selected.borrow_mut() = node.selected_style();
167            },
168            LayoutBox::BlockLevel(block_level_box) => {
169                block_level_box
170                    .borrow_mut()
171                    .repair_style(context, node, new_style);
172            },
173            LayoutBox::InlineLevel(inline_item) => {
174                inline_item.repair_style(context, node, new_style);
175            },
176            LayoutBox::FlexLevel(flex_level_box) => flex_level_box
177                .borrow_mut()
178                .repair_style(context, node, new_style),
179            LayoutBox::TableLevelBox(table_level_box) => {
180                table_level_box.repair_style(context, node, new_style)
181            },
182            LayoutBox::TaffyItemBox(taffy_item_box) => taffy_item_box
183                .borrow_mut()
184                .repair_style(context, node, new_style),
185        }
186    }
187}

203pub struct BoxSlot<'dom> {
204    pub(crate) slot: ArcRefCell<Option<LayoutBox>>,
205    pub(crate) marker: PhantomData<&'dom ()>,
206}
207
208impl From<ArcRefCell<Option<LayoutBox>>> for BoxSlot<'_> {
209    fn from(slot: ArcRefCell<Option<LayoutBox>>) -> Self {
210        Self {
211            slot,
212            marker: PhantomData,
213        }
214    }
215}
216
217/// A mutable reference to a `LayoutBox` stored in a DOM element.
218impl BoxSlot<'_> {
219    pub(crate) fn set(self, layout_box: LayoutBox) {
220        *self.slot.borrow_mut() = Some(layout_box);
221    }
222
223    pub(crate) fn take_layout_box_if_undamaged(&self, damage: LayoutDamage) -> Option<LayoutBox> {
224        if damage.has_box_damage() {
225            return None;
226        }
227        self.slot.borrow_mut().take()
228    }
229}
230
231impl Drop for BoxSlot<'_> {
232    fn drop(&mut self) {
233        if !std::thread::panicking() {
234            assert!(self.slot.borrow().is_some(), "failed to set a layout box");
235        }
236    }
237}

266impl<'dom> NodeExt<'dom> for ServoThreadSafeLayoutNode<'dom> {
267    fn as_image(&self) -> Option<(Option<Image>, PhysicalSize<f64>)> {
268        let (resource, metadata) = self.image_data()?;
269        let width = metadata.map(|metadata| metadata.width).unwrap_or_default();
270        let height = metadata.map(|metadata| metadata.height).unwrap_or_default();
271        let (mut width, mut height) = (width as f64, height as f64);
272        if let Some(density) = self.image_density().filter(|density| *density != 1.) {
273            width /= density;
274            height /= density;
275        }
276        Some((resource, PhysicalSize::new(width, height)))
277    }
278
279    fn as_svg(&self) -> Option<SVGElementData> {
280        self.svg_data()
281    }
282
283    fn as_video(&self) -> Option<(Option<webrender_api::ImageKey>, Option<PhysicalSize<f64>>)> {
284        let data = self.media_data()?;
285        let natural_size = if let Some(frame) = data.current_frame {
286            Some(PhysicalSize::new(frame.width.into(), frame.height.into()))
287        } else {
288            data.metadata
289                .map(|meta| PhysicalSize::new(meta.width.into(), meta.height.into()))
290        };
291        Some((
292            data.current_frame.map(|frame| frame.image_key),
293            natural_size,
294        ))
295    }
296
297    fn as_canvas(&self) -> Option<(CanvasInfo, PhysicalSize<f64>)> {
298        let canvas_data = self.canvas_data()?;
299        let source = canvas_data.image_key;
300        Some((
301            CanvasInfo { source },
302            PhysicalSize::new(canvas_data.width.into(), canvas_data.height.into()),
303        ))
304    }
305
306    fn as_iframe(&self) -> Option<(PipelineId, BrowsingContextId)> {
307        match (self.iframe_pipeline_id(), self.iframe_browsing_context_id()) {
308            (Some(pipeline_id), Some(browsing_context_id)) => {
309                Some((pipeline_id, browsing_context_id))
310            },
311            _ => None,
312        }
313    }
314
315    fn as_typeless_object_with_data_attribute(&self) -> Option<String> {
316        if self.type_id() !=
317            Some(ScriptLayoutNodeType::Element(
318                LayoutElementType::HTMLObjectElement,
319            ))
320        {
321            return None;
322        }
323
324        // TODO: This is the what the legacy layout system did, but really if Servo
325        // supports any `<object>` that's an image, it should support those with URLs
326        // and `type` attributes with image mime types.
327        let element = self.as_element()?;
328        if element.get_attr(&ns!(), &local_name!("type")).is_some() {
329            return None;
330        }
331        element
332            .get_attr(&ns!(), &local_name!("data"))
333            .map(|string| string.to_owned())
334    }
335
336    fn ensure_inner_layout_data(&self) -> AtomicRefMut<'dom, InnerDOMLayoutData> {
337        if self.layout_data().is_none() {
338            self.initialize_layout_data::<DOMLayoutData>();
339        }
340        self.layout_data()
341            .unwrap()
342            .as_any()
343            .downcast_ref::<DOMLayoutData>()
344            .unwrap()
345            .0
346            .borrow_mut()
347    }
348
349    fn inner_layout_data(&self) -> Option<AtomicRef<'dom, InnerDOMLayoutData>> {
350        self.layout_data().map(|data| {
351            data.as_any()
352                .downcast_ref::<DOMLayoutData>()
353                .unwrap()
354                .0
355                .borrow()
356        })
357    }
358
359    fn box_slot(&self) -> BoxSlot<'dom> {
360        let pseudo_element_chain = self.pseudo_element_chain();
361        let Some(primary) = pseudo_element_chain.primary else {
362            return self.ensure_inner_layout_data().self_box.clone().into();
363        };
364
365        let Some(secondary) = pseudo_element_chain.secondary else {
366            let primary_layout_data = self
367                .ensure_inner_layout_data()
368                .create_pseudo_layout_data(primary);
369            return primary_layout_data.borrow().self_box.clone().into();
370        };
371
372        // It's *very* important that this not borrow the element's main
373        // `InnerLayoutData`. Primary pseudo-elements are processed at the same recursion
374        // level as the main data, so the `BoxSlot` is created sequentially with other
375        // primary pseudo-elements and the element itself. The secondary pseudo-element is
376        // one level deep, so could be happening in parallel with the primary
377        // pseudo-elements or main element layout.
378        let primary_layout_data = self
379            .inner_layout_data()
380            .expect("Should already have element InnerLayoutData here.")
381            .pseudo_layout_data(primary)
382            .expect("Should already have primary pseudo-element InnerLayoutData here");
383        let secondary_layout_data = primary_layout_data
384            .borrow_mut()
385            .create_pseudo_layout_data(secondary);
386        secondary_layout_data.borrow().self_box.clone().into()
387    }
388
389    fn unset_all_boxes(&self) {
390        let mut layout_data = self.ensure_inner_layout_data();
391        *layout_data.self_box.borrow_mut() = None;
392        layout_data.pseudo_boxes.clear();
393
394        // Stylo already takes care of removing all layout data
395        // for DOM descendants of elements with `display: none`.
396    }
397
398    fn unset_all_pseudo_boxes(&self) {
399        self.ensure_inner_layout_data().pseudo_boxes.clear();
400    }
401
402    fn with_layout_box_base_including_pseudos(&self, callback: impl Fn(&LayoutBoxBase)) {
403        if let Some(inner_layout_data) = self.inner_layout_data() {
404            inner_layout_data.with_layout_box_base_including_pseudos(callback);
405        }
406    }
407
408    fn fragments_for_pseudo(&self, pseudo_element: Option<PseudoElement>) -> Vec<Fragment> {
409        let Some(layout_data) = self.inner_layout_data() else {
410            return vec![];
411        };
412        match pseudo_element {
413            Some(pseudo_element) => layout_data
414                .pseudo_layout_data(pseudo_element)
415                .map(|pseudo_layout_data| pseudo_layout_data.borrow().fragments())
416                .unwrap_or_default(),
417            None => layout_data.fragments(),
418        }
419    }
420
421    fn repair_style(&self, context: &SharedStyleContext) {
422        if let Some(layout_data) = self.inner_layout_data() {
423            layout_data.repair_style(self, context);
424        }
425    }
426
427    fn take_restyle_damage(&self) -> LayoutDamage {
428        let damage = self
429            .style_data()
430            .map(|style_data| std::mem::take(&mut style_data.element_data.borrow_mut().damage))
431            .unwrap_or_else(RestyleDamage::reconstruct);
432        LayoutDamage::from_bits_retain(damage.bits())
433    }
