29/// A data structure used to pass and store related layout information together to
30/// avoid having to repeat the same arguments in argument lists.
31#[derive(Clone)]
32pub(crate) struct NodeAndStyleInfo<'dom> {
33    pub node: ServoThreadSafeLayoutNode<'dom>,
34    pub style: ServoArc<ComputedValues>,
35    pub damage: LayoutDamage,
36}
37
38impl<'dom> NodeAndStyleInfo<'dom> {
39    pub(crate) fn new(
40        node: ServoThreadSafeLayoutNode<'dom>,
41        style: ServoArc<ComputedValues>,
42        damage: LayoutDamage,
43    ) -> Self {
44        Self {
45            node,
46            style,
47            damage,
48        }
49    }
50
51    pub(crate) fn pseudo_element_chain(&self) -> PseudoElementChain {
52        self.node.pseudo_element_chain()
53    }
54
55    pub(crate) fn with_pseudo_element(
56        &self,
57        context: &LayoutContext,
58        pseudo_element_type: PseudoElement,
59    ) -> Option<Self> {
60        let element = self.node.as_element()?.with_pseudo(pseudo_element_type)?;
61        let style = element.style(&context.style_context);
62        Some(NodeAndStyleInfo {
63            node: element.as_node(),
64            style,
65            damage: self.damage,
66        })
67    }
68
69    pub(crate) fn get_selection_range(&self) -> Option<TextByteRange> {
70        self.node.selection()
71    }
72}

74#[derive(Debug)]
75pub(super) enum Contents {
76    /// Any kind of content that is not replaced nor a widget, including the contents of pseudo-elements.
77    NonReplaced(NonReplacedContents),
78    /// A widget with native appearance. This has several behavior in common with replaced elements,
79    /// but isn't fully replaced (see discussion in <https://github.com/w3c/csswg-drafts/issues/12876>).
80    /// Examples: `<input>`, `<textarea>`, `<select>`...
81    /// <https://drafts.csswg.org/css-ui/#widget>
82    Widget(NonReplacedContents),
83    /// Example: an `<img src=â€¦>` element.
84    /// <https://drafts.csswg.org/css2/conform.html#replaced-element>
85    Replaced(ReplacedContents),
86}
87
88#[derive(Debug)]
89pub(super) enum NonReplacedContents {
90    /// Refers to a DOM subtree, plus `::before` and `::after` pseudo-elements.
91    OfElement,
92    /// Content of a `::before` or `::after` pseudo-element that is being generated.
93    /// <https://drafts.csswg.org/css2/generate.html#content>
94    OfPseudoElement(Vec<PseudoElementContentItem>),
95}

97#[derive(Debug)]
98pub(super) enum PseudoElementContentItem {
99    Text(String),
100    Replaced(ReplacedContents),
101}

103pub(super) trait TraversalHandler<'dom> {
104    fn handle_text(&mut self, info: &NodeAndStyleInfo<'dom>, text: Cow<'dom, str>);
105
106    /// Or pseudo-element
107    fn handle_element(
108        &mut self,
109        info: &NodeAndStyleInfo<'dom>,
110        display: DisplayGeneratingBox,
111        contents: Contents,
112        box_slot: BoxSlot<'dom>,
113    );
114
115    /// Notify the handler that we are about to recurse into a `display: contents` element.
116    fn enter_display_contents(&mut self, _: SharedInlineStyles) {}
117
118    /// Notify the handler that we have finished a `display: contents` element.
119    fn leave_display_contents(&mut self) {}
120}

122fn traverse_children_of<'dom>(
123    parent_element_info: &NodeAndStyleInfo<'dom>,
124    context: &LayoutContext,
125    handler: &mut impl TraversalHandler<'dom>,
126) {
127    parent_element_info
128        .node
129        .set_uses_content_attribute_with_attr(false);
130
131    let is_element = parent_element_info.pseudo_element_chain().is_empty();
132    if is_element {
133        traverse_eager_pseudo_element(PseudoElement::Before, parent_element_info, context, handler);
134    }
135
136    for child in parent_element_info.node.children() {
137        if child.is_text_node() {
138            let info = NodeAndStyleInfo::new(
139                child,
140                child.style(&context.style_context),
141                child.take_restyle_damage(),
142            );
143            handler.handle_text(&info, child.text_content());
144        } else if child.is_element() {
145            traverse_element(child, context, handler);
146        }
147    }
148
149    if is_element {
150        traverse_eager_pseudo_element(PseudoElement::After, parent_element_info, context, handler);
151    }
152}

154fn traverse_element<'dom>(
155    element: ServoThreadSafeLayoutNode<'dom>,
156    context: &LayoutContext,
157    handler: &mut impl TraversalHandler<'dom>,
158) {
159    let damage = element.take_restyle_damage();
160    if damage.has_box_damage() {
161        element.unset_all_pseudo_boxes();
162    }
163
164    let style = element.style(&context.style_context);
165    let info = NodeAndStyleInfo::new(element, style, damage);
166
167    match Display::from(info.style.get_box().display) {
168        Display::None => element.unset_all_boxes(),
169        Display::Contents => {
170            if ReplacedContents::for_element(element, context).is_some() {
171                // `display: content` on a replaced element computes to `display: none`
172                // <https://drafts.csswg.org/css-display-3/#valdef-display-contents>
173                element.unset_all_boxes()
174            } else {
175                let shared_inline_styles: SharedInlineStyles = (&info).into();
176                element
177                    .box_slot()
178                    .set(LayoutBox::DisplayContents(shared_inline_styles.clone()));
179
180                handler.enter_display_contents(shared_inline_styles);
181                traverse_children_of(&info, context, handler);
182                handler.leave_display_contents();
183            }
184        },
185        Display::GeneratingBox(display) => {
186            let contents = Contents::for_element(element, context);
187            let display = display.used_value_for_contents(&contents);
188            let box_slot = element.box_slot();
189            handler.handle_element(&info, display, contents, box_slot);
190        },
191    }
192}

194fn traverse_eager_pseudo_element<'dom>(
195    pseudo_element_type: PseudoElement,
196    node_info: &NodeAndStyleInfo<'dom>,
197    context: &LayoutContext,
198    handler: &mut impl TraversalHandler<'dom>,
199) {
200    assert!(pseudo_element_type.is_eager());
201
202    // If this node doesn't have this eager pseudo-element, exit early. This depends on
203    // the style applied to the element.
204    let Some(pseudo_element_info) = node_info.with_pseudo_element(context, pseudo_element_type)
205    else {
206        return;
207    };
208    if pseudo_element_info.style.ineffective_content_property() {
209        return;
210    }
211
212    match Display::from(pseudo_element_info.style.get_box().display) {
213        Display::None => {},
214        Display::Contents => {
215            let items = generate_pseudo_element_content(&pseudo_element_info, context);
216            let box_slot = pseudo_element_info.node.box_slot();
217            let shared_inline_styles: SharedInlineStyles = (&pseudo_element_info).into();
218            box_slot.set(LayoutBox::DisplayContents(shared_inline_styles.clone()));
219
220            handler.enter_display_contents(shared_inline_styles);
221            traverse_pseudo_element_contents(&pseudo_element_info, context, handler, items);
222            handler.leave_display_contents();
223        },
224        Display::GeneratingBox(display) => {
225            let items = generate_pseudo_element_content(&pseudo_element_info, context);
226            let box_slot = pseudo_element_info.node.box_slot();
227            let contents = Contents::for_pseudo_element(items);
228            handler.handle_element(&pseudo_element_info, display, contents, box_slot);
229        },
230    }
231}

233fn traverse_pseudo_element_contents<'dom>(
234    info: &NodeAndStyleInfo<'dom>,
235    context: &LayoutContext,
236    handler: &mut impl TraversalHandler<'dom>,
237    items: Vec<PseudoElementContentItem>,
238) {
239    let mut anonymous_info = None;
240    for item in items {
241        match item {
242            PseudoElementContentItem::Text(text) => handler.handle_text(info, text.into()),
243            PseudoElementContentItem::Replaced(contents) => {
244                let anonymous_info = anonymous_info.get_or_insert_with(|| {
245                    info.with_pseudo_element(context, PseudoElement::ServoAnonymousBox)
246                        .unwrap_or_else(|| info.clone())
247                });
248                let display_inline = DisplayGeneratingBox::OutsideInside {
249                    outside: DisplayOutside::Inline,
250                    inside: DisplayInside::Flow {
251                        is_list_item: false,
252                    },
253                };
254                // `display` is not inherited, so we get the initial value
255                debug_assert!(
256                    Display::from(anonymous_info.style.get_box().display) ==
257                        Display::GeneratingBox(display_inline)
258                );
259                handler.handle_element(
260                    anonymous_info,
261                    display_inline,
262                    Contents::Replaced(contents),
263                    anonymous_info.node.box_slot(),
264                )
265            },
266        }
267    }
268}

270impl Contents {
271    /// Returns true iff the `try_from` impl below would return `Err(_)`
272    pub fn is_replaced(&self) -> bool {
273        matches!(self, Contents::Replaced(_))
274    }
275
276    pub(crate) fn for_element(
277        node: ServoThreadSafeLayoutNode<'_>,
278        context: &LayoutContext,
279    ) -> Self {
280        if let Some(replaced) = ReplacedContents::for_element(node, context) {
281            return Self::Replaced(replaced);
282        }
283        let is_widget = matches!(
284            node.type_id(),
285            Some(LayoutNodeType::Element(
286                LayoutElementType::HTMLInputElement |
287                    LayoutElementType::HTMLSelectElement |
288                    LayoutElementType::HTMLTextAreaElement
289            ))
290        );
291        if is_widget {
292            Self::Widget(NonReplacedContents::OfElement)
293        } else {
294            Self::NonReplaced(NonReplacedContents::OfElement)
295        }
296    }
297
298    pub(crate) fn for_pseudo_element(contents: Vec<PseudoElementContentItem>) -> Self {
299        Self::NonReplaced(NonReplacedContents::OfPseudoElement(contents))
300    }
301
302    pub(crate) fn non_replaced_contents(self) -> Option<NonReplacedContents> {
303        match self {
304            Self::NonReplaced(contents) | Self::Widget(contents) => Some(contents),
305            Self::Replaced(_) => None,
306        }
307    }
308}
309
310impl NonReplacedContents {
311    pub(crate) fn traverse<'dom>(
312        self,
313        context: &LayoutContext,
314        info: &NodeAndStyleInfo<'dom>,
315        handler: &mut impl TraversalHandler<'dom>,
316    ) {
317        match self {
318            NonReplacedContents::OfElement => traverse_children_of(info, context, handler),
319            NonReplacedContents::OfPseudoElement(items) => {
320                traverse_pseudo_element_contents(info, context, handler, items)
321            },
322        }
323    }
324}

326fn get_quote_from_pair<I, S>(item: &ContentItem<I>, opening: &S, closing: &S) -> String
327where
328    S: ToString + ?Sized,
329{
330    match item {
331        ContentItem::OpenQuote => opening.to_string(),
332        ContentItem::CloseQuote => closing.to_string(),
333        _ => unreachable!("Got an unexpected ContentItem type when processing quotes."),
334    }
335}

337/// <https://www.w3.org/TR/CSS2/generate.html#propdef-content>
338fn generate_pseudo_element_content(
339    pseudo_element_info: &NodeAndStyleInfo,
340    context: &LayoutContext,
341) -> Vec<PseudoElementContentItem> {
342    match &pseudo_element_info.style.get_counters().content {
343        Content::Items(items) => {
344            let mut vec = vec![];
345            for item in items.items.iter() {
346                match item {
347                    ContentItem::String(s) => {
348                        vec.push(PseudoElementContentItem::Text(s.to_string()));
349                    },
350                    ContentItem::Attr(attr) => {
351                        let element = pseudo_element_info
352                            .node
353                            .as_element()
354                            .expect("Expected an element");
355
356                        // From
357                        // <https://html.spec.whatwg.org/multipage/#case-sensitivity-of-the-css-%27attr%28%29%27-function>
358                        //
359                        // > CSS Values and Units leaves the case-sensitivity of attribute names for
360                        // > the purpose of the `attr()` function to be defined by the host language.
361                        // > [[CSSVALUES]].
362                        // >
363                        // > When comparing the attribute name part of a CSS `attr()`function to the
364                        // > names of namespace-less attributes on HTML elements in HTML documents,
365                        // > the name part of the CSS `attr()` function must first be converted to
366                        // > ASCII lowercase. The same function when compared to other attributes must
367                        // > be compared according to its original case. In both cases, to match the
368                        // > values must be identical to each other (and therefore the comparison is
369                        // > case sensitive).
370                        let attr_name = match element.is_html_element_in_html_document() {
371                            true => &*attr.attribute.to_ascii_lowercase(),
372                            false => &*attr.attribute,
373                        };
374
375                        pseudo_element_info
376                            .node
377                            .set_uses_content_attribute_with_attr(true);
378                        let attr_val =
379                            element.get_attr(&attr.namespace_url, &LocalName::from(attr_name));
380                        vec.push(PseudoElementContentItem::Text(
381                            attr_val.map_or("".to_string(), |s| s.to_string()),
382                        ));
383                    },
384                    ContentItem::Image(image) => {
385                        if let Some(replaced_content) =
386                            ReplacedContents::from_image(pseudo_element_info.node, context, image)
387                        {
388                            vec.push(PseudoElementContentItem::Replaced(replaced_content));
389                        }
390                    },
391                    ContentItem::OpenQuote | ContentItem::CloseQuote => {
392                        // TODO(xiaochengh): calculate quote depth
393                        let maybe_quote = match &pseudo_element_info.style.get_list().quotes {
394                            Quotes::QuoteList(quote_list) => {
395                                quote_list.0.first().map(|quote_pair| {
396                                    get_quote_from_pair(
397                                        item,
398                                        &*quote_pair.opening,
399                                        &*quote_pair.closing,
400                                    )
401                                })
402                            },
403                            Quotes::Auto => {
404                                let lang = &pseudo_element_info.style.get_font()._x_lang;
405                                let quotes = quotes_for_lang(lang.0.as_ref(), 0);
406                                Some(get_quote_from_pair(item, &quotes.opening, &quotes.closing))
407                            },
408                        };
409                        if let Some(quote) = maybe_quote {
410                            vec.push(PseudoElementContentItem::Text(quote));
411                        }
412                    },
413                    ContentItem::Counter(_, _) |
414                    ContentItem::Counters(_, _, _) |
415                    ContentItem::NoOpenQuote |
416                    ContentItem::NoCloseQuote => {
417                        // TODO: Add support for counters and quotes.
418                    },
419                }
420            }
421            vec
422        },
423        Content::Normal | Content::None => unreachable!(),
424    }
