150        self.layout_result
151            .baseline_relative_to_margin_box

158    #[allow(clippy::too_many_arguments)]
159    fn collect_fragment(
160        mut self,
161        initial_flex_layout: &InitialFlexLineLayout,
162        item_used_size: FlexRelativeVec2<Au>,
163        item_margin: FlexRelativeSides<Au>,
164        item_main_interval: Au,
165        final_line_cross_size: Au,
166        shared_alignment_baseline: &Option<Au>,
167        flex_context: &mut FlexContext,
168        all_baselines: &mut Baselines,
169        main_position_cursor: &mut Au,
170    ) -> (ArcRefCell<BoxFragment>, PositioningContext) {
171        // https://drafts.csswg.org/css-flexbox/#algo-main-align
172        // “Align the items along the main-axis”
173        *main_position_cursor +=
174            item_margin.main_start + self.item.border.main_start + self.item.padding.main_start;
175        let item_content_main_start_position = *main_position_cursor;
176
177        *main_position_cursor += item_used_size.main +
178            self.item.padding.main_end +
179            self.item.border.main_end +
180            item_margin.main_end +
181            item_main_interval;
182
183        // https://drafts.csswg.org/css-flexbox/#algo-cross-align
184        let item_content_cross_start_position = self.item.align_along_cross_axis(
185            &item_margin,
186            &item_used_size.cross,
187            final_line_cross_size,
188            self.layout_result
189                .baseline_relative_to_margin_box
190                .unwrap_or_default(),
191            shared_alignment_baseline.unwrap_or_default(),
192            flex_context.config.flex_wrap_is_reversed,
193        );
194
195        let start_corner = FlexRelativeVec2 {
196            main: item_content_main_start_position,
197            cross: item_content_cross_start_position,
198        };
199
200        // Need to collect both baselines from baseline participation and other baselines.
201        let final_line_size = FlexRelativeVec2 {
202            main: initial_flex_layout.line_size.main,
203            cross: final_line_cross_size,
204        };
205        let content_rect = flex_context.rect_to_flow_relative(
206            final_line_size,
207            FlexRelativeRect {
208                start_corner,
209                size: item_used_size,
210            },
211        );
212
213        if let Some(item_baseline) = self.layout_result.baseline_relative_to_margin_box.as_ref() {
214            let item_baseline = *item_baseline + item_content_cross_start_position -
215                self.item.border.cross_start -
216                self.item.padding.cross_start -
217                item_margin.cross_start;
218            all_baselines.first.get_or_insert(item_baseline);
219            all_baselines.last = Some(item_baseline);
220        }
221
222        let mut fragment_info = self.item.box_.base_fragment_info();
223        fragment_info
224            .flags
225            .insert(FragmentFlags::IS_FLEX_OR_GRID_ITEM);
226        if self.item.depends_on_block_constraints {
227            fragment_info.flags.insert(
228                FragmentFlags::SIZE_DEPENDS_ON_BLOCK_CONSTRAINTS_AND_CAN_BE_CHILD_OF_FLEX_ITEM,
229            );
230        }
231        let flags = fragment_info.flags;
232
233        let containing_block = flex_context.containing_block;
234        let container_writing_mode = containing_block.style.writing_mode;
235        let style = self.item.box_.style();
236
237        let mut fragment = BoxFragment::new(
238            fragment_info,
239            style.clone(),
240            self.layout_result.fragments,
241            content_rect.as_physical(Some(flex_context.containing_block)),
242            flex_context
243                .sides_to_flow_relative(self.item.padding)
244                .to_physical(container_writing_mode),
245            flex_context
246                .sides_to_flow_relative(self.item.border)
247                .to_physical(container_writing_mode),
248            flex_context
249                .sides_to_flow_relative(item_margin)
250                .to_physical(container_writing_mode),
251            self.layout_result.specific_layout_info,
252        );
253
254        // If this flex item establishes a containing block for absolutely-positioned
255        // descendants, then lay out any relevant absolutely-positioned children. This
256        // will remove those children from `self.positioning_context`.
257        if style.establishes_containing_block_for_absolute_descendants(flags) {
258            self.layout_result
259                .positioning_context
260                .layout_collected_children(flex_context.layout_context, &mut fragment);
261        }
262
263        if style.clone_position() == Position::Relative {
264            fragment.base.rect.origin += relative_adjustement(style, containing_block)
265                .to_physical_size(containing_block.style.writing_mode)
266        }
267
268        let fragment = ArcRefCell::new(fragment);
269        self.item
270            .box_
271            .independent_formatting_context
272            .base
273            .set_fragment(Fragment::Box(fragment.clone()));
274        (fragment, self.layout_result.positioning_context)
275    }

396impl FlexContainer {
397    fn cross_content_sizes(
398        &self,
399        layout_context: &LayoutContext,
400        containing_block_for_children: &IndefiniteContainingBlock,
401    ) -> InlineContentSizesResult {
402        // <https://drafts.csswg.org/css-flexbox/#intrinsic-cross-sizes>
403        assert_eq!(
404            self.config.flex_axis,
405            FlexAxis::Column,
406            "The cross axis should be the inline one"
407        );
408        let mut sizes = ContentSizes::zero();
409        let mut depends_on_block_constraints = false;
410        for kid in self.children.iter() {
411            let kid = &*kid.borrow();
412            match kid {
413                FlexLevelBox::FlexItem(item) => {
414                    // TODO: For the max-content size we should distribute items into
415                    // columns, and sum the column sizes and gaps.
416                    // TODO: Use the proper automatic minimum size.
417                    let ifc = &item.independent_formatting_context;
418                    let result = ifc.outer_inline_content_sizes(
419                        layout_context,
420                        containing_block_for_children,
421                        &LogicalVec2::zero(),
422                        false, /* auto_block_size_stretches_to_containing_block */
423                    );
424                    sizes.max_assign(result.sizes);
425                    depends_on_block_constraints |= result.depends_on_block_constraints;
426                },
427                FlexLevelBox::OutOfFlowAbsolutelyPositionedBox(_) => {},
428            }
429        }
430        InlineContentSizesResult {
431            sizes,
432            depends_on_block_constraints,
433        }
434    }
435
436    fn main_content_sizes<'a>(
437        &self,
438        layout_context: &LayoutContext,
439        containing_block_for_children: &IndefiniteContainingBlock,
440        flex_context_getter: impl Fn() -> &'a FlexContext<'a>,
441    ) -> InlineContentSizesResult {
442        // - TODO: calculate intrinsic cross sizes when container is a column
443        // (and check for ‘writing-mode’?)
444        // - TODO: Collapsed flex items need to be skipped for intrinsic size calculation.
445
446        // <https://drafts.csswg.org/css-flexbox-1/#intrinsic-main-sizes>
447        // > It is calculated, considering only non-collapsed flex items, by:
448        // > 1. For each flex item, subtract its outer flex base size from its max-content
449        // > contribution size.
450        let mut chosen_max_flex_fraction = f32::NEG_INFINITY;
451        let mut chosen_min_flex_fraction = f32::NEG_INFINITY;
452        let mut sum_of_flex_grow_factors = 0.0;
453        let mut sum_of_flex_shrink_factors = 0.0;
454        let mut item_infos = vec![];
455
456        for kid in self.children.iter() {
457            let kid = &*kid.borrow();
458            match kid {
459                FlexLevelBox::FlexItem(item) => {
460                    sum_of_flex_grow_factors += item.style().get_position().flex_grow.0;
461                    sum_of_flex_shrink_factors += item.style().get_position().flex_shrink.0;
462
463                    let info = item.main_content_size_info(
464                        layout_context,
465                        containing_block_for_children,
466                        &self.config,
467                        &flex_context_getter,
468                    );
469
470                    // > 2. Place all flex items into lines of infinite length. Within
471                    // > each line, find the greatest (most positive) desired flex
472                    // > fraction among all the flex items. This is the line’s chosen flex
473                    // > fraction.
474                    chosen_max_flex_fraction =
475                        chosen_max_flex_fraction.max(info.max_flex_factors.desired_flex_fraction);
476                    chosen_min_flex_fraction =
477                        chosen_min_flex_fraction.max(info.min_flex_factors.desired_flex_fraction);
478
479                    item_infos.push(info)
480                },
481                FlexLevelBox::OutOfFlowAbsolutelyPositionedBox(_) => {},
482            }
483        }
484
485        let normalize_flex_fraction = |chosen_flex_fraction| {
486            if chosen_flex_fraction > 0.0 && sum_of_flex_grow_factors < 1.0 {
487                // > 3. If the chosen flex fraction is positive, and the sum of the line’s
488                // > flex grow factors is less than 1, > divide the chosen flex fraction by that
489                // > sum.
490                chosen_flex_fraction / sum_of_flex_grow_factors
491            } else if chosen_flex_fraction < 0.0 && sum_of_flex_shrink_factors < 1.0 {
492                // > If the chosen flex fraction is negative, and the sum of the line’s flex
493                // > shrink factors is less than 1, > multiply the chosen flex fraction by that
494                // > sum.
495                chosen_flex_fraction * sum_of_flex_shrink_factors
496            } else {
497                chosen_flex_fraction
498            }
499        };
500
501        let chosen_min_flex_fraction = normalize_flex_fraction(chosen_min_flex_fraction);
502        let chosen_max_flex_fraction = normalize_flex_fraction(chosen_max_flex_fraction);
503
504        let main_gap = match self.config.flex_axis {
505            FlexAxis::Row => self.style.clone_column_gap(),
506            FlexAxis::Column => self.style.clone_row_gap(),
507        };
508        let main_gap = match main_gap {
509            LengthPercentageOrNormal::LengthPercentage(length_percentage) => {
510                length_percentage.to_used_value(Au::zero())
511            },
512            LengthPercentageOrNormal::Normal => Au::zero(),
513        };
514        let extra_space_from_main_gap = main_gap * (item_infos.len() as i32 - 1);
515        let mut container_max_content_size = extra_space_from_main_gap;
516        let mut container_min_content_size = if self.config.flex_wrap == FlexWrap::Nowrap {
517            extra_space_from_main_gap
518        } else {
519            Au::zero()
520        };
521        let mut container_depends_on_block_constraints = false;
522
523        for FlexItemBoxInlineContentSizesInfo {
524            outer_flex_base_size,
525            outer_min_main_size,
526            outer_max_main_size,
527            min_flex_factors,
528            max_flex_factors,
529            min_content_main_size_for_multiline_container,
530            depends_on_block_constraints,
531        } in item_infos.iter()
532        {
533            // > 4. Add each item’s flex base size to the product of its flex grow factor (scaled flex shrink
534            // > factor, if shrinking) and the chosen flex fraction, then clamp that result by the max main size
535            // > floored by the min main size.
536            // > 5. The flex container’s max-content size is the largest sum (among all the lines) of the
537            // > afore-calculated sizes of all items within a single line.
538            container_max_content_size += (*outer_flex_base_size +
539                Au::from_f32_px(
540                    max_flex_factors.flex_grow_or_shrink_factor * chosen_max_flex_fraction,
541                ))
542            .clamp_between_extremums(*outer_min_main_size, *outer_max_main_size);
543
544            // > The min-content main size of a single-line flex container is calculated
545            // > identically to the max-content main size, except that the flex items’
546            // > min-content contributions are used instead of their max-content contributions.
547            //
548            // > However, for a multi-line container, the min-content main size is simply the
549            // > largest min-content contribution of all the non-collapsed flex items in the
550            // > flex container. For this purpose, each item’s contribution is capped by the
551            // > item’s flex base size if the item is not growable, floored by the item’s flex
552            // > base size if the item is not shrinkable, and then further clamped by the item’s
553            // > min and max main sizes.
554            if self.config.flex_wrap == FlexWrap::Nowrap {
555                container_min_content_size += (*outer_flex_base_size +
556                    Au::from_f32_px(
557                        min_flex_factors.flex_grow_or_shrink_factor * chosen_min_flex_fraction,
558                    ))
559                .clamp_between_extremums(*outer_min_main_size, *outer_max_main_size);
560            } else {
561                container_min_content_size
562                    .max_assign(*min_content_main_size_for_multiline_container);
563            }
564
565            container_depends_on_block_constraints |= depends_on_block_constraints;
566        }
567
568        InlineContentSizesResult {
569            sizes: ContentSizes {
570                min_content: container_min_content_size,
571                max_content: container_max_content_size,
572            },
573            depends_on_block_constraints: container_depends_on_block_constraints,
574        }
575    }

577    /// <https://drafts.csswg.org/css-flexbox/#layout-algorithm>
578    #[servo_tracing::instrument(
579        name = "FlexContainer::layout",
580        skip_all,
581        fields(self_address = self as *const _ as usize)

583    pub(crate) fn layout(
584        &self,
585        layout_context: &LayoutContext,
586        positioning_context: &mut PositioningContext,
587        containing_block: &ContainingBlock,
588        lazy_block_size: &LazySize,
589    ) -> CacheableLayoutResult {
590        let mut flex_context = FlexContext {
591            config: self.config.clone(),
592            layout_context,
593            containing_block,
594            // https://drafts.csswg.org/css-flexbox/#definite-sizes
595            container_inner_size_constraint: self.config.flex_axis.vec2_to_flex_relative(
596                LogicalVec2 {
597                    inline: SizeConstraint::Definite(containing_block.size.inline),
598                    block: containing_block.size.block,
599                },
600            ),
601        };
602
603        // “Determine the main size of the flex container”
604        // https://drafts.csswg.org/css-flexbox/#algo-main-container
605        let container_main_size = match self.config.flex_axis {
606            FlexAxis::Row => containing_block.size.inline,
607            FlexAxis::Column => lazy_block_size.resolve(|| {
608                let mut containing_block = IndefiniteContainingBlock::from(containing_block);
609                containing_block.size.block = None;
610                self.main_content_sizes(layout_context, &containing_block, || &flex_context)
611                    .sizes
612                    .max_content
613            }),
614        };
615
616        // Actual length may be less, but we guess that usually not by a lot
617        let mut flex_items = Vec::with_capacity(self.children.len());
618
619        // Absolutely-positioned children of the flex container may be interleaved
620        // with flex items. We need to preserve their relative order for correct painting order,
621        // which is the order of `Fragment`s in this function’s return value.
622        //
623        // Example:
624        // absolutely_positioned_items_with_original_order = [Some(item), Some(item), None, Some(item), None]
625        // flex_items                                      =                         [item,             item]
626        let absolutely_positioned_items_with_original_order = self
627            .children
628            .iter()
629            .map(|arcrefcell| {
630                let borrowed = arcrefcell.borrow();
631                match &*borrowed {
632                    FlexLevelBox::OutOfFlowAbsolutelyPositionedBox(absolutely_positioned) => {
633                        FlexContent::AbsolutelyPositionedBox(absolutely_positioned.clone())
634                    },
635                    FlexLevelBox::FlexItem(_) => {
636                        let item = AtomicRef::map(borrowed, |child| match child {
637                            FlexLevelBox::FlexItem(item) => item,
638                            _ => unreachable!(),
639                        });
640                        flex_items.push(item);
641                        FlexContent::FlexItemPlaceholder
642                    },
643                }
644            })
645            .collect::<Vec<_>>();
646
647        let flex_item_boxes = flex_items.iter().map(|child| &**child);
648        let flex_items = flex_item_boxes
649            .map(|flex_item_box| FlexItem::new(&flex_context, flex_item_box))
650            .collect::<Vec<_>>();
651
652        let row_gap = self.style.clone_row_gap();
653        let column_gap = self.style.clone_column_gap();
654        let (cross_gap, main_gap) = match flex_context.config.flex_axis {
655            FlexAxis::Row => (row_gap, column_gap),
656            FlexAxis::Column => (column_gap, row_gap),
657        };
658        let cross_gap = match cross_gap {
659            LengthPercentageOrNormal::LengthPercentage(length_percent) => length_percent
660                .maybe_to_used_value(
661                    flex_context
662                        .container_inner_size_constraint
663                        .cross
664                        .to_definite(),
665                )
666                .unwrap_or_default(),
667            LengthPercentageOrNormal::Normal => Au::zero(),
668        };
669        let main_gap = match main_gap {
670            LengthPercentageOrNormal::LengthPercentage(length_percent) => length_percent
671                .maybe_to_used_value(
672                    flex_context
673                        .container_inner_size_constraint
674                        .main
675                        .to_definite(),
676                )
677                .unwrap_or_default(),
678            LengthPercentageOrNormal::Normal => Au::zero(),
679        };
680
681        // “Resolve the flexible lengths of all the flex items to find their *used main size*.”
682        // https://drafts.csswg.org/css-flexbox/#algo-flex
683        let initial_line_layouts = do_initial_flex_line_layout(
684            &mut flex_context,
685            container_main_size,
686            flex_items,
687            main_gap,
688        );
689
690        let line_count = initial_line_layouts.len();
691        let content_cross_size = initial_line_layouts
692            .iter()
693            .map(|layout| layout.line_size.cross)
694            .sum::<Au>() +
695            cross_gap * (line_count as i32 - 1);
696        let content_block_size = match self.config.flex_axis {
697            FlexAxis::Row => content_cross_size,
698            FlexAxis::Column => container_main_size,
699        };
700
701        // https://drafts.csswg.org/css-flexbox/#algo-cross-container
702        let container_cross_size = match self.config.flex_axis {
703            FlexAxis::Row => lazy_block_size.resolve(|| content_cross_size),
704            FlexAxis::Column => containing_block.size.inline,
705        };
706
707        let container_size = FlexRelativeVec2 {
708            main: container_main_size,
709            cross: container_cross_size,
710        };
711
712        let mut remaining_free_cross_space = container_cross_size - content_cross_size;
713
714        // Implement fallback alignment.
715        //
716        // In addition to the spec at https://www.w3.org/TR/css-align-3/ this implementation follows
717        // the resolution of https://github.com/w3c/csswg-drafts/issues/10154
718        let num_lines = initial_line_layouts.len();
719        let resolved_align_content: AlignFlags = {
720            // Computed value from the style system
721            let align_content_style = flex_context.config.align_content.primary();
722            let mut is_safe = align_content_style.flags() == AlignFlags::SAFE;
723
724            // From https://drafts.csswg.org/css-align/#distribution-flex
725            // > `normal` behaves as `stretch`.
726            let mut resolved_align_content = match align_content_style.value() {
727                AlignFlags::NORMAL => AlignFlags::STRETCH,
728                align_content => align_content,
729            };
730
731            // From https://drafts.csswg.org/css-flexbox/#algo-line-align:
732            // > Some alignments can only be fulfilled in certain situations or are
733            // > limited in how much space they can consume; for example, space-between
734            // > can only operate when there is more than one alignment subject, and
735            // > baseline alignment, once fulfilled, might not be enough to absorb all
736            // > the excess space. In these cases a fallback alignment takes effect (as
737            // > defined below) to fully consume the excess space.
738            let fallback_is_needed = match resolved_align_content {
739                _ if remaining_free_cross_space <= Au::zero() => true,
740                AlignFlags::STRETCH => num_lines < 1,
741                AlignFlags::SPACE_BETWEEN | AlignFlags::SPACE_AROUND | AlignFlags::SPACE_EVENLY => {
742                    num_lines < 2
743                },
744                _ => false,
745            };
746
747            if fallback_is_needed {
748                (resolved_align_content, is_safe) = match resolved_align_content {
749                    AlignFlags::STRETCH => (AlignFlags::FLEX_START, false),
750                    AlignFlags::SPACE_BETWEEN => (AlignFlags::FLEX_START, true),
751                    AlignFlags::SPACE_AROUND => (AlignFlags::CENTER, true),
752                    AlignFlags::SPACE_EVENLY => (AlignFlags::CENTER, true),
753                    _ => (resolved_align_content, is_safe),
754                }
755            };
756
757            // 2. If free space is negative the "safe" alignment variants all fallback to Start alignment
758            if remaining_free_cross_space <= Au::zero() && is_safe {
759                resolved_align_content = AlignFlags::START;
760            }
761
762            resolved_align_content
763        };
764
765        // Implement "unsafe" alignment. "safe" alignment is handled by the fallback process above.
766        let flex_wrap_is_reversed = flex_context.config.flex_wrap_is_reversed;
767        let resolved_align_content = self
768            .config
769            .resolve_reversable_flex_alignment(resolved_align_content, flex_wrap_is_reversed);
770        let mut cross_start_position_cursor = match resolved_align_content {
771            AlignFlags::START if flex_wrap_is_reversed => remaining_free_cross_space,
772            AlignFlags::START => Au::zero(),
773            AlignFlags::END if flex_wrap_is_reversed => Au::zero(),
774            AlignFlags::END => remaining_free_cross_space,
775            AlignFlags::CENTER => remaining_free_cross_space / 2,
776            AlignFlags::STRETCH => Au::zero(),
777            AlignFlags::SPACE_BETWEEN => Au::zero(),
778            AlignFlags::SPACE_AROUND => remaining_free_cross_space / num_lines as i32 / 2,
779            AlignFlags::SPACE_EVENLY => remaining_free_cross_space / (num_lines as i32 + 1),
780
781            // TODO: Implement all alignments. Note: not all alignment values are valid for content distribution
782            _ => Au::zero(),
783        };
784
785        let inline_axis_is_main_axis = self.config.flex_axis == FlexAxis::Row;
786        let mut baseline_alignment_participating_baselines = Baselines::default();
787        let mut all_baselines = Baselines::default();
788        let flex_item_fragments: Vec<_> = initial_line_layouts
789            .into_iter()
790            .enumerate()
791            .flat_map(|(index, initial_line_layout)| {
792                // We call `allocate_free_cross_space_for_flex_line` for each line to avoid having
793                // leftover space when the number of lines doesn't evenly divide the total free space,
794                // considering the precision of app units.
795                let (space_to_add_to_line, space_to_add_after_line) =
796                    allocate_free_cross_space_for_flex_line(
797                        resolved_align_content,
798                        remaining_free_cross_space,
799                        (num_lines - index) as i32,
800                    );
801                remaining_free_cross_space -= space_to_add_to_line + space_to_add_after_line;
802
803                let final_line_cross_size =
804                    initial_line_layout.line_size.cross + space_to_add_to_line;
805                let mut final_line_layout = initial_line_layout.finish_with_final_cross_size(
806                    &mut flex_context,
807                    main_gap,
808                    final_line_cross_size,
809                );
810
811                let line_cross_start_position = cross_start_position_cursor;
812                cross_start_position_cursor = line_cross_start_position +
813                    final_line_cross_size +
814                    space_to_add_after_line +
815                    cross_gap;
816
817                let flow_relative_line_position =
818                    match (self.config.flex_axis, flex_wrap_is_reversed) {
819                        (FlexAxis::Row, false) => LogicalVec2 {
820                            block: line_cross_start_position,
821                            inline: Au::zero(),
822                        },
823                        (FlexAxis::Row, true) => LogicalVec2 {
824                            block: container_cross_size -
825                                line_cross_start_position -
826                                final_line_layout.cross_size,
827                            inline: Au::zero(),
828                        },
829                        (FlexAxis::Column, false) => LogicalVec2 {
830                            block: Au::zero(),
831                            inline: line_cross_start_position,
832                        },
833                        (FlexAxis::Column, true) => LogicalVec2 {
834                            block: Au::zero(),
835                            inline: container_cross_size -
836                                line_cross_start_position -
837                                final_line_cross_size,
838                        },
839                    };
840
841                if inline_axis_is_main_axis {
842                    let line_shared_alignment_baseline = final_line_layout
843                        .shared_alignment_baseline
844                        .map(|baseline| baseline + flow_relative_line_position.block);
845                    if index == 0 {
846                        baseline_alignment_participating_baselines.first =
847                            line_shared_alignment_baseline;
848                    }
849                    if index == num_lines - 1 {
850                        baseline_alignment_participating_baselines.last =
851                            line_shared_alignment_baseline;
852                    }
853                }
854
855                let line_all_baselines = final_line_layout
856                    .all_baselines
857                    .offset(flow_relative_line_position.block);
858                if index == 0 {
859                    all_baselines.first = line_all_baselines.first;
860                }
861                if index == num_lines - 1 {
862                    all_baselines.last = line_all_baselines.last;
863                }
864
865                let physical_line_position =
866                    flow_relative_line_position.to_physical_size(self.style.writing_mode);
867                for (fragment, _) in &mut final_line_layout.item_fragments {
868                    fragment.borrow_mut().base.rect.origin += physical_line_position;
869                }
870                final_line_layout.item_fragments
871            })
872            .collect();
873
874        let mut flex_item_fragments = flex_item_fragments.into_iter();
875        let fragments = absolutely_positioned_items_with_original_order
876            .into_iter()
877            .map(|child_as_abspos| match child_as_abspos {
878                FlexContent::AbsolutelyPositionedBox(absolutely_positioned_box) => self
879                    .create_absolutely_positioned_flex_child_fragment(
880                        absolutely_positioned_box,
881                        containing_block,
882                        container_size,
883                        positioning_context,
884                    ),
885                FlexContent::FlexItemPlaceholder => {
886                    // The `flex_item_fragments` iterator yields one fragment
887                    // per flex item, in the original order.
888                    let (fragment, mut child_positioning_context) =
889                        flex_item_fragments.next().unwrap();
890                    let fragment = Fragment::Box(fragment);
891                    child_positioning_context.adjust_static_position_of_hoisted_fragments(
892                        &fragment,
893                        PositioningContextLength::zero(),
894                    );
895                    positioning_context.append(child_positioning_context);
896                    fragment
897                },
898            })
899            .collect::<Vec<_>>();
900
901        // There should be no more flex items
902        assert!(flex_item_fragments.next().is_none());
903
904        let baselines = Baselines {
905            first: baseline_alignment_participating_baselines
906                .first
907                .or(all_baselines.first),
908            last: baseline_alignment_participating_baselines
909                .last
910                .or(all_baselines.last),
911        };
912
913        // TODO: `depends_on_block_constraints` could be false in some corner cases
914        // in order to improve performance.
915        // - In a single-line column container where all items have the grow and shrink
916        //   factors set to zero and the flex basis doesn't depend on block constraints,
917        //   and `justify-content` is `start` or equivalent.
918        //   This is unlikely because the flex shrink factor defaults to 1.
919        // - In a single-line row container where all items have `align-self: start` or
920        //   equivalent, and the cross size doesn't depend on block constraints.
921        //   This is unlikely because `align-self` stretches by default.
922        // - In a multi-line row container where `align-content` is `start` or equivalent,
923        //   and no item cross size depends on block constraints.
924        //   This is unlikely because `align-content` defaults to `stretch`.
925        let depends_on_block_constraints = true;
926
927        CacheableLayoutResult {
928            fragments,
929            content_block_size,
930            content_inline_size_for_table: None,
931            baselines,
932            depends_on_block_constraints,
933            specific_layout_info: None,
934            collapsible_margins_in_children: CollapsedBlockMargins::zero(),
935        }
936    }
937
938    /// Create a absolutely positioned flex child fragment, using the rules the
939    /// specification dictates. This should take into account the alignment and
940    /// justification values of the container and the child to position it within a
941    /// "inset-modified containing block," which may be either the "static-position
942    /// rectangle" that's calculated below or a modified version of the absolute's
943    /// containing block adjusted by the insets specified in the item's style.
944    ///
945    /// From <https://drafts.csswg.org/css-flexbox/#abspos-items>:
946    /// > The cross-axis edges of the static-position rectangle of an
947    /// > absolutely-positioned child of a flex container are the content edges of the
948    /// > flex container The main-axis edges of the static-position rectangle are where
949    /// > the margin edges of the child would be positioned if it were the sole flex item
950    /// > in the flex container, assuming both the child and the flex container were
951    /// > fixed-size boxes of their used size. (For this purpose, auto margins are
952    /// > treated as zero.)
953    fn create_absolutely_positioned_flex_child_fragment(
954        &self,
955        absolutely_positioned_box: ArcRefCell<AbsolutelyPositionedBox>,
956        containing_block: &ContainingBlock,
957        container_size: FlexRelativeVec2<Au>,
958        positioning_context: &mut PositioningContext,
959    ) -> Fragment {
960        let alignment = {
961            let fragment = absolutely_positioned_box.borrow();
962            let make_flex_only_values_directional_for_absolutes =
963                |value: AlignFlags, reversed: bool| match (value.value(), reversed) {
964                    (AlignFlags::NORMAL | AlignFlags::AUTO | AlignFlags::STRETCH, true) => {
965                        AlignFlags::END | AlignFlags::SAFE
966                    },
967                    (AlignFlags::STRETCH, false) => AlignFlags::START | AlignFlags::SAFE,
968                    (AlignFlags::SPACE_BETWEEN, false) => AlignFlags::START | AlignFlags::SAFE,
969                    (AlignFlags::SPACE_BETWEEN, true) => AlignFlags::END | AlignFlags::SAFE,
970                    _ => value,
971                };
972            let cross = make_flex_only_values_directional_for_absolutes(
973                self.config
974                    .resolve_align_self_for_child(fragment.context.style()),
975                self.config.flex_wrap_is_reversed,
976            );
977            let main = make_flex_only_values_directional_for_absolutes(
978                self.config.resolve_justify_content_for_child(),
979                self.config.flex_direction_is_reversed,
980            );
981
982            FlexRelativeVec2 { cross, main }
983        };
984        let logical_alignment = self.config.flex_axis.vec2_to_flow_relative(alignment);
985
986        let static_position_rect = LogicalRect {
987            start_corner: LogicalVec2::zero(),
988            size: self.config.flex_axis.vec2_to_flow_relative(container_size),
989        }
990        .as_physical(Some(containing_block));
991
992        let hoisted_box = AbsolutelyPositionedBox::to_hoisted(
993            absolutely_positioned_box,
994            static_position_rect,
995            logical_alignment,
996            self.config.writing_mode,
997        );
998        let hoisted_fragment = hoisted_box.fragment.clone();
999        positioning_context.push(hoisted_box);
1000        Fragment::AbsoluteOrFixedPositioned(hoisted_fragment)
1001    }

1009/// Align all flex lines per `align-content` according to
1010/// <https://drafts.csswg.org/css-flexbox/#algo-line-align>. Returns the space to add to
1011/// each line or the space to add after each line.
1012fn allocate_free_cross_space_for_flex_line(
1013    resolved_align_content: AlignFlags,
1014    remaining_free_cross_space: Au,
1015    remaining_line_count: i32,
1016) -> (Au, Au) {
1017    if remaining_free_cross_space.is_zero() {
1018        return (Au::zero(), Au::zero());
1019    }
1020
1021    match resolved_align_content {
1022        AlignFlags::START => (Au::zero(), Au::zero()),
1023        AlignFlags::FLEX_START => (Au::zero(), Au::zero()),
1024        AlignFlags::END => (Au::zero(), Au::zero()),
1025        AlignFlags::FLEX_END => (Au::zero(), Au::zero()),
1026        AlignFlags::CENTER => (Au::zero(), Au::zero()),
1027        AlignFlags::STRETCH => (
1028            remaining_free_cross_space / remaining_line_count,
1029            Au::zero(),
1030        ),
1031        AlignFlags::SPACE_BETWEEN => {
1032            if remaining_line_count > 1 {
1033                (
1034                    Au::zero(),
1035                    remaining_free_cross_space / (remaining_line_count - 1),
1036                )
1037            } else {
1038                (Au::zero(), Au::zero())
1039            }
1040        },
1041        AlignFlags::SPACE_AROUND => (
1042            Au::zero(),
1043            remaining_free_cross_space / remaining_line_count,
1044        ),
1045        AlignFlags::SPACE_EVENLY => (
1046            Au::zero(),
1047            remaining_free_cross_space / (remaining_line_count + 1),
1048        ),
1049
1050        // TODO: Implement all alignments. Note: not all alignment values are valid for content distribution
1051        _ => (Au::zero(), Au::zero()),
1052    }
1053}

1095/// “Collect flex items into flex lines”
1096/// <https://drafts.csswg.org/css-flexbox/#algo-line-break>
1097fn do_initial_flex_line_layout<'items>(
1098    flex_context: &mut FlexContext,
1099    container_main_size: Au,
1100    mut items: Vec<FlexItem<'items>>,
1101    main_gap: Au,
1102) -> Vec<InitialFlexLineLayout<'items>> {
1103    let construct_line = |(items, outer_hypothetical_main_size)| {
1104        InitialFlexLineLayout::new(
1105            flex_context,
1106            items,
1107            outer_hypothetical_main_size,
1108            container_main_size,
1109            main_gap,
1110        )
1111    };
1112
1113    if flex_context.config.container_is_single_line {
1114        let outer_hypothetical_main_sizes_sum = items
1115            .iter()
1116            .map(|item| item.hypothetical_main_size + item.pbm_auto_is_zero.main)
1117            .sum();
1118        return vec![construct_line((items, outer_hypothetical_main_sizes_sum))];
1119    }
1120
1121    let mut lines = Vec::new();
1122    let mut line_size_so_far = Au::zero();
1123    let mut line_so_far_is_empty = true;
1124    let mut index = 0;
1125
1126    while let Some(item) = items.get(index) {
1127        let item_size = item.hypothetical_main_size + item.pbm_auto_is_zero.main;
1128        let mut line_size_would_be = line_size_so_far + item_size;
1129        if !line_so_far_is_empty {
1130            line_size_would_be += main_gap;
1131        }
1132        let item_fits = line_size_would_be <= container_main_size;
1133        if item_fits || line_so_far_is_empty {
1134            line_size_so_far = line_size_would_be;
1135            line_so_far_is_empty = false;
1136            index += 1;
1137            continue;
1138        }
1139
1140        // We found something that doesn’t fit. This line ends *before* this item.
1141        let remaining = items.split_off(index);
1142        lines.push((items, line_size_so_far));
1143        items = remaining;
1144
1145        // The next line has this item.
1146        line_size_so_far = item_size;
1147        index = 1;
1148    }
1149
1150    // We didn't reach the end of the last line, so add all remaining items there.
1151    lines.push((items, line_size_so_far));
1152
1153    if flex_context.layout_context.use_rayon {
1154        lines.par_drain(..).map(construct_line).collect()
1155    } else {
1156        lines.drain(..).map(construct_line).collect()
1157    }
1158}

1160/// The result of splitting the flex items into lines using their intrinsic sizes and doing an
1161/// initial layout of each item. A final layout still needs to happen after this is produced to
1162/// handle stretching.
1163struct InitialFlexLineLayout<'a> {
1164    /// The items that are placed in this line.
1165    items: Vec<FlexLineItem<'a>>,
1166
1167    /// The initial size of this flex line, not taking into account `align-content: stretch`.
1168    line_size: FlexRelativeVec2<Au>,
1169
1170    /// The free space available to this line after the initial layout.
1171    free_space_in_main_axis: Au,
1172}

1175    fn new<'items>(
1176        flex_context: &FlexContext,
1177        items: Vec<FlexItem<'items>>,
1178        outer_hypothetical_main_sizes_sum: Au,
1179        container_main_size: Au,
1180        main_gap: Au,
1181    ) -> InitialFlexLineLayout<'items> {
1182        let item_count = items.len();
1183        let (item_used_main_sizes, free_space_in_main_axis) = Self::resolve_flexible_lengths(
1184            &items,
1185            outer_hypothetical_main_sizes_sum,
1186            container_main_size - main_gap * (item_count as i32 - 1),
1187        );
1188
1189        // https://drafts.csswg.org/css-flexbox/#algo-cross-item
1190        let layout_results: Vec<_> = if flex_context.layout_context.use_rayon {
1191            items
1192                .par_iter()
1193                .zip(&item_used_main_sizes)
1194                .map(|(item, used_main_size)| item.layout(*used_main_size, flex_context, None))
1195                .collect()
1196        } else {
1197            items
1198                .iter()
1199                .zip(&item_used_main_sizes)
1200                .map(|(item, used_main_size)| item.layout(*used_main_size, flex_context, None))
1201                .collect()
1202        };
1203
1204        let items: Vec<_> = izip!(
1205            items.into_iter(),
1206            layout_results.into_iter(),
1207            item_used_main_sizes.into_iter()
1208        )
1209        .map(|(item, layout_result, used_main_size)| FlexLineItem {
1210            item,
1211            layout_result,
1212            used_main_size,
1213        })
1214        .collect();
1215
1216        // https://drafts.csswg.org/css-flexbox/#algo-cross-line
1217        let line_cross_size = Self::cross_size(&items, flex_context);
1218        let line_size = FlexRelativeVec2 {
1219            main: container_main_size,
1220            cross: line_cross_size,
1221        };
1222
1223        InitialFlexLineLayout {
1224            items,
1225            line_size,
1226            free_space_in_main_axis,
1227        }
1228    }
1229
1230    /// Return the *main size* of each item, and the line’s remainaing free space
1231    /// <https://drafts.csswg.org/css-flexbox/#resolve-flexible-lengths>
1232    fn resolve_flexible_lengths<'items>(
1233        items: &'items [FlexItem<'items>],
1234        outer_hypothetical_main_sizes_sum: Au,
1235        container_main_size: Au,
1236    ) -> (Vec<Au>, Au) {
1237        struct FlexibleLengthResolutionItem<'items> {
1238            item: &'items FlexItem<'items>,
1239            frozen: Cell<bool>,
1240            target_main_size: Cell<Au>,
1241            flex_factor: f32,
1242            min_max_violation_kind: Cell<Ordering>,
1243        }
1244
1245        // > 1. Determine the used flex factor. Sum the outer hypothetical main sizes of all
1246        // > items on the line. If the sum is less than the flex container’s inner main
1247        // > size, use the flex grow factor for the rest of this algorithm; otherwise, use
1248        // > the flex shrink factor.
1249        let grow = outer_hypothetical_main_sizes_sum < container_main_size;
1250
1251        let mut frozen_count = 0;
1252        let items: Vec<_> = items
1253            .iter()
1254            .map(|item| {
1255                // > 2. Each item in the flex line has a target main size, initially set to its
1256                // > flex base size. Each item is initially unfrozen and may become frozen.
1257                let target_main_size = Cell::new(item.flex_base_size);
1258
1259                // > 3. Size inflexible items. Freeze, setting its target main size to its hypothetical main size…
1260                // > - any item that has a flex factor of zero
1261                // > - if using the flex grow factor: any item that has a flex base size
1262                // >   greater than its hypothetical main size
1263                // > - if using the flex shrink factor: any item that has a flex base size
1264                // >   smaller than its hypothetical main size
1265                let flex_factor = if grow {
1266                    item.box_.style().get_position().flex_grow.0
1267                } else {
1268                    item.box_.style().get_position().flex_shrink.0
1269                };
1270
1271                let is_inflexible = flex_factor == 0. ||
1272                    if grow {
1273                        item.flex_base_size > item.hypothetical_main_size
1274                    } else {
1275                        item.flex_base_size < item.hypothetical_main_size
1276                    };
1277
1278                let frozen = Cell::new(false);
1279                if is_inflexible {
1280                    frozen_count += 1;
1281                    frozen.set(true);
1282                    target_main_size.set(item.hypothetical_main_size);
1283                }
1284
1285                FlexibleLengthResolutionItem {
1286                    item,
1287                    frozen,
1288                    target_main_size,
1289                    flex_factor,
1290                    // The actual violation will be computed later.
1291                    min_max_violation_kind: Cell::new(Ordering::Equal),
1292                }
1293            })
1294            .collect();
1295
1296        let unfrozen_items = || items.iter().filter(|item| !item.frozen.get());
1297        let main_sizes = |items: Vec<FlexibleLengthResolutionItem>| {
1298            items
1299                .into_iter()
1300                .map(|item| item.target_main_size.get())
1301                .collect()
1302        };
1303
1304        // https://drafts.csswg.org/css-flexbox/#initial-free-space
1305        // > 4. Calculate initial free space. Sum the outer sizes of all items on the line, and
1306        // > subtract this from the flex container’s inner main size. For frozen items, use
1307        // > their outer target main size; for other items, use their outer flex base size.
1308        let free_space = |all_items_frozen| {
1309            let items_size = items
1310                .iter()
1311                .map(|item| {
1312                    item.item.pbm_auto_is_zero.main +
1313                        if all_items_frozen || item.frozen.get() {
1314                            item.target_main_size.get()
1315                        } else {
1316                            item.item.flex_base_size
1317                        }
1318                })
1319                .sum();
1320            container_main_size - items_size
1321        };
1322
1323        let initial_free_space = free_space(false);
1324        loop {
1325            // https://drafts.csswg.org/css-flexbox/#remaining-free-space
1326            let mut remaining_free_space = free_space(false);
1327            // > 5. a. Check for flexible items. If all the flex items on the line are
1328            // >       frozen, free space has been distributed; exit this loop.
1329            if frozen_count >= items.len() {
1330                return (main_sizes(items), remaining_free_space);
1331            }
1332
1333            // > 5. b. Calculate the remaining free space as for initial free space, above. If the
1334            // > sum of the unfrozen flex items’ flex factors is less than one, multiply the
1335            // > initial free space by this sum. If the magnitude of this value is less than
1336            // > the magnitude of the remaining free space, use this as the remaining free
1337            // > space.
1338            let unfrozen_items_flex_factor_sum =
1339                unfrozen_items().map(|item| item.flex_factor).sum();
1340            if unfrozen_items_flex_factor_sum < 1. {
1341                let multiplied = initial_free_space.scale_by(unfrozen_items_flex_factor_sum);
1342                if multiplied.abs() < remaining_free_space.abs() {
1343                    remaining_free_space = multiplied
1344                }
1345            }
1346
1347            // > 5. c. If the remaining free space is non-zero, distribute it proportional
1348            // to the flex factors:
1349            //
1350            // FIXME: is it a problem if floating point precision errors accumulate
1351            // and we get not-quite-zero remaining free space when we should get zero here?
1352            if !remaining_free_space.is_zero() {
1353                // > If using the flex grow factor:
1354                // > For every unfrozen item on the line, find the ratio of the item’s flex grow factor to
1355                // > the sum of the flex grow factors of all unfrozen items on the line. Set the item’s
1356                // > target main size to its flex base size plus a fraction of the remaining free space
1357                // > proportional to the ratio.
1358                if grow {
1359                    for item in unfrozen_items() {
1360                        let ratio = item.flex_factor / unfrozen_items_flex_factor_sum;
1361                        item.target_main_size
1362                            .set(item.item.flex_base_size + remaining_free_space.scale_by(ratio));
1363                    }
1364                // > If using the flex shrink factor
1365                // > For every unfrozen item on the line, multiply its flex shrink factor by its inner flex
1366                // > base size, and note this as its scaled flex shrink factor. Find the ratio of the
1367                // > item’s scaled flex shrink factor to the sum of the scaled flex shrink factors of all
1368                // > unfrozen items on the line. Set the item’s target main size to its flex base size
1369                // > minus a fraction of the absolute value of the remaining free space proportional to the
1370                // > ratio. Note this may result in a negative inner main size; it will be corrected in the
1371                // > next step.
1372                } else {
1373                    // https://drafts.csswg.org/css-flexbox/#scaled-flex-shrink-factor
1374                    let scaled_shrink_factor = |item: &FlexibleLengthResolutionItem| {
1375                        item.item.flex_base_size.scale_by(item.flex_factor)
1376                    };
1377                    let scaled_shrink_factors_sum: Au =
1378                        unfrozen_items().map(scaled_shrink_factor).sum();
1379                    if scaled_shrink_factors_sum > Au::zero() {
1380                        for item in unfrozen_items() {
1381                            let ratio = scaled_shrink_factor(item).0 as f32 /
1382                                scaled_shrink_factors_sum.0 as f32;
1383                            item.target_main_size.set(
1384                                item.item.flex_base_size -
1385                                    remaining_free_space.abs().scale_by(ratio),
1386                            );
1387                        }
1388                    }
1389                }
1390            }
1391
1392            // > 5. d. Fix min/max violations. Clamp each non-frozen item’s target main size
1393            // > by its used min and max main sizes and floor its content-box size at zero.
1394            // > If the item’s target main size was made smaller by this, it’s a max
1395            // > violation. If the item’s target main size was made larger by this, it’s a
1396            // > min violation.
1397            let mut total_violation = Au::zero();
1398            for item in unfrozen_items() {
1399                let unclamped = item.target_main_size.get();
1400                let clamped = unclamped.clamp_between_extremums(
1401                    item.item.content_min_main_size,
1402                    item.item.content_max_main_size,
1403                );
1404                item.target_main_size.set(clamped);
1405                // We represent min violations with Ordering::Greater, and max violations
1406                // with Ordering::Less.
1407                item.min_max_violation_kind.set(clamped.cmp(&unclamped));
1408                total_violation += clamped - unclamped;
1409            }
1410
1411            // > 5. e. Freeze over-flexed items. The total violation is the sum of the
1412            // > adjustments from the previous step ∑(clamped size - unclamped size). If the
1413            // > total violation is:
1414            // > - Zero:  Freeze all items.
1415            // > - Positive: Freeze all the items with min violations.
1416            // > - Negative:  Freeze all the items with max violations.
1417            match total_violation.cmp(&Au::zero()) {
1418                Ordering::Equal => {
1419                    // “Freeze all items.”
1420                    // Return instead, as that’s what the next loop iteration would do.
1421                    let remaining_free_space = free_space(true);
1422                    return (main_sizes(items), remaining_free_space);
1423                },
1424                total_violation_kind => {
1425                    for item in unfrozen_items() {
1426                        if item.min_max_violation_kind.get() == total_violation_kind {
1427                            item.frozen.set(true);
1428                            frozen_count += 1;
1429                        }
1430                    }
1431                },
1432            }
1433        }
1434    }

1436    /// <https://drafts.csswg.org/css-flexbox/#algo-cross-line>
1437    fn cross_size<'items>(items: &'items [FlexLineItem<'items>], flex_context: &FlexContext) -> Au {
1438        if flex_context.config.container_is_single_line {
1439            if let SizeConstraint::Definite(size) =
1440                flex_context.container_inner_size_constraint.cross
1441            {
1442                return size;
1443            }
1444        }
1445
1446        let mut max_ascent = Au::zero();
1447        let mut max_descent = Au::zero();
1448        let mut max_outer_hypothetical_cross_size = Au::zero();
1449        for item in items.iter() {
1450            // TODO: check inline-axis is parallel to main axis, check no auto cross margins
1451            if matches!(
1452                item.item.align_self.0.value(),
1453                AlignFlags::BASELINE | AlignFlags::LAST_BASELINE
1454            ) {
1455                let baseline = item.get_or_synthesize_baseline_with_cross_size(
1456                    item.layout_result.hypothetical_cross_size,
1457                );
1458                let hypothetical_margin_box_cross_size =
1459                    item.layout_result.hypothetical_cross_size + item.item.pbm_auto_is_zero.cross;
1460                max_ascent = max_ascent.max(baseline);
1461                max_descent = max_descent.max(hypothetical_margin_box_cross_size - baseline);
1462            } else {
1463                max_outer_hypothetical_cross_size = max_outer_hypothetical_cross_size.max(
1464                    item.layout_result.hypothetical_cross_size + item.item.pbm_auto_is_zero.cross,
1465                );
1466            }
1467        }
1468
1469        // https://drafts.csswg.org/css-flexbox/#baseline-participation
1470        let largest = max_outer_hypothetical_cross_size.max(max_ascent + max_descent);
1471        match flex_context.container_inner_size_constraint.cross {
1472            SizeConstraint::MinMax(min, max) if flex_context.config.container_is_single_line => {
1473                largest.clamp_between_extremums(min, max)
1474            },
1475            _ => largest,
1476        }
1477    }

1479    fn finish_with_final_cross_size(
1480        mut self,
1481        flex_context: &mut FlexContext,
1482        main_gap: Au,
1483        final_line_cross_size: Au,
1484    ) -> FinalFlexLineLayout {
1485        // FIXME: Collapse `visibility: collapse` items
1486        // This involves “restart layout from the beginning” with a modified second round,
1487        // which will make structuring the code… interesting.
1488        // https://drafts.csswg.org/css-flexbox/#algo-visibility
1489
1490        // Distribute any remaining main free space to auto margins according to
1491        // https://drafts.csswg.org/css-flexbox/#algo-main-align.
1492        let auto_margins_count = self
1493            .items
1494            .iter()
1495            .map(|item| {
1496                item.item.margin.main_start.is_auto() as u32 +
1497                    item.item.margin.main_end.is_auto() as u32
1498            })
1499            .sum::<u32>();
1500        let (space_distributed_to_auto_main_margins, free_space_in_main_axis) =
1501            if self.free_space_in_main_axis > Au::zero() && auto_margins_count > 0 {
1502                (
1503                    self.free_space_in_main_axis / auto_margins_count as i32,
1504                    Au::zero(),
1505                )
1506            } else {
1507                (Au::zero(), self.free_space_in_main_axis)
1508            };
1509
1510        // Determine the used cross size of each flex item
1511        // https://drafts.csswg.org/css-flexbox/#algo-stretch
1512        let item_count = self.items.len();
1513        let mut shared_alignment_baseline = None;
1514        let mut item_used_cross_sizes = Vec::with_capacity(item_count);
1515        let mut item_margins = Vec::with_capacity(item_count);
1516        for item in self.items.iter_mut() {
1517            let cross_axis = match flex_context.config.flex_axis {
1518                FlexAxis::Row => Direction::Block,
1519                FlexAxis::Column => Direction::Inline,
1520            };
1521            let layout = &mut item.layout_result;
1522            let get_content_size = || match cross_axis {
1523                Direction::Block => layout.content_block_size.into(),
1524                Direction::Inline => item
1525                    .item
1526                    .inline_content_sizes(flex_context, item.used_main_size),
1527            };
1528            let used_cross_size = item.item.content_cross_sizes.resolve(
1529                cross_axis,
1530                item.item.automatic_cross_size,
1531                Au::zero,
1532                Some(Au::zero().max(final_line_cross_size - item.item.pbm_auto_is_zero.cross)),
1533                get_content_size,
1534                // Tables have a special sizing in the block axis in that handles collapsed rows,
1535                // but it would prevent stretching. So we only recognize tables in the inline axis.
1536                // The interaction of collapsed table tracks and the flexbox algorithms is unclear,
1537                // see https://github.com/w3c/csswg-drafts/issues/11408.
1538                item.item.box_.independent_formatting_context.is_table() &&
1539                    cross_axis == Direction::Inline,
1540            );
1541            item_used_cross_sizes.push(used_cross_size);
1542
1543            // “If the flex item has `align-self: stretch`, redo layout for its contents,
1544            // treating this used size as its definite cross size so that percentage-sized
1545            // children can be resolved.”
1546            // However, as resolved in https://github.com/w3c/csswg-drafts/issues/11784,
1547            // we do that when the cross size is `stretch`. We also need to do it if the
1548            // inline size changes, which may happen with a `fit-content` cross size.
1549            let needs_new_layout = match cross_axis {
1550                Direction::Block => {
1551                    (match item.item.content_cross_sizes.preferred {
1552                        Size::Initial => item.item.automatic_cross_size == Size::Stretch,
1553                        Size::Stretch => true,
1554                        _ => false,
1555                    }) && SizeConstraint::Definite(used_cross_size) !=
1556                        layout.containing_block_size.block &&
1557                        layout.depends_on_block_constraints
1558                },
1559                Direction::Inline => used_cross_size != layout.containing_block_size.inline,
1560            };
1561            if needs_new_layout {
1562                #[cfg(feature = "tracing")]
1563                tracing::warn!(
1564                    name: "Flex item stretch cache miss",
1565                    cached_inline = ?layout.containing_block_size.inline,
1566                    cached_block = ?layout.containing_block_size.block,
1567                    required_cross_size = ?used_cross_size,
1568                    cross_axis = ?cross_axis,
1569                    depends_on_block_constraints = layout.depends_on_block_constraints,
1570                );
1571                *layout =
1572                    item.item
1573                        .layout(item.used_main_size, flex_context, Some(used_cross_size));
1574            }
1575
1576            let baseline = item.get_or_synthesize_baseline_with_cross_size(used_cross_size);
1577            if matches!(
1578                item.item.align_self.0.value(),
1579                AlignFlags::BASELINE | AlignFlags::LAST_BASELINE
1580            ) {
1581                shared_alignment_baseline =
1582                    Some(shared_alignment_baseline.unwrap_or(baseline).max(baseline));
1583            }
1584            item.layout_result.baseline_relative_to_margin_box = Some(baseline);
1585
1586            item_margins.push(item.item.resolve_auto_margins(
1587                flex_context,
1588                final_line_cross_size,
1589                used_cross_size,
1590                space_distributed_to_auto_main_margins,
1591            ));
1592        }
1593
1594        // Align the items along the main-axis per justify-content.
1595        // Implement fallback alignment.
1596        //
1597        // In addition to the spec at https://www.w3.org/TR/css-align-3/ this implementation follows
1598        // the resolution of https://github.com/w3c/csswg-drafts/issues/10154
1599        let resolved_justify_content: AlignFlags = {
1600            let justify_content_style = flex_context.config.justify_content.primary();
1601
1602            // Inital values from the style system
1603            let mut resolved_justify_content = justify_content_style.value();
1604            let mut is_safe = justify_content_style.flags() == AlignFlags::SAFE;
1605
1606            // Fallback occurs in two cases:
1607
1608            // 1. If there is only a single item being aligned and alignment is a distributed alignment keyword
1609            //    https://www.w3.org/TR/css-align-3/#distribution-values
1610            if item_count <= 1 || free_space_in_main_axis <= Au::zero() {
1611                (resolved_justify_content, is_safe) = match resolved_justify_content {
1612                    AlignFlags::STRETCH => (AlignFlags::FLEX_START, true),
1613                    AlignFlags::SPACE_BETWEEN => (AlignFlags::FLEX_START, true),
1614                    AlignFlags::SPACE_AROUND => (AlignFlags::CENTER, true),
1615                    AlignFlags::SPACE_EVENLY => (AlignFlags::CENTER, true),
1616                    _ => (resolved_justify_content, is_safe),
1617                }
1618            };
1619
1620            // 2. If free space is negative the "safe" alignment variants all fallback to Start alignment
1621            if free_space_in_main_axis <= Au::zero() && is_safe {
1622                resolved_justify_content = AlignFlags::START;
1623            }
1624
1625            resolved_justify_content
1626        };
1627
1628        // Implement "unsafe" alignment. "safe" alignment is handled by the fallback process above.
1629        let main_start_position = match resolved_justify_content {
1630            AlignFlags::START => Au::zero(),
1631            AlignFlags::FLEX_START => {
1632                if flex_context.config.flex_direction_is_reversed {
1633                    free_space_in_main_axis
1634                } else {
1635                    Au::zero()
1636                }
1637            },
1638            AlignFlags::END => free_space_in_main_axis,
1639            AlignFlags::FLEX_END => {
1640                if flex_context.config.flex_direction_is_reversed {
1641                    Au::zero()
1642                } else {
1643                    free_space_in_main_axis
1644                }
1645            },
1646            AlignFlags::CENTER => free_space_in_main_axis / 2,
1647            AlignFlags::STRETCH => Au::zero(),
1648            AlignFlags::SPACE_BETWEEN => Au::zero(),
1649            AlignFlags::SPACE_AROUND => (free_space_in_main_axis / item_count as i32) / 2,
1650            AlignFlags::SPACE_EVENLY => free_space_in_main_axis / (item_count + 1) as i32,
1651
1652            // TODO: Implement all alignments. Note: not all alignment values are valid for content distribution
1653            _ => Au::zero(),
1654        };
1655
1656        let item_main_interval = match resolved_justify_content {
1657            AlignFlags::START => Au::zero(),
1658            AlignFlags::FLEX_START => Au::zero(),
1659            AlignFlags::END => Au::zero(),
1660            AlignFlags::FLEX_END => Au::zero(),
1661            AlignFlags::CENTER => Au::zero(),
1662            AlignFlags::STRETCH => Au::zero(),
1663            AlignFlags::SPACE_BETWEEN => free_space_in_main_axis / (item_count - 1) as i32,
1664            AlignFlags::SPACE_AROUND => free_space_in_main_axis / item_count as i32,
1665            AlignFlags::SPACE_EVENLY => free_space_in_main_axis / (item_count + 1) as i32,
1666
1667            // TODO: Implement all alignments. Note: not all alignment values are valid for content distribution
1668            _ => Au::zero(),
1669        };
1670        let item_main_interval = item_main_interval + main_gap;
1671
1672        let mut all_baselines = Baselines::default();
1673        let mut main_position_cursor = main_start_position;
1674
1675        let items = std::mem::take(&mut self.items);
1676        let item_fragments = izip!(items, item_margins, item_used_cross_sizes.iter())
1677            .map(|(item, item_margin, item_used_cross_size)| {
1678                let item_used_size = FlexRelativeVec2 {
1679                    main: item.used_main_size,
1680                    cross: *item_used_cross_size,
1681                };
1682                item.collect_fragment(
1683                    &self,
1684                    item_used_size,
1685                    item_margin,
1686                    item_main_interval,
1687                    final_line_cross_size,
1688                    &shared_alignment_baseline,
1689                    flex_context,
1690                    &mut all_baselines,
1691                    &mut main_position_cursor,
1692                )
1693            })
1694            .collect();
1695
1696        FinalFlexLineLayout {
1697            cross_size: final_line_cross_size,
1698            item_fragments,
1699            all_baselines,
1700            shared_alignment_baseline,
1701        }
1702    }
1703}

1706    /// Return the hypothetical cross size together with laid out contents of the fragment.
1707    /// From <https://drafts.csswg.org/css-flexbox/#algo-cross-item>:
1708    /// > performing layout as if it were an in-flow block-level box with the used main

1720        &self,

1722        flex_context: &FlexContext,
1723        used_cross_size_override: Option<Au>,
1724    ) -> FlexItemLayoutResult {

1726        let independent_formatting_context = &self.box_.independent_formatting_context;
1727        let is_table = independent_formatting_context.is_table();

1730        let item_is_horizontal = item_writing_mode.is_horizontal();
1731        let flex_axis = flex_context.config.flex_axis;
1732        let cross_axis_is_item_block_axis = cross_axis_is_item_block_axis(
1733            containing_block.style.writing_mode.is_horizontal(),
1734            item_is_horizontal,
1735            flex_axis,
1736        );
1737
1738        let (inline_size, block_size) = if cross_axis_is_item_block_axis {
1739            let cross_size = match used_cross_size_override {
1740                Some(s) => SizeConstraint::Definite(s),
1741                None => {
1742                    let stretch_size = containing_block
1743                        .size
1744                        .block
1745                        .to_definite()
1746                        .map(|size| Au::zero().max(size - self.pbm_auto_is_zero.cross));
1747                    let tentative_block_content_size = independent_formatting_context
1748                        .tentative_block_content_size(self.preferred_aspect_ratio);
1749                    if let Some(block_content_size) = tentative_block_content_size {

1751                            Direction::Block,
1752                            Size::FitContent,
1753                            Au::zero,
1754                            stretch_size,
1755                            || block_content_size,
1756                            is_table,
1757                        ))
1758                    } else {
1759                        self.content_cross_sizes.resolve_extrinsic(
1760                            Size::FitContent,
1761                            Au::zero(),
1762                            stretch_size,
1763                        )
1764                    }
1765                },
1766            };
1767            (used_main_size, cross_size)
1768        } else {
1769            let cross_size = used_cross_size_override.unwrap_or_else(|| {
1770                let stretch_size =
1771                    Au::zero().max(containing_block.size.inline - self.pbm_auto_is_zero.cross);
1772                self.content_cross_sizes.resolve(
1773                    Direction::Inline,
1774                    Size::FitContent,
1775                    Au::zero,
1776                    Some(stretch_size),
1777                    || self.inline_content_sizes(flex_context, used_main_size),
1778                    is_table,
1779                )
1780            });
1781            // The main size of a flex item is considered to be definite if its flex basis is definite
1782            // or the flex container has a definite main size.
1783            // <https://drafts.csswg.org/css-flexbox-1/#definite-sizes>
1784            let main_size = if self.flex_base_size_is_definite ||
1785                flex_context
1786                    .container_inner_size_constraint
1787                    .main
1788                    .is_definite()
1789            {
1790                SizeConstraint::Definite(used_main_size)
1791            } else {
1792                SizeConstraint::default()
1793            };
1794            (cross_size, main_size)
1795        };
1796
1797        let item_style = independent_formatting_context.style();
1798        let item_as_containing_block = ContainingBlock {
1799            size: ContainingBlockSize {
1800                inline: inline_size,
1801                block: block_size,
1802            },
1803            style: item_style,
1804        };
1805
1806        let lazy_block_size = if !cross_axis_is_item_block_axis {
1807            used_main_size.into()
1808        } else if let Some(cross_size) = used_cross_size_override {
1809            cross_size.into()
1810        } else {
1811            let stretch_size = containing_block
1812                .size
1813                .block
1814                .to_definite()
1815                .map(|size| Au::zero().max(size - self.pbm_auto_is_zero.cross));
1816            LazySize::new(
1817                &self.content_cross_sizes,
1818                Direction::Block,
1819                Size::FitContent,
1820                Au::zero,
1821                stretch_size,
1822                is_table,
1823            )
1824        };
1825
1826        let layout = independent_formatting_context.layout(
1827            flex_context.layout_context,
1828            &mut positioning_context,
1829            &item_as_containing_block,
1830            containing_block,
1831            self.preferred_aspect_ratio,
1832            &lazy_block_size,
1833        );

1835            fragments,
1836            content_block_size,
1837            baselines: content_box_baselines,
1838            depends_on_block_constraints,
1839            specific_layout_info,
1840            ..
1841        } = layout;
1842

1844            lazy_block_size.resolve(|| content_block_size)
1845        } else {
1846            inline_size
1847        };
1848
1849        let item_writing_mode_is_orthogonal_to_container_writing_mode =
1850            flex_context.config.writing_mode.is_horizontal() !=
1851                item_style.writing_mode.is_horizontal();
1852        let has_compatible_baseline = match flex_axis {
1853            FlexAxis::Row => !item_writing_mode_is_orthogonal_to_container_writing_mode,
1854            FlexAxis::Column => item_writing_mode_is_orthogonal_to_container_writing_mode,
1855        };
1856
1857        let baselines_relative_to_margin_box = if has_compatible_baseline {
1858            content_box_baselines.offset(
1859                self.margin.cross_start.auto_is(Au::zero) +
1860                    self.padding.cross_start +
1861                    self.border.cross_start,
1862            )
1863        } else {
1864            Baselines::default()
1865        };
1866
1867        let baseline_relative_to_margin_box = match self.align_self.0.value() {
1868            // ‘baseline’ computes to ‘first baseline’.
1869            AlignFlags::BASELINE => baselines_relative_to_margin_box.first,
1870            AlignFlags::LAST_BASELINE => baselines_relative_to_margin_box.last,
1871            _ => None,
1872        };
1873
1874        FlexItemLayoutResult {
1875            hypothetical_cross_size,
1876            fragments,
1877            positioning_context,
1878            baseline_relative_to_margin_box,
1879            content_block_size,
1880            containing_block_size: item_as_containing_block.size,
1881            depends_on_block_constraints,
1882            specific_layout_info,
1883        }

1887        // If the item does not have a baseline in the necessary axis,
1888        // then one is synthesized from the flex item’s border box.
1889        // https://drafts.csswg.org/css-flexbox/#valdef-align-items-baseline
1890        content_size +
1891            self.margin.cross_start.auto_is(Au::zero) +

1894            self.border.cross_end +
1895            self.padding.cross_end

1898    /// Return the cross-start, cross-end, main-start, and main-end margins, with `auto` values resolved.
1899    /// See:
1900    ///
1901    /// - <https://drafts.csswg.org/css-flexbox/#algo-cross-margins>
1902    /// - <https://drafts.csswg.org/css-flexbox/#algo-main-align>
1903    fn resolve_auto_margins(
1904        &self,
1905        flex_context: &FlexContext,
1906        line_cross_size: Au,
1907        item_cross_content_size: Au,
1908        space_distributed_to_auto_main_margins: Au,
1909    ) -> FlexRelativeSides<Au> {
1910        let main_start = self
1911            .margin
1912            .main_start
1913            .auto_is(|| space_distributed_to_auto_main_margins);
1914        let main_end = self
1915            .margin
1916            .main_end
1917            .auto_is(|| space_distributed_to_auto_main_margins);
1918
1919        let auto_count = match (self.margin.cross_start, self.margin.cross_end) {
1920            (AuOrAuto::LengthPercentage(cross_start), AuOrAuto::LengthPercentage(cross_end)) => {
1921                return FlexRelativeSides {
1922                    cross_start,
1923                    cross_end,
1924                    main_start,
1925                    main_end,
1926                };
1927            },
1928            (AuOrAuto::Auto, AuOrAuto::Auto) => 2,
1929            _ => 1,
1930        };
1931        let outer_cross_size = self.pbm_auto_is_zero.cross + item_cross_content_size;
1932        let available = line_cross_size - outer_cross_size;
1933        let cross_start;
1934        let cross_end;
1935        if available > Au::zero() {
1936            let each_auto_margin = available / auto_count;
1937            cross_start = self.margin.cross_start.auto_is(|| each_auto_margin);
1938            cross_end = self.margin.cross_end.auto_is(|| each_auto_margin);
1939        } else {
1940            // “the block-start or inline-start margin (whichever is in the cross axis)”
1941            // This margin is the cross-end on iff `flex-wrap` is `wrap-reverse`,
1942            // cross-start otherwise.
1943            // We know this because:
1944            // https://drafts.csswg.org/css-flexbox/#flex-wrap-property
1945            // “For the values that are not wrap-reverse,
1946            //  the cross-start direction is equivalent to
1947            //  either the inline-start or block-start direction of the current writing mode
1948            //  (whichever is in the cross axis)
1949            //  and the cross-end direction is the opposite direction of cross-start.
1950            //  When flex-wrap is wrap-reverse,
1951            //  the cross-start and cross-end directions are swapped.”
1952            let flex_wrap = flex_context.containing_block.style.get_position().flex_wrap;
1953            let flex_wrap_reverse = match flex_wrap {
1954                FlexWrap::Nowrap | FlexWrap::Wrap => false,
1955                FlexWrap::WrapReverse => true,
1956            };
1957            // “if the block-start or inline-start margin (whichever is in the cross axis) is auto,
1958            //  set it to zero. Set the opposite margin so that the outer cross size of the item
1959            //  equals the cross size of its flex line.”
1960            if flex_wrap_reverse {
1961                cross_start = self.margin.cross_start.auto_is(|| available);
1962                cross_end = self.margin.cross_end.auto_is(Au::zero);
1963            } else {
1964                cross_start = self.margin.cross_start.auto_is(Au::zero);
1965                cross_end = self.margin.cross_end.auto_is(|| available);
1966            }
1967        }
1968
1969        FlexRelativeSides {
1970            cross_start,
1971            cross_end,
1972            main_start,
1973            main_end,
1974        }
1975    }

1977    /// Return the coordinate of the cross-start side of the content area
1978    fn align_along_cross_axis(
1979        &self,
1980        margin: &FlexRelativeSides<Au>,
1981        used_cross_size: &Au,
1982        line_cross_size: Au,
1983        propagated_baseline: Au,
1984        max_propagated_baseline: Au,
1985        wrap_reverse: bool,
1986    ) -> Au {
1987        let ending_alignment = line_cross_size - *used_cross_size - self.pbm_auto_is_zero.cross;
1988        let outer_cross_start =
1989            if self.margin.cross_start.is_auto() || self.margin.cross_end.is_auto() {
1990                Au::zero()
1991            } else {
1992                match self.align_self.0.value() {
1993                    AlignFlags::STRETCH => Au::zero(),
1994                    AlignFlags::CENTER => ending_alignment / 2,
1995                    AlignFlags::BASELINE | AlignFlags::LAST_BASELINE => {
1996                        max_propagated_baseline - propagated_baseline
1997                    },
1998                    AlignFlags::START => {
1999                        if !wrap_reverse {
2000                            Au::zero()
2001                        } else {
2002                            ending_alignment
2003                        }
2004                    },
2005                    AlignFlags::END => {
2006                        if !wrap_reverse {
2007                            ending_alignment
2008                        } else {
2009                            Au::zero()
2010                        }
2011                    },
2012                    _ => Au::zero(),
2013                }
2014            };
2015        outer_cross_start + margin.cross_start + self.border.cross_start + self.padding.cross_start
2016    }

2277    fn main_content_size_info<'a>(
2278        &self,
2279        layout_context: &LayoutContext,
2280        containing_block: &IndefiniteContainingBlock,
2281        config: &FlexContainerConfig,
2282        flex_context_getter: &impl Fn() -> &'a FlexContext<'a>,
2283    ) -> FlexItemBoxInlineContentSizesInfo {
2284        let content_box_sizes_and_pbm = self
2285            .independent_formatting_context
2286            .layout_style()
2287            .content_box_sizes_and_padding_border_margin(containing_block);
2288
2289        // TODO: when laying out a column container with an indefinite main size,
2290        // we compute the base sizes of the items twice. We should consider caching.
2291        let FlexItem {
2292            flex_base_size,
2293            content_min_main_size,
2294            content_max_main_size,
2295            pbm_auto_is_zero,
2296            preferred_aspect_ratio,
2297            automatic_cross_size_for_intrinsic_sizing,
2298            ..
2299        } = self.to_flex_item(
2300            layout_context,
2301            containing_block,
2302            &content_box_sizes_and_pbm,
2303            config,
2304            flex_context_getter,
2305        );
2306
2307        // Compute the min-content and max-content contributions of the item.
2308        // <https://drafts.csswg.org/css-flexbox/#intrinsic-item-contributions>
2309        let (content_contribution_sizes, depends_on_block_constraints) = match config.flex_axis {
2310            FlexAxis::Row => {
2311                let auto_minimum = LogicalVec2 {
2312                    inline: content_min_main_size,
2313                    block: Au::zero(),
2314                };
2315                let InlineContentSizesResult {
2316                    sizes,
2317                    depends_on_block_constraints,
2318                } = self
2319                    .independent_formatting_context
2320                    .outer_inline_content_sizes(
2321                        layout_context,
2322                        containing_block,
2323                        &auto_minimum,
2324                        automatic_cross_size_for_intrinsic_sizing == Size::Stretch,
2325                    );
2326                (sizes, depends_on_block_constraints)
2327            },
2328            FlexAxis::Column => {
2329                let size = self.layout_for_block_content_size(
2330                    flex_context_getter(),
2331                    &pbm_auto_is_zero,
2332                    &content_box_sizes_and_pbm.content_box_sizes,
2333                    preferred_aspect_ratio,
2334                    automatic_cross_size_for_intrinsic_sizing,
2335                    IntrinsicSizingMode::Contribution,
2336                );
2337                (size.into(), true)
2338            },
2339        };
2340
2341        let outer_flex_base_size = flex_base_size + pbm_auto_is_zero.main;
2342        let outer_min_main_size = content_min_main_size + pbm_auto_is_zero.main;
2343        let outer_max_main_size = content_max_main_size.map(|v| v + pbm_auto_is_zero.main);
2344        let max_flex_factors = self.desired_flex_factors_for_preferred_width(
2345            content_contribution_sizes.max_content,
2346            flex_base_size,
2347            outer_flex_base_size,
2348        );
2349
2350        // > The min-content main size of a single-line flex container is calculated
2351        // > identically to the max-content main size, except that the flex items’
2352        // > min-content contributions are used instead of their max-content contributions.
2353        let min_flex_factors = self.desired_flex_factors_for_preferred_width(
2354            content_contribution_sizes.min_content,
2355            flex_base_size,
2356            outer_flex_base_size,
2357        );
2358
2359        // > However, for a multi-line container, the min-content main size is simply the
2360        // > largest min-content contribution of all the non-collapsed flex items in the
2361        // > flex container. For this purpose, each item’s contribution is capped by the
2362        // > item’s flex base size if the item is not growable, floored by the item’s flex
2363        // > base size if the item is not shrinkable, and then further clamped by the item’s
2364        // > min and max main sizes.
2365        let mut min_content_main_size_for_multiline_container =
2366            content_contribution_sizes.min_content;
2367        let style_position = &self.style().get_position();
2368        if style_position.flex_grow.is_zero() {
2369            min_content_main_size_for_multiline_container.min_assign(outer_flex_base_size);
2370        }
2371        if style_position.flex_shrink.is_zero() {
2372            min_content_main_size_for_multiline_container.max_assign(outer_flex_base_size);
2373        }
2374        min_content_main_size_for_multiline_container =
2375            min_content_main_size_for_multiline_container
2376                .clamp_between_extremums(outer_min_main_size, outer_max_main_size);
2377
2378        FlexItemBoxInlineContentSizesInfo {
2379            outer_flex_base_size,
2380            outer_min_main_size,
2381            outer_max_main_size,
2382            min_flex_factors,
2383            max_flex_factors,
2384            min_content_main_size_for_multiline_container,
2385            depends_on_block_constraints,
2386        }
2387    }

2389    fn desired_flex_factors_for_preferred_width(
2390        &self,
2391        preferred_width: Au,
2392        flex_base_size: Au,
2393        outer_flex_base_size: Au,
2394    ) -> DesiredFlexFractionAndGrowOrShrinkFactor {
2395        let difference = (preferred_width - outer_flex_base_size).to_f32_px();
2396        let (flex_grow_or_scaled_flex_shrink_factor, desired_flex_fraction) = if difference > 0.0 {
2397            // > If that result is positive, divide it by the item’s flex
2398            // > grow factor if the flex grow > factor is ≥ 1, or multiply
2399            // > it by the flex grow factor if the flex grow factor is < 1;
2400            let flex_grow_factor = self.style().get_position().flex_grow.0;
2401
2402            (
2403                flex_grow_factor,
2404                if flex_grow_factor >= 1.0 {
2405                    difference / flex_grow_factor
2406                } else {
2407                    difference * flex_grow_factor
2408                },
2409            )
2410        } else if difference < 0.0 {
2411            // > if the result is negative, divide it by the item’s scaled
2412            // > flex shrink factor (if dividing > by zero, treat the result
2413            // > as negative infinity).
2414            let flex_shrink_factor = self.style().get_position().flex_shrink.0;
2415            let scaled_flex_shrink_factor = flex_shrink_factor * flex_base_size.to_f32_px();
2416
2417            (
2418                scaled_flex_shrink_factor,
2419                if scaled_flex_shrink_factor != 0.0 {
2420                    difference / scaled_flex_shrink_factor
2421                } else {
2422                    f32::NEG_INFINITY
2423                },
2424            )
2425        } else {
2426            (0.0, 0.0)
2427        };
2428
2429        DesiredFlexFractionAndGrowOrShrinkFactor {
2430            desired_flex_fraction,
2431            flex_grow_or_shrink_factor: flex_grow_or_scaled_flex_shrink_factor,
2432        }
2433    }

2435    /// <https://drafts.csswg.org/css-flexbox-1/#flex-basis-property>
2436    /// Returns the used value of the `flex-basis` property, after resolving percentages,
2437    /// resolving `auto`, and taking `box-sizing` into account.
2438    /// Note that a return value of `Size::Initial` represents `flex-basis: content`,
2439    /// not `flex-basis: auto`, since the latter always resolves to something else.
2440    fn flex_basis(
2441        &self,
2442        container_definite_main_size: Option<Au>,
2443        main_preferred_size: Size<Au>,
2444        main_padding_border_sum: Au,
2445    ) -> Size<Au> {
2446        let style_position = &self.independent_formatting_context.style().get_position();
2447        match &style_position.flex_basis {
2448            // https://drafts.csswg.org/css-flexbox-1/#valdef-flex-basis-content
2449            // > Indicates an automatic size based on the flex item’s content.
2450            FlexBasis::Content => Size::Initial,
2451
2452            FlexBasis::Size(size) => match Size::<LengthPercentage>::from(size.clone()) {
2453                // https://drafts.csswg.org/css-flexbox-1/#valdef-flex-basis-auto
2454                // > When specified on a flex item, the `auto` keyword retrieves
2455                // > the value of the main size property as the used `flex-basis`.
2456                Size::Initial => main_preferred_size,
2457
2458                // https://drafts.csswg.org/css-flexbox-1/#flex-basis-property
2459                // > For all values other than `auto` and `content` (defined above),
2460                // > `flex-basis` is resolved the same way as `width` in horizontal
2461                // > writing modes, except that if a value would resolve to `auto`
2462                // > for `width`, it instead resolves to `content` for `flex-basis`.
2463                // > For example, percentage values of `flex-basis` are resolved
2464                // > against the flex item’s containing block (i.e. its flex container);
2465                // > and if that containing block’s size is indefinite,
2466                // > the used value for `flex-basis` is `content`.
2467                size => {
2468                    let apply_box_sizing = |length: Au| {
2469                        match style_position.box_sizing {
2470                            BoxSizing::ContentBox => length,
2471                            // This may make `length` negative,
2472                            // but it will be clamped in the hypothetical main size
2473                            BoxSizing::BorderBox => length - main_padding_border_sum,
2474                        }
2475                    };
2476                    size.resolve_percentages_for_preferred(container_definite_main_size)
2477                        .map(apply_box_sizing)
2478                },
2479            },
2480        }
2481    }

2483    #[allow(clippy::too_many_arguments)]
2484    #[servo_tracing::instrument(name = "FlexContainer::layout_for_block_content_size", skip_all)]
2485    fn layout_for_block_content_size(
2486        &self,
2487        flex_context: &FlexContext,
2488        pbm_auto_is_zero: &FlexRelativeVec2<Au>,
2489        content_box_sizes: &LogicalVec2<Sizes>,
2490        preferred_aspect_ratio: Option<AspectRatio>,
2491        automatic_inline_size: Size<Au>,
2492        intrinsic_sizing_mode: IntrinsicSizingMode,
2493    ) -> Au {
2494        let content_block_size = || {
2495            let mut positioning_context = PositioningContext::default();
2496            let style = self.independent_formatting_context.style();
2497
2498            // We are computing the intrinsic block size, so the tentative block size that we use
2499            // as an input to the intrinsic inline sizes needs to ignore the values of the sizing
2500            // properties in the block axis.
2501            let tentative_block_size = SizeConstraint::default();
2502
2503            // TODO: This is wrong if the item writing mode is different from the flex
2504            // container's writing mode.
2505            let inline_size = {
2506                let stretch_size =
2507                    flex_context.containing_block.size.inline - pbm_auto_is_zero.cross;
2508                let get_content_size = || {
2509                    self.inline_content_sizes(
2510                        flex_context,
2511                        tentative_block_size,
2512                        preferred_aspect_ratio,
2513                    )
2514                };
2515                content_box_sizes.inline.resolve(
2516                    Direction::Inline,
2517                    automatic_inline_size,
2518                    Au::zero,
2519                    Some(stretch_size),
2520                    get_content_size,
2521                    false,
2522                )
2523            };
2524            let item_as_containing_block = ContainingBlock {
2525                size: ContainingBlockSize {
2526                    inline: inline_size,
2527                    block: tentative_block_size,
2528                },
2529                style,
2530            };
2531            self.independent_formatting_context
2532                .layout(
2533                    flex_context.layout_context,
2534                    &mut positioning_context,
2535                    &item_as_containing_block,
2536                    flex_context.containing_block,
2537                    preferred_aspect_ratio,
2538                    &LazySize::intrinsic(),
2539                )
2540                .content_block_size
2541        };
2542        match intrinsic_sizing_mode {
2543            IntrinsicSizingMode::Contribution => {
2544                let stretch_size = flex_context
2545                    .containing_block
2546                    .size
2547                    .block
2548                    .to_definite()
2549                    .map(|block_size| block_size - pbm_auto_is_zero.main);
2550                let inner_block_size = content_box_sizes.block.resolve(
2551                    Direction::Block,
2552                    Size::FitContent,
2553                    Au::zero,
2554                    stretch_size,
2555                    || ContentSizes::from(content_block_size()),
2556                    // Tables have a special sizing in the block axis that handles collapsed rows
2557                    // by ignoring the sizing properties and instead relying on the content block size,
2558                    // which should indirectly take sizing properties into account.
2559                    // However, above we laid out the table with a SizeConstraint::default() block size,
2560                    // so the content block size doesn't take sizing properties into account.
2561                    // Therefore, pretending that it's never a table tends to provide a better result.
2562                    false, /* is_table */
2563                );
2564                inner_block_size + pbm_auto_is_zero.main
2565            },
2566            IntrinsicSizingMode::Size => content_block_size(),
2567        }
2568    }
