41// There are two reasons why we might want to break at the start:
42//
43//  1. The line breaker told us that a break was necessary between two separate
44//     instances of sending text to it.
45//  2. We are following replaced content ie `have_deferred_soft_wrap_opportunity`.
46//
47// In both cases, we don't want to do this if the first character prevents a
48// soft wrap opportunity.
49#[derive(PartialEq)]
50enum SegmentStartSoftWrapPolicy {
51    Force,
52    FollowLinebreaker,
53}

79    fn new(font: FontRef, script: Script, bidi_level: Level, start_offset: usize) -> Self {
80        Self {
81            font,
82            script,
83            bidi_level,
84            range: start_offset..start_offset,
85            runs: Vec::new(),
86            break_at_start: false,
87        }

89
90    /// Update this segment if the Font and Script are compatible. The update will only
91    /// ever make the Script specific. Returns true if the new Font and Script are
92    /// compatible with this segment or false otherwise.
93    fn update_if_compatible(
94        &mut self,
95        layout_context: &LayoutContext,
96        new_font: &FontRef,
97        script: Script,
98        bidi_level: Level,
99    ) -> bool {
100        fn is_specific(script: Script) -> bool {
101            script != Script::Common && script != Script::Inherited
102        }
103
104        if bidi_level != self.bidi_level {
105            return false;
106        }
107
108        let painter_id = layout_context.painter_id;
109        let font_context = &layout_context.font_context;
110        if new_font.key(painter_id, font_context) !=
111            self.font
112                .key(layout_context.painter_id, &layout_context.font_context) ||
113            new_font.descriptor.pt_size != self.font.descriptor.pt_size
114        {
115            return false;
116        }
117
118        if !is_specific(self.script) && is_specific(script) {
119            self.script = script;
120        }
121        script == self.script || !is_specific(script)
122    }
123
124    fn layout_into_line_items(
125        &self,
126        text_run: &TextRun,
127        mut soft_wrap_policy: SegmentStartSoftWrapPolicy,
128        ifc: &mut InlineFormattingContextLayout,
129    ) {
130        if self.break_at_start && soft_wrap_policy == SegmentStartSoftWrapPolicy::FollowLinebreaker
131        {
132            soft_wrap_policy = SegmentStartSoftWrapPolicy::Force;
133        }
134
135        let mut byte_processed = ByteIndex(0);
136        for (run_index, run) in self.runs.iter().enumerate() {
137            ifc.possibly_flush_deferred_forced_line_break();
138
139            // If this whitespace forces a line break, queue up a hard line break the next time we
140            // see any content. We don't line break immediately, because we'd like to finish processing
141            // any ongoing inline boxes before ending the line.
142            if run.is_single_preserved_newline() {
143                byte_processed = byte_processed + run.range.len();
144                ifc.defer_forced_line_break();
145                continue;
146            }
147            // Break before each unbreakable run in this TextRun, except the first unless the
148            // linebreaker was set to break before the first run.
149            if run_index != 0 || soft_wrap_policy == SegmentStartSoftWrapPolicy::Force {
150                ifc.process_soft_wrap_opportunity();
151            }
152
153            let range_start = byte_processed + ByteIndex(self.range.start as isize);
154            ifc.push_glyph_store_to_unbreakable_segment(
155                run.glyph_store.clone(),
156                text_run,
157                &self.font,
158                self.bidi_level,
159                TextByteRange::new(range_start, range_start + run.range.len()),
160            );
161            byte_processed = byte_processed + run.range.len();
162        }
163    }
164
165    fn shape_and_push_range(
166        &mut self,
167        range: &Range<usize>,
168        formatting_context_text: &str,
169        options: &ShapingOptions,
170    ) {
171        self.runs.push(GlyphRun {
172            glyph_store: self
173                .font
174                .shape_text(&formatting_context_text[range.clone()], options),
175            range: TextByteRange::new(
176                ByteIndex(range.start as isize),
177                ByteIndex(range.end as isize),
178            ),
179        });
180    }
181
182    /// Shape the text of this [`TextRunSegment`], first finding "words" for the shaper by processing
183    /// the linebreaks found in the owning [`super::InlineFormattingContext`]. Linebreaks are filtered,
184    /// based on the style of the parent inline box.
185    fn shape_text(
186        &mut self,
187        parent_style: &ComputedValues,
188        formatting_context_text: &str,
189        linebreaker: &mut LineBreaker,
190        shaping_options: &ShapingOptions,
191    ) {
192        // Gather the linebreaks that apply to this segment from the inline formatting context's collection
193        // of line breaks. Also add a simulated break at the end of the segment in order to ensure the final
194        // piece of text is processed.
195        let range = self.range.clone();
196        let linebreaks = linebreaker.advance_to_linebreaks_in_range(self.range.clone());
197        let linebreak_iter = linebreaks.iter().chain(std::iter::once(&range.end));
198
199        self.runs.clear();
200        self.runs.reserve(linebreaks.len());
201        self.break_at_start = false;
202
203        let text_style = parent_style.get_inherited_text().clone();
204        let can_break_anywhere = text_style.word_break == WordBreak::BreakAll ||
205            text_style.overflow_wrap == OverflowWrap::Anywhere ||
206            text_style.overflow_wrap == OverflowWrap::BreakWord;
207
208        let mut last_slice = self.range.start..self.range.start;
209        for break_index in linebreak_iter {
210            if *break_index == self.range.start {
211                self.break_at_start = true;
212                continue;
213            }
214
215            let mut options = *shaping_options;
216
217            // Extend the slice to the next UAX#14 line break opportunity.
218            let mut slice = last_slice.end..*break_index;
219            let word = &formatting_context_text[slice.clone()];
220
221            // Split off any trailing whitespace into a separate glyph run.
222            let mut whitespace = slice.end..slice.end;
223            let mut rev_char_indices = word.char_indices().rev().peekable();
224
225            let mut ends_with_whitespace = false;
226            let ends_with_newline = rev_char_indices
227                .peek()
228                .is_some_and(|&(_, character)| character == '\n');
229            if let Some((first_white_space_index, first_white_space_character)) = rev_char_indices
230                .take_while(|&(_, character)| char_is_whitespace(character))
231                .last()
232            {
233                ends_with_whitespace = true;
234                whitespace.start = slice.start + first_white_space_index;
235
236                // If line breaking for a piece of text that has `white-space-collapse: break-spaces` there
237                // is a line break opportunity *after* every preserved space, but not before. This means
238                // that we should not split off the first whitespace, unless that white-space is a preserved
239                // newline.
240                //
241                // An exception to this is if the style tells us that we can break in the middle of words.
242                if text_style.white_space_collapse == WhiteSpaceCollapse::BreakSpaces &&
243                    first_white_space_character != '\n' &&
244                    !can_break_anywhere
245                {
246                    whitespace.start += first_white_space_character.len_utf8();
247                    options
248                        .flags
249                        .insert(ShapingFlags::ENDS_WITH_WHITESPACE_SHAPING_FLAG);
250                }
251
252                slice.end = whitespace.start;
253            }
254
255            // If there's no whitespace and `word-break` is set to `keep-all`, try increasing the slice.
256            // TODO: This should only happen for CJK text.
257            if !ends_with_whitespace &&
258                *break_index != self.range.end &&
259                text_style.word_break == WordBreak::KeepAll &&
260                !can_break_anywhere
261            {
262                continue;
263            }
264
265            // Only advance the last slice if we are not going to try to expand the slice.
266            last_slice = slice.start..*break_index;
267
268            // Push the non-whitespace part of the range.
269            if !slice.is_empty() {
270                self.shape_and_push_range(&slice, formatting_context_text, &options);
271            }
272
273            if whitespace.is_empty() {
274                continue;
275            }
276
277            options.flags.insert(
278                ShapingFlags::IS_WHITESPACE_SHAPING_FLAG |
279                    ShapingFlags::ENDS_WITH_WHITESPACE_SHAPING_FLAG,
280            );
281
282            // If `white-space-collapse: break-spaces` is active, insert a line breaking opportunity
283            // between each white space character in the white space that we trimmed off.
284            if text_style.white_space_collapse == WhiteSpaceCollapse::BreakSpaces {
285                let start_index = whitespace.start;
286                for (index, character) in formatting_context_text[whitespace].char_indices() {
287                    let index = start_index + index;
288                    self.shape_and_push_range(
289                        &(index..index + character.len_utf8()),
290                        formatting_context_text,
291                        &options,
292                    );
293                }
294                continue;
295            }
296
297            // The breaker breaks after every newline, so either there is none,
298            // or there is exactly one at the very end. In the latter case,
299            // split it into a different run. That's because shaping considers
300            // a newline to have the same advance as a space, but during layout
301            // we want to treat the newline as having no advance.
302            if ends_with_newline && whitespace.len() > 1 {
303                self.shape_and_push_range(
304                    &(whitespace.start..whitespace.end - 1),
305                    formatting_context_text,
306                    &options,
307                );
308                self.shape_and_push_range(
309                    &(whitespace.end - 1..whitespace.end),
310                    formatting_context_text,
311                    &options,
312                );
313            } else {
314                self.shape_and_push_range(&whitespace, formatting_context_text, &options);
315            }
316        }
317    }
318}

320/// A single [`TextRun`] for the box tree. These are all descendants of
321/// [`super::InlineBox`] or the root of the [`super::InlineFormattingContext`].  During
322/// box tree construction, text is split into [`TextRun`]s based on their font, script,
323/// etc. When these are created text is already shaped.
324///
325/// <https://www.w3.org/TR/css-display-3/#css-text-run>
326#[derive(Debug, MallocSizeOf)]
327pub(crate) struct TextRun {
328    /// The [`BaseFragmentInfo`] for this [`TextRun`]. Usually this comes from the
329    /// original text node in the DOM for the text.
330    pub base_fragment_info: BaseFragmentInfo,
331
332    /// The [`crate::SharedStyle`] from this [`TextRun`]s parent element. This is
333    /// shared so that incremental layout can simply update the parent element and
334    /// this [`TextRun`] will be updated automatically.
335    pub inline_styles: SharedInlineStyles,
336
337    /// The range of text in [`super::InlineFormattingContext::text_content`] of the
338    /// [`super::InlineFormattingContext`] that owns this [`TextRun`]. These are UTF-8 offsets.
339    pub text_range: Range<usize>,
340
341    /// The text of this [`TextRun`] with a font selected, broken into unbreakable
342    /// segments, and shaped.
343    pub shaped_text: Vec<TextRunSegment>,
344
345    /// The selection range for the DOM text node that originated this [`TextRun`]. This
346    /// comes directly from the DOM.
347    pub selection_range: Option<TextByteRange>,
348}

350impl TextRun {
351    pub(crate) fn new(
352        base_fragment_info: BaseFragmentInfo,
353        inline_styles: SharedInlineStyles,
354        text_range: Range<usize>,
355        selection_range: Option<TextByteRange>,
356    ) -> Self {
357        Self {
358            base_fragment_info,
359            inline_styles,
360            text_range,
361            shaped_text: Vec::new(),
362            selection_range,
363        }
364    }

366    pub(super) fn segment_and_shape(
367        &mut self,
368        formatting_context_text: &str,
369        layout_context: &LayoutContext,
370        linebreaker: &mut LineBreaker,
371        bidi_info: &BidiInfo,
372    ) {
373        let parent_style = self.inline_styles.style.borrow().clone();
374        let inherited_text_style = parent_style.get_inherited_text().clone();
375        let letter_spacing = inherited_text_style
376            .letter_spacing
377            .0
378            .resolve(parent_style.clone_font().font_size.computed_size());
379        let letter_spacing = if letter_spacing.px() != 0. {
380            Some(app_units::Au::from(letter_spacing))
381        } else {
382            None
383        };
384
385        let mut flags = ShapingFlags::empty();
386        if letter_spacing.is_some() {
387            flags.insert(ShapingFlags::IGNORE_LIGATURES_SHAPING_FLAG);
388        }
389        if inherited_text_style.text_rendering == TextRendering::Optimizespeed {
390            flags.insert(ShapingFlags::IGNORE_LIGATURES_SHAPING_FLAG);
391            flags.insert(ShapingFlags::DISABLE_KERNING_SHAPING_FLAG)
392        }
393
394        let specified_word_spacing = &inherited_text_style.word_spacing;
395        let style_word_spacing: Option<Au> = specified_word_spacing.to_length().map(|l| l.into());
396
397        let segments = self
398            .segment_text_by_font(
399                layout_context,
400                formatting_context_text,
401                bidi_info,
402                &parent_style,
403            )
404            .into_iter()
405            .map(|mut segment| {
406                let word_spacing = style_word_spacing.unwrap_or_else(|| {
407                    let space_width = segment
408                        .font
409                        .glyph_index(' ')
410                        .map(|glyph_id| segment.font.glyph_h_advance(glyph_id))
411                        .unwrap_or(LAST_RESORT_GLYPH_ADVANCE);
412                    specified_word_spacing.to_used_value(Au::from_f64_px(space_width))
413                });
414
415                let mut flags = flags;
416                if segment.bidi_level.is_rtl() {
417                    flags.insert(ShapingFlags::RTL_FLAG);
418                }
419                let shaping_options = ShapingOptions {
420                    letter_spacing,
421                    word_spacing,
422                    script: segment.script,
423                    flags,
424                };
425
426                segment.shape_text(
427                    &parent_style,
428                    formatting_context_text,
429                    linebreaker,
430                    &shaping_options,
431                );
432
433                segment
434            })
435            .collect();
436
437        let _ = std::mem::replace(&mut self.shaped_text, segments);
438    }
439
440    /// Take the [`TextRun`]'s text and turn it into [`TextRunSegment`]s. Each segment has a matched
441    /// font and script. Fonts may differ when glyphs are found in fallback fonts.
442    /// [`super::InlineFormattingContext`].
443    fn segment_text_by_font(
444        &mut self,
445        layout_context: &LayoutContext,
446        formatting_context_text: &str,
447        bidi_info: &BidiInfo,
448        parent_style: &Arc<ComputedValues>,
449    ) -> Vec<TextRunSegment> {
450        let font_group = layout_context
451            .font_context
452            .font_group(parent_style.clone_font());
453        let mut current: Option<TextRunSegment> = None;
454        let mut results = Vec::new();
455
456        let text_run_text = &formatting_context_text[self.text_range.clone()];
457        let char_iterator = TwoCharsAtATimeIterator::new(text_run_text.chars());
458        let mut next_byte_index = self.text_range.start;
459        for (character, next_character) in char_iterator {
460            let current_byte_index = next_byte_index;
461            next_byte_index += character.len_utf8();
462
463            if char_does_not_change_font(character) {
464                continue;
465            }
466
467            // If the script and BiDi level do not change, use the current font as the first fallback. This
468            // can potentially speed up fallback on long font lists or with uncommon scripts which might be
469            // at the bottom of the list.
470            let script = Script::from(character);
471            let bidi_level = bidi_info.levels[current_byte_index];
472            let current_font = current.as_ref().and_then(|text_run_segment| {
473                if text_run_segment.bidi_level == bidi_level && text_run_segment.script == script {
474                    Some(text_run_segment.font.clone())
475                } else {
476                    None
477                }
478            });
479
480            let lang = parent_style.get_font()._x_lang.clone();
481
482            let Some(font) = font_group.find_by_codepoint(
483                &layout_context.font_context,
484                character,
485                next_character,
486                current_font,
487                Some(lang.0.as_ref().to_string()),
488            ) else {
489                continue;
490            };
491
492            // If the existing segment is compatible with the character, keep going.
493            if let Some(current) = current.as_mut() {
494                if current.update_if_compatible(layout_context, &font, script, bidi_level) {
495                    continue;
496                }
497            }
498
499            // Add the new segment and finish the existing one, if we had one. If the first
500            // characters in the run were control characters we may be creating the first
501            // segment in the middle of the run (ie the start should be the start of this
502            // text run's text).
503            let start_byte_index = match current {
504                Some(_) => current_byte_index,
505                None => self.text_range.start,
506            };
507            let new = TextRunSegment::new(font, script, bidi_level, start_byte_index);
508            if let Some(mut finished) = current.replace(new) {
509                // The end of the previous segment is the start of the next one.
510                finished.range.end = current_byte_index;
511                results.push(finished);
512            }
513        }
514
515        // Either we have a current segment or we only had control character and whitespace. In both
516        // of those cases, just use the first font.
517        if current.is_none() {
518            current = font_group.first(&layout_context.font_context).map(|font| {
519                TextRunSegment::new(font, Script::Common, Level::ltr(), self.text_range.start)
520            })
521        }
522
523        // Extend the last segment to the end of the string and add it to the results.
524        if let Some(mut last_segment) = current.take() {
525            last_segment.range.end = self.text_range.end;
526            results.push(last_segment);
527        }
528
529        results
530    }

532    pub(super) fn layout_into_line_items(&self, ifc: &mut InlineFormattingContextLayout) {
533        if self.text_range.is_empty() {
534            return;
535        }
536
537        // If we are following replaced content, we should have a soft wrap opportunity, unless the
538        // first character of this `TextRun` prevents that soft wrap opportunity. If we see such a
539        // character it should also override the LineBreaker's indication to break at the start.
540        let have_deferred_soft_wrap_opportunity =
541            mem::replace(&mut ifc.have_deferred_soft_wrap_opportunity, false);
542        let mut soft_wrap_policy = match have_deferred_soft_wrap_opportunity {
543            true => SegmentStartSoftWrapPolicy::Force,
544            false => SegmentStartSoftWrapPolicy::FollowLinebreaker,
545        };
546
547        for segment in self.shaped_text.iter() {
548            segment.layout_into_line_items(self, soft_wrap_policy, ifc);
549            soft_wrap_policy = SegmentStartSoftWrapPolicy::FollowLinebreaker;
550        }
551    }
552}

554/// Whether or not this character should be able to change the font during segmentation.  Certain
555/// character are not rendered at all, so it doesn't matter what font we use to render them. They
556/// should just be added to the current segment.
557fn char_does_not_change_font(character: char) -> bool {
558    if character.is_control() {
559        return true;
560    }
561    if character == '\u{00A0}' {
562        return true;
563    }
564    if is_bidi_control(character) {
565        return false;
566    }
567
568    let class = linebreak_property(character);
569    class == XI_LINE_BREAKING_CLASS_CM ||
570        class == XI_LINE_BREAKING_CLASS_GL ||
571        class == XI_LINE_BREAKING_CLASS_ZW ||
572        class == XI_LINE_BREAKING_CLASS_WJ ||
573        class == XI_LINE_BREAKING_CLASS_ZWJ
574}

576pub(super) fn get_font_for_first_font_for_style(
577    style: &ComputedValues,
578    font_context: &FontContext,
579) -> Option<FontRef> {
580    let font = font_context
581        .font_group(style.clone_font())
582        .first(font_context);
583    if font.is_none() {
584        warn!("Could not find font for style: {:?}", style.clone_font());
585    }
586    font

588pub(crate) struct TwoCharsAtATimeIterator<InputIterator> {
589    /// The input character iterator.
590    iterator: InputIterator,
591    /// The first character to produce in the next run of the iterator.
592    next_character: Option<char>,
593}
594
595impl<InputIterator> TwoCharsAtATimeIterator<InputIterator> {
596    fn new(iterator: InputIterator) -> Self {
597        Self {
598            iterator,
599            next_character: None,
600        }
601    }
602}
603
604impl<InputIterator> Iterator for TwoCharsAtATimeIterator<InputIterator>
605where
606    InputIterator: Iterator<Item = char>,
607{
608    type Item = (char, Option<char>);
609
610    fn next(&mut self) -> Option<Self::Item> {
611        // If the iterator isn't initialized do that now.
612        if self.next_character.is_none() {
613            self.next_character = self.iterator.next();
614        }
615        let character = self.next_character?;
616        self.next_character = self.iterator.next();
617        Some((character, self.next_character))
618    }
