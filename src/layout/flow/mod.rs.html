107        &mut self,
108        context: &SharedStyleContext,
109        node: &ServoThreadSafeLayoutNode,

112        self.with_base_mut(|base| {

175    fn find_block_margin_collapsing_with_parent(
176        &self,
177        layout_context: &LayoutContext,
178        collected_margin: &mut CollapsedMargin,
179        containing_block: &ContainingBlock,
180    ) -> bool {
181        let layout_style = match self {
182            BlockLevelBox::SameFormattingContextBlock { base, contents, .. } => {
183                contents.layout_style(base)
184            },
185            BlockLevelBox::OutOfFlowAbsolutelyPositionedBox(_) |
186            BlockLevelBox::OutOfFlowFloatBox(_) => return true,
187            BlockLevelBox::OutsideMarker(_) => return false,
188            BlockLevelBox::Independent(context) => {
189                // FIXME: If the element doesn't fit next to floats, it will get clearance.
190                // In that case this should be returning false.
191                context.layout_style()
192            },
193        };
194
195        // FIXME: This should only return false when 'clear' causes clearance.
196        let style = layout_style.style();
197        if style.get_box().clear != StyleClear::None {
198            return false;
199        }
200
201        let ContentBoxSizesAndPBM {
202            content_box_sizes,
203            pbm,
204            ..
205        } = layout_style.content_box_sizes_and_padding_border_margin(&containing_block.into());
206        let margin = pbm.margin.auto_is(Au::zero);
207        collected_margin.adjoin_assign(&CollapsedMargin::new(margin.block_start));
208
209        let BlockLevelBox::SameFormattingContextBlock { contents, .. } = self else {
210            return false;
211        };
212
213        if !pbm.padding.block_start.is_zero() || !pbm.border.block_start.is_zero() {
214            return false;
215        }
216
217        let available_inline_size =
218            containing_block.size.inline - pbm.padding_border_sums.inline - margin.inline_sum();
219        let available_block_size = containing_block.size.block.to_definite().map(|block_size| {
220            Au::zero().max(block_size - pbm.padding_border_sums.block - margin.block_sum())
221        });
222
223        let tentative_block_size = content_box_sizes.block.resolve_extrinsic(
224            Size::FitContent,
225            Au::zero(),
226            available_block_size,
227        );
228
229        let get_inline_content_sizes = || {
230            let constraint_space = ConstraintSpace::new(
231                tentative_block_size,
232                style,
233                None, /* TODO: support preferred aspect ratios on non-replaced boxes */
234            );
235            self.inline_content_sizes(layout_context, &constraint_space)
236                .sizes
237        };
238        let inline_size = content_box_sizes.inline.resolve(
239            Direction::Inline,
240            Size::Stretch,
241            Au::zero,
242            Some(available_inline_size),
243            get_inline_content_sizes,
244            false, /* is_table */
245        );
246
247        let containing_block_for_children = ContainingBlock {
248            size: ContainingBlockSize {
249                inline: inline_size,
250                block: tentative_block_size,
251            },
252            style,
253        };
254
255        if !contents.find_block_margin_collapsing_with_parent(
256            layout_context,
257            collected_margin,
258            &containing_block_for_children,
259        ) {
260            return false;
261        }
262
263        if !tentative_block_size.definite_or_min().is_zero() ||
264            !pbm.padding_border_sums.block.is_zero()
265        {
266            return false;
267        }
268
269        collected_margin.adjoin_assign(&CollapsedMargin::new(margin.block_end));
270
271        true
272    }

300    fn layout(
301        &self,
302        layout_context: &LayoutContext<'_>,
303        containing_block: &ContainingBlock<'_>,
304        positioning_context: &mut PositioningContext,
305    ) -> Fragment {
306        let constraint_space = ConstraintSpace::new_for_style_and_ratio(
307            &self.base.style,
308            None, /* TODO: support preferred aspect ratios on non-replaced boxes */
309        );
310        let content_sizes = self.inline_content_sizes(layout_context, &constraint_space);
311        let containing_block_for_children = ContainingBlock {
312            size: ContainingBlockSize {
313                inline: content_sizes.sizes.max_content,
314                block: SizeConstraint::default(),
315            },
316            style: &self.base.style,
317        };
318
319        let flow_layout = self.block_formatting_context.layout(
320            layout_context,
321            positioning_context,
322            &containing_block_for_children,
323        );
324
325        let max_inline_size = flow_layout
326            .fragments
327            .iter()
328            .map(|fragment| {
329                fragment
330                    .base()
331                    .map(|base| base.rect)
332                    .unwrap_or_default()
333                    .to_logical(&containing_block_for_children)
334                    .max_inline_position()
335            })
336            .max()
337            .unwrap_or_default();
338
339        // Position the marker beyond the inline start of the border box list item. This needs to
340        // take into account the border and padding of the item.
341        //
342        // TODO: This is the wrong containing block, as it should be the containing block of the
343        // parent of this list item. What this means in practice is that the writing mode could be
344        // wrong and padding defined as a percentage will be resolved incorrectly.
345        //
346        // TODO: This should use the LayoutStyle of the list item, not the default one. Currently
347        // they are the same, but this could change in the future.
348        let pbm_of_list_item =
349            LayoutStyle::Default(&self.list_item_style).padding_border_margin(containing_block);
350        let content_rect = LogicalRect {
351            start_corner: LogicalVec2 {
352                inline: -max_inline_size -
353                    (pbm_of_list_item.border.inline_start +
354                        pbm_of_list_item.padding.inline_start),
355                block: Zero::zero(),
356            },
357            size: LogicalVec2 {
358                inline: max_inline_size,
359                block: flow_layout.content_block_size,
360            },
361        };
362
363        let mut base_fragment_info = BaseFragmentInfo::anonymous();
364        base_fragment_info.flags |= FragmentFlags::IS_OUTSIDE_LIST_ITEM_MARKER;
365
366        Fragment::Box(ArcRefCell::new(BoxFragment::new(
367            base_fragment_info,
368            self.base.style.clone(),
369            flow_layout.fragments,
370            content_rect.as_physical(Some(containing_block)),
371            PhysicalSides::zero(),
372            PhysicalSides::zero(),
373            PhysicalSides::zero(),
374            flow_layout.specific_layout_info,
375        )))
376    }

389impl BlockFormattingContext {
390    pub(super) fn layout(
391        &self,
392        layout_context: &LayoutContext,
393        positioning_context: &mut PositioningContext,
394        containing_block: &ContainingBlock,
395    ) -> CacheableLayoutResult {
396        let mut sequential_layout_state = if self.contains_floats || !layout_context.use_rayon {
397            Some(SequentialLayoutState::new(containing_block.size.inline))
398        } else {
399            None
400        };
401
402        // Since this is an independent formatting context, we don't ignore block margins when
403        // resolving a stretch block size of the children.
404        // https://drafts.csswg.org/css-sizing-4/#stretch-fit-sizing
405        let ignore_block_margins_for_stretch = LogicalSides1D::new(false, false);
406
407        let flow_layout = self.contents.layout(
408            layout_context,
409            positioning_context,
410            containing_block,
411            sequential_layout_state.as_mut(),
412            CollapsibleWithParentStartMargin(false),
413            ignore_block_margins_for_stretch,
414        );
415        debug_assert!(
416            !flow_layout
417                .collapsible_margins_in_children
418                .collapsed_through
419        );
420
421        // The content height of a BFC root should include any float participating in that BFC
422        // (https://drafts.csswg.org/css2/#root-height), we implement this by imagining there is
423        // an element with `clear: both` after the actual contents.
424        let clearance = sequential_layout_state.and_then(|sequential_layout_state| {
425            sequential_layout_state.calculate_clearance(Clear::Both, &CollapsedMargin::zero())
426        });
427
428        CacheableLayoutResult {
429            fragments: flow_layout.fragments,
430            content_block_size: flow_layout.content_block_size +
431                flow_layout.collapsible_margins_in_children.end.solve() +
432                clearance.unwrap_or_default(),
433            content_inline_size_for_table: None,
434            baselines: flow_layout.baselines,
435            depends_on_block_constraints: flow_layout.depends_on_block_constraints,
436            specific_layout_info: None,
437            collapsible_margins_in_children: CollapsedBlockMargins::zero(),
438        }
439    }

455/// Finds the min/max-content inline size of the block-level children of a block container.
456/// The in-flow boxes will stack vertically, so we only need to consider the maximum size.
457/// But floats can flow horizontally depending on 'clear', so we may need to sum their sizes.
458/// CSS 2 does not define the exact algorithm, this logic is based on the behavior observed
459/// on Gecko and Blink.
460fn compute_inline_content_sizes_for_block_level_boxes(
461    boxes: &[ArcRefCell<BlockLevelBox>],
462    layout_context: &LayoutContext,
463    containing_block: &IndefiniteContainingBlock,
464) -> InlineContentSizesResult {
465    let get_box_info = |box_: &ArcRefCell<BlockLevelBox>| {
466        match &*box_.borrow() {
467            BlockLevelBox::OutOfFlowAbsolutelyPositionedBox(_) |
468            BlockLevelBox::OutsideMarker { .. } => None,
469            BlockLevelBox::OutOfFlowFloatBox(float_box) => {
470                let inline_content_sizes_result = float_box.contents.outer_inline_content_sizes(
471                    layout_context,
472                    containing_block,
473                    &LogicalVec2::zero(),
474                    false, /* auto_block_size_stretches_to_containing_block */
475                );
476                let style = &float_box.contents.style();
477                let container_writing_mode = containing_block.style.writing_mode;
478                Some((
479                    inline_content_sizes_result,
480                    FloatSide::from_style_and_container_writing_mode(style, container_writing_mode),
481                    Clear::from_style_and_container_writing_mode(style, container_writing_mode),
482                ))
483            },
484            BlockLevelBox::SameFormattingContextBlock { base, contents, .. } => {
485                let is_anonymous_block =
486                    matches!(base.style.pseudo(), Some(PseudoElement::ServoAnonymousBox));
487                let inline_content_sizes_result = sizing::outer_inline(
488                    base,
489                    &contents.layout_style(base),
490                    containing_block,
491                    &LogicalVec2::zero(),
492                    false,               /* auto_block_size_stretches_to_containing_block */
493                    false,               /* is_replaced */
494                    !is_anonymous_block, /* establishes_containing_block */
495                    |_| None, /* TODO: support preferred aspect ratios on non-replaced boxes */
496                    |constraint_space| {
497                        base.inline_content_sizes(layout_context, constraint_space, contents)
498                    },
499                    |_aspect_ratio| None,
500                );
501                // A block in the same BFC can overlap floats, it's not moved next to them,
502                // so we shouldn't add its size to the size of the floats.
503                // Instead, we treat it like an independent block with 'clear: both',
504                // except if it's an anonymous block.
505                // Presumably, the exception is because an anonymous block will always have
506                // inline-level contents, which don't overlap floats. However, the same might
507                // also happen with a non-anonymous block, so the logic is a bit arbitrary,
508                // but matches other browsers (see #41280).
509                let clear = if is_anonymous_block {
510                    Clear::None
511                } else {
512                    Clear::Both
513                };
514                Some((inline_content_sizes_result, None, clear))
515            },
516            BlockLevelBox::Independent(independent) => {
517                let inline_content_sizes_result = independent.outer_inline_content_sizes(
518                    layout_context,
519                    containing_block,
520                    &LogicalVec2::zero(),
521                    false, /* auto_block_size_stretches_to_containing_block */
522                );
523                Some((
524                    inline_content_sizes_result,
525                    None,
526                    Clear::from_style_and_container_writing_mode(
527                        independent.style(),
528                        containing_block.style.writing_mode,
529                    ),
530                ))
531            },
532        }
533    };
534
535    /// When iterating the block-level boxes to compute the inline content sizes,
536    /// this struct contains the data accumulated up to the current box.
537    #[derive(Default)]
538    struct AccumulatedData {
539        /// Whether the inline size depends on the block one.
540        depends_on_block_constraints: bool,
541        /// The maximum size seen so far, not including trailing uncleared floats.
542        max_size: ContentSizes,
543        /// The size of the trailing uncleared floats on the inline-start and
544        /// inline-end sides of the containing block.
545        floats: LogicalSides1D<ContentSizes>,
546    }
547
548    impl AccumulatedData {
549        fn max_size_including_uncleared_floats(&self) -> ContentSizes {
550            self.max_size.max(self.floats.start.union(&self.floats.end))
551        }
552        fn clear_floats(&mut self, clear: Clear) {
553            match clear {
554                Clear::InlineStart => {
555                    self.max_size = self.max_size_including_uncleared_floats();
556                    self.floats.start = ContentSizes::default();
557                },
558                Clear::InlineEnd => {
559                    self.max_size = self.max_size_including_uncleared_floats();
560                    self.floats.end = ContentSizes::default();
561                },
562                Clear::Both => {
563                    self.max_size = self.max_size_including_uncleared_floats();
564                    self.floats = LogicalSides1D::default();
565                },
566                Clear::None => {},
567            };
568        }
569    }
570
571    let accumulate =
572        |mut data: AccumulatedData,
573         (inline_content_sizes_result, float, clear): (InlineContentSizesResult, _, _)| {
574            let size = inline_content_sizes_result.sizes.max(ContentSizes::zero());
575            let depends_on_block_constraints =
576                inline_content_sizes_result.depends_on_block_constraints;
577            data.depends_on_block_constraints |= depends_on_block_constraints;
578            data.clear_floats(clear);
579            match float {
580                Some(FloatSide::InlineStart) => data.floats.start.union_assign(&size),
581                Some(FloatSide::InlineEnd) => data.floats.end.union_assign(&size),
582                None => {
583                    data.max_size
584                        .max_assign(data.floats.start.union(&data.floats.end).union(&size));
585                    data.floats = LogicalSides1D::default();
586                },
587            }
588            data
589        };
590    let data = if layout_context.use_rayon {
591        boxes
592            .par_iter()
593            .filter_map(get_box_info)
594            .collect::<Vec<_>>()
595            .into_iter()
596            .fold(AccumulatedData::default(), accumulate)
597    } else {
598        boxes
599            .iter()
600            .filter_map(get_box_info)
601            .fold(AccumulatedData::default(), accumulate)
602    };
603    InlineContentSizesResult {
604        depends_on_block_constraints: data.depends_on_block_constraints,
605        sizes: data.max_size_including_uncleared_floats(),
606    }
607}

648        containing_block_for_children: &ContainingBlock,
649    ) -> bool {
650        match self {
651            BlockContainer::BlockLevelBoxes(boxes) => boxes.iter().all(|block_level_box| {
652                block_level_box
653                    .borrow()
654                    .find_block_margin_collapsing_with_parent(
655                        layout_context,
656                        collected_margin,
657                        containing_block_for_children,
658                    )
659            }),
660            BlockContainer::InlineFormattingContext(context) => context
661                .find_block_margin_collapsing_with_parent(
662                    layout_context,
663                    collected_margin,
664                    containing_block_for_children,
665                ),
666        }
667    }
668}

689fn layout_block_level_children(
690    layout_context: &LayoutContext,
691    positioning_context: &mut PositioningContext,
692    child_boxes: &[ArcRefCell<BlockLevelBox>],
693    containing_block: &ContainingBlock,
694    mut sequential_layout_state: Option<&mut SequentialLayoutState>,
695    collapsible_with_parent_start_margin: CollapsibleWithParentStartMargin,
696    ignore_block_margins_for_stretch: LogicalSides1D<bool>,
697) -> CacheableLayoutResult {
698    let mut placement_state =
699        PlacementState::new(collapsible_with_parent_start_margin, containing_block);
700
701    let fragments = match sequential_layout_state {
702        Some(ref mut sequential_layout_state) => layout_block_level_children_sequentially(
703            layout_context,
704            positioning_context,
705            child_boxes,
706            containing_block,
707            sequential_layout_state,
708            &mut placement_state,
709            ignore_block_margins_for_stretch,
710        ),
711        None => layout_block_level_children_in_parallel(
712            layout_context,
713            positioning_context,
714            child_boxes,
715            containing_block,
716            &mut placement_state,
717            ignore_block_margins_for_stretch,
718        ),
719    };
720
721    let depends_on_block_constraints = fragments.iter().any(|fragment| {
722        fragment.base().is_some_and(|base| {
723            base.flags.contains(
724                FragmentFlags::SIZE_DEPENDS_ON_BLOCK_CONSTRAINTS_AND_CAN_BE_CHILD_OF_FLEX_ITEM,
725            )
726        })
727    });
728
729    let (content_block_size, collapsible_margins_in_children, baselines) = placement_state.finish();
730    CacheableLayoutResult {
731        fragments,
732        content_block_size,
733        collapsible_margins_in_children,
734        baselines,
735        depends_on_block_constraints,
736        content_inline_size_for_table: None,
737        specific_layout_info: None,
738    }
739}

741fn layout_block_level_children_in_parallel(
742    layout_context: &LayoutContext,
743    positioning_context: &mut PositioningContext,
744    child_boxes: &[ArcRefCell<BlockLevelBox>],
745    containing_block: &ContainingBlock,
746    placement_state: &mut PlacementState,
747    ignore_block_margins_for_stretch: LogicalSides1D<bool>,
748) -> Vec<Fragment> {
749    let mut layout_results: Vec<(Fragment, PositioningContext)> =
750        Vec::with_capacity(child_boxes.len());
751
752    child_boxes
753        .par_iter()
754        .map(|child_box| {
755            let mut child_positioning_context = PositioningContext::default();
756            let fragment = child_box.borrow().layout(
757                layout_context,
758                &mut child_positioning_context,
759                containing_block,
760                /* sequential_layout_state = */ None,
761                /* collapsible_with_parent_start_margin = */ None,
762                ignore_block_margins_for_stretch,
763            );
764            (fragment, child_positioning_context)
765        })
766        .collect_into_vec(&mut layout_results);
767
768    layout_results
769        .into_iter()
770        .map(|(mut fragment, mut child_positioning_context)| {
771            placement_state.place_fragment_and_update_baseline(&mut fragment, None);
772            child_positioning_context.adjust_static_position_of_hoisted_fragments(
773                &fragment,
774                PositioningContextLength::zero(),
775            );
776            positioning_context.append(child_positioning_context);
777            fragment
778        })
779        .collect()
780}

782fn layout_block_level_children_sequentially(
783    layout_context: &LayoutContext,
784    positioning_context: &mut PositioningContext,
785    child_boxes: &[ArcRefCell<BlockLevelBox>],
786    containing_block: &ContainingBlock,
787    sequential_layout_state: &mut SequentialLayoutState,
788    placement_state: &mut PlacementState,
789    ignore_block_margins_for_stretch: LogicalSides1D<bool>,
790) -> Vec<Fragment> {
791    // Because floats are involved, we do layout for this block formatting context in tree
792    // order without parallelism. This enables mutable access to a `SequentialLayoutState` that
793    // tracks every float encountered so far (again in tree order).
794    child_boxes
795        .iter()
796        .map(|child_box| {
797            let positioning_context_length_before_layout = positioning_context.len();
798            let mut fragment = child_box.borrow().layout(
799                layout_context,
800                positioning_context,
801                containing_block,
802                Some(&mut *sequential_layout_state),
803                Some(CollapsibleWithParentStartMargin(
804                    placement_state.next_in_flow_margin_collapses_with_parent_start_margin,
805                )),
806                ignore_block_margins_for_stretch,
807            );
808
809            placement_state
810                .place_fragment_and_update_baseline(&mut fragment, Some(sequential_layout_state));
811            positioning_context.adjust_static_position_of_hoisted_fragments(
812                &fragment,
813                positioning_context_length_before_layout,
814            );
815
816            fragment
817        })
818        .collect()
819}

821impl BlockLevelBox {
822    fn layout(
823        &self,
824        layout_context: &LayoutContext,
825        positioning_context: &mut PositioningContext,
826        containing_block: &ContainingBlock,
827        sequential_layout_state: Option<&mut SequentialLayoutState>,
828        collapsible_with_parent_start_margin: Option<CollapsibleWithParentStartMargin>,
829        ignore_block_margins_for_stretch: LogicalSides1D<bool>,
830    ) -> Fragment {
831        let fragment = match self {
832            BlockLevelBox::SameFormattingContextBlock { base, contents, .. } => Fragment::Box(
833                ArcRefCell::new(positioning_context.layout_maybe_position_relative_fragment(
834                    layout_context,
835                    containing_block,
836                    base,
837                    |positioning_context| {
838                        layout_in_flow_non_replaced_block_level_same_formatting_context(
839                            layout_context,
840                            positioning_context,
841                            containing_block,
842                            base,
843                            contents,
844                            sequential_layout_state,
845                            collapsible_with_parent_start_margin,
846                            ignore_block_margins_for_stretch,
847                        )
848                    },
849                )),
850            ),
851            BlockLevelBox::Independent(independent) => Fragment::Box(ArcRefCell::new(
852                positioning_context.layout_maybe_position_relative_fragment(
853                    layout_context,
854                    containing_block,
855                    &independent.base,
856                    |positioning_context| {
857                        independent.layout_in_flow_block_level(
858                            layout_context,
859                            positioning_context,
860                            containing_block,
861                            sequential_layout_state,
862                            ignore_block_margins_for_stretch,
863                        )
864                    },
865                ),
866            )),
867            BlockLevelBox::OutOfFlowAbsolutelyPositionedBox(box_) => {
868                // The static position of zero here is incorrect, however we do not know
869                // the correct positioning until later, in place_block_level_fragment, and
870                // this value will be adjusted there.
871                let hoisted_box = AbsolutelyPositionedBox::to_hoisted(
872                    box_.clone(),
873                    // This is incorrect, however we do not know the correct positioning
874                    // until later, in PlacementState::place_fragment, and this value will be
875                    // adjusted there
876                    PhysicalRect::zero(),
877                    LogicalVec2 {
878                        inline: AlignFlags::START,
879                        block: AlignFlags::START,
880                    },
881                    containing_block.style.writing_mode,
882                );
883                let hoisted_fragment = hoisted_box.fragment.clone();
884                positioning_context.push(hoisted_box);
885                Fragment::AbsoluteOrFixedPositioned(hoisted_fragment)
886            },
887            BlockLevelBox::OutOfFlowFloatBox(float_box) => Fragment::Float(ArcRefCell::new(
888                float_box.layout(layout_context, positioning_context, containing_block),
889            )),
890            BlockLevelBox::OutsideMarker(outside_marker) => {
891                outside_marker.layout(layout_context, containing_block, positioning_context)
892            },
893        };
894
895        self.with_base(|base| base.set_fragment(fragment.clone()));
896
897        fragment
898    }

900    fn inline_content_sizes(
901        &self,
902        layout_context: &LayoutContext,
903        constraint_space: &ConstraintSpace,
904    ) -> InlineContentSizesResult {
905        let independent_formatting_context = match self {
906            BlockLevelBox::Independent(independent) => independent,
907            BlockLevelBox::OutOfFlowAbsolutelyPositionedBox(box_) => &box_.borrow().context,
908            BlockLevelBox::OutOfFlowFloatBox(float_box) => &float_box.contents,
909            BlockLevelBox::OutsideMarker(outside_marker) => {
910                return outside_marker.inline_content_sizes(layout_context, constraint_space);
911            },
912            BlockLevelBox::SameFormattingContextBlock { base, contents, .. } => {
913                return base.inline_content_sizes(layout_context, constraint_space, contents);
914            },
915        };
916        independent_formatting_context.inline_content_sizes(layout_context, constraint_space)
917    }

920/// Lay out a normal flow non-replaced block that does not establish a new formatting
921/// context.
922///
923/// - <https://drafts.csswg.org/css2/visudet.html#blockwidth>
924/// - <https://drafts.csswg.org/css2/visudet.html#normal-block>
925#[allow(clippy::too_many_arguments)]
926pub(crate) fn layout_in_flow_non_replaced_block_level_same_formatting_context(
927    layout_context: &LayoutContext,
928    positioning_context: &mut PositioningContext,
929    containing_block: &ContainingBlock,
930    base: &LayoutBoxBase,
931    contents: &BlockContainer,
932    mut sequential_layout_state: Option<&mut SequentialLayoutState>,
933    collapsible_with_parent_start_margin: Option<CollapsibleWithParentStartMargin>,
934    ignore_block_margins_for_stretch: LogicalSides1D<bool>,
935) -> BoxFragment {
936    let style = &base.style;
937    let layout_style = contents.layout_style(base);
938    let containing_block_writing_mode = containing_block.style.writing_mode;
939    let get_inline_content_sizes = |constraint_space: &ConstraintSpace| {
940        base.inline_content_sizes(layout_context, constraint_space, contents)
941            .sizes
942    };
943    let ContainingBlockPaddingAndBorder {
944        containing_block: containing_block_for_children,
945        pbm,
946        block_sizes,
947        depends_on_block_constraints,
948        available_block_size,
949        justify_self,
950        ..
951    } = solve_containing_block_padding_and_border_for_in_flow_box(
952        containing_block,
953        &layout_style,
954        get_inline_content_sizes,
955        ignore_block_margins_for_stretch,
956        None,
957    );
958    let ResolvedMargins {
959        margin,
960        effective_margin_inline_start,
961    } = solve_margins(
962        containing_block,
963        &pbm,
964        containing_block_for_children.size.inline,
965        justify_self,
966    );
967
968    let start_margin_can_collapse_with_children =
969        pbm.padding.block_start.is_zero() && pbm.border.block_start.is_zero();
970
971    let mut clearance = None;
972    let parent_containing_block_position_info;
973    match sequential_layout_state {
974        None => parent_containing_block_position_info = None,
975        Some(ref mut sequential_layout_state) => {
976            let clear =
977                Clear::from_style_and_container_writing_mode(style, containing_block_writing_mode);
978            let mut block_start_margin = CollapsedMargin::new(margin.block_start);
979
980            // The block start margin may collapse with content margins,
981            // compute the resulting one in order to place floats correctly.
982            // Only need to do this if the element isn't also collapsing with its parent,
983            // otherwise we should have already included the margin in an ancestor.
984            // Note this lookahead stops when finding a descendant whose `clear` isn't `none`
985            // (since clearance prevents collapsing margins with the parent).
986            // But then we have to decide whether to actually add clearance or not,
987            // so look forward again regardless of `collapsible_with_parent_start_margin`.
988            // TODO: This isn't completely right: if we don't add actual clearance,
989            // the margin should have been included in the parent (or some ancestor).
990            // The lookahead should stop for actual clearance, not just for `clear`.
991            let collapsible_with_parent_start_margin = collapsible_with_parent_start_margin.expect(
992                "We should know whether we are collapsing the block start margin with the parent \
993                when laying out sequentially",
994            ).0 && clear == Clear::None;
995            if !collapsible_with_parent_start_margin && start_margin_can_collapse_with_children {
996                contents.find_block_margin_collapsing_with_parent(
997                    layout_context,
998                    &mut block_start_margin,
999                    &containing_block_for_children,
1000                );
1001            }
1002
1003            // Introduce clearance if necessary.
1004            clearance = sequential_layout_state.calculate_clearance(clear, &block_start_margin);
1005            if clearance.is_some() {
1006                sequential_layout_state.collapse_margins();
1007            }
1008            sequential_layout_state.adjoin_assign(&block_start_margin);
1009            if !start_margin_can_collapse_with_children {
1010                sequential_layout_state.collapse_margins();
1011            }
1012
1013            // NB: This will be a no-op if we're collapsing margins with our children since that
1014            // can only happen if we have no block-start padding and border.
1015            sequential_layout_state.advance_block_position(
1016                pbm.padding.block_start +
1017                    pbm.border.block_start +
1018                    clearance.unwrap_or_else(Au::zero),
1019            );
1020
1021            // We are about to lay out children. Update the offset between the block formatting
1022            // context and the containing block that we create for them. This offset is used to
1023            // ajust BFC relative coordinates to coordinates that are relative to our content box.
1024            // Our content box establishes the containing block for non-abspos children, including
1025            // floats.
1026            let inline_start = sequential_layout_state
1027                .floats
1028                .containing_block_info
1029                .inline_start +
1030                pbm.padding.inline_start +
1031                pbm.border.inline_start +
1032                effective_margin_inline_start;
1033            let new_cb_offsets = ContainingBlockPositionInfo {
1034                block_start: sequential_layout_state.bfc_relative_block_position,
1035                block_start_margins_not_collapsed: sequential_layout_state.current_margin,
1036                inline_start,
1037                inline_end: inline_start + containing_block_for_children.size.inline,
1038            };
1039            parent_containing_block_position_info = Some(
1040                sequential_layout_state.replace_containing_block_position_info(new_cb_offsets),
1041            );
1042        },
1043    };
1044
1045    // https://drafts.csswg.org/css-sizing-4/#stretch-fit-sizing
1046    // > If this is a block axis size, and the element is in a Block Layout formatting context,
1047    // > and the parent element does not have a block-start border or padding and is not an
1048    // > independent formatting context, treat the elements block-start margin as zero
1049    // > for the purpose of calculating this size. Do the same for the block-end margin.
1050    let ignore_block_margins_for_stretch = LogicalSides1D::new(
1051        pbm.border.block_start.is_zero() && pbm.padding.block_start.is_zero(),
1052        pbm.border.block_end.is_zero() && pbm.padding.block_end.is_zero(),
1053    );
1054
1055    let flow_layout = contents.layout(
1056        layout_context,
1057        positioning_context,
1058        &containing_block_for_children,
1059        sequential_layout_state.as_deref_mut(),
1060        CollapsibleWithParentStartMargin(start_margin_can_collapse_with_children),
1061        ignore_block_margins_for_stretch,
1062    );
1063    let mut content_block_size = flow_layout.content_block_size;
1064
1065    // Update margins.
1066    let mut block_margins_collapsed_with_children = CollapsedBlockMargins::from_margin(&margin);
1067    let mut collapsible_margins_in_children = flow_layout.collapsible_margins_in_children;
1068    if start_margin_can_collapse_with_children {
1069        block_margins_collapsed_with_children
1070            .start
1071            .adjoin_assign(&collapsible_margins_in_children.start);
1072        if collapsible_margins_in_children.collapsed_through {
1073            block_margins_collapsed_with_children
1074                .start
1075                .adjoin_assign(&std::mem::replace(
1076                    &mut collapsible_margins_in_children.end,
1077                    CollapsedMargin::zero(),
1078                ));
1079        }
1080    }
1081
1082    let tentative_block_size = &containing_block_for_children.size.block;
1083    let collapsed_through = collapsible_margins_in_children.collapsed_through &&
1084        pbm.padding_border_sums.block.is_zero() &&
1085        tentative_block_size.definite_or_min().is_zero();
1086    block_margins_collapsed_with_children.collapsed_through = collapsed_through;
1087
1088    let end_margin_can_collapse_with_children =
1089        pbm.padding.block_end.is_zero() && pbm.border.block_end.is_zero();
1090    if !end_margin_can_collapse_with_children {
1091        content_block_size += collapsible_margins_in_children.end.solve();
1092    }
1093
1094    let block_size = block_sizes.resolve(
1095        Direction::Block,
1096        Size::FitContent,
1097        Au::zero,
1098        available_block_size,
1099        || content_block_size.into(),
1100        false, /* is_table */
1101    );
1102
1103    // If the final block size is different than the intrinsic size of the contents,
1104    // then we can't actually collapse the end margins. This can happen due to min
1105    // or max block sizes, or due to `calc-size()` once we implement it.
1106    //
1107    // We also require `block-size` to have an intrinsic value, by checking whether
1108    // the containing block established for the contents has an indefinite block size.
1109    // However, even if `block-size: 0px` is extrinsic (so it would normally prevent
1110    // collapsing the end margin with children), it doesn't prevent the top and end
1111    // margins from collapsing through. If that happens, allow collapsing end margins.
1112    //
1113    // This is being discussed in https://github.com/w3c/csswg-drafts/issues/12218.
1114    // It would probably make more sense to check the definiteness of the containing
1115    // block in the logic above (when we check if there is some block-end padding or
1116    // border), or maybe drop the condition altogether. But for now, we match Blink.
1117    let end_margin_can_collapse_with_children = end_margin_can_collapse_with_children &&
1118        block_size == content_block_size &&
1119        (collapsed_through || !tentative_block_size.is_definite());
1120    if end_margin_can_collapse_with_children {
1121        block_margins_collapsed_with_children
1122            .end
1123            .adjoin_assign(&collapsible_margins_in_children.end);
1124    }
1125
1126    if let Some(ref mut sequential_layout_state) = sequential_layout_state {
1127        // Now that we're done laying out our children, we can restore the
1128        // parent's containing block position information.
1129        sequential_layout_state
1130            .replace_containing_block_position_info(parent_containing_block_position_info.unwrap());
1131
1132        // Account for padding and border. We also might have to readjust the
1133        // `bfc_relative_block_position` if it was different from the content size (i.e. was
1134        // non-`auto` and/or was affected by min/max block size).
1135        //
1136        // If this adjustment is positive, that means that a block size was specified, but
1137        // the content inside had a smaller block size. If this adjustment is negative, a
1138        // block size was specified, but the content inside overflowed this container in
1139        // the block direction. In that case, the ceiling for floats is effectively raised
1140        // as long as no floats in the overflowing content lowered it.
1141        sequential_layout_state.advance_block_position(
1142            block_size - content_block_size + pbm.padding.block_end + pbm.border.block_end,
1143        );
1144
1145        if !end_margin_can_collapse_with_children {
1146            sequential_layout_state.collapse_margins();
1147        }
1148        sequential_layout_state.adjoin_assign(&CollapsedMargin::new(margin.block_end));
1149    }
1150
1151    let content_rect = LogicalRect {
1152        start_corner: LogicalVec2 {
1153            block: (pbm.padding.block_start +
1154                pbm.border.block_start +
1155                clearance.unwrap_or_else(Au::zero)),
1156            inline: pbm.padding.inline_start +
1157                pbm.border.inline_start +
1158                effective_margin_inline_start,
1159        },
1160        size: LogicalVec2 {
1161            block: block_size,
1162            inline: containing_block_for_children.size.inline,
1163        },
1164    };
1165
1166    let mut base_fragment_info = base.base_fragment_info;
1167
1168    // An anonymous block doesn't establish a containing block for its contents. Therefore,
1169    // if its contents depend on block constraints, its block size (which is intrinsic) also
1170    // depends on block constraints.
1171    let is_anonymous = matches!(base.style.pseudo(), Some(PseudoElement::ServoAnonymousBox));
1172    if depends_on_block_constraints || (is_anonymous && flow_layout.depends_on_block_constraints) {
1173        base_fragment_info
1174            .flags
1175            .insert(FragmentFlags::SIZE_DEPENDS_ON_BLOCK_CONSTRAINTS_AND_CAN_BE_CHILD_OF_FLEX_ITEM);
1176    }
1177
1178    BoxFragment::new(
1179        base_fragment_info,
1180        style.clone(),
1181        flow_layout.fragments,
1182        content_rect.as_physical(Some(containing_block)),
1183        pbm.padding.to_physical(containing_block_writing_mode),
1184        pbm.border.to_physical(containing_block_writing_mode),
1185        margin.to_physical(containing_block_writing_mode),
1186        flow_layout.specific_layout_info,
1187    )
1188    .with_baselines(flow_layout.baselines)
1189    .with_block_level_layout_info(block_margins_collapsed_with_children, clearance)
1190}

1192impl IndependentFormattingContext {
1193    /// Lay out an in-flow block-level box that establishes an independent
1194    /// formatting context in its containing formatting context.
1195    ///
1196    /// - <https://drafts.csswg.org/css2/visudet.html#blockwidth>
1197    /// - <https://drafts.csswg.org/css2/visudet.html#block-replaced-width>
1198    /// - <https://drafts.csswg.org/css2/visudet.html#normal-block>
1199    /// - <https://drafts.csswg.org/css2/visudet.html#inline-replaced-height>
1200    pub(crate) fn layout_in_flow_block_level(
1201        &self,
1202        layout_context: &LayoutContext,
1203        positioning_context: &mut PositioningContext,
1204        containing_block: &ContainingBlock,
1205        sequential_layout_state: Option<&mut SequentialLayoutState>,
1206        ignore_block_margins_for_stretch: LogicalSides1D<bool>,
1207    ) -> BoxFragment {
1208        if let Some(sequential_layout_state) = sequential_layout_state {
1209            return self.layout_in_flow_block_level_sequentially(
1210                layout_context,
1211                positioning_context,
1212                containing_block,
1213                sequential_layout_state,
1214                ignore_block_margins_for_stretch,
1215            );
1216        }
1217
1218        let get_inline_content_sizes = |constraint_space: &ConstraintSpace| {
1219            self.inline_content_sizes(layout_context, constraint_space)
1220                .sizes
1221        };
1222        let layout_style = self.layout_style();
1223        let ContainingBlockPaddingAndBorder {
1224            containing_block: containing_block_for_children,
1225            pbm,
1226            block_sizes,
1227            depends_on_block_constraints,
1228            available_block_size,
1229            justify_self,
1230            preferred_aspect_ratio,
1231        } = solve_containing_block_padding_and_border_for_in_flow_box(
1232            containing_block,
1233            &layout_style,
1234            get_inline_content_sizes,
1235            ignore_block_margins_for_stretch,
1236            Some(self),
1237        );
1238
1239        let lazy_block_size = LazySize::new(
1240            &block_sizes,
1241            Direction::Block,
1242            Size::FitContent,
1243            Au::zero,
1244            available_block_size,
1245            layout_style.is_table(),
1246        );
1247
1248        let layout = self.layout(
1249            layout_context,
1250            positioning_context,
1251            &containing_block_for_children,
1252            containing_block,
1253            preferred_aspect_ratio,
1254            &lazy_block_size,
1255        );
1256
1257        let inline_size = layout
1258            .content_inline_size_for_table
1259            .unwrap_or(containing_block_for_children.size.inline);
1260        let block_size = lazy_block_size.resolve(|| layout.content_block_size);
1261
1262        let ResolvedMargins {
1263            margin,
1264            effective_margin_inline_start,
1265        } = solve_margins(containing_block, &pbm, inline_size, justify_self);
1266
1267        let content_rect = LogicalRect {
1268            start_corner: LogicalVec2 {
1269                block: pbm.padding.block_start + pbm.border.block_start,
1270                inline: pbm.padding.inline_start +
1271                    pbm.border.inline_start +
1272                    effective_margin_inline_start,
1273            },
1274            size: LogicalVec2 {
1275                block: block_size,
1276                inline: inline_size,
1277            },
1278        };
1279
1280        let block_margins_collapsed_with_children = CollapsedBlockMargins::from_margin(&margin);
1281        let containing_block_writing_mode = containing_block.style.writing_mode;
1282
1283        let mut base_fragment_info = self.base.base_fragment_info;
1284        if depends_on_block_constraints {
1285            base_fragment_info.flags.insert(
1286                FragmentFlags::SIZE_DEPENDS_ON_BLOCK_CONSTRAINTS_AND_CAN_BE_CHILD_OF_FLEX_ITEM,
1287            );
1288        }
1289        BoxFragment::new(
1290            base_fragment_info,
1291            self.base.style.clone(),
1292            layout.fragments,
1293            content_rect.as_physical(Some(containing_block)),
1294            pbm.padding.to_physical(containing_block_writing_mode),
1295            pbm.border.to_physical(containing_block_writing_mode),
1296            margin.to_physical(containing_block_writing_mode),
1297            layout.specific_layout_info,
1298        )
1299        .with_baselines(layout.baselines)
1300        .with_block_level_layout_info(block_margins_collapsed_with_children, None)
1301    }
1302
1303    /// Lay out a normal in flow non-replaced block that establishes an independent
1304    /// formatting context in its containing formatting context but handling sequential
1305    /// layout concerns, such clearing and placing the content next to floats.
1306    fn layout_in_flow_block_level_sequentially(
1307        &self,
1308        layout_context: &LayoutContext<'_>,
1309        positioning_context: &mut PositioningContext,
1310        containing_block: &ContainingBlock<'_>,
1311        sequential_layout_state: &mut SequentialLayoutState,
1312        ignore_block_margins_for_stretch: LogicalSides1D<bool>,
1313    ) -> BoxFragment {
1314        let style = &self.base.style;
1315        let containing_block_writing_mode = containing_block.style.writing_mode;
1316        let ContentBoxSizesAndPBM {
1317            content_box_sizes,
1318            pbm,
1319            depends_on_block_constraints,
1320            ..
1321        } = self
1322            .layout_style()
1323            .content_box_sizes_and_padding_border_margin(&containing_block.into());
1324
1325        let (margin_block_start, margin_block_end) =
1326            solve_block_margins_for_in_flow_block_level(&pbm);
1327        let collapsed_margin_block_start = CollapsedMargin::new(margin_block_start);
1328
1329        // From https://drafts.csswg.org/css2/#floats:
1330        // "The border box of a table, a block-level replaced element, or an element in
1331        //  the normal flow that establishes a new block formatting context (such as an
1332        //  element with overflow other than visible) must not overlap the margin box of
1333        //  any floats in the same block formatting context as the element itself. If
1334        //  necessary, implementations should clear the said element by placing it below
1335        //  any preceding floats, but may place it adjacent to such floats if there is
1336        //  sufficient space. They may even make the border box of said element narrower
1337        //  than defined by section 10.3.3. CSS 2 does not define when a UA may put said
1338        //  element next to the float or by how much said element may become narrower."
1339        let mut content_size;
1340        let mut layout;
1341        let mut placement_rect;
1342
1343        // First compute the clear position required by the 'clear' property.
1344        // The code below may then add extra clearance when the element can't fit
1345        // next to floats not covered by 'clear'.
1346        let clear_position = sequential_layout_state.calculate_clear_position(
1347            Clear::from_style_and_container_writing_mode(style, containing_block_writing_mode),
1348            &collapsed_margin_block_start,
1349        );
1350        let ceiling = clear_position.unwrap_or_else(|| {
1351            sequential_layout_state.position_without_clearance(&collapsed_margin_block_start)
1352        });
1353
1354        // Then compute a tentative block size.
1355        let pbm_sums = pbm.sums_auto_is_zero(ignore_block_margins_for_stretch);
1356        let available_block_size = containing_block
1357            .size
1358            .block
1359            .to_definite()
1360            .map(|block_size| Au::zero().max(block_size - pbm_sums.block));
1361        let is_table = self.is_table();
1362        let preferred_aspect_ratio = self.preferred_aspect_ratio(&pbm.padding_border_sums);
1363        let tentative_block_content_size =
1364            self.tentative_block_content_size(preferred_aspect_ratio);
1365        let (preferred_block_size, min_block_size, max_block_size) =
1366            if let Some(block_content_size) = tentative_block_content_size {
1367                let (preferred, min, max) = content_box_sizes.block.resolve_each(
1368                    Size::FitContent,
1369                    Au::zero,
1370                    available_block_size,
1371                    || block_content_size,
1372                    is_table,
1373                );
1374                (Some(preferred), min, max)
1375            } else {
1376                content_box_sizes.block.resolve_each_extrinsic(
1377                    Size::FitContent,
1378                    Au::zero(),
1379                    available_block_size,
1380                )
1381            };
1382        let tentative_block_size =
1383            SizeConstraint::new(preferred_block_size, min_block_size, max_block_size);
1384
1385        // With the tentative block size we can compute the inline min/max-content sizes.
1386        let get_inline_content_sizes = || {
1387            let constraint_space =
1388                ConstraintSpace::new(tentative_block_size, style, preferred_aspect_ratio);
1389            self.inline_content_sizes(layout_context, &constraint_space)
1390                .sizes
1391        };
1392
1393        let justify_self = resolve_justify_self(style, containing_block.style);
1394        let automatic_inline_size = automatic_inline_size(justify_self, Some(self));
1395        let compute_inline_size = |stretch_size| {
1396            content_box_sizes.inline.resolve(
1397                Direction::Inline,
1398                automatic_inline_size,
1399                Au::zero,
1400                Some(stretch_size),
1401                get_inline_content_sizes,
1402                is_table,
1403            )
1404        };
1405
1406        let get_lazy_block_size = || {
1407            LazySize::new(
1408                &content_box_sizes.block,
1409                Direction::Block,
1410                Size::FitContent,
1411                Au::zero,
1412                available_block_size,
1413                is_table,
1414            )
1415        };
1416
1417        // The final inline size can depend on the available space, which depends on where
1418        // we are placing the box, since floats reduce the available space.
1419        // Here we assume that `compute_inline_size()` is a monotonically increasing function
1420        // with respect to the available space. Therefore, if we get the same result for 0
1421        // and for MAX_AU, it means that the function is constant.
1422        // TODO: `compute_inline_size()` may not be monotonic with `calc-size()`. For example,
1423        // `calc-size(stretch, (1px / (size + 1px) + sign(size)) * 1px)` would result in 1px
1424        // both when the available space is zero and infinity, but it's not constant.
1425        let inline_size_with_no_available_space = compute_inline_size(Au::zero());
1426        if inline_size_with_no_available_space == compute_inline_size(MAX_AU) {
1427            // If the inline size doesn't depend on the available inline space, we can just
1428            // compute it with an available inline space of zero. Then, after layout we can
1429            // compute the block size, and finally place among floats.
1430            let inline_size = inline_size_with_no_available_space;
1431            let lazy_block_size = get_lazy_block_size();
1432            layout = self.layout(
1433                layout_context,
1434                positioning_context,
1435                &ContainingBlock {
1436                    size: ContainingBlockSize {
1437                        inline: inline_size,
1438                        block: tentative_block_size,
1439                    },
1440                    style,
1441                },
1442                containing_block,
1443                preferred_aspect_ratio,
1444                &lazy_block_size,
1445            );
1446
1447            content_size = LogicalVec2 {
1448                block: lazy_block_size.resolve(|| layout.content_block_size),
1449                inline: layout.content_inline_size_for_table.unwrap_or(inline_size),
1450            };
1451
1452            let mut placement = PlacementAmongFloats::new(
1453                &sequential_layout_state.floats,
1454                ceiling,
1455                content_size + pbm.padding_border_sums,
1456                &pbm,
1457            );
1458            placement_rect = placement.place();
1459        } else {
1460            // If the inline size depends on the available space, then we need to iterate
1461            // the various placement candidates, resolve both the inline and block sizes
1462            // on each one placement area, and then check if the box actually fits it.
1463            // As an optimization, we first compute a lower bound of the final box size,
1464            // and skip placement candidates where not even the lower bound would fit.
1465            let minimum_size_of_block = LogicalVec2 {
1466                // For the lower bound of the inline size, simply assume no available space.
1467                // TODO: this won't work for things like `calc-size(stretch, 100px - size)`,
1468                // which should result in a bigger size when the available space gets smaller.
1469                inline: inline_size_with_no_available_space,
1470                block: match tentative_block_size {
1471                    // If we were able to resolve the preferred and maximum block sizes,
1472                    // use the tentative block size (it takes the 3 sizes into account).
1473                    SizeConstraint::Definite(size) if max_block_size.is_some() => size,
1474                    // Oherwise the preferred or maximum block size might end up being zero,
1475                    // so can only rely on the minimum block size.
1476                    _ => min_block_size,
1477                },
1478            } + pbm.padding_border_sums;
1479            let mut placement = PlacementAmongFloats::new(
1480                &sequential_layout_state.floats,
1481                ceiling,
1482                minimum_size_of_block,
1483                &pbm,
1484            );
1485
1486            loop {
1487                // First try to place the block using the minimum size as the object size.
1488                placement_rect = placement.place();
1489                let available_inline_size =
1490                    placement_rect.size.inline - pbm.padding_border_sums.inline;
1491                let proposed_inline_size = compute_inline_size(available_inline_size);
1492
1493                // Now lay out the block using the inline size we calculated from the placement.
1494                // Later we'll check to see if the resulting block size is compatible with the
1495                // placement.
1496                let positioning_context_length = positioning_context.len();
1497                let lazy_block_size = get_lazy_block_size();
1498                layout = self.layout(
1499                    layout_context,
1500                    positioning_context,
1501                    &ContainingBlock {
1502                        size: ContainingBlockSize {
1503                            inline: proposed_inline_size,
1504                            block: tentative_block_size,
1505                        },
1506                        style,
1507                    },
1508                    containing_block,
1509                    preferred_aspect_ratio,
1510                    &lazy_block_size,
1511                );
1512
1513                let inline_size = if let Some(inline_size) = layout.content_inline_size_for_table {
1514                    // This is a table that ended up being smaller than predicted because of
1515                    // collapsed columns. Note we don't backtrack to consider areas that we
1516                    // previously thought weren't big enough.
1517                    // TODO: Should `minimum_size_of_block.inline` be zero for tables?
1518                    debug_assert!(inline_size < proposed_inline_size);
1519                    inline_size
1520                } else {
1521                    proposed_inline_size
1522                };
1523                content_size = LogicalVec2 {
1524                    block: lazy_block_size.resolve(|| layout.content_block_size),
1525                    inline: inline_size,
1526                };
1527
1528                // Now we know the block size of this attempted layout of a box with block
1529                // size of auto. Try to fit it into our precalculated placement among the
1530                // floats. If it fits, then we can stop trying layout candidates.
1531                if placement.try_to_expand_for_auto_block_size(
1532                    content_size.block + pbm.padding_border_sums.block,
1533                    &placement_rect.size,
1534                ) {
1535                    break;
1536                }
1537
1538                // The previous attempt to lay out this independent formatting context
1539                // among the floats did not work, so we must unhoist any boxes from that
1540                // attempt.
1541                positioning_context.truncate(&positioning_context_length);
1542            }
1543        }
1544
1545        // Only set clearance if we would have cleared or the placement among floats moves
1546        // the block further in the block direction. These two situations are the ones that
1547        // prevent margin collapse.
1548        let has_clearance = clear_position.is_some() || placement_rect.start_corner.block > ceiling;
1549        let clearance = has_clearance.then(|| {
1550            placement_rect.start_corner.block -
1551                sequential_layout_state
1552                    .position_with_zero_clearance(&collapsed_margin_block_start)
1553        });
1554
1555        let ((margin_inline_start, margin_inline_end), effective_margin_inline_start) =
1556            solve_inline_margins_avoiding_floats(
1557                sequential_layout_state,
1558                containing_block,
1559                &pbm,
1560                content_size.inline + pbm.padding_border_sums.inline,
1561                placement_rect,
1562                justify_self,
1563            );
1564
1565        let margin = LogicalSides {
1566            inline_start: margin_inline_start,
1567            inline_end: margin_inline_end,
1568            block_start: margin_block_start,
1569            block_end: margin_block_end,
1570        };
1571
1572        // Clearance prevents margin collapse between this block and previous ones,
1573        // so in that case collapse margins before adjoining them below.
1574        if clearance.is_some() {
1575            sequential_layout_state.collapse_margins();
1576        }
1577        sequential_layout_state.adjoin_assign(&collapsed_margin_block_start);
1578
1579        // Margins can never collapse into independent formatting contexts.
1580        sequential_layout_state.collapse_margins();
1581        sequential_layout_state.advance_block_position(
1582            pbm.padding_border_sums.block + content_size.block + clearance.unwrap_or_else(Au::zero),
1583        );
1584        sequential_layout_state.adjoin_assign(&CollapsedMargin::new(margin.block_end));
1585
1586        let content_rect = LogicalRect {
1587            start_corner: LogicalVec2 {
1588                block: pbm.padding.block_start +
1589                    pbm.border.block_start +
1590                    clearance.unwrap_or_else(Au::zero),
1591                inline: pbm.padding.inline_start +
1592                    pbm.border.inline_start +
1593                    effective_margin_inline_start,
1594            },
1595            size: content_size,
1596        };
1597
1598        let mut base_fragment_info = self.base.base_fragment_info;
1599        if depends_on_block_constraints {
1600            base_fragment_info.flags.insert(
1601                FragmentFlags::SIZE_DEPENDS_ON_BLOCK_CONSTRAINTS_AND_CAN_BE_CHILD_OF_FLEX_ITEM,
1602            );
1603        }
1604
1605        BoxFragment::new(
1606            base_fragment_info,
1607            style.clone(),
1608            layout.fragments,
1609            content_rect.as_physical(Some(containing_block)),
1610            pbm.padding.to_physical(containing_block_writing_mode),
1611            pbm.border.to_physical(containing_block_writing_mode),
1612            margin.to_physical(containing_block_writing_mode),
1613            layout.specific_layout_info,
1614        )
1615        .with_baselines(layout.baselines)
1616        .with_block_level_layout_info(CollapsedBlockMargins::from_margin(&margin), clearance)
1617    }
1618}

1620struct ContainingBlockPaddingAndBorder<'a> {
1621    containing_block: ContainingBlock<'a>,
1622    pbm: PaddingBorderMargin,
1623    block_sizes: Sizes,
1624    depends_on_block_constraints: bool,
1625    available_block_size: Option<Au>,
1626    justify_self: AlignFlags,
1627    preferred_aspect_ratio: Option<AspectRatio>,
1628}
1629
1630struct ResolvedMargins {
1631    /// Used value for the margin properties, as exposed in getComputedStyle().
1632    pub margin: LogicalSides<Au>,
1633
1634    /// Distance between the border box and the containing block on the inline-start side.
1635    /// This is typically the same as the inline-start margin, but can be greater when
1636    /// the box is justified within the free space in the containing block.
1637    /// The reason we aren't just adjusting the used margin-inline-start is that
1638    /// this shouldn't be observable via getComputedStyle().
1639    /// <https://drafts.csswg.org/css-align/#justify-self-property>
1640    pub effective_margin_inline_start: Au,
1641}

1643/// Given the style for an in-flow box and its containing block, determine the containing
1644/// block for its children.
1645/// Note that in the presence of floats, this shouldn't be used for a block-level box
1646/// that establishes an independent formatting context (or is replaced), since the
1647/// inline size could then be incorrect.
1648fn solve_containing_block_padding_and_border_for_in_flow_box<'a>(
1649    containing_block: &ContainingBlock<'_>,
1650    layout_style: &'a LayoutStyle,
1651    get_inline_content_sizes: impl FnOnce(&ConstraintSpace) -> ContentSizes,
1652    ignore_block_margins_for_stretch: LogicalSides1D<bool>,
1653    context: Option<&IndependentFormattingContext>,
1654) -> ContainingBlockPaddingAndBorder<'a> {
1655    let style = layout_style.style();
1656    if matches!(style.pseudo(), Some(PseudoElement::ServoAnonymousBox)) {
1657        // <https://drafts.csswg.org/css2/#anonymous-block-level>
1658        // > Anonymous block boxes are ignored when resolving percentage values that would
1659        // > refer to it: the closest non-anonymous ancestor box is used instead.
1660        let containing_block_for_children = ContainingBlock {
1661            size: ContainingBlockSize {
1662                inline: containing_block.size.inline,
1663                block: containing_block.size.block,
1664            },
1665            style,
1666        };
1667        // <https://drafts.csswg.org/css2/#anonymous-block-level>
1668        // > Non-inherited properties have their initial value.
1669        return ContainingBlockPaddingAndBorder {
1670            containing_block: containing_block_for_children,
1671            pbm: PaddingBorderMargin::zero(),
1672            block_sizes: Sizes::default(),
1673            depends_on_block_constraints: false,
1674            // The available block size may actually be definite, but it should be irrelevant
1675            // since the sizing properties are set to their initial value.
1676            available_block_size: None,
1677            // The initial `justify-self` is `auto`, but use `normal` (behaving as `stretch`).
1678            // This is being discussed in <https://github.com/w3c/csswg-drafts/issues/11461>.
1679            justify_self: AlignFlags::NORMAL,
1680            preferred_aspect_ratio: None,
1681        };
1682    }
1683
1684    let ContentBoxSizesAndPBM {
1685        content_box_sizes,
1686        pbm,
1687        depends_on_block_constraints,
1688        ..
1689    } = layout_style.content_box_sizes_and_padding_border_margin(&containing_block.into());
1690
1691    let pbm_sums = pbm.sums_auto_is_zero(ignore_block_margins_for_stretch);
1692    let available_inline_size = Au::zero().max(containing_block.size.inline - pbm_sums.inline);
1693    let available_block_size = containing_block
1694        .size
1695        .block
1696        .to_definite()
1697        .map(|block_size| Au::zero().max(block_size - pbm_sums.block));
1698
1699    // TODO: support preferred aspect ratios on boxes that don't establish an independent
1700    // formatting context.
1701    let preferred_aspect_ratio =
1702        context.and_then(|context| context.preferred_aspect_ratio(&pbm.padding_border_sums));
1703    let is_table = layout_style.is_table();
1704
1705    // https://drafts.csswg.org/css2/#the-height-property
1706    // https://drafts.csswg.org/css2/visudet.html#min-max-heights
1707    let tentative_block_content_size =
1708        context.and_then(|context| context.tentative_block_content_size(preferred_aspect_ratio));
1709    let tentative_block_size = if let Some(block_content_size) = tentative_block_content_size {
1710        SizeConstraint::Definite(content_box_sizes.block.resolve(
1711            Direction::Block,
1712            Size::FitContent,
1713            Au::zero,
1714            available_block_size,
1715            || block_content_size,
1716            is_table,
1717        ))
1718    } else {
1719        content_box_sizes.block.resolve_extrinsic(
1720            Size::FitContent,
1721            Au::zero(),
1722            available_block_size,
1723        )
1724    };
1725
1726    // https://drafts.csswg.org/css2/#the-width-property
1727    // https://drafts.csswg.org/css2/visudet.html#min-max-widths
1728    let get_inline_content_sizes = || {
1729        get_inline_content_sizes(&ConstraintSpace::new(
1730            tentative_block_size,
1731            style,
1732            preferred_aspect_ratio,
1733        ))
1734    };
1735    let justify_self = resolve_justify_self(style, containing_block.style);
1736    let inline_size = content_box_sizes.inline.resolve(
1737        Direction::Inline,
1738        automatic_inline_size(justify_self, context),
1739        Au::zero,
1740        Some(available_inline_size),
1741        get_inline_content_sizes,
1742        is_table,
1743    );
1744
1745    let containing_block_for_children = ContainingBlock {
1746        size: ContainingBlockSize {
1747            inline: inline_size,
1748            block: tentative_block_size,
1749        },
1750        style,
1751    };
1752    // https://drafts.csswg.org/css-writing-modes/#orthogonal-flows
1753    assert_eq!(
1754        containing_block.style.writing_mode.is_horizontal(),
1755        containing_block_for_children
1756            .style
1757            .writing_mode
1758            .is_horizontal(),
1759        "Vertical writing modes are not supported yet"
1760    );
1761    ContainingBlockPaddingAndBorder {
1762        containing_block: containing_block_for_children,
1763        pbm,
1764        block_sizes: content_box_sizes.block,
1765        depends_on_block_constraints,
1766        available_block_size,
1767        justify_self,
1768        preferred_aspect_ratio,
1769    }
1770}

1772/// Given the containing block and size of an in-flow box, determine the margins.
1773/// Note that in the presence of floats, this shouldn't be used for a block-level box
1774/// that establishes an independent formatting context (or is replaced), since the
1775/// margins could then be incorrect.
1776fn solve_margins(
1777    containing_block: &ContainingBlock<'_>,
1778    pbm: &PaddingBorderMargin,
1779    inline_size: Au,
1780    justify_self: AlignFlags,
1781) -> ResolvedMargins {
1782    let (inline_margins, effective_margin_inline_start) =
1783        solve_inline_margins_for_in_flow_block_level(
1784            containing_block,
1785            pbm,

1793            inline_end: inline_margins.1,

1795            block_end: block_margins.1,

1797        effective_margin_inline_start,
1798    }

1811/// Resolves the `justify-self` value, preserving flags.
1812fn resolve_justify_self(style: &ComputedValues, parent_style: &ComputedValues) -> AlignFlags {
1813    let is_ltr = |style: &ComputedValues| style.writing_mode.line_left_is_inline_start();
1814    let alignment = match style.clone_justify_self().0 {
1815        AlignFlags::AUTO => parent_style.clone_justify_items().computed.0.0,
1816        alignment => alignment,
1817    };
1818    let alignment_value = match alignment.value() {
1819        AlignFlags::LEFT if is_ltr(parent_style) => AlignFlags::START,
1820        AlignFlags::LEFT => AlignFlags::END,
1821        AlignFlags::RIGHT if is_ltr(parent_style) => AlignFlags::END,
1822        AlignFlags::RIGHT => AlignFlags::START,
1823        AlignFlags::SELF_START if is_ltr(parent_style) == is_ltr(style) => AlignFlags::START,
1824        AlignFlags::SELF_START => AlignFlags::END,
1825        AlignFlags::SELF_END if is_ltr(parent_style) == is_ltr(style) => AlignFlags::END,
1826        AlignFlags::SELF_END => AlignFlags::START,
1827        alignment_value => alignment_value,
1828    };
1829    alignment.flags() | alignment_value
1830}

1856/// Justifies a block-level box, distributing the free space according to `justify-self`.
1857/// Note `<center>` and `<div align>` are implemented via internal 'text-align' values,
1858/// which are also handled here.
1859/// The provided free space should already take margins into account. In particular,
1860/// it should be zero if there is an auto margin.
1861/// <https://drafts.csswg.org/css-align/#justify-block>
1862fn justify_self_alignment(
1863    containing_block: &ContainingBlock,
1864    free_space: Au,
1865    justify_self: AlignFlags,
1866) -> Au {
1867    let mut alignment = justify_self.value();
1868    let is_safe = justify_self.flags() == AlignFlags::SAFE || alignment == AlignFlags::NORMAL;
1869    if is_safe && free_space <= Au::zero() {
1870        alignment = AlignFlags::START
1871    }
1872    match alignment {
1873        AlignFlags::NORMAL => {},
1874        AlignFlags::CENTER => return free_space / 2,
1875        AlignFlags::END => return free_space,
1876        _ => return Au::zero(),
1877    }
1878
1879    // For `justify-self: normal`, fall back to the special 'text-align' values.
1880    let style = containing_block.style;
1881    match style.clone_text_align() {
1882        TextAlignKeyword::MozCenter => free_space / 2,
1883        TextAlignKeyword::MozLeft if !style.writing_mode.line_left_is_inline_start() => free_space,
1884        TextAlignKeyword::MozRight if style.writing_mode.line_left_is_inline_start() => free_space,
1885        _ => Au::zero(),
1886    }
1887}

1889/// Resolves 'auto' margins of an in-flow block-level box in the inline axis,
1890/// distributing the free space in the containing block.
1891///
1892/// This is based on CSS2.1  10.3.3 <https://drafts.csswg.org/css2/#blockwidth>
1893/// but without adjusting the margins in "over-contrained" cases, as mandated by
1894/// <https://drafts.csswg.org/css-align/#justify-block>.
1895///
1896/// Note that in the presence of floats, this shouldn't be used for a block-level box
1897/// that establishes an independent formatting context (or is replaced).
1898///
1899/// In addition to the used margins, it also returns the effective margin-inline-start
1900/// (see ContainingBlockPaddingAndBorder).
1901fn solve_inline_margins_for_in_flow_block_level(
1902    containing_block: &ContainingBlock,
1903    pbm: &PaddingBorderMargin,
1904    inline_size: Au,
1905    justify_self: AlignFlags,
1906) -> ((Au, Au), Au) {
1907    let free_space = containing_block.size.inline - pbm.padding_border_sums.inline - inline_size;
1908    let mut justification = Au::zero();
1909    let inline_margins = match (pbm.margin.inline_start, pbm.margin.inline_end) {
1910        (AuOrAuto::Auto, AuOrAuto::Auto) => {
1911            let start = Au::zero().max(free_space / 2);
1912            (start, free_space - start)
1913        },
1914        (AuOrAuto::Auto, AuOrAuto::LengthPercentage(end)) => {
1915            (Au::zero().max(free_space - end), end)
1916        },
1917        (AuOrAuto::LengthPercentage(start), AuOrAuto::Auto) => (start, free_space - start),
1918        (AuOrAuto::LengthPercentage(start), AuOrAuto::LengthPercentage(end)) => {
1919            // In the cases above, the free space is zero after taking 'auto' margins into account.
1920            // But here we may still have some free space to perform 'justify-self' alignment.
1921            // This aligns the margin box within the containing block, or in other words,
1922            // aligns the border box within the margin-shrunken containing block.
1923            justification =
1924                justify_self_alignment(containing_block, free_space - start - end, justify_self);
1925            (start, end)
1926        },
1927    };
1928    let effective_margin_inline_start = inline_margins.0 + justification;
1929    (inline_margins, effective_margin_inline_start)
1930}

1932/// Resolves 'auto' margins of an in-flow block-level box in the inline axis
1933/// similarly to |solve_inline_margins_for_in_flow_block_level|. However,
1934/// they align within the provided rect (instead of the containing block),
1935/// to avoid overlapping floats.
1936/// In addition to the used margins, it also returns the effective
1937/// margin-inline-start (see ContainingBlockPaddingAndBorder).
1938/// It may differ from the used inline-start margin if the computed value
1939/// wasn't 'auto' and there are floats to avoid or the box is justified.
1940/// See <https://github.com/w3c/csswg-drafts/issues/9174>
1941fn solve_inline_margins_avoiding_floats(
1942    sequential_layout_state: &SequentialLayoutState,
1943    containing_block: &ContainingBlock,
1944    pbm: &PaddingBorderMargin,
1945    inline_size: Au,
1946    placement_rect: LogicalRect<Au>,
1947    justify_self: AlignFlags,
1948) -> ((Au, Au), Au) {
1949    // PlacementAmongFloats should guarantee that the inline size of the placement rect
1950    // is at least as big as `inline_size`. However, that may fail when dealing with
1951    // huge sizes that need to be saturated to MAX_AU, so floor by zero. See #37312.
1952    let free_space = Au::zero().max(placement_rect.size.inline - inline_size);
1953    let cb_info = &sequential_layout_state.floats.containing_block_info;
1954    let start_adjustment = placement_rect.start_corner.inline - cb_info.inline_start;
1955    let end_adjustment = cb_info.inline_end - placement_rect.max_inline_position();
1956    let mut justification = Au::zero();
1957    let inline_margins = match (pbm.margin.inline_start, pbm.margin.inline_end) {
1958        (AuOrAuto::Auto, AuOrAuto::Auto) => {
1959            let half = free_space / 2;
1960            (start_adjustment + half, end_adjustment + free_space - half)
1961        },
1962        (AuOrAuto::Auto, AuOrAuto::LengthPercentage(end)) => (start_adjustment + free_space, end),
1963        (AuOrAuto::LengthPercentage(start), AuOrAuto::Auto) => (start, end_adjustment + free_space),
1964        (AuOrAuto::LengthPercentage(start), AuOrAuto::LengthPercentage(end)) => {
1965            // The spec says 'justify-self' aligns the margin box within the float-shrunken
1966            // containing block. That's wrong (https://github.com/w3c/csswg-drafts/issues/9963),
1967            // and Blink and WebKit are broken anyways. So we match Gecko instead: this aligns
1968            // the border box within the instersection of the float-shrunken containing-block
1969            // and the margin-shrunken containing-block.
1970            justification = justify_self_alignment(containing_block, free_space, justify_self);
1971            (start, end)
1972        },
1973    };
1974    let effective_margin_inline_start = inline_margins.0.max(start_adjustment) + justification;
1975    (inline_margins, effective_margin_inline_start)
1976}

1978/// State that we maintain when placing blocks.
1979///
1980/// In parallel mode, this placement is done after all child blocks are laid out. In
1981/// sequential mode, this is done right after each block is laid out.
1982struct PlacementState<'container> {
1983    next_in_flow_margin_collapses_with_parent_start_margin: bool,
1984    last_in_flow_margin_collapses_with_parent_end_margin: bool,
1985    start_margin: CollapsedMargin,
1986    current_margin: CollapsedMargin,
1987    current_block_direction_position: Au,
1988    inflow_baselines: Baselines,
1989    is_inline_block_context: bool,
1990
1991    /// If this [`PlacementState`] is laying out a list item with an outside marker. Record the
1992    /// block size of that marker, because the content block size of the list item needs to be at
1993    /// least as tall as the marker size -- even though the marker doesn't advance the block
1994    /// position of the placement.
1995    marker_block_size: Option<Au>,
1996
1997    /// The [`ContainingBlock`] of the container into which this [`PlacementState`] is laying out
1998    /// fragments. This is used to convert between physical and logical geometry.
1999    containing_block: &'container ContainingBlock<'container>,
2000}
2001
2002impl<'container> PlacementState<'container> {
2003    fn new(
2004        collapsible_with_parent_start_margin: CollapsibleWithParentStartMargin,
2005        containing_block: &'container ContainingBlock<'container>,
2006    ) -> PlacementState<'container> {

2008            containing_block.style.get_box().clone_display() == Display::InlineBlock;
2009        PlacementState {
2010            next_in_flow_margin_collapses_with_parent_start_margin:
2011                collapsible_with_parent_start_margin.0,
2012            last_in_flow_margin_collapses_with_parent_end_margin: true,
2013            start_margin: CollapsedMargin::zero(),
2014            current_margin: CollapsedMargin::zero(),
2015            current_block_direction_position: Au::zero(),
2016            inflow_baselines: Baselines::default(),
2017            is_inline_block_context,
2018            marker_block_size: None,
2019            containing_block,
2020        }
2021    }
2022
2023    fn place_fragment_and_update_baseline(
2024        &mut self,
2025        fragment: &mut Fragment,
2026        sequential_layout_state: Option<&mut SequentialLayoutState>,
2027    ) {
2028        self.place_fragment(fragment, sequential_layout_state);
2029
2030        let box_fragment = match fragment {
2031            Fragment::Box(box_fragment) => box_fragment,
2032            _ => return,
2033        };
2034        let box_fragment = box_fragment.borrow();
2035
2036        // From <https://drafts.csswg.org/css-align-3/#baseline-export>:
2037        // > When finding the first/last baseline set of an inline-block, any baselines
2038        // > contributed by table boxes must be skipped. (This quirk is a legacy behavior from
2039        // > [CSS2].)
2040        if self.is_inline_block_context && box_fragment.is_table_wrapper() {
2041            return;
2042        }
2043
2044        let box_block_offset = box_fragment
2045            .content_rect()
2046            .origin
2047            .to_logical(self.containing_block)
2048            .block;
2049        let box_fragment_baselines =
2050            box_fragment.baselines(self.containing_block.style.writing_mode);
2051        if let (None, Some(first)) = (self.inflow_baselines.first, box_fragment_baselines.first) {
2052            self.inflow_baselines.first = Some(first + box_block_offset);
2053        }
2054        if let Some(last) = box_fragment_baselines.last {
2055            self.inflow_baselines.last = Some(last + box_block_offset);
2056        }
2057    }

2059    /// Place a single [Fragment] in a block level context using the state so far and
2060    /// information gathered from the [Fragment] itself.
2061    fn place_fragment(
2062        &mut self,
2063        fragment: &mut Fragment,
2064        sequential_layout_state: Option<&mut SequentialLayoutState>,
2065    ) {
2066        match fragment {
2067            Fragment::Box(fragment) => {
2068                // If this child is a marker positioned outside of a list item, then record its
2069                // size, but also ensure that it doesn't advance the block position of the placment.
2070                // This ensures item content is placed next to the marker.
2071                //
2072                // This is a pretty big hack because it doesn't properly handle all interactions
2073                // between the marker and the item. For instance the marker should be positioned at
2074                // the baseline of list item content and the first line of the item content should
2075                // be at least as tall as the marker -- not the entire list item itself.
2076                let fragment = &mut *fragment.borrow_mut();
2077                let is_outside_marker = fragment
2078                    .base
2079                    .flags
2080                    .contains(FragmentFlags::IS_OUTSIDE_LIST_ITEM_MARKER);
2081                if is_outside_marker {
2082                    assert!(self.marker_block_size.is_none());
2083                    self.marker_block_size = Some(
2084                        fragment
2085                            .content_rect()
2086                            .size
2087                            .to_logical(self.containing_block.style.writing_mode)
2088                            .block,
2089                    );
2090                    return;
2091                }
2092
2093                let BlockLevelLayoutInfo {
2094                    clearance,
2095                    block_margins_collapsed_with_children: fragment_block_margins,
2096                } = &**fragment
2097                    .block_level_layout_info
2098                    .as_ref()
2099                    .expect("A block-level fragment should have a BlockLevelLayoutInfo.");
2100                let mut fragment_block_size = fragment
2101                    .border_rect()
2102                    .size
2103                    .to_logical(self.containing_block.style.writing_mode)
2104                    .block;
2105
2106                // We use `last_in_flow_margin_collapses_with_parent_end_margin` to implement
2107                // this quote from https://drafts.csswg.org/css2/#collapsing-margins
2108                // > If the top and bottom margins of an element with clearance are adjoining,
2109                // > its margins collapse with the adjoining margins of following siblings but that
2110                // > resulting margin does not collapse with the bottom margin of the parent block.
2111                if let Some(clearance) = *clearance {
2112                    fragment_block_size += clearance;
2113                    // Margins can't be adjoining if they are separated by clearance.
2114                    // Setting `next_in_flow_margin_collapses_with_parent_start_margin` to false
2115                    // prevents collapsing with the start margin of the parent, and will set
2116                    // `collapsed_through` to false, preventing the parent from collapsing through.
2117                    self.current_block_direction_position += self.current_margin.solve();
2118                    self.current_margin = CollapsedMargin::zero();
2119                    self.next_in_flow_margin_collapses_with_parent_start_margin = false;
2120                    if fragment_block_margins.collapsed_through {
2121                        self.last_in_flow_margin_collapses_with_parent_end_margin = false;
2122                    }
2123                } else if !fragment_block_margins.collapsed_through {
2124                    self.last_in_flow_margin_collapses_with_parent_end_margin = true;
2125                }
2126
2127                if self.next_in_flow_margin_collapses_with_parent_start_margin {
2128                    debug_assert!(self.current_margin.solve().is_zero());
2129                    self.start_margin
2130                        .adjoin_assign(&fragment_block_margins.start);
2131                    if fragment_block_margins.collapsed_through {
2132                        self.start_margin.adjoin_assign(&fragment_block_margins.end);
2133                        return;
2134                    }
2135                    self.next_in_flow_margin_collapses_with_parent_start_margin = false;
2136                } else {
2137                    self.current_margin
2138                        .adjoin_assign(&fragment_block_margins.start);
2139                }
2140
2141                fragment.base.rect.origin += LogicalVec2 {
2142                    inline: Au::zero(),
2143                    block: self.current_margin.solve() + self.current_block_direction_position,
2144                }
2145                .to_physical_size(self.containing_block.style.writing_mode);
2146
2147                if fragment_block_margins.collapsed_through {
2148                    // `fragment_block_size` is typically zero when collapsing through,
2149                    // but we still need to consider it in case there is clearance.
2150                    self.current_block_direction_position += fragment_block_size;
2151                    self.current_margin
2152                        .adjoin_assign(&fragment_block_margins.end);
2153                } else {
2154                    self.current_block_direction_position +=
2155                        self.current_margin.solve() + fragment_block_size;
2156                    self.current_margin = fragment_block_margins.end;
2157                }
2158            },
2159            Fragment::AbsoluteOrFixedPositioned(fragment) => {
2160                // The alignment of absolutes in block flow layout is always "start", so the size of
2161                // the static position rectangle does not matter.
2162                fragment.borrow_mut().original_static_position_rect = LogicalRect {
2163                    start_corner: LogicalVec2 {
2164                        block: (self.current_margin.solve() +
2165                            self.current_block_direction_position),
2166                        inline: Au::zero(),
2167                    },
2168                    size: LogicalVec2::zero(),
2169                }
2170                .as_physical(Some(self.containing_block));
2171            },
2172            Fragment::Float(box_fragment) => {
2173                let sequential_layout_state = sequential_layout_state
2174                    .expect("Found float fragment without SequentialLayoutState");
2175                let block_offset_from_containing_block_top =
2176                    self.current_block_direction_position + self.current_margin.solve();
2177                let box_fragment = &mut *box_fragment.borrow_mut();
2178                sequential_layout_state.place_float_fragment(
2179                    box_fragment,
2180                    self.containing_block,
2181                    self.start_margin,
2182                    block_offset_from_containing_block_top,
2183                );
2184            },
2185            Fragment::Positioning(_) => {},
2186            _ => unreachable!(),
2187        }
2188    }
2189
2190    fn finish(mut self) -> (Au, CollapsedBlockMargins, Baselines) {
2191        if !self.last_in_flow_margin_collapses_with_parent_end_margin {
2192            self.current_block_direction_position += self.current_margin.solve();
2193            self.current_margin = CollapsedMargin::zero();
2194        }
2195        let (total_block_size, collapsed_through) = match self.marker_block_size {
2196            Some(marker_block_size) => (
2197                self.current_block_direction_position.max(marker_block_size),
2198                // If this is a list item (even empty) with an outside marker, then it
2199                // should not collapse through.
2200                false,
2201            ),
2202            None => (
2203                self.current_block_direction_position,
2204                self.next_in_flow_margin_collapses_with_parent_start_margin,
2205            ),
2206        };
2207
2208        (
2209            total_block_size,
2210            CollapsedBlockMargins {
2211                collapsed_through,
2212                start: self.start_margin,
2213                end: self.current_margin,
2214            },
2215            self.inflow_baselines,
2216        )
2217    }
2218}
