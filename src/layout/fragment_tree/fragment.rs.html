27#[derive(Clone, MallocSizeOf)]
28pub(crate) enum Fragment {
29    Box(ArcRefCell<BoxFragment>),
30    /// Floating content. A floated fragment is very similar to a normal
31    /// [BoxFragment] but it isn't positioned using normal in block flow
32    /// positioning rules (margin collapse, etc). Instead, they are laid
33    /// out by the [crate::flow::float::SequentialLayoutState] of their
34    /// float containing block formatting context.
35    Float(ArcRefCell<BoxFragment>),
36    Positioning(ArcRefCell<PositioningFragment>),
37    /// Absolute and fixed position fragments are hoisted up so that they
38    /// are children of the BoxFragment that establishes their containing
39    /// blocks, so that they can be laid out properly. When this happens
40    /// an `AbsoluteOrFixedPositioned` fragment is left at the original tree
41    /// position. This allows these hoisted fragments to be painted with
42    /// regard to their original tree order during stacking context tree /
43    /// display list construction.
44    AbsoluteOrFixedPositioned(ArcRefCell<HoistedSharedFragment>),
45    Text(ArcRefCell<TextFragment>),
46    Image(ArcRefCell<ImageFragment>),
47    IFrame(ArcRefCell<IFrameFragment>),
48}

224        match self {

241    pub(crate) fn client_rect(&self) -> Rect<i32, UnknownUnit> {
242        let rect = match self {
243            Fragment::Box(fragment) | Fragment::Float(fragment) => {
244                // https://drafts.csswg.org/cssom-view/#dom-element-clienttop
245                // " If the element has no associated CSS layout box or if the
246                //   CSS layout box is inline, return zero." For this check we
247                // also explicitly ignore the list item portion of the display
248                // style.
249                let fragment = fragment.borrow();
250                if fragment.is_inline_box() {
251                    return Rect::zero();
252                }
253
254                if fragment.is_table_wrapper() {
255                    // For tables the border actually belongs to the table grid box,
256                    // so we need to include it in the dimension of the table wrapper box.
257                    let mut rect = fragment.border_rect();
258                    rect.origin = PhysicalPoint::zero();
259                    rect
260                } else {
261                    let mut rect = fragment.padding_rect();
262                    rect.origin = PhysicalPoint::new(fragment.border.left, fragment.border.top);
263                    rect
264                }
265            },
266            _ => return Rect::zero(),
267        }
268        .to_untyped();
269
270        let rect = Rect::new(
271            Point2D::new(rect.origin.x.to_f32_px(), rect.origin.y.to_f32_px()),
272            Size2D::new(rect.size.width.to_f32_px(), rect.size.height.to_f32_px()),
273        );
274        rect.round().to_i32()
275    }

277    pub(crate) fn find<T>(
278        &self,
279        manager: &ContainingBlockManager<PhysicalRect<Au>>,
280        level: usize,
281        process_func: &mut impl FnMut(&Fragment, usize, &PhysicalRect<Au>) -> Option<T>,
282    ) -> Option<T> {
283        let containing_block = manager.get_containing_block_for_fragment(self);
284        if let Some(result) = process_func(self, level, containing_block) {
285            return Some(result);
286        }
287
288        match self {
289            Fragment::Box(fragment) | Fragment::Float(fragment) => {
290                let fragment = fragment.borrow();
291                let style = fragment.style();
292                let content_rect = fragment
293                    .content_rect()
294                    .translate(containing_block.origin.to_vector());
295                let padding_rect = fragment
296                    .padding_rect()
297                    .translate(containing_block.origin.to_vector());
298                let new_manager = if style
299                    .establishes_containing_block_for_all_descendants(fragment.base.flags)
300                {
301                    manager.new_for_absolute_and_fixed_descendants(&content_rect, &padding_rect)
302                } else if style
303                    .establishes_containing_block_for_absolute_descendants(fragment.base.flags)
304                {
305                    manager.new_for_absolute_descendants(&content_rect, &padding_rect)
306                } else {
307                    manager.new_for_non_absolute_descendants(&content_rect)
308                };
309
310                fragment
311                    .children
312                    .iter()
313                    .find_map(|child| child.find(&new_manager, level + 1, process_func))
314            },
315            Fragment::Positioning(fragment) => {
316                let fragment = fragment.borrow();
317                let content_rect = fragment
318                    .base
319                    .rect
320                    .translate(containing_block.origin.to_vector());
321                let new_manager = manager.new_for_non_absolute_descendants(&content_rect);
322                fragment
323                    .children
324                    .iter()
325                    .find_map(|child| child.find(&new_manager, level + 1, process_func))
326            },
327            _ => None,
328        }
329    }

394impl CollapsedMargin {
395    pub fn zero() -> Self {
396        Self {
397            max_positive: Au::zero(),
398            min_negative: Au::zero(),
399        }
400    }
401
402    pub fn new(margin: Au) -> Self {
403        Self {
404            max_positive: margin.max(Au::zero()),
405            min_negative: margin.min(Au::zero()),
406        }
407    }
408
409    pub fn adjoin(&self, other: &Self) -> Self {
410        Self {
411            max_positive: self.max_positive.max(other.max_positive),
412            min_negative: self.min_negative.min(other.min_negative),
413        }
414    }
415
416    pub fn adjoin_assign(&mut self, other: &Self) {
417        *self = self.adjoin(other);
418    }
419
420    pub fn solve(&self) -> Au {
421        self.max_positive + self.min_negative
422    }
