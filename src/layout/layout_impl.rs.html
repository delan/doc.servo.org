252            return false;
253        }

255        device.set_viewport_size(viewport_details.size);
256        device.set_device_pixel_ratio(device_pixel_ratio);
257        self.device_has_changed = true;
258        true

620    /// <https://drafts.css-houdini.org/css-properties-values-api-1/#the-registerproperty-function>
621    fn register_custom_property(
622        &mut self,
623        property_registration: PropertyRegistration,
624    ) -> Result<(), RegisterPropertyError> {
625        // Step 2. If name is not a custom property name string, throw a SyntaxError and exit this algorithm.
626        // If property set already contains an entry with name as its property name
627        // (compared codepoint-wise), throw an InvalidModificationError and exit this algorithm.
628        let Ok(name) = parse_name(&property_registration.name) else {
629            return Err(RegisterPropertyError::InvalidName);
630        };
631        let name = Atom::from(name);
632
633        if self
634            .stylist
635            .custom_property_script_registry()
636            .get(&name)
637            .is_some()
638        {
639            return Err(RegisterPropertyError::AlreadyRegistered);
640        }
641
642        // Step 3. Attempt to consume a syntax definition from syntax. If it returns failure,
643        // throw a SyntaxError. Otherwise, let syntax definition be the returned syntax definition.
644        let syntax = Descriptor::from_str(&property_registration.syntax, false)
645            .map_err(|_| RegisterPropertyError::InvalidSyntax)?;
646
647        // Step 4 - Parse and validate initial value
648        let initial_value = match property_registration.initial_value {
649            Some(value) => {
650                let mut input = ParserInput::new(&value);
651                let parsed = Parser::new(&mut input)
652                    .parse_entirely(|input| {
653                        input.skip_whitespace();
654                        SpecifiedValue::parse(input, &property_registration.url_data)
655                            .map(servo_arc::Arc::new)
656                    })
657                    .ok();
658                if parsed.is_none() {
659                    return Err(RegisterPropertyError::InvalidInitialValue);
660                }
661                parsed
662            },
663            None => None,
664        };
665
666        StyloPropertyRegistration::validate_initial_value(&syntax, initial_value.as_deref())
667            .map_err(|error| match error {
668                PropertyRegistrationError::InitialValueNotComputationallyIndependent => {
669                    RegisterPropertyError::InitialValueNotComputationallyIndependent
670                },
671                PropertyRegistrationError::InvalidInitialValue => {
672                    RegisterPropertyError::InvalidInitialValue
673                },
674                PropertyRegistrationError::NoInitialValue => RegisterPropertyError::NoInitialValue,
675            })?;
676
677        // Step 5. Set inherit flag to the value of inherits.
678        let inherits = if property_registration.inherits {
679            Inherits::True
680        } else {
681            Inherits::False
682        };
683
684        // Step 6. Let registered property be a struct with a property name of name, a syntax of
685        // syntax definition, an initial value of parsed initial value, and an inherit flag of inherit flag.
686        // Append registered property to property set.
687        let property_registration = StyloPropertyRegistration {
688            name: PropertyRuleName(name),
689            data: PropertyRegistrationData {
690                syntax,
691                initial_value,
692                inherits,
693            },
694            url_data: property_registration.url_data,
695            source_location: SourceLocation { line: 0, column: 0 },
696        };
697
698        self.stylist
699            .custom_property_script_registry_mut()
700            .register(property_registration);
701        self.stylist.rebuild_initial_values_for_custom_properties();
702
703        Ok(())
704    }

812    /// In some cases, if a restyle isn't necessary we can skip doing any work for layout
813    /// entirely. This check allows us to return early from layout without doing any work
814    /// at all.
815    fn can_skip_reflow_request_entirely(&self, reflow_request: &ReflowRequest) -> bool {
816        // If a restyle is necessary, restyle and reflow is a necessity.
817        if reflow_request.restyle.is_some() {
818            return false;
819        }
820        // We always need to at least build a fragment tree.
821        if self.fragment_tree.borrow().is_none() {
822            return false;
823        }
824
825        // If we have a fragment tree and it's up-to-date and this reflow
826        // doesn't need more reflow results, we can skip the rest of layout.
827        let necessary_phases = ReflowPhases::necessary(&reflow_request.reflow_goal);
828        if necessary_phases.is_empty() {
829            return true;
830        }
831
832        // If only the stacking context tree is required, and it's up-to-date,
833        // layout is unnecessary, otherwise a layout is necessary.
834        if necessary_phases == ReflowPhases::StackingContextTreeConstruction {
835            return self.stacking_context_tree.borrow().is_some() &&
836                !self.need_new_stacking_context_tree.get();
837        }
838
839        // Otherwise, the only interesting thing is whether the current display
840        // list is up-to-date.
841        assert_eq!(
842            necessary_phases,
843            ReflowPhases::StackingContextTreeConstruction | ReflowPhases::DisplayListConstruction
844        );
845        !self.need_new_display_list.get()
846    }

876    /// The high-level routine that performs layout.
877    #[servo_tracing::instrument(skip_all)]
878    fn handle_reflow(&mut self, mut reflow_request: ReflowRequest) -> Option<ReflowResult> {
879        self.maybe_print_reflow_event(&reflow_request);
880
881        if self.can_skip_reflow_request_entirely(&reflow_request) {
882            // We can skip layout, but we might need to update a scroll node.
883            return self
884                .handle_update_scroll_node_request(&reflow_request)
885                .then(|| ReflowResult {
886                    reflow_phases_run: ReflowPhasesRun::UpdatedScrollNodeOffset,
887                    ..Default::default()
888                });
889        }
890
891        let document = unsafe { ServoLayoutNode::new(&reflow_request.document) };
892        let document = document.as_document().unwrap();
893        let Some(root_element) = document.root_element() else {
894            debug!("layout: No root node: bailing");
895            return None;
896        };
897
898        let image_resolver = Arc::new(ImageResolver {
899            origin: reflow_request.origin.clone(),
900            image_cache: self.image_cache.clone(),
901            resolved_images_cache: self.resolved_images_cache.clone(),
902            pending_images: Mutex::default(),
903            pending_rasterization_images: Mutex::default(),
904            pending_svg_elements_for_serialization: Mutex::default(),
905            animating_images: reflow_request.animating_images.clone(),
906            animation_timeline_value: reflow_request.animation_timeline_value,
907        });
908
909        let (mut reflow_phases_run, iframe_sizes) = self.restyle_and_build_trees(
910            &mut reflow_request,
911            document,
912            root_element,
913            &image_resolver,
914        );
915        if self.calculate_overflow() {
916            reflow_phases_run.insert(ReflowPhasesRun::CalculatedOverflow);
917        }
918        if self.build_stacking_context_tree_for_reflow(&reflow_request) {
919            reflow_phases_run.insert(ReflowPhasesRun::BuiltStackingContextTree);
920        }
921        if self.build_display_list(&reflow_request, &image_resolver) {
922            reflow_phases_run.insert(ReflowPhasesRun::BuiltDisplayList);
923        }
924        if self.handle_update_scroll_node_request(&reflow_request) {
925            reflow_phases_run.insert(ReflowPhasesRun::UpdatedScrollNodeOffset);
926        }
927
928        let pending_images = std::mem::take(&mut *image_resolver.pending_images.lock());
929        let pending_rasterization_images =
930            std::mem::take(&mut *image_resolver.pending_rasterization_images.lock());
931        let pending_svg_elements_for_serialization =
932            std::mem::take(&mut *image_resolver.pending_svg_elements_for_serialization.lock());
933
934        Some(ReflowResult {
935            reflow_phases_run,
936            pending_images,
937            pending_rasterization_images,
938            pending_svg_elements_for_serialization,
939            iframe_sizes: Some(iframe_sizes),
940        })
941    }

990    #[servo_tracing::instrument(skip_all)]
991    fn restyle_and_build_trees(
992        &mut self,
993        reflow_request: &mut ReflowRequest,
994        document: ServoLayoutDocument<'_>,
995        root_element: ServoLayoutElement<'_>,
996        image_resolver: &Arc<ImageResolver>,
997    ) -> (ReflowPhasesRun, IFrameSizes) {
998        let mut snapshot_map = SnapshotMap::new();
999        let _snapshot_setter = match reflow_request.restyle.as_mut() {
1000            Some(restyle) => SnapshotSetter::new(restyle, &mut snapshot_map),
1001            None => return Default::default(),
1002        };
1003
1004        let document_shared_lock = document.style_shared_lock();
1005        let author_guard = document_shared_lock.read();
1006        let ua_stylesheets = &*UA_STYLESHEETS;
1007        let ua_or_user_guard = ua_stylesheets.shared_lock.read();
1008        let guards = StylesheetGuards {
1009            author: &author_guard,
1010            ua_or_user: &ua_or_user_guard,
1011        };
1012
1013        let rayon_pool = STYLE_THREAD_POOL.lock();
1014        let rayon_pool = rayon_pool.pool();
1015        let rayon_pool = rayon_pool.as_ref();
1016
1017        let device_has_changed = std::mem::replace(&mut self.device_has_changed, false);
1018        if device_has_changed {
1019            let sheet_origins_affected_by_device_change = self
1020                .stylist
1021                .media_features_change_changed_style(&guards, self.device());
1022            self.stylist
1023                .force_stylesheet_origins_dirty(sheet_origins_affected_by_device_change);
1024
1025            if let Some(mut data) = root_element.mutate_data() {
1026                data.hint.insert(RestyleHint::recascade_subtree());
1027            }
1028        }
1029
1030        self.prepare_stylist_for_reflow(
1031            reflow_request,
1032            document,
1033            root_element,
1034            &guards,
1035            ua_stylesheets,
1036            &snapshot_map,
1037        );
1038
1039        if self.previously_highlighted_dom_node.get() != reflow_request.highlighted_dom_node {
1040            // Need to manually force layout to build a new display list regardless of whether the box tree
1041            // changed or not.
1042            self.need_new_display_list.set(true);
1043        }
1044
1045        let layout_context = LayoutContext {
1046            style_context: self.build_shared_style_context(
1047                guards,
1048                &snapshot_map,
1049                reflow_request.animation_timeline_value,
1050                &reflow_request.animations,
1051                match reflow_request.stylesheets_changed() {
1052                    true => TraversalFlags::ForCSSRuleChanges,
1053                    false => TraversalFlags::empty(),
1054                },
1055            ),
1056            font_context: self.font_context.clone(),
1057            iframe_sizes: Mutex::default(),
1058            use_rayon: rayon_pool.is_some(),
1059            image_resolver: image_resolver.clone(),
1060            painter_id: self.webview_id.into(),
1061        };
1062
1063        let restyle = reflow_request
1064            .restyle
1065            .as_ref()
1066            .expect("Should not get here if there is not restyle.");
1067
1068        let recalc_style_traversal;
1069        let dirty_root;
1070        {
1071            let _span = profile_traits::trace_span!("Styling").entered();
1072
1073            let original_dirty_root = unsafe {
1074                ServoLayoutNode::new(&restyle.dirty_root.unwrap())
1075                    .as_element()
1076                    .unwrap()
1077            };
1078
1079            recalc_style_traversal = RecalcStyle::new(&layout_context);
1080            let token = {
1081                let shared =
1082                    DomTraversal::<ServoLayoutElement>::shared_context(&recalc_style_traversal);
1083                RecalcStyle::pre_traverse(original_dirty_root, shared)
1084            };
1085
1086            if !token.should_traverse() {
1087                layout_context.style_context.stylist.rule_tree().maybe_gc();
1088                return Default::default();
1089            }
1090
1091            dirty_root = driver::traverse_dom(&recalc_style_traversal, token, rayon_pool).as_node();
1092        }
1093
1094        let root_node = root_element.as_node();
1095        let damage_from_environment = if device_has_changed {
1096            RestyleDamage::RELAYOUT
1097        } else {
1098            Default::default()
1099        };
1100
1101        let damage = compute_damage_and_repair_style(
1102            &layout_context.style_context,
1103            root_node.to_threadsafe(),
1104            damage_from_environment,
1105        );
1106        if damage.contains(RestyleDamage::RECALCULATE_OVERFLOW) {
1107            self.need_overflow_calculation.set(true);
1108        }
1109        if damage.contains(RestyleDamage::REBUILD_STACKING_CONTEXT) {
1110            self.need_new_stacking_context_tree.set(true);
1111        }
1112        if damage.contains(RestyleDamage::REPAINT) {
1113            self.need_new_display_list.set(true);
1114        }
1115
1116        if !damage.contains(RestyleDamage::RELAYOUT) {
1117            layout_context.style_context.stylist.rule_tree().maybe_gc();
1118            return (ReflowPhasesRun::empty(), IFrameSizes::default());
1119        }
1120
1121        let mut box_tree = self.box_tree.borrow_mut();
1122        let box_tree = &mut *box_tree;
1123        let layout_damage: LayoutDamage = damage.into();
1124        if box_tree.is_none() || layout_damage.has_box_damage() {
1125            let mut build_box_tree = || {
1126                if !BoxTree::update(recalc_style_traversal.context(), dirty_root) {
1127                    *box_tree = Some(Arc::new(BoxTree::construct(
1128                        recalc_style_traversal.context(),
1129                        root_node,
1130                    )));
1131                }
1132            };
1133            if let Some(pool) = rayon_pool {
1134                pool.install(build_box_tree)
1135            } else {
1136                build_box_tree()
1137            };
1138        }
1139
1140        let viewport_size = self.stylist.device().au_viewport_size();
1141        let run_layout = || {
1142            box_tree
1143                .as_ref()
1144                .unwrap()
1145                .layout(recalc_style_traversal.context(), viewport_size)
1146        };
1147        let fragment_tree = Rc::new(if let Some(pool) = rayon_pool {
1148            pool.install(run_layout)
1149        } else {
1150            run_layout()
1151        });
1152
1153        *self.fragment_tree.borrow_mut() = Some(fragment_tree);
1154
1155        if self.debug.style_tree {
1156            println!(
1157                "{:?}",
1158                ShowSubtreeDataAndPrimaryValues(root_element.as_node())
1159            );
1160        }
1161        if self.debug.rule_tree {
1162            recalc_style_traversal
1163                .context()
1164                .style_context
1165                .stylist
1166                .rule_tree()
1167                .dump_stdout(&layout_context.style_context.guards);
1168        }
1169
1170        // GC the rule tree if some heuristics are met.
1171        layout_context.style_context.stylist.rule_tree().maybe_gc();
1172
1173        let mut iframe_sizes = layout_context.iframe_sizes.lock();
1174        (
1175            ReflowPhasesRun::RanLayout,
1176            std::mem::take(&mut *iframe_sizes),
1177        )
1178    }

1180    #[servo_tracing::instrument(name = "Overflow Calculation", skip_all)]
1181    fn calculate_overflow(&self) -> bool {
1182        if !self.need_overflow_calculation.get() {
1183            return false;
1184        }

1186        if let Some(fragment_tree) = &*self.fragment_tree.borrow() {

1189                fragment_tree.print();

1195        assert!(self.need_new_stacking_context_tree.get());

1220        let old_scroll_offsets = stacking_context_tree
1221            .as_ref()
1222            .map(|tree| tree.paint_info.scroll_tree.scroll_offsets());
1223
1224        // Build the StackingContextTree. This turns the `FragmentTree` into a
1225        // tree of fragments in CSS painting order and also creates all
1226        // applicable spatial and clip nodes.
1227        let mut new_stacking_context_tree = StackingContextTree::new(
1228            fragment_tree,
1229            viewport_details,
1230            self.id.into(),
1231            !self.have_ever_generated_display_list.get(),
1232            &self.debug,
1233        );
1234
1235        // When a new StackingContextTree is built, it contains a freshly built
1236        // ScrollTree. We want to preserve any existing scroll offsets in that tree,
1237        // adjusted by any new scroll constraints.
1238        if let Some(old_scroll_offsets) = old_scroll_offsets {
1239            new_stacking_context_tree
1240                .paint_info
1241                .scroll_tree
1242                .set_all_scroll_offsets(&old_scroll_offsets);
1243        }
1244
1245        if self.debug.scroll_tree {
1246            new_stacking_context_tree
1247                .paint_info
1248                .scroll_tree
1249                .debug_print();
1250        }
1251
1252        *stacking_context_tree = Some(new_stacking_context_tree);
1253
1254        // The stacking context tree is up-to-date again.
1255        self.need_new_stacking_context_tree.set(false);
1256        assert!(self.need_new_display_list.get());
1257
1258        true
1259    }
1260
1261    /// Build the display list for the current layout and send it to the renderer. If no display
1262    /// list is built, returns false.
1263    #[servo_tracing::instrument(name = "Display List Construction", skip_all)]
1264    fn build_display_list(
1265        &self,
1266        reflow_request: &ReflowRequest,
1267        image_resolver: &Arc<ImageResolver>,
1268    ) -> bool {
1269        if !ReflowPhases::necessary(&reflow_request.reflow_goal)
1270            .contains(ReflowPhases::DisplayListConstruction)
1271        {
1272            return false;
1273        }
1274        let Some(fragment_tree) = &*self.fragment_tree.borrow() else {
1275            return false;
1276        };
1277        let mut stacking_context_tree = self.stacking_context_tree.borrow_mut();
1278        let Some(stacking_context_tree) = stacking_context_tree.as_mut() else {
1279            return false;
1280        };
1281
1282        // If a non-display-list-generating reflow updated layout in a previous refow, we
1283        // cannot skip display list generation here the next time a display list is
1284        // requested.
1285        if !self.need_new_display_list.get() {
1286            return false;
1287        }
1288
1289        // TODO: Eventually this should be set when `paint_info` is created, but that requires
1290        // ensuring that the Epoch is passed to any method that can creates `StackingContextTree`.
1291        stacking_context_tree.paint_info.epoch = reflow_request.epoch;
1292
1293        let mut lcp_candidate_collector = self.lcp_candidate_collector.borrow_mut();
1294        if pref!(largest_contentful_paint_enabled) {
1295            // This ensures that we only create the LCP collector once per layout thread.
1296            if lcp_candidate_collector.is_none() {
1297                *lcp_candidate_collector = Some(LargestContentfulPaintCandidateCollector::new(
1298                    stacking_context_tree
1299                        .paint_info
1300                        .viewport_details
1301                        .layout_size(),
1302                ));
1303            }
1304        } else {
1305            *lcp_candidate_collector = None;
1306        }
1307
1308        let built_display_list = DisplayListBuilder::build(
1309            stacking_context_tree,
1310            fragment_tree,
1311            image_resolver.clone(),
1312            self.device().device_pixel_ratio(),
1313            reflow_request.highlighted_dom_node,
1314            &self.debug,
1315            lcp_candidate_collector.as_mut(),
1316        );
1317        self.paint_api.send_display_list(
1318            self.webview_id,
1319            &stacking_context_tree.paint_info,
1320            built_display_list,
1321        );
1322        if let Some(lcp_candidate_collector) = lcp_candidate_collector.as_mut() {
1323            if lcp_candidate_collector.did_lcp_candidate_update {
1324                if let Some(lcp_candidate) = lcp_candidate_collector.largest_contentful_paint() {
1325                    self.paint_api.send_lcp_candidate(
1326                        lcp_candidate,
1327                        self.webview_id,
1328                        self.id,
1329                        stacking_context_tree.paint_info.epoch,
1330                    );
1331                    lcp_candidate_collector.did_lcp_candidate_update = false;
1332                }
1333            }
1334        }
1335
1336        let (keys, instance_keys) = self
1337            .font_context
1338            .collect_unused_webrender_resources(false /* all */);
1339        self.paint_api
1340            .remove_unused_font_resources(self.webview_id.into(), keys, instance_keys);
1341
1342        self.have_ever_generated_display_list.set(true);
1343        self.need_new_display_list.set(false);
1344        self.previously_highlighted_dom_node
1345            .set(reflow_request.highlighted_dom_node);
1346        true
1347    }

1527        font: &Font,
1528        base_size: CSSPixelLength,
1529        _flags: QueryFontMetricsFlags,
1530    ) -> FontMetrics {

1533            .0

1537            .first(font_context)

1557            .map(CSSPixelLength::from);
1558
1559        let ic_width = first_font_metrics
1560            .ic_horizontal_advance
1561            .or_else(|| {
1562                font_group
1563                    .find_by_codepoint(font_context, '\u{6C34}', None, None, None)?
1564                    .metrics
1565                    .ic_horizontal_advance
1566            })
1567            .map(CSSPixelLength::from);
1568
1569        FontMetrics {
1570            x_height,
1571            zero_advance_measure,
1572            cap_height: None,

1595struct SnapshotSetter<'dom> {
1596    elements_with_snapshot: Vec<ServoLayoutElement<'dom>>,
1597}
1598
1599impl SnapshotSetter<'_> {
1600    fn new(restyle: &mut ReflowRequestRestyle, snapshot_map: &mut SnapshotMap) -> Self {
1601        debug!("Draining restyles: {}", restyle.pending_restyles.len());
1602        let restyles = std::mem::take(&mut restyle.pending_restyles);
1603
1604        let elements_with_snapshot: Vec<_> = restyles
1605            .iter()
1606            .filter(|r| r.1.snapshot.is_some())
1607            .map(|r| unsafe { ServoLayoutNode::new(&r.0).as_element().unwrap() })
1608            .collect();
1609
1610        for (element, restyle) in restyles {
1611            let element = unsafe { ServoLayoutNode::new(&element).as_element().unwrap() };
1612
1613            // If we haven't styled this node yet, we don't need to track a
1614            // restyle.
1615            let Some(mut style_data) = element.mutate_data() else {
1616                unsafe { element.unset_snapshot_flags() };
1617                continue;
1618            };
1619
1620            debug!("Noting restyle for {:?}: {:?}", element, style_data);
1621            if let Some(s) = restyle.snapshot {
1622                unsafe { element.set_has_snapshot() };
1623                snapshot_map.insert(element.as_node().opaque(), s);
1624            }
1625
1626            // Stash the data on the element for processing by the style system.
1627            style_data.hint.insert(restyle.hint);
1628            style_data.damage = restyle.damage;
1629        }
1630        Self {
1631            elements_with_snapshot,
1632        }
1633    }
1634}
1635
1636impl Drop for SnapshotSetter<'_> {
1637    fn drop(&mut self) {
1638        for element in &self.elements_with_snapshot {
1639            unsafe { element.unset_snapshot_flags() }
1640        }
1641    }
1642}

1645    #[derive(Clone, Copy, Debug, Eq, PartialEq)]
1646    pub struct ReflowPhases: u8 {
1647        const StackingContextTreeConstruction = 1 << 0;
1648        const DisplayListConstruction = 1 << 1;
1649    }

1651
1652impl ReflowPhases {
1653    /// Return the necessary phases of layout for the given [`ReflowGoal`]. Note that all
1654    /// [`ReflowGoals`] need the basic restyle + box tree layout + fragment tree layout,
1655    /// so [`ReflowPhases::empty()`] implies that.
1656    fn necessary(reflow_goal: &ReflowGoal) -> Self {
1657        match reflow_goal {
1658            ReflowGoal::LayoutQuery(query) => match query {
1659                QueryMsg::NodesFromPointQuery => {
1660                    Self::StackingContextTreeConstruction | Self::DisplayListConstruction
1661                },
1662                QueryMsg::BoxArea |
1663                QueryMsg::BoxAreas |
1664                QueryMsg::ElementsFromPoint |
1665                QueryMsg::OffsetParentQuery |
1666                QueryMsg::ResolvedStyleQuery |
1667                QueryMsg::ScrollingAreaOrOffsetQuery => Self::StackingContextTreeConstruction,
1668                QueryMsg::ClientRectQuery |
1669                QueryMsg::CurrentCSSZoomQuery |
1670                QueryMsg::ElementInnerOuterTextQuery |
1671                QueryMsg::InnerWindowDimensionsQuery |
1672                QueryMsg::PaddingQuery |
1673                QueryMsg::ResolvedFontStyleQuery |
1674                QueryMsg::ScrollParentQuery |
1675                QueryMsg::StyleQuery |
1676                QueryMsg::TextIndexQuery => Self::empty(),
1677            },
1678            ReflowGoal::UpdateScrollNode(..) | ReflowGoal::UpdateTheRendering => {
1679                Self::StackingContextTreeConstruction | Self::DisplayListConstruction
1680            },
1681        }
1682    }
