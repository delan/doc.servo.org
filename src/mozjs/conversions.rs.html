97/// Similar to num_traits, but we use need to be able to customize values
98pub trait Number {
99    /// Zero value of this type
100    const ZERO: Self;
101    /// Smallest finite number this type can represent
102    const MIN: Self;
103    /// Largest finite number this type can represent
104    const MAX: Self;
105}

107macro_rules! impl_num {
108    ($N:ty, $zero:expr, $min:expr, $max:expr) => {
109        impl Number for $N {
110            const ZERO: $N = $zero;
111            const MIN: $N = $min;
112            const MAX: $N = $max;
113        }
114    };
115}

137/// An enum to better support enums through FromJSValConvertible::from_jsval.
138#[derive(PartialEq, Eq, Clone, Debug)]
139pub enum ConversionResult<T> {
140    /// Everything went fine.
141    Success(T),
142    /// Conversion failed, without a pending exception.
143    Failure(Cow<'static, str>),
144}

146impl<T> ConversionResult<T> {
147    /// Returns Some(value) if it is `ConversionResult::Success`.
148    pub fn get_success_value(&self) -> Option<&T> {
149        match *self {
150            ConversionResult::Success(ref v) => Some(v),
151            _ => None,
152        }
153    }

195/// Behavior for converting out-of-range integers.
196#[derive(PartialEq, Eq, Clone)]
197pub enum ConversionBehavior {
198    /// Wrap into the integer's range.
199    Default,
200    /// Throw an exception.
201    EnforceRange,
202    /// Clamp into the integer's range.
203    Clamp,
204}

206/// Try to cast the number to a smaller type, but
207/// if it doesn't fit, it will return an error.
208// https://searchfox.org/mozilla-esr128/rev/1aa97f9d67f7a7231e62af283eaa02a6b31380e1/dom/bindings/PrimitiveConversions.h#166
209unsafe fn enforce_range<D>(cx: *mut JSContext, d: f64) -> Result<ConversionResult<D>, ()>
210where
211    D: Number + As<f64>,
212    f64: As<D>,
213{
214    if d.is_infinite() {
215        throw_type_error(cx, "value out of range in an EnforceRange argument");
216        return Err(());
217    }
218
219    let rounded = d.signum() * d.abs().floor();
220    if D::MIN.cast() <= rounded && rounded <= D::MAX.cast() {
221        Ok(ConversionResult::Success(rounded.cast()))
222    } else {
223        throw_type_error(cx, "value out of range in an EnforceRange argument");
224        Err(())
225    }
226}

228/// WebIDL ConvertToInt (Clamp) conversion.
229/// Spec: <https://webidl.spec.whatwg.org/#abstract-opdef-converttoint>
230///
231/// This function is ported from Geckoâ€™s
232/// [`PrimitiveConversionTraits_Clamp`](https://searchfox.org/firefox-main/rev/aee7c0f24f488cd7f5a835803b48dd0c0cb2fd5f/dom/bindings/PrimitiveConversions.h#226).
233///
234/// # Warning
235/// This function must only be used when the target type `D` represents an
236/// integer WebIDL type. Using it with non-integer types would be incorrect.
237fn clamp_to<D>(d: f64) -> D
238where
239    D: Number + PrimInt + As<f64>,
240    f64: As<D>,
241{
242    // NaN maps to zero.
243    if d.is_nan() {
244        return D::ZERO;
245    }
246
247    if d >= D::MAX.cast() {
248        return D::MAX;
249    }
250    if d <= D::MIN.cast() {
251        return D::MIN;
252    }
253
254    debug_assert!(d.is_finite());
255
256    // Banker's rounding (round ties towards even).
257    // We move away from 0 by 0.5 and then truncate. That gets us the right
258    // answer for any starting value except plus or minus N.5. With a starting
259    // value of that form, we now have plus or minus N+1. If N is odd, this is
260    // the correct result. If N is even, plus or minus N is the correct result.
261    let to_truncate = if d < 0.0 { d - 0.5 } else { d + 0.5 };
262
263    let mut truncated: D = to_truncate.cast();
264
265    if truncated.cast() == to_truncate {
266        // It was a tie (since moving away from 0 by 0.5 gave us the exact integer
267        // we want). Since we rounded away from 0, we either already have an even
268        // number or we have an odd number but the number we want is one closer to
269        // 0. So just unconditionally masking out the ones bit should do the trick
270        // to get us the value we want.
271        truncated = truncated & !D::one();
272    }
273
274    truncated
275}

320#[inline]
321unsafe fn convert_int_from_jsval<T, M>(
322    cx: *mut JSContext,
323    value: HandleValue,
324    option: ConversionBehavior,
325    convert_fn: unsafe fn(*mut JSContext, HandleValue) -> Result<M, ()>,
326) -> Result<ConversionResult<T>, ()>
327where
328    T: Number + As<f64> + PrimInt,
329    M: Number + As<T>,
330    f64: As<T>,
331{
332    match option {
333        ConversionBehavior::Default => Ok(ConversionResult::Success(convert_fn(cx, value)?.cast())),
334        ConversionBehavior::EnforceRange => enforce_range(cx, ToNumber(cx, value)?),
335        ConversionBehavior::Clamp => Ok(ConversionResult::Success(clamp_to(ToNumber(cx, value)?))),
336    }
337}

560/// Converts a `JSString`, encoded in "Latin1" (i.e. U+0000-U+00FF encoded as 0x00-0xFF) into a
561/// `String`.
562pub unsafe fn latin1_to_string(cx: *mut JSContext, s: NonNull<JSString>) -> String {
563    assert!(JS_DeprecatedStringHasLatin1Chars(s.as_ptr()));
564
565    let mut length = 0;
566    let chars = unsafe {
567        let chars = JS_GetLatin1StringCharsAndLength(cx, ptr::null(), s.as_ptr(), &mut length);
568        assert!(!chars.is_null());
569
570        slice::from_raw_parts(chars, length as usize)
571    };
572    // The `encoding.rs` documentation for `convert_latin1_to_utf8` states that:
573    // > The length of the destination buffer must be at least the length of the source
574    // > buffer times two.
575    let mut v = vec![0; chars.len() * 2];
576    let real_size = encoding_rs::mem::convert_latin1_to_utf8(chars, v.as_mut_slice());
577
578    v.truncate(real_size);
579
580    // Safety: convert_latin1_to_utf8 converts the raw bytes to utf8 and the
581    // buffer is the size specified in the documentation, so this should be safe.
582    unsafe { String::from_utf8_unchecked(v) }
583}

585/// Converts a `JSString` into a `String`, regardless of used encoding.
586pub unsafe fn jsstr_to_string(cx: *mut JSContext, jsstr: NonNull<JSString>) -> String {
587    if JS_DeprecatedStringHasLatin1Chars(jsstr.as_ptr()) {
588        return latin1_to_string(cx, jsstr);
589    }
590
591    let mut length = 0;
592    let chars = JS_GetTwoByteStringCharsAndLength(cx, ptr::null(), jsstr.as_ptr(), &mut length);
593    assert!(!chars.is_null());
594    let char_vec = slice::from_raw_parts(chars, length as usize);
595    String::from_utf16_lossy(char_vec)
596}

599impl ToJSValConvertible for str {
600    #[inline]
601    #[deny(unsafe_op_in_unsafe_fn)]
602    unsafe fn to_jsval(&self, cx: *mut JSContext, mut rval: MutableHandleValue) {
603        // Spidermonkey will automatically only copy latin1
604        // or similar if the given encoding can be small enough.
605        // So there is no need to distinguish between ascii only or similar.
606        let s = Utf8Chars::from(self);
607        let jsstr = unsafe { JS_NewStringCopyUTF8N(cx, &*s as *const _) };
608        if jsstr.is_null() {
609            panic!("JS String copy routine failed");
610        }
611        unsafe {
612            rval.set(StringValue(&*jsstr));
613        }
614    }
615}

626impl FromJSValConvertible for String {
627    type Config = ();
628    unsafe fn from_jsval(
629        cx: *mut JSContext,
630        value: HandleValue,
631        _: (),
632    ) -> Result<ConversionResult<String>, ()> {
633        let jsstr = ToString(cx, value);
634        let Some(jsstr) = NonNull::new(jsstr) else {
635            debug!("ToString failed");

723/// Rooting guard for the iterator field of ForOfIterator.
724/// Behaves like RootedGuard (roots on creation, unroots on drop),
725/// but borrows and allows access to the whole ForOfIterator, so
726/// that methods on ForOfIterator can still be used through it.
727struct ForOfIteratorGuard<'a> {
728    root: &'a mut ForOfIterator,
729}
730
731impl<'a> ForOfIteratorGuard<'a> {
732    fn new(cx: *mut JSContext, root: &'a mut ForOfIterator) -> Self {
733        unsafe {
734            Rooted::add_to_root_stack(&raw mut root.iterator, cx);
735        }
736        ForOfIteratorGuard { root }
737    }
738}
739
740impl<'a> Drop for ForOfIteratorGuard<'a> {
741    fn drop(&mut self) {
742        unsafe {
743            self.root.iterator.remove_from_root_stack();
744        }
745    }
746}
747
748impl<C: Clone, T: FromJSValConvertible<Config = C>> FromJSValConvertible for Vec<T> {
749    type Config = C;
750
751    unsafe fn from_jsval(
752        cx: *mut JSContext,
753        value: HandleValue,
754        option: C,
755    ) -> Result<ConversionResult<Vec<T>>, ()> {
756        if !value.is_object() {
757            return Ok(ConversionResult::Failure("Value is not an object".into()));
758        }
759
760        // Depending on the version of LLVM in use, bindgen can end up including
761        // a padding field in the ForOfIterator. To support multiple versions of
762        // LLVM that may not have the same fields as a result, we create an empty
763        // iterator instance and initialize a non-empty instance using the empty
764        // instance as a base value.
765        let zero = mem::zeroed();
766        let mut iterator = ForOfIterator {
767            cx_: cx,
768            iterator: RootedObject::new_unrooted(ptr::null_mut()),
769            nextMethod: RootedValue::new_unrooted(JSVal { asBits_: 0 }),
770            index: ::std::u32::MAX, // NOT_ARRAY
771            ..zero
772        };
773        let iterator = ForOfIteratorGuard::new(cx, &mut iterator);
774        let iterator: &mut ForOfIterator = &mut *iterator.root;
775
776        if !iterator.init(
777            value.into(),
778            ForOfIterator_NonIterableBehavior::AllowNonIterable,
779        ) {
780            return Err(());
781        }
782
783        if iterator.iterator.data.is_null() {
784            return Ok(ConversionResult::Failure("Value is not iterable".into()));
785        }
786
787        let mut ret = vec![];
788
789        loop {
790            let mut done = false;
791            rooted!(in(cx) let mut val = UndefinedValue());
792            if !iterator.next(val.handle_mut().into(), &mut done) {
793                return Err(());
794            }
795
796            if done {
797                break;
798            }
799
800            ret.push(match T::from_jsval(cx, val.handle(), option.clone())? {
801                ConversionResult::Success(v) => v,
802                ConversionResult::Failure(e) => {
803                    throw_type_error(cx, &e);
804                    return Err(());
805                }
806            });
807        }
808
809        Ok(ret).map(ConversionResult::Success)
810    }
811}

884/// A wrapper type over [`crate::jsapi::UTF8Chars`]. This is created to help transferring
885/// a rust string to mozjs. The inner [`crate::jsapi::UTF8Chars`] can be accessed via the
886/// [`std::ops::Deref`] trait.
887pub struct Utf8Chars<'a> {
888    lt_marker: std::marker::PhantomData<&'a ()>,
889    inner: crate::jsapi::UTF8Chars,
890}
891
892impl<'a> std::ops::Deref for Utf8Chars<'a> {
893    type Target = crate::jsapi::UTF8Chars;
894
895    fn deref(&self) -> &Self::Target {
896        &self.inner
897    }
898}
899
900impl<'a> From<&'a str> for Utf8Chars<'a> {
901    #[allow(unsafe_code)]
902    fn from(value: &'a str) -> Self {
903        use std::marker::PhantomData;
904
905        use crate::jsapi::mozilla::{Range, RangedPtr};
906        use crate::jsapi::UTF8Chars;
907
908        let range = value.as_bytes().as_ptr_range();
909        let range_start = range.start as *mut _;
910        let range_end = range.end as *mut _;
911        let start = RangedPtr {
912            _phantom_0: PhantomData,
913            mPtr: range_start,
914            #[cfg(feature = "debugmozjs")]
915            mRangeStart: range_start,
916            #[cfg(feature = "debugmozjs")]
917            mRangeEnd: range_end,
918        };
919        let end = RangedPtr {
920            _phantom_0: PhantomData,
921            mPtr: range_end,
922            #[cfg(feature = "debugmozjs")]
923            mRangeStart: range_start,
924            #[cfg(feature = "debugmozjs")]
925            mRangeEnd: range_end,
926        };
927        let base = Range {
928            _phantom_0: PhantomData,
929            mStart: start,
930            mEnd: end,
931        };
932        let inner = UTF8Chars { _base: base };
933        Self {
934            lt_marker: PhantomData,
935            inner,
936        }
937    }
938}
