47/// <https://fetch.spec.whatwg.org/#concept-body-clone>
48pub(crate) fn clone_body_stream_for_dom_body(
49    original_body_stream: &MutNullableDom<ReadableStream>,
50    cloned_body_stream: &MutNullableDom<ReadableStream>,
51    can_gc: CanGc,
52) -> Fallible<()> {
53    // To clone a body *body*, run these steps:
54
55    let Some(stream) = original_body_stream.get() else {
56        return Ok(());
57    };
58
59    // step 1. Let « out1, out2 » be the result of teeing body’s stream.
60    let branches = stream.tee(true, can_gc)?;
61    let out1 = &*branches[0];
62    let out2 = &*branches[1];
63
64    // step 2. Set body’s stream to out1.
65    // step 3. Return a body whose stream is out2 and other members are copied from body.
66    original_body_stream.set(Some(out1));
67    cloned_body_stream.set(Some(out2));
68
69    Ok(())
70}

72/// The Dom object, or ReadableStream, that is the source of a body.
73/// <https://fetch.spec.whatwg.org/#concept-body-source>
74#[derive(Clone, PartialEq)]
75pub(crate) enum BodySource {
76    /// A ReadableStream comes with a null-source.
77    Null,
78    /// Another Dom object as source,
79    /// TODO: store the actual object
80    /// and re-extract a stream on re-direct.
81    Object,
82}

84/// The reason to stop reading from the body.
85enum StopReading {
86    /// The stream has errored.
87    Error,
88    /// The stream is done.
89    Done,
90}

92/// The IPC route handler
93/// for <https://fetch.spec.whatwg.org/#concept-request-transmit-body>.
94/// This route runs in the script process,
95/// and will queue tasks to perform operations
96/// on the stream and transmit body chunks over IPC.
97#[derive(Clone)]
98struct TransmitBodyConnectHandler {
99    stream: Trusted<ReadableStream>,
100    task_source: SendableTaskSource,
101    bytes_sender: Option<IpcSender<BodyChunkResponse>>,
102    control_sender: IpcSender<BodyChunkRequest>,
103    in_memory: Option<GenericSharedMemory>,
104    in_memory_done: bool,
105    source: BodySource,
106}
107
108impl TransmitBodyConnectHandler {
109    pub(crate) fn new(
110        stream: Trusted<ReadableStream>,
111        task_source: SendableTaskSource,
112        control_sender: IpcSender<BodyChunkRequest>,
113        in_memory: Option<GenericSharedMemory>,
114        source: BodySource,
115    ) -> TransmitBodyConnectHandler {
116        TransmitBodyConnectHandler {
117            stream,
118            task_source,
119            bytes_sender: None,
120            control_sender,
121            in_memory,
122            in_memory_done: false,
123            source,
124        }
125    }
126
127    /// Reset `in_memory_done`, called when a stream is
128    /// re-extracted from the source to support a re-direct.
129    pub(crate) fn reset_in_memory_done(&mut self) {
130        self.in_memory_done = false;
131    }
132
133    /// Re-extract the source to support streaming it again for a re-direct.
134    /// TODO: actually re-extract the source, instead of just cloning data, to support Blob.
135    fn re_extract(&mut self, chunk_request_receiver: IpcReceiver<BodyChunkRequest>) {
136        let mut body_handler = self.clone();
137        body_handler.reset_in_memory_done();
138
139        ROUTER.add_typed_route(
140            chunk_request_receiver,
141            Box::new(move |message| {
142                let request = message.unwrap();
143                match request {
144                    BodyChunkRequest::Connect(sender) => {
145                        body_handler.start_reading(sender);
146                    },
147                    BodyChunkRequest::Extract(receiver) => {
148                        body_handler.re_extract(receiver);
149                    },
150                    BodyChunkRequest::Chunk => body_handler.transmit_source(),
151                    // Note: this is actually sent from this process
152                    // by the TransmitBodyPromiseHandler when reading stops.
153                    BodyChunkRequest::Done => {
154                        body_handler.stop_reading(StopReading::Done);
155                    },
156                    // Note: this is actually sent from this process
157                    // by the TransmitBodyPromiseHandler when the stream errors.
158                    BodyChunkRequest::Error => {
159                        body_handler.stop_reading(StopReading::Error);
160                    },
161                }
162            }),
163        );
164    }
165
166    /// In case of re-direct, and of a source available in memory,
167    /// send it all in one chunk.
168    ///
169    /// TODO: this method should be deprecated
170    /// in favor of making `re_extract` actually re-extract a stream from the source.
171    /// See #26686
172    fn transmit_source(&mut self) {
173        if self.in_memory_done {
174            // Step 5.1.3
175            self.stop_reading(StopReading::Done);
176            return;
177        }
178
179        if let BodySource::Null = self.source {
180            panic!("ReadableStream(Null) sources should not re-direct.");
181        }
182
183        if let Some(bytes) = self.in_memory.clone() {
184            // The memoized bytes are sent so we mark it as done again
185            self.in_memory_done = true;
186            let _ = self
187                .bytes_sender
188                .as_ref()
189                .expect("No bytes sender to transmit source.")
190                .send(BodyChunkResponse::Chunk(bytes));
191            return;
192        }
193        warn!("Re-directs for file-based Blobs not supported yet.");
194    }
195
196    /// Take the IPC sender sent by `net`, so we can send body chunks with it.
197    /// Also the entry point to <https://fetch.spec.whatwg.org/#concept-request-transmit-body>
198    fn start_reading(&mut self, sender: IpcSender<BodyChunkResponse>) {
199        self.bytes_sender = Some(sender);
200
201        // If we're using an actual ReadableStream, acquire a reader for it.
202        if self.source == BodySource::Null {
203            let stream = self.stream.clone();
204            self.task_source
205                .queue(task!(start_reading_request_body_stream: move || {
206                    // Step 1, Let body be request’s body.
207                    let rooted_stream = stream.root();
208
209                    // TODO: Step 2, If body is null.
210
211                    // Step 3, get a reader for stream.
212                    rooted_stream.acquire_default_reader(CanGc::note())
213                        .expect("Couldn't acquire a reader for the body stream.");
214
215                    // Note: this algorithm continues when the first chunk is requested by `net`.
216                }));
217        }
218    }
219
220    /// Drop the IPC sender sent by `net`
221    fn stop_reading(&mut self, reason: StopReading) {
222        let bytes_sender = self
223            .bytes_sender
224            .take()
225            .expect("Stop reading called multiple times on TransmitBodyConnectHandler.");
226        match reason {
227            StopReading::Error => {
228                let _ = bytes_sender.send(BodyChunkResponse::Error);
229            },
230            StopReading::Done => {
231                let _ = bytes_sender.send(BodyChunkResponse::Done);
232            },
233        }
234    }
235
236    /// Step 4 and following of <https://fetch.spec.whatwg.org/#concept-request-transmit-body>
237    fn transmit_body_chunk(&mut self) {
238        if self.in_memory_done {
239            // Step 5.1.3
240            self.stop_reading(StopReading::Done);
241            return;
242        }
243
244        let stream = self.stream.clone();
245        let control_sender = self.control_sender.clone();
246        let bytes_sender = self
247            .bytes_sender
248            .clone()
249            .expect("No bytes sender to transmit chunk.");
250
251        // In case of the data being in-memory, send everything in one chunk, by-passing SpiderMonkey.
252        if let Some(bytes) = self.in_memory.clone() {
253            let _ = bytes_sender.send(BodyChunkResponse::Chunk(bytes));
254            // Mark this body as `done` so that we can stop reading in the next tick,
255            // matching the behavior of the promise-based flow
256            self.in_memory_done = true;
257            return;
258        }
259
260        self.task_source.queue(
261            task!(setup_native_body_promise_handler: move || {
262                let rooted_stream = stream.root();
263                let global = rooted_stream.global();
264                let cx = GlobalScope::get_cx();
265
266                // Step 4, the result of reading a chunk from body’s stream with reader.
267                let promise = rooted_stream.read_a_chunk(CanGc::note());
268
269                // Step 5, the parallel steps waiting for and handling the result of the read promise,
270                // are a combination of the promise native handler here,
271                // and the corresponding IPC route in `component::net::http_loader`.
272                rooted!(in(*cx) let mut promise_handler = Some(TransmitBodyPromiseHandler {
273                    bytes_sender: bytes_sender.clone(),
274                    stream: Dom::from_ref(&rooted_stream.clone()),
275                    control_sender: control_sender.clone(),
276                }));
277
278                rooted!(in(*cx) let mut rejection_handler = Some(TransmitBodyPromiseRejectionHandler {
279                    bytes_sender,
280                    stream: Dom::from_ref(&rooted_stream.clone()),
281                    control_sender,
282                }));
283
284                let handler =
285                    PromiseNativeHandler::new(&global, promise_handler.take().map(|h| Box::new(h) as Box<_>), rejection_handler.take().map(|h| Box::new(h) as Box<_>), CanGc::note());
286
287                let realm = enter_realm(&*global);
288                let comp = InRealm::Entered(&realm);
289                promise.append_native_handler(&handler, comp, CanGc::note());
290            })
291        );
292    }
293}

295/// The handler of read promises of body streams used in
296/// <https://fetch.spec.whatwg.org/#concept-request-transmit-body>.
297#[derive(Clone, JSTraceable, MallocSizeOf)]
298#[cfg_attr(crown, crown::unrooted_must_root_lint::must_root)]
299struct TransmitBodyPromiseHandler {
300    #[ignore_malloc_size_of = "Channels are hard"]
301    #[no_trace]
302    bytes_sender: IpcSender<BodyChunkResponse>,
303    stream: Dom<ReadableStream>,
304    #[ignore_malloc_size_of = "Channels are hard"]
305    #[no_trace]
306    control_sender: IpcSender<BodyChunkRequest>,
307}
308
309impl js::gc::Rootable for TransmitBodyPromiseHandler {}
310
311impl Callback for TransmitBodyPromiseHandler {
312    /// Step 5 of <https://fetch.spec.whatwg.org/#concept-request-transmit-body>
313    fn callback(&self, cx: &mut CurrentRealm, v: HandleValue) {
314        let can_gc = CanGc::from_cx(cx);
315        let _realm = InRealm::Already(&cx.into());
316        let cx = cx.into();
317        let is_done = match get_read_promise_done(cx, &v, can_gc) {
318            Ok(is_done) => is_done,
319            Err(_) => {
320                // Step 5.5, the "otherwise" steps.
321                // TODO: terminate fetch.
322                let _ = self.control_sender.send(BodyChunkRequest::Done);
323                return self.stream.stop_reading(can_gc);
324            },
325        };
326
327        if is_done {
328            // Step 5.3, the "done" steps.
329            // TODO: queue a fetch task on request to process request end-of-body.
330            let _ = self.control_sender.send(BodyChunkRequest::Done);
331            return self.stream.stop_reading(can_gc);
332        }
333
334        let chunk = match get_read_promise_bytes(cx, &v, can_gc) {
335            Ok(chunk) => chunk,
336            Err(_) => {
337                // Step 5.5, the "otherwise" steps.
338                let _ = self.control_sender.send(BodyChunkRequest::Error);
339                return self.stream.stop_reading(can_gc);
340            },
341        };
342
343        // Step 5.1 and 5.2, transmit chunk.
344        // Send the chunk to the body transmitter in net::http_loader::obtain_response.
345        // TODO: queue a fetch task on request to process request body for request.
346        let _ = self
347            .bytes_sender
348            .send(BodyChunkResponse::Chunk(GenericSharedMemory::from_bytes(
349                &chunk,
350            )));
351    }
352}

379/// <https://fetch.spec.whatwg.org/#body-with-type>
380pub(crate) struct ExtractedBody {
381    /// <https://fetch.spec.whatwg.org/#concept-body-stream>
382    pub(crate) stream: DomRoot<ReadableStream>,
383    /// <https://fetch.spec.whatwg.org/#concept-body-source>
384    pub(crate) source: BodySource,
385    /// <https://fetch.spec.whatwg.org/#concept-body-total-bytes>
386    pub(crate) total_bytes: Option<usize>,
387    /// <https://fetch.spec.whatwg.org/#body-with-type-type>
388    pub(crate) content_type: Option<DOMString>,
389}
390
391impl ExtractedBody {
392    /// Build a request body from the extracted body,
393    /// to be sent over IPC to net to use with `concept-request-transmit-body`,
394    /// see <https://fetch.spec.whatwg.org/#concept-request-transmit-body>.
395    ///
396    /// Also returning the corresponding readable stream,
397    /// to be stored on the request in script,
398    /// and potentially used as part of `consume_body`,
399    /// see <https://fetch.spec.whatwg.org/#concept-body-consume-body>
400    ///
401    /// Transmitting a body over fetch, and consuming it in script,
402    /// are mutually exclusive operations, since each will lock the stream to a reader.
403    pub(crate) fn into_net_request_body(self) -> (RequestBody, DomRoot<ReadableStream>) {
404        let ExtractedBody {
405            stream,
406            total_bytes,
407            content_type: _,
408            source,
409        } = self;
410
411        // First, setup some infra to be used to transmit body
412        //  from `components::script` to `components::net`.
413        let (chunk_request_sender, chunk_request_receiver) = ipc::channel().unwrap();
414
415        let trusted_stream = Trusted::new(&*stream);
416
417        let global = stream.global();
418        let task_source = global.task_manager().networking_task_source();
419
420        // In case of the data being in-memory, send everything in one chunk, by-passing SM.
421        let in_memory = stream.get_in_memory_bytes();
422
423        let net_source = match source {
424            BodySource::Null => NetBodySource::Null,
425            _ => NetBodySource::Object,
426        };
427
428        let mut body_handler = TransmitBodyConnectHandler::new(
429            trusted_stream,
430            task_source.into(),
431            chunk_request_sender.clone(),
432            in_memory,
433            source,
434        );
435
436        ROUTER.add_typed_route(
437            chunk_request_receiver,
438            Box::new(move |message| {
439                match message.unwrap() {
440                    BodyChunkRequest::Connect(sender) => {
441                        body_handler.start_reading(sender);
442                    },
443                    BodyChunkRequest::Extract(receiver) => {
444                        body_handler.re_extract(receiver);
445                    },
446                    BodyChunkRequest::Chunk => body_handler.transmit_body_chunk(),
447                    // Note: this is actually sent from this process
448                    // by the TransmitBodyPromiseHandler when reading stops.
449                    BodyChunkRequest::Done => {
450                        body_handler.stop_reading(StopReading::Done);
451                    },
452                    // Note: this is actually sent from this process
453                    // by the TransmitBodyPromiseHandler when the stream errors.
454                    BodyChunkRequest::Error => {
455                        body_handler.stop_reading(StopReading::Error);
456                    },
457                }
458            }),
459        );
460
461        // Return `components::net` view into this request body,
462        // which can be used by `net` to transmit it over the network.
463        let request_body = RequestBody::new(chunk_request_sender, net_source, total_bytes);
464
465        // Also return the stream for this body, which can be used by script to consume it.
466        (request_body, stream)
467    }
468
469    /// Is the data of the stream of this extracted body available in memory?
470    pub(crate) fn in_memory(&self) -> bool {
471        self.stream.in_memory()
472    }
473}

477    fn extract(
478        &self,
479        global: &GlobalScope,
480        keep_alive: bool,
481        can_gc: CanGc,
482    ) -> Fallible<ExtractedBody>;
483}

485impl Extractable for BodyInit {
486    /// <https://fetch.spec.whatwg.org/#concept-bodyinit-extract>
487    fn extract(
488        &self,
489        global: &GlobalScope,
490        keep_alive: bool,
491        can_gc: CanGc,
492    ) -> Fallible<ExtractedBody> {
493        match self {
494            BodyInit::String(s) => s.extract(global, keep_alive, can_gc),
495            BodyInit::URLSearchParams(usp) => usp.extract(global, keep_alive, can_gc),
496            BodyInit::Blob(b) => b.extract(global, keep_alive, can_gc),
497            BodyInit::FormData(formdata) => formdata.extract(global, keep_alive, can_gc),
498            BodyInit::ArrayBuffer(typedarray) => {
499                let bytes = typedarray.to_vec();
500                let total_bytes = bytes.len();
501                let stream = ReadableStream::new_from_bytes(global, bytes, can_gc)?;
502                Ok(ExtractedBody {
503                    stream,
504                    total_bytes: Some(total_bytes),
505                    content_type: None,
506                    source: BodySource::Object,
507                })
508            },
509            BodyInit::ArrayBufferView(typedarray) => {
510                let bytes = typedarray.to_vec();
511                let total_bytes = bytes.len();
512                let stream = ReadableStream::new_from_bytes(global, bytes, can_gc)?;
513                Ok(ExtractedBody {
514                    stream,
515                    total_bytes: Some(total_bytes),
516                    content_type: None,
517                    source: BodySource::Object,
518                })
519            },
520            BodyInit::ReadableStream(stream) => {
521                // If keepalive is true, then throw a TypeError.
522                if keep_alive {
523                    return Err(Error::Type(
524                        "The body's stream is for a keepalive request".to_string(),
525                    ));
526                }
527                // If object is disturbed or locked, then throw a TypeError.
528                if stream.is_locked() || stream.is_disturbed() {
529                    return Err(Error::Type(
530                        "The body's stream is disturbed or locked".to_string(),
531                    ));
532                }
533
534                Ok(ExtractedBody {
535                    stream: stream.clone(),
536                    total_bytes: None,
537                    content_type: None,
538                    source: BodySource::Null,
539                })
540            },
541        }
542    }
543}

650            source: BodySource::Object,
651        })
652    }
653}

675/// <https://fetch.spec.whatwg.org/#concept-body-consume-body>
676/// <https://fetch.spec.whatwg.org/#body-fully-read>
677/// A combination of parts of both algorithms,
678/// `body-fully-read` can be fully implemented, and separated, later,
679/// see #36049.
680pub(crate) fn consume_body<T: BodyMixin + DomObject>(
681    object: &T,
682    body_type: BodyType,
683    can_gc: CanGc,
684) -> Rc<Promise> {
685    let global = object.global();
686    let cx = GlobalScope::get_cx();
687
688    // Enter the realm of the object whose body is being consumed.
689    let realm = enter_realm(&*global);
690    let comp = InRealm::Entered(&realm);
691
692    // Let promise be a new promise.
693    // Note: re-ordered so we can return the promise below.
694    let promise = Promise::new_in_current_realm(comp, can_gc);
695
696    // If object is unusable, then return a promise rejected with a TypeError.
697    if object.is_unusable() {
698        promise.reject_error(
699            Error::Type("The body's stream is disturbed or locked".to_string()),
700            can_gc,
701        );
702        return promise;
703    }
704
705    let stream = match object.body() {
706        Some(stream) => stream,
707        None => {
708            // If object’s body is null, then run successSteps with an empty byte sequence.
709            resolve_result_promise(
710                body_type,
711                &promise,
712                object.get_mime_type(can_gc),
713                Vec::with_capacity(0),
714                cx,
715                can_gc,
716            );
717            return promise;
718        },
719    };
720
721    // Note: from `fully_read`.
722    // Let reader be the result of getting a reader for body’s stream.
723    // If that threw an exception,
724    // then run errorSteps with that exception and return.
725    let reader = match stream.acquire_default_reader(can_gc) {
726        Ok(r) => r,
727        Err(e) => {
728            promise.reject_error(e, can_gc);
729            return promise;
730        },
731    };
732
733    // Let errorSteps given error be to reject promise with error.
734    let error_promise = promise.clone();
735
736    // Let successSteps given a byte sequence data be to resolve promise
737    // with the result of running convertBytesToJSValue with data.
738    // If that threw an exception, then run errorSteps with that exception.
739    let mime_type = object.get_mime_type(can_gc);
740    let success_promise = promise.clone();
741
742    // Read all bytes from reader, given successSteps and errorSteps.
743    // Note: spec uses an intermediary concept of `fully_read`,
744    // which seems useful when invoking fetch from other places.
745    // TODO: #36049
746    reader.read_all_bytes(
747        cx,
748        Rc::new(move |bytes: &[u8]| {
749            resolve_result_promise(
750                body_type,
751                &success_promise,
752                mime_type.clone(),
753                bytes.to_vec(),
754                cx,
755                can_gc,
756            );
757        }),
758        Rc::new(move |cx, v| {
759            error_promise.reject(cx, v, can_gc);
760        }),
761        can_gc,
762    );
763
764    promise
765}
766
767/// The success steps of
768/// <https://fetch.spec.whatwg.org/#concept-body-consume-body>.
769fn resolve_result_promise(
770    body_type: BodyType,
771    promise: &Promise,
772    mime_type: Vec<u8>,
773    body: Vec<u8>,
774    cx: JSContext,
775    can_gc: CanGc,
776) {
777    let pkg_data_results = run_package_data_algorithm(cx, body, body_type, mime_type, can_gc);
778
779    match pkg_data_results {
780        Ok(results) => {
781            match results {
782                FetchedData::Text(s) => promise.resolve_native(&USVString(s), can_gc),
783                FetchedData::Json(j) => promise.resolve_native(&j, can_gc),
784                FetchedData::BlobData(b) => promise.resolve_native(&b, can_gc),
785                FetchedData::FormData(f) => promise.resolve_native(&f, can_gc),
786                FetchedData::Bytes(b) => promise.resolve_native(&b, can_gc),
787                FetchedData::ArrayBuffer(a) => promise.resolve_native(&a, can_gc),
788                FetchedData::JSException(e) => promise.reject_native(&e.handle(), can_gc),
789            };
790        },
791        Err(err) => promise.reject_error(err, can_gc),
792    }
793}

795/// The algorithm that takes a byte sequence
796/// and returns a JavaScript value or throws an exception of
797/// <https://fetch.spec.whatwg.org/#concept-body-consume-body>.
798fn run_package_data_algorithm(
799    cx: JSContext,
800    bytes: Vec<u8>,
801    body_type: BodyType,
802    mime_type: Vec<u8>,
803    can_gc: CanGc,
804) -> Fallible<FetchedData> {
805    let mime = &*mime_type;
806    let in_realm_proof = AlreadyInRealm::assert_for_cx(cx);
807    let global = GlobalScope::from_safe_context(cx, InRealm::Already(&in_realm_proof));
808    match body_type {
809        BodyType::Text => run_text_data_algorithm(bytes),
810        BodyType::Json => run_json_data_algorithm(cx, bytes),
811        BodyType::Blob => run_blob_data_algorithm(&global, bytes, mime, can_gc),
812        BodyType::FormData => run_form_data_algorithm(&global, bytes, mime, can_gc),
813        BodyType::ArrayBuffer => run_array_buffer_data_algorithm(cx, bytes, can_gc),
814        BodyType::Bytes => run_bytes_data_algorithm(cx, bytes, can_gc),
815    }
816}
817
818/// <https://fetch.spec.whatwg.org/#ref-for-concept-body-consume-body%E2%91%A4>
819fn run_text_data_algorithm(bytes: Vec<u8>) -> Fallible<FetchedData> {
820    // This implements the Encoding standard's "decode UTF-8", which removes the
821    // BOM if present.
822    let no_bom_bytes = if bytes.starts_with(b"\xEF\xBB\xBF") {
823        &bytes[3..]
824    } else {
825        &bytes
826    };
827    Ok(FetchedData::Text(
828        String::from_utf8_lossy(no_bom_bytes).into_owned(),
829    ))
830}
831
832#[expect(unsafe_code)]
833/// <https://fetch.spec.whatwg.org/#ref-for-concept-body-consume-body%E2%91%A3>
834fn run_json_data_algorithm(cx: JSContext, bytes: Vec<u8>) -> Fallible<FetchedData> {
835    // The JSON spec allows implementations to either ignore UTF-8 BOM or treat it as an error.
836    // `JS_ParseJSON` treats this as an error, so it is necessary for us to strip it if present.
837    //
838    // https://datatracker.ietf.org/doc/html/rfc8259#section-8.1
839    let json_text = decode_to_utf16_with_bom_removal(&bytes, UTF_8);
840    rooted!(in(*cx) let mut rval = UndefinedValue());
841    unsafe {
842        if !JS_ParseJSON(
843            *cx,
844            json_text.as_ptr(),
845            json_text.len() as u32,
846            rval.handle_mut(),
847        ) {
848            rooted!(in(*cx) let mut exception = UndefinedValue());
849            assert!(JS_GetPendingException(*cx, exception.handle_mut()));
850            JS_ClearPendingException(*cx);
851            return Ok(FetchedData::JSException(RootedTraceableBox::from_box(
852                Heap::boxed(exception.get()),
853            )));
854        }
855        let rooted_heap = RootedTraceableBox::from_box(Heap::boxed(rval.get()));
856        Ok(FetchedData::Json(rooted_heap))
857    }
858}

860/// <https://fetch.spec.whatwg.org/#ref-for-concept-body-consume-body%E2%91%A0>
861fn run_blob_data_algorithm(
862    root: &GlobalScope,
863    bytes: Vec<u8>,
864    mime: &[u8],
865    can_gc: CanGc,
866) -> Fallible<FetchedData> {
867    let mime_string = if let Ok(s) = String::from_utf8(mime.to_vec()) {
868        s
869    } else {
870        "".to_string()
871    };
872    let blob = Blob::new(
873        root,
874        BlobImpl::new_from_bytes(bytes, normalize_type_string(&mime_string)),
875        can_gc,
876    );
877    Ok(FetchedData::BlobData(blob))
878}

880/// <https://fetch.spec.whatwg.org/#ref-for-concept-body-consume-body%E2%91%A2>
881fn run_form_data_algorithm(
882    root: &GlobalScope,
883    bytes: Vec<u8>,
884    mime: &[u8],
885    can_gc: CanGc,
886) -> Fallible<FetchedData> {
887    let mime_str = str::from_utf8(mime).unwrap_or_default();
888    let mime: Mime = mime_str
889        .parse()
890        .map_err(|_| Error::Type("Inappropriate MIME-type for Body".to_string()))?;
891
892    // TODO
893    // ... Parser for Mime(TopLevel::Multipart, SubLevel::FormData, _)
894    // ... is not fully determined yet.
895    if mime.type_() == mime::APPLICATION && mime.subtype() == mime::WWW_FORM_URLENCODED {
896        let entries = form_urlencoded::parse(&bytes);
897        let formdata = FormData::new(None, root, can_gc);
898        for (k, e) in entries {
899            formdata.Append(USVString(k.into_owned()), USVString(e.into_owned()));
900        }
901        return Ok(FetchedData::FormData(formdata));
902    }
903
904    Err(Error::Type("Inappropriate MIME-type for Body".to_string()))
905}

907/// <https://fetch.spec.whatwg.org/#ref-for-concept-body-consume-body%E2%91%A1>
908fn run_bytes_data_algorithm(cx: JSContext, bytes: Vec<u8>, can_gc: CanGc) -> Fallible<FetchedData> {
909    rooted!(in(*cx) let mut array_buffer_ptr = ptr::null_mut::<JSObject>());
910
911    create_buffer_source::<Uint8>(cx, &bytes, array_buffer_ptr.handle_mut(), can_gc)
912        .map_err(|_| Error::JSFailed)?;
913
914    let rooted_heap = RootedTraceableBox::from_box(Heap::boxed(array_buffer_ptr.get()));
915    Ok(FetchedData::Bytes(rooted_heap))
916}

933#[expect(unsafe_code)]
934pub(crate) fn decode_to_utf16_with_bom_removal(
935    bytes: &[u8],
936    encoding: &'static Encoding,
937) -> Vec<u16> {
938    let mut decoder = encoding.new_decoder_with_bom_removal();
939    let capacity = decoder
940        .max_utf16_buffer_length(bytes.len())
941        .expect("Overflow");
942    let mut utf16 = Vec::with_capacity(capacity);
943    let extra = unsafe { slice::from_raw_parts_mut(utf16.as_mut_ptr(), capacity) };
944    let (_, read, written, _) = decoder.decode_to_utf16(bytes, extra, true);
945    assert_eq!(read, bytes.len());
946    unsafe { utf16.set_len(written) }
947    utf16
948}
