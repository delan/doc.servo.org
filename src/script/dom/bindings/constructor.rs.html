57/// <https://html.spec.whatwg.org/multipage/#htmlconstructor>
58fn html_constructor(
59    cx: &mut js::context::JSContext,
60    global: &GlobalScope,
61    call_args: &CallArgs,
62    check_type: fn(&Element) -> bool,
63    proto_id: PrototypeList::ID,
64    creator: unsafe fn(SafeJSContext, HandleObject, *mut ProtoOrIfaceArray),
65) -> Result<(), ()> {
66    let window = global.downcast::<Window>().unwrap();
67    let document = window.Document();
68
69    // Step 1. Let registry be current global object's custom element registry.
70    let registry = window.CustomElements();
71
72    // Step 2 https://html.spec.whatwg.org/multipage/#htmlconstructor
73    // The custom element definition cannot use an element interface as its constructor
74
75    // The new_target might be a cross-compartment wrapper. Get the underlying object
76    // so we can do the spec's object-identity checks.
77    rooted!(&in(cx) let new_target_unwrapped = unsafe {
78        UnwrapObjectDynamic(call_args.new_target().to_object(), cx.raw_cx(), true)
79    });
80    if new_target_unwrapped.is_null() {
81        throw_dom_exception(
82            cx.into(),
83            global,
84            Error::Type("new.target is null".to_owned()),
85            CanGc::from_cx(cx),
86        );
87        return Err(());
88    }
89    if call_args.callee() == new_target_unwrapped.get() {
90        throw_dom_exception(
91            cx.into(),
92            global,
93            Error::Type("new.target must not be the active function object".to_owned()),
94            CanGc::from_cx(cx),
95        );
96        return Err(());
97    }
98
99    // Step 3. Let definition be the item in registry's custom element definition set with constructor
100    // equal to NewTarget. If there is no such item, then throw a TypeError.
101    rooted!(&in(cx) let new_target = call_args.new_target().to_object());
102    let definition = match registry.lookup_definition_by_constructor(new_target.handle()) {
103        Some(definition) => definition,
104        None => {
105            throw_dom_exception(
106                cx.into(),
107                global,
108                Error::Type("No custom element definition found for new.target".to_owned()),
109                CanGc::from_cx(cx),
110            );
111            return Err(());
112        },
113    };
114
115    // Step 4. Let isValue be null.
116    let mut is_value = None;
117
118    rooted!(&in(cx) let callee = unsafe { UnwrapObjectStatic(call_args.callee()) });
119    if callee.is_null() {
120        throw_dom_exception(cx.into(), global, Error::Security(None), CanGc::from_cx(cx));
121        return Err(());
122    }
123
124    {
125        let mut realm = AutoRealm::new_from_handle(cx, callee.handle());
126        let (global_object, cx) = realm.global_and_reborrow();
127        rooted!(&in(cx) let mut constructor = ptr::null_mut::<JSObject>());
128
129        // Step 5. If definition's local name is equal to definition's name
130        // (i.e., definition is for an autonomous custom element):
131        if definition.is_autonomous() {
132            // Since this element is autonomous, its active function object must be the HTMLElement
133            // Retrieve the constructor object for HTMLElement
134            HTMLElementBinding::GetConstructorObject(
135                cx.into(),
136                global_object,
137                constructor.handle_mut(),
138            );
139        }
140        // Step 6. Otherwise (i.e., if definition is for a customized built-in element):
141        else {
142            get_constructor_object_from_local_name(
143                definition.local_name.clone(),
144                cx.into(),
145                global_object,
146                constructor.handle_mut(),
147            );
148
149            // Step 6.3 Set isValue to definition's name.
150            is_value = Some(definition.name.clone());
151        }
152        // Callee must be the same as the element interface's constructor object.
153        if constructor.get() != callee.get() {
154            throw_dom_exception(
155                cx.into(),
156                global,
157                Error::Type("Custom element does not extend the proper interface".to_owned()),
158                CanGc::from_cx(cx),
159            );
160            return Err(());
161        }
162    }
163
164    // Step 6
165    rooted!(&in(cx) let mut prototype = ptr::null_mut::<JSObject>());
166    get_desired_proto(
167        cx.into(),
168        call_args,
169        proto_id,
170        creator,
171        prototype.handle_mut(),
172    )?;
173
174    let entry = definition.construction_stack.borrow().last().cloned();
175    let result = match entry {
176        // Step 7. If definition's construction stack is empty:
177        None => {
178            // Step 7.1
179            let name = QualName::new(None, ns!(html), definition.local_name.clone());
180            // Any prototype used to create these elements will be overwritten before returning
181            // from this function, so we don't bother overwriting the defaults here.
182            let element = if definition.is_autonomous() {
183                DomRoot::upcast(HTMLElement::new(
184                    name.local,
185                    None,
186                    &document,
187                    None,
188                    CanGc::from_cx(cx),
189                ))
190            } else {
191                create_native_html_element(
192                    name,
193                    None,
194                    &document,
195                    ElementCreator::ScriptCreated,
196                    None,
197                )
198            };
199
200            // Step 7.2-7.5 are performed in the generated caller code.
201
202            // Step 7.6 Set element's custom element state to "custom".
203            element.set_custom_element_state(CustomElementState::Custom);
204
205            // Step 7.7 Set element's custom element definition to definition.
206            element.set_custom_element_definition(definition.clone());
207
208            // Step 7.8 Set element's is value to isValue.
209            if let Some(is_value) = is_value {
210                element.set_is(is_value);
211            }
212
213            if !check_type(&element) {
214                throw_dom_exception(
215                    cx.into(),
216                    global,
217                    Error::InvalidState(None),
218                    CanGc::from_cx(cx),
219                );
220                return Err(());
221            } else {
222                // Step 7.9 Return element.
223                element
224            }
225        },
226        // Step 9
227        Some(ConstructionStackEntry::Element(element)) => {
228            // Step 11 is performed in the generated caller code.
229
230            // Step 12
231            let mut construction_stack = definition.construction_stack.borrow_mut();
232            construction_stack.pop();
233            construction_stack.push(ConstructionStackEntry::AlreadyConstructedMarker);
234
235            // Step 13
236            if !check_type(&element) {
237                throw_dom_exception(
238                    cx.into(),
239                    global,
240                    Error::InvalidState(None),
241                    CanGc::from_cx(cx),
242                );
243                return Err(());
244            } else {
245                element
246            }
247        },
248        // Step 10
249        Some(ConstructionStackEntry::AlreadyConstructedMarker) => {
250            let s = "Top of construction stack marked AlreadyConstructed due to \
251                     a custom element constructor constructing itself after super()"
252                .to_string();
253            throw_dom_exception(cx.into(), global, Error::Type(s), CanGc::from_cx(cx));
254            return Err(());
255        },
256    };
257
258    rooted!(&in(cx) let mut element = result.reflector().get_jsobject().get());
259    unsafe {
260        if !JS_WrapObject(cx, element.handle_mut()) {
261            return Err(());
262        }
263
264        JS_SetPrototype(cx, element.handle(), prototype.handle());
265
266        result.safe_to_jsval(
267            cx.into(),
268            MutableHandleValue::from_raw(call_args.rval()),
269            CanGc::from_cx(cx),
270        );
271    }
272    Ok(())
273}

419pub(crate) fn call_html_constructor<T: DerivedFrom<Element> + DomObject>(
420    cx: &mut js::context::JSContext,
421    args: &CallArgs,
422    global: &GlobalScope,
423    proto_id: PrototypeList::ID,
424    creator: unsafe fn(SafeJSContext, HandleObject, *mut ProtoOrIfaceArray),
425) -> bool {
426    fn element_derives_interface<T: DerivedFrom<Element>>(element: &Element) -> bool {
427        element.is::<T>()
428    }
429
430    html_constructor(
431        cx,
432        global,
433        args,
434        element_derives_interface::<T>,
435        proto_id,
436        creator,
437    )
438    .is_ok()
