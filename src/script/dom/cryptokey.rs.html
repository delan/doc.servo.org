68/// <https://w3c.github.io/webcrypto/#cryptokey-interface>
69#[dom_struct]
70pub(crate) struct CryptoKey {
71    reflector_: Reflector,
72
73    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-type>
74    key_type: KeyType,
75
76    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-extractable>
77    extractable: Cell<bool>,
78
79    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-algorithm>
80    ///
81    /// The contents of the [[algorithm]] internal slot shall be, or be derived from, a
82    /// KeyAlgorithm.
83    #[no_trace]
84    algorithm: KeyAlgorithmAndDerivatives,
85
86    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-algorithm_cached>
87    #[ignore_malloc_size_of = "Defined in mozjs"]
88    algorithm_cached: Heap<*mut JSObject>,
89
90    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-usages>
91    ///
92    /// The contents of the [[usages]] internal slot shall be of type Sequence<KeyUsage>.
93    usages: DomRefCell<Vec<KeyUsage>>,
94
95    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-usages_cached>
96    #[ignore_malloc_size_of = "Defined in mozjs"]
97    usages_cached: Heap<*mut JSObject>,
98
99    /// <https://w3c.github.io/webcrypto/#dfn-CryptoKey-slot-handle>
100    #[no_trace]
101    handle: Handle,
102}
103
104impl CryptoKey {
105    fn new_inherited(
106        key_type: KeyType,
107        extractable: bool,
108        algorithm: KeyAlgorithmAndDerivatives,
109        usages: Vec<KeyUsage>,
110        handle: Handle,
111    ) -> CryptoKey {
112        CryptoKey {
113            reflector_: Reflector::new(),
114            key_type,
115            extractable: Cell::new(extractable),
116            algorithm,
117            algorithm_cached: Heap::default(),
118            usages: DomRefCell::new(usages),
119            usages_cached: Heap::default(),
120            handle,
121        }
122    }
123
124    pub(crate) fn new(
125        global: &GlobalScope,
126        key_type: KeyType,
127        extractable: bool,
128        algorithm: KeyAlgorithmAndDerivatives,
129        usages: Vec<KeyUsage>,
130        handle: Handle,
131        can_gc: CanGc,
132    ) -> DomRoot<CryptoKey> {
133        let crypto_key = reflect_dom_object(
134            Box::new(CryptoKey::new_inherited(
135                key_type,
136                extractable,
137                algorithm.clone(),
138                usages.clone(),
139                handle,
140            )),
141            global,
142            can_gc,
143        );
144
145        let cx = GlobalScope::get_cx();
146
147        // Create and store a cached object of algorithm
148        rooted!(in(*cx) let mut algorithm_object_value: Value);
149        algorithm.safe_to_jsval(cx, algorithm_object_value.handle_mut(), can_gc);
150        crypto_key
151            .algorithm_cached
152            .set(algorithm_object_value.to_object());
153
154        // Create and store a cached object of usages
155        rooted!(in(*cx) let mut usages_object_value: Value);
156        usages.safe_to_jsval(cx, usages_object_value.handle_mut(), can_gc);
157        crypto_key
158            .usages_cached
159            .set(usages_object_value.to_object());
160
161        crypto_key
162    }
163
164    pub(crate) fn algorithm(&self) -> &KeyAlgorithmAndDerivatives {
165        &self.algorithm
166    }
167
168    pub(crate) fn usages(&self) -> Vec<KeyUsage> {
169        self.usages.borrow().clone()
170    }
171
172    pub(crate) fn handle(&self) -> &Handle {
173        &self.handle
174    }
175
176    pub(crate) fn set_extractable(&self, extractable: bool) {
177        self.extractable.set(extractable);
178    }
179
180    pub(crate) fn set_usages(&self, usages: &[KeyUsage]) {
181        *self.usages.borrow_mut() = usages.to_owned();
182
183        // Create and store a cached object of usages
184        let cx = GlobalScope::get_cx();
185        rooted!(in(*cx) let mut usages_object_value: Value);
186        usages.safe_to_jsval(cx, usages_object_value.handle_mut(), CanGc::note());
187        self.usages_cached.set(usages_object_value.to_object());
188    }
189}

191impl CryptoKeyMethods<crate::DomTypeHolder> for CryptoKey {
192    /// <https://w3c.github.io/webcrypto/#dom-cryptokey-type>
193    fn Type(&self) -> KeyType {
194        // Reflects the [[type]] internal slot, which contains the type of the underlying key.
195        self.key_type
196    }
197
198    /// <https://w3c.github.io/webcrypto/#dom-cryptokey-extractable>
199    fn Extractable(&self) -> bool {
200        // Reflects the [[extractable]] internal slot, which indicates whether or not the raw
201        // keying material may be exported by the application.
202        self.extractable.get()
203    }
204
205    /// <https://w3c.github.io/webcrypto/#dom-cryptokey-algorithm>
206    fn Algorithm(&self, _cx: JSContext) -> NonNull<JSObject> {
207        // Returns the cached ECMAScript object associated with the [[algorithm]] internal slot.
208        NonNull::new(self.algorithm_cached.get()).unwrap()
209    }
210
211    /// <https://w3c.github.io/webcrypto/#dom-cryptokey-usages>
212    fn Usages(&self, _cx: JSContext) -> NonNull<JSObject> {
213        // Returns the cached ECMAScript object associated with the [[usages]] internal slot, which
214        // indicates which cryptographic operations are permissible to be used with this key.
215        NonNull::new(self.usages_cached.get()).unwrap()
216    }
217}

219impl Handle {
220    pub(crate) fn as_bytes(&self) -> &[u8] {
221        match self {
222            Self::Aes128(bytes) => bytes,
223            Self::Aes192(bytes) => bytes,
224            Self::Aes256(bytes) => bytes,
225            Self::Pbkdf2(bytes) => bytes,
226            Self::Hmac(bytes) => bytes,
227            Self::Ed25519(bytes) => bytes,
228            _ => unreachable!(),
229        }
230    }
231}

233impl MallocSizeOf for Handle {
234    fn size_of(&self, ops: &mut malloc_size_of::MallocSizeOfOps) -> usize {
235        match self {
236            Handle::RsaPrivateKey(private_key) => private_key.size_of(ops),
237            Handle::RsaPublicKey(public_key) => public_key.size_of(ops),
238            Handle::P256PrivateKey(private_key) => private_key.size_of(ops),
239            Handle::P384PrivateKey(private_key) => private_key.size_of(ops),
240            Handle::P521PrivateKey(private_key) => private_key.size_of(ops),
241            Handle::P256PublicKey(public_key) => public_key.size_of(ops),
242            Handle::P384PublicKey(public_key) => public_key.size_of(ops),
243            Handle::P521PublicKey(public_key) => public_key.size_of(ops),
244            Handle::X25519PrivateKey(private_key) => private_key.size_of(ops),
245            Handle::X25519PublicKey(public_key) => public_key.size_of(ops),
246            Handle::Aes128(bytes) => bytes.size_of(ops),
247            Handle::Aes192(bytes) => bytes.size_of(ops),
248            Handle::Aes256(bytes) => bytes.size_of(ops),
249            Handle::Aes128Key(key) => key.size_of(ops),
250            Handle::Aes192Key(key) => key.size_of(ops),
251            Handle::Aes256Key(key) => key.size_of(ops),
252            Handle::HkdfSecret(secret) => secret.size_of(ops),
253            Handle::Pbkdf2(bytes) => bytes.size_of(ops),
254            Handle::Hmac(bytes) => bytes.size_of(ops),
255            Handle::Ed25519(bytes) => bytes.size_of(ops),
256            Handle::MlKem512PrivateKey(seed) => seed.0.size_of(ops) + seed.1.size_of(ops),
257            Handle::MlKem768PrivateKey(seed) => seed.0.size_of(ops) + seed.1.size_of(ops),
258            Handle::MlKem1024PrivateKey(seed) => seed.0.size_of(ops) + seed.1.size_of(ops),
259            Handle::MlKem512PublicKey(public_key) => public_key.size_of(ops),
260            Handle::MlKem768PublicKey(public_key) => public_key.size_of(ops),
261            Handle::MlKem1024PublicKey(public_key) => public_key.size_of(ops),
262            Handle::MlDsa44PrivateKey(seed) => seed.size_of(ops),
263            Handle::MlDsa65PrivateKey(seed) => seed.size_of(ops),
264            Handle::MlDsa87PrivateKey(seed) => seed.size_of(ops),
265            Handle::MlDsa44PublicKey(public_key) => public_key.size_of(ops),
266            Handle::MlDsa65PublicKey(public_key) => public_key.size_of(ops),
267            Handle::MlDsa87PublicKey(public_key) => public_key.size_of(ops),
268            Handle::ChaCha20Poly1305Key(key) => key.size_of(ops),
269            Handle::Argon2Password(password) => password.size_of(ops),
270        }
271    }
