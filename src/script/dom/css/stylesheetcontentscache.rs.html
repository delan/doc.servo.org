21const MAX_LENGTH_OF_TEXT_INSERTED_INTO_TABLE: usize = 1024;
22const UNIQUE_OWNED: usize = 2;
23
24/// Using [`Atom`] as a cache key to avoid inefficient string content comparison. Although
25/// the [`Atom`] is already based on reference counting, an extra [`Rc`] is introduced
26/// to trace how many [`style::stylesheets::Stylesheet`]s of style elements are sharing
27/// same [`StylesheetContents`], based on the following considerations:
28/// * The reference count within [`Atom`] is dedicated to lifecycle management and is not
29///   suitable for tracking the number of [`StylesheetContents`]s owners.
30/// * The reference count within [`Atom`] is not publicly acessible.
31#[derive(Clone, Eq, Hash, MallocSizeOf, PartialEq)]
32pub(crate) struct StylesheetContentsCacheKey {
33    #[conditional_malloc_size_of]
34    stylesheet_text: Rc<Atom>,
35    base_url: Atom,
36    #[ignore_malloc_size_of = "defined in style crate"]
37    quirks_mode: QuirksMode,
38}
39
40impl StylesheetContentsCacheKey {
41    fn new(stylesheet_text: &str, base_url: &str, quirks_mode: QuirksMode) -> Self {
42        // The stylesheet text may be quite lengthy, exceeding hundreds of kilobytes.
43        // Instead of directly inserting such a huge string into AtomicString table,
44        // take its hash value and use that. (This is not a cryptographic hash, so a
45        // page could cause collisions if it wanted to.)
46        let contents_atom = if stylesheet_text.len() > MAX_LENGTH_OF_TEXT_INSERTED_INTO_TABLE {
47            let mut hasher = DefaultHasher::new();
48            stylesheet_text.hash(&mut hasher);
49            Atom::from(hasher.finish().to_string().as_str())
50        } else {
51            Atom::from(stylesheet_text)
52        };
53
54        Self {
55            stylesheet_text: Rc::new(contents_atom),
56            base_url: Atom::from(base_url),
57            quirks_mode,
58        }
59    }
60
61    pub(crate) fn is_uniquely_owned(&self) -> bool {
62        // The cache itself already holds one reference.
63        Rc::strong_count(&self.stylesheet_text) <= UNIQUE_OWNED
64    }
65}

67thread_local! {
68    static STYLESHEETCONTENTS_CACHE: RefCell<HashMap<StylesheetContentsCacheKey, ServoArc<StylesheetContents>>> =
69       RefCell::default();
70}
71
72pub(crate) struct StylesheetContentsCache;
73
74impl StylesheetContentsCache {
75    fn contents_can_be_cached(contents: &StylesheetContents, shared_lock: &SharedRwLock) -> bool {
76        let guard = shared_lock.read();
77        let rules = contents.rules(&guard);
78        // The copy-on-write can not be performed when the modification happens on the
79        // imported stylesheet, because it containing cssom has no owner dom node.
80        !(rules.is_empty() || rules.iter().any(|rule| matches!(rule, CssRule::Import(_))))
81    }
82
83    pub(crate) fn get_or_insert_with(
84        stylesheet_text: &str,
85        shared_lock: &SharedRwLock,
86        url_data: UrlExtraData,
87        quirks_mode: QuirksMode,
88        element: &HTMLElement,
89    ) -> (
90        Option<StylesheetContentsCacheKey>,
91        ServoArc<StylesheetContents>,
92    ) {
93        let cache_key =
94            StylesheetContentsCacheKey::new(stylesheet_text, url_data.as_str(), quirks_mode);
95        STYLESHEETCONTENTS_CACHE.with_borrow_mut(|stylesheetcontents_cache| {
96            let entry = stylesheetcontents_cache.entry(cache_key);
97            match entry {
98                Entry::Occupied(occupied_entry) => {
99                    // Use a copy of the cache key from `Entry` instead of the newly created one above
100                    // to correctly update and track to owner count of `StylesheetContents`.
101                    (
102                        Some(occupied_entry.key().clone()),
103                        occupied_entry.get().clone(),
104                    )
105                },
106                Entry::Vacant(vacant_entry) => {
107                    let contents = {
108                        let _span = profile_traits::trace_span!("ParseStylesheet").entered();
109                        StylesheetContents::from_str(
110                            stylesheet_text,
111                            url_data,
112                            Origin::Author,
113                            shared_lock,
114                            Some(&ElementStylesheetLoader::new(element)),
115                            Some(element.owner_window().css_error_reporter()),
116                            quirks_mode,
117                            AllowImportRules::Yes,
118                            /* sanitized_output = */ None,
119                        )
120                    };
121                    if Self::contents_can_be_cached(&contents, shared_lock) {
122                        let occupied_entry = vacant_entry.insert_entry(contents.clone());
123                        // Use a copy of the cache key from `Entry` instead of the newly created one above
124                        // to correctly update and track to owner count of `StylesheetContents`.
125                        (Some(occupied_entry.key().clone()), contents)
126                    } else {
127                        (None, contents)
128                    }
129                },
130            }
131        })
132    }
133
134    pub(crate) fn remove(cache_key: StylesheetContentsCacheKey) {
135        STYLESHEETCONTENTS_CACHE.with_borrow_mut(|stylesheetcontents_cache| {
136            stylesheetcontents_cache.remove(&cache_key)
137        });
138    }
