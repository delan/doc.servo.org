41#[dom_struct]
42/// Global scope for interacting with the devtools Debugger API.
43///
44/// <https://firefox-source-docs.mozilla.org/js/Debugger/>
45pub(crate) struct DebuggerGlobalScope {
46    global_scope: GlobalScope,
47    #[no_trace]
48    devtools_to_script_sender: GenericSender<DevtoolScriptControlMsg>,
49    #[no_trace]
50    get_possible_breakpoints_result_sender:
51        RefCell<Option<GenericSender<Vec<devtools_traits::RecommendedBreakpointLocation>>>>,
52}
53
54impl DebuggerGlobalScope {
55    /// Create a new heap-allocated `DebuggerGlobalScope`.
56    ///
57    /// `debugger_pipeline_id` is the pipeline id to use when creating the debugger’s [`GlobalScope`]:
58    /// - in normal script threads, it should be set to `PipelineId::new()`, because those threads can generate
59    ///   pipeline ids, and they may contain debuggees from more than one pipeline
60    /// - in web worker threads, it should be set to the pipeline id of the page that created the thread, because
61    ///   those threads can’t generate pipeline ids, and they only contain one debuggee from one pipeline
62    #[expect(unsafe_code, clippy::too_many_arguments)]
63    pub(crate) fn new(
64        debugger_pipeline_id: PipelineId,
65        script_to_devtools_sender: Option<GenericCallback<ScriptToDevtoolsControlMsg>>,
66        devtools_to_script_sender: GenericSender<DevtoolScriptControlMsg>,
67        mem_profiler_chan: mem::ProfilerChan,
68        time_profiler_chan: time::ProfilerChan,
69        script_to_constellation_chan: ScriptToConstellationChan,
70        script_to_embedder_chan: ScriptToEmbedderChan,
71        resource_threads: ResourceThreads,
72        storage_threads: StorageThreads,
73        #[cfg(feature = "webgpu")] gpu_id_hub: std::sync::Arc<IdentityHub>,
74        cx: &mut js::context::JSContext,
75    ) -> DomRoot<Self> {
76        let global = Box::new(Self {
77            global_scope: GlobalScope::new_inherited(
78                debugger_pipeline_id,
79                script_to_devtools_sender,
80                mem_profiler_chan,
81                time_profiler_chan,
82                script_to_constellation_chan,
83                script_to_embedder_chan,
84                resource_threads,
85                storage_threads,
86                MutableOrigin::new(ImmutableOrigin::new_opaque()),
87                ServoUrl::parse_with_base(None, "about:internal/debugger")
88                    .expect("Guaranteed by argument"),
89                None,
90                #[cfg(feature = "webgpu")]
91                gpu_id_hub,
92                None,
93                false,
94                None, // font_context
95            ),
96            devtools_to_script_sender,
97            get_possible_breakpoints_result_sender: RefCell::new(None),
98        });
99        let global = DebuggerGlobalScopeBinding::Wrap::<crate::DomTypeHolder>(cx.into(), global);
100
101        let mut realm = enter_auto_realm(cx, &*global);
102        let mut realm = realm.current_realm();
103        define_all_exposed_interfaces(&mut realm, global.upcast());
104        assert!(unsafe {
105            // Invariants: `obj` must be a handle to a JS global object.
106            JS_DefineDebuggerObject(&mut realm, global.global_scope.reflector().get_jsobject())
107        });
108
109        global
110    }
111
112    /// Get the JS context.
113    pub(crate) fn get_cx() -> JSContext {
114        GlobalScope::get_cx()
115    }
116
117    pub(crate) fn as_global_scope(&self) -> &GlobalScope {
118        self.upcast::<GlobalScope>()
119    }
120
121    fn evaluate_js(
122        &self,
123        script: Cow<'_, str>,
124        can_gc: CanGc,
125    ) -> Result<(), JavaScriptEvaluationError> {
126        rooted!(in (*Self::get_cx()) let mut rval = UndefinedValue());
127        self.global_scope
128            .evaluate_js_on_global(script, "", None, rval.handle_mut(), can_gc)
129    }
130
131    pub(crate) fn execute(&self, can_gc: CanGc) {
132        if self
133            .evaluate_js(resources::read_string(Resource::DebuggerJS).into(), can_gc)
134            .is_err()
135        {
136            let ar = enter_realm(self);
137            report_pending_exception(Self::get_cx(), true, InRealm::Entered(&ar), can_gc);
138        }
139    }
140
141    pub(crate) fn fire_add_debuggee(
142        &self,
143        can_gc: CanGc,
144        debuggee_global: &GlobalScope,
145        debuggee_pipeline_id: PipelineId,
146        debuggee_worker_id: Option<WorkerId>,
147    ) {
148        let _realm = enter_realm(self);
149        let debuggee_pipeline_id =
150            crate::dom::pipelineid::PipelineId::new(self.upcast(), debuggee_pipeline_id, can_gc);
151        let event = DomRoot::upcast::<Event>(DebuggerAddDebuggeeEvent::new(
152            self.upcast(),
153            debuggee_global,
154            &debuggee_pipeline_id,
155            debuggee_worker_id.map(|id| id.to_string().into()),
156            can_gc,
157        ));
158        assert!(
159            DomRoot::upcast::<Event>(event).fire(self.upcast(), can_gc),
160            "Guaranteed by DebuggerAddDebuggeeEvent::new"
161        );
162    }
163
164    pub(crate) fn fire_get_possible_breakpoints(
165        &self,
166        can_gc: CanGc,
167        spidermonkey_id: u32,
168        result_sender: GenericSender<Vec<devtools_traits::RecommendedBreakpointLocation>>,
169    ) {
170        assert!(
171            self.get_possible_breakpoints_result_sender
172                .replace(Some(result_sender))
173                .is_none()
174        );
175        let _realm = enter_realm(self);
176        let event = DomRoot::upcast::<Event>(DebuggerGetPossibleBreakpointsEvent::new(
177            self.upcast(),
178            spidermonkey_id,
179            can_gc,
180        ));
181        assert!(
182            DomRoot::upcast::<Event>(event).fire(self.upcast(), can_gc),
183            "Guaranteed by DebuggerGetPossibleBreakpointsEvent::new"
184        );
185    }
186}

188impl DebuggerGlobalScopeMethods<crate::DomTypeHolder> for DebuggerGlobalScope {
189    // check-tidy: no specs after this line
190    fn NotifyNewSource(&self, args: &NotifyNewSource) {
191        let Some(devtools_chan) = self.as_global_scope().devtools_chan() else {
192            return;
193        };
194        let pipeline_id = PipelineId {
195            namespace_id: PipelineNamespaceId(args.pipelineId.namespaceId),
196            index: Index::new(args.pipelineId.index).expect("`pipelineId.index` must not be zero"),
197        };
198
199        if let Some(introduction_type) = args.introductionType.as_ref() {
200            // Check the `introductionType` and `url`, decide whether or not to create a source actor, and if so,
201            // tell the devtools server to create a source actor. Based on the Firefox impl in:
202            // - getDebuggerSourceURL() <https://searchfox.org/mozilla-central/rev/85667ab51e4b2a3352f7077a9ee43513049ed2d6/devtools/server/actors/utils/source-url.js#7-42>
203            // - getSourceURL() <https://searchfox.org/mozilla-central/rev/85667ab51e4b2a3352f7077a9ee43513049ed2d6/devtools/server/actors/source.js#67-109>
204            // - resolveSourceURL() <https://searchfox.org/mozilla-central/rev/85667ab51e4b2a3352f7077a9ee43513049ed2d6/devtools/server/actors/source.js#48-66>
205            // - SourceActor#_isInlineSource <https://searchfox.org/mozilla-central/rev/85667ab51e4b2a3352f7077a9ee43513049ed2d6/devtools/server/actors/source.js#130-143>
206            // - SourceActor#url <https://searchfox.org/mozilla-central/rev/85667ab51e4b2a3352f7077a9ee43513049ed2d6/devtools/server/actors/source.js#157-162>
207
208            // Firefox impl: getDebuggerSourceURL(), getSourceURL()
209            // TODO: handle `about:srcdoc` case (see Firefox getDebuggerSourceURL())
210            // TODO: remove trailing details that may have been appended by SpiderMonkey
211            // (currently impossible to do robustly due to <https://bugzilla.mozilla.org/show_bug.cgi?id=1982001>)
212            let url_original = args.url.str();
213            // FIXME: use page/worker url as base here
214            let url_original = ServoUrl::parse(&url_original).ok();
215
216            // If the source has a `urlOverride` (aka `displayURL` aka `//# sourceURL`), it should be a valid url,
217            // possibly relative to the page/worker url, and we should treat the source as coming from that url for
218            // devtools purposes, including the file tree in the Sources tab.
219            // Firefox impl: getSourceURL()
220            let url_override = args
221                .urlOverride
222                .as_ref()
223                .map(|url| url.str())
224                // FIXME: use page/worker url as base here, not `url_original`
225                .and_then(|url| ServoUrl::parse_with_base(url_original.as_ref(), &url).ok());
226
227            // If the `introductionType` is “eval or eval-like”, the `url` won’t be meaningful, so ignore these
228            // sources unless we have a `urlOverride` (aka `displayURL` aka `//# sourceURL`).
229            // Firefox impl: getDebuggerSourceURL(), getSourceURL()
230            if [
231                IntroductionType::INJECTED_SCRIPT_STR,
232                IntroductionType::EVAL_STR,
233                IntroductionType::DEBUGGER_EVAL_STR,
234                IntroductionType::FUNCTION_STR,
235                IntroductionType::JAVASCRIPT_URL_STR,
236                IntroductionType::EVENT_HANDLER_STR,
237                IntroductionType::DOM_TIMER_STR,
238            ]
239            .contains(&&*introduction_type.str()) &&
240                url_override.is_none()
241            {
242                debug!(
243                    "Not creating debuggee: `introductionType` is `{introduction_type}` but no valid url"
244                );
245                return;
246            }
247
248            // Sources with an `introductionType` of `inlineScript` are generally inline, meaning their contents
249            // are a substring of the page markup (hence not known to SpiderMonkey, requiring plumbing in Servo).
250            // But sources with a `urlOverride` are not inline, since they get their own place in the Sources tree.
251            // nor are sources created for `<iframe srcdoc>`, since they are not necessarily a substring of the
252            // page markup as originally sent by the server.
253            // Firefox impl: SourceActor#_isInlineSource
254            // TODO: handle `about:srcdoc` case (see Firefox SourceActor#_isInlineSource)
255            let inline = introduction_type.str() == "inlineScript" && url_override.is_none();
256            let Some(url) = url_override.or(url_original) else {
257                debug!("Not creating debuggee: no valid url");
258                return;
259            };
260
261            let worker_id = args.workerId.as_ref().map(|id| id.parse().unwrap());
262
263            let source_info = SourceInfo {
264                url,
265                introduction_type: introduction_type.str().to_owned(),
266                inline,
267                worker_id,
268                content: (!inline).then(|| args.text.to_string()),
269                content_type: None, // TODO
270                spidermonkey_id: args.spidermonkeyId,
271            };
272            if let Err(error) = devtools_chan.send(ScriptToDevtoolsControlMsg::CreateSourceActor(
273                self.devtools_to_script_sender.clone(),
274                pipeline_id,
275                source_info,
276            )) {
277                warn!("Failed to send to devtools server: {error:?}");
278            }
279        } else {
280            debug!("Not creating debuggee for script with no `introductionType`");
281        }
282    }

284    fn GetPossibleBreakpointsResult(
285        &self,
286        event: &DebuggerGetPossibleBreakpointsEvent,
287        result: Vec<RecommendedBreakpointLocation>,
288    ) {
289        info!("GetPossibleBreakpointsResult: {event:?} {result:?}");
290        let sender = self
291            .get_possible_breakpoints_result_sender
292            .take()
293            .expect("Guaranteed by Self::fire_get_possible_breakpoints()");
294        let _ = sender.send(
295            result
296                .into_iter()
297                .map(|entry| devtools_traits::RecommendedBreakpointLocation {
298                    offset: entry.offset,
299                    line_number: entry.lineNumber,
300                    column_number: entry.columnNumber,
301                    is_step_start: entry.isStepStart,
302                })
303                .collect(),
304        );
305    }
