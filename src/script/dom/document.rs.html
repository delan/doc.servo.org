274/// Reasons why a [`Document`] might need a rendering update that is otherwise
275/// untracked via other [`Document`] properties.
276#[derive(Clone, Copy, Debug, Default, JSTraceable, MallocSizeOf)]
277pub(crate) struct RenderingUpdateReason(u8);
278
279bitflags! {
280    impl RenderingUpdateReason: u8 {
281        /// When a `ResizeObserver` starts observing a target, this becomes true, which in turn is a
282        /// signal to the [`ScriptThread`] that a rendering update should happen.
283        const ResizeObserverStartedObservingTarget = 1 << 0;
284        /// When an `IntersectionObserver` starts observing a target, this becomes true, which in turn is a
285        /// signal to the [`ScriptThread`] that a rendering update should happen.
286        const IntersectionObserverStartedObservingTarget = 1 << 1;
287        /// All web fonts have loaded and `fonts.ready` promise has been fulfilled. We want to trigger
288        /// one more rendering update possibility after this happens, so that any potential screenshot
289        /// reflects the up-to-date contents.
290        const FontReadyPromiseFulfilled = 1 << 2;
291    }

657    pub(crate) fn note_node_with_dirty_descendants(&self, node: &Node) {
658        debug_assert!(*node.owner_doc() == *self);
659        if !node.is_connected() {
660            return;
661        }
662
663        let parent = match node.parent_in_flat_tree() {
664            Some(parent) => parent,
665            None => {
666                // There is no parent so this is the Document node, so we
667                // behave as if we were called with the document element.
668                let document_element = match self.GetDocumentElement() {
669                    Some(element) => element,
670                    None => return,
671                };
672                if let Some(dirty_root) = self.dirty_root.get() {
673                    // There was an existing dirty root so we mark its
674                    // ancestors as dirty until the document element.
675                    for ancestor in dirty_root
676                        .upcast::<Node>()
677                        .inclusive_ancestors_in_flat_tree()
678                    {
679                        if ancestor.is::<Element>() {
680                            ancestor.set_flag(NodeFlags::HAS_DIRTY_DESCENDANTS, true);
681                        }
682                    }
683                }
684                self.dirty_root.set(Some(&document_element));
685                return;
686            },
687        };
688
689        if parent.is::<Element>() {
690            if !parent.is_styled() {
691                return;
692            }
693
694            if parent.is_display_none() {
695                return;
696            }
697        }
698
699        let element_parent: DomRoot<Element>;
700        let element = match node.downcast::<Element>() {
701            Some(element) => element,
702            None => {
703                // Current node is not an element, it's probably a text node,
704                // we try to get its element parent.
705                match DomRoot::downcast::<Element>(parent) {
706                    Some(parent) => {
707                        element_parent = parent;
708                        &element_parent
709                    },
710                    None => {
711                        // Parent is not an element so it must be a document,
712                        // and this is not an element either, so there is
713                        // nothing to do.
714                        return;
715                    },
716                }
717            },
718        };
719
720        let dirty_root = match self.dirty_root.get() {
721            Some(root) if root.is_connected() => root,
722            _ => {
723                element
724                    .upcast::<Node>()
725                    .set_flag(NodeFlags::HAS_DIRTY_DESCENDANTS, true);
726                self.dirty_root.set(Some(element));
727                return;
728            },
729        };
730
731        for ancestor in element.upcast::<Node>().inclusive_ancestors_in_flat_tree() {
732            if ancestor.get_flag(NodeFlags::HAS_DIRTY_DESCENDANTS) {
733                return;
734            }
735
736            if ancestor.is::<Element>() {
737                ancestor.set_flag(NodeFlags::HAS_DIRTY_DESCENDANTS, true);
738            }
739        }
740
741        let new_dirty_root = element
742            .upcast::<Node>()
743            .common_ancestor_in_flat_tree(dirty_root.upcast())
744            .expect("Couldn't find common ancestor");
745
746        let mut has_dirty_descendants = true;
747        for ancestor in dirty_root
748            .upcast::<Node>()
749            .inclusive_ancestors_in_flat_tree()
750        {
751            ancestor.set_flag(NodeFlags::HAS_DIRTY_DESCENDANTS, has_dirty_descendants);
752            has_dirty_descendants &= *ancestor != *new_dirty_root;
753        }
754
755        self.dirty_root
756            .set(Some(new_dirty_root.downcast::<Element>().unwrap()));
757    }

823    pub(crate) fn set_activity(&self, activity: DocumentActivity, can_gc: CanGc) {
824        // This function should only be called on documents with a browsing context
825        assert!(self.has_browsing_context);
826        if activity == self.activity.get() {
827            return;
828        }
829
830        // Set the document's activity level, reflow if necessary, and suspend or resume timers.
831        self.activity.set(activity);
832        let media = ServoMedia::get();
833        let pipeline_id = self.window().pipeline_id();
834        let client_context_id =
835            ClientContextId::build(pipeline_id.namespace_id.0, pipeline_id.index.0.get());
836
837        if activity != DocumentActivity::FullyActive {
838            self.window().suspend(can_gc);
839            media.suspend(&client_context_id);
840            return;
841        }
842
843        self.title_changed();
844        self.notify_embedder_favicon();
845        self.dirty_all_nodes();
846        self.window().resume(can_gc);
847        media.resume(&client_context_id);
848
849        if self.ready_state.get() != DocumentReadyState::Complete {
850            return;
851        }
852
853        // This step used to be Step 4.6 in html.spec.whatwg.org/multipage/#history-traversal
854        // But it's now Step 4 in https://html.spec.whatwg.org/multipage/#reactivate-a-document
855        // TODO: See #32687 for more information.
856        let document = Trusted::new(self);
857        self.owner_global()
858            .task_manager()
859            .dom_manipulation_task_source()
860            .queue(task!(fire_pageshow_event: move || {
861                let document = document.root();
862                let window = document.window();
863                // Step 4.6.1
864                if document.page_showing.get() {
865                    return;
866                }
867                // Step 4.6.2 Set document's page showing flag to true.
868                document.page_showing.set(true);
869                // Step 4.6.3 Update the visibility state of document to "visible".
870                document.update_visibility_state(DocumentVisibilityState::Visible, CanGc::note());
871                // Step 4.6.4 Fire a page transition event named pageshow at document's relevant
872                // global object with true.
873                let event = PageTransitionEvent::new(
874                    window,
875                    atom!("pageshow"),
876                    false, // bubbles
877                    false, // cancelable
878                    true, // persisted
879                    CanGc::note(),
880                );
881                let event = event.upcast::<Event>();
882                event.set_trusted(true);
883                window.dispatch_event_with_target_override(event, CanGc::note());
884            }))
885    }

904    /// <https://html.spec.whatwg.org/multipage/#fallback-base-url>
905    pub(crate) fn fallback_base_url(&self) -> ServoUrl {
906        // Step 1: If document is an iframe srcdoc document:
907        let document_url = self.url();
908        if document_url.as_str() == "about:srcdoc" {
909            let base_url = self
910                .browsing_context()
911                .and_then(|browsing_context| browsing_context.creator_base_url());
912
913            // Step 1.1: Assert: document's about base URL is non-null.
914            if base_url.is_none() {
915                error!("about:srcdoc page should always have a creator base URL");
916            }
917
918            // Step 1.2: Return document's about base URL.
919            return base_url.unwrap_or(document_url);
920        }
921
922        // Step 2: If document's URL matches about:blank and document's about base URL is
923        // non-null, then return document's about base URL.
924        if document_url.matches_about_blank() {
925            return self
926                .browsing_context()
927                .and_then(|browsing_context| browsing_context.creator_base_url())
928                .unwrap_or(document_url);
929        }
930
931        // Step 3: Return document's URL.
932        document_url
933    }

1126    /// Scroll to the target element, and when we do not find a target
1127    /// and the fragment is empty or "top", scroll to the top.
1128    /// <https://html.spec.whatwg.org/multipage/#scroll-to-the-fragment-identifier>
1129    pub(crate) fn check_and_scroll_fragment(&self, fragment: &str) {
1130        let target = self.find_fragment_node(fragment);
1131
1132        // Step 1
1133        self.set_target_element(target.as_deref());
1134
1135        let point = target
1136            .as_ref()
1137            .map(|element| {
1138                // TODO: This strategy is completely wrong if the element we are scrolling to in
1139                // inside other scrollable containers. Ideally this should use an implementation of
1140                // `scrollIntoView` when that is available:
1141                // See https://github.com/servo/servo/issues/24059.
1142                let rect = element.upcast::<Node>().border_box().unwrap_or_default();
1143
1144                // In order to align with element edges, we snap to unscaled pixel boundaries, since
1145                // the paint thread currently does the same for drawing elements. This is important
1146                // for pages that require pixel perfect scroll positioning for proper display
1147                // (like Acid2).
1148                let device_pixel_ratio = self.window.device_pixel_ratio().get();
1149                (
1150                    rect.origin.x.to_nearest_pixel(device_pixel_ratio),
1151                    rect.origin.y.to_nearest_pixel(device_pixel_ratio),
1152                )
1153            })
1154            .or_else(|| {
1155                if fragment.is_empty() || fragment.eq_ignore_ascii_case("top") {
1156                    // FIXME(stshine): this should be the origin of the stacking context space,
1157                    // which may differ under the influence of writing mode.
1158                    Some((0.0, 0.0))
1159                } else {
1160                    None
1161                }
1162            });
1163
1164        if let Some((x, y)) = point {
1165            self.window.scroll(x, y, ScrollBehavior::Instant)
1166        }
1167    }

1347    /// Update the local focus state accordingly after being notified that the
1348    /// document's container is removed from the top-level browsing context's
1349    /// focus chain (not considering system focus).
1350    pub(crate) fn handle_container_unfocus(&self, can_gc: CanGc) {

1352            warn!("Top-level document cannot be unfocused");

1354        }

1376    /// Reassign the focus context to the element that last requested focus during this
1377    /// transaction, or the document if no elements requested it.
1378    pub(crate) fn commit_focus_transaction(&self, focus_initiator: FocusInitiator, can_gc: CanGc) {
1379        let (mut new_focused, new_focus_state, prevent_scroll) = {
1380            let focus_transaction = self.focus_transaction.borrow();
1381            let focus_transaction = focus_transaction
1382                .as_ref()
1383                .expect("no focus transaction in progress");
1384            (
1385                focus_transaction
1386                    .element
1387                    .as_ref()
1388                    .map(|e| DomRoot::from_ref(&**e)),
1389                focus_transaction.has_focus,
1390                focus_transaction.focus_options.preventScroll,
1391            )
1392        };
1393        *self.focus_transaction.borrow_mut() = None;
1394
1395        if !new_focus_state {
1396            // In many browsers, a document forgets its focused area when the
1397            // document is removed from the top-level BC's focus chain
1398            if new_focused.take().is_some() {
1399                trace!(
1400                    "Forgetting the document's focused area because the \
1401                    document's container was removed from the top-level BC's \
1402                    focus chain"
1403                );
1404            }
1405        }
1406
1407        let old_focused = self.focused.get();
1408        let old_focus_state = self.has_focus.get();
1409
1410        debug!(
1411            "Committing focus transaction: {:?}  {:?}",
1412            (&old_focused, old_focus_state),
1413            (&new_focused, new_focus_state),
1414        );
1415
1416        // `*_focused_filtered` indicates the local element (if any) included in
1417        // the top-level BC's focus chain.
1418        let old_focused_filtered = old_focused.as_ref().filter(|_| old_focus_state);
1419        let new_focused_filtered = new_focused.as_ref().filter(|_| new_focus_state);
1420
1421        let trace_focus_chain = |name, element, doc| {
1422            trace!(
1423                "{} local focus chain: {}",
1424                name,
1425                match (element, doc) {
1426                    (Some(e), _) => format!("[{:?}, document]", e),
1427                    (None, true) => "[document]".to_owned(),
1428                    (None, false) => "[]".to_owned(),
1429                }
1430            );
1431        };
1432
1433        trace_focus_chain("Old", old_focused_filtered, old_focus_state);
1434        trace_focus_chain("New", new_focused_filtered, new_focus_state);
1435
1436        if old_focused_filtered != new_focused_filtered {
1437            if let Some(elem) = &old_focused_filtered {
1438                let node = elem.upcast::<Node>();
1439                elem.set_focus_state(false);
1440                // FIXME: pass appropriate relatedTarget
1441                if node.is_connected() {
1442                    self.fire_focus_event(FocusEventType::Blur, node.upcast(), None, can_gc);
1443                }
1444            }
1445        }
1446
1447        if old_focus_state != new_focus_state && !new_focus_state {
1448            self.fire_focus_event(FocusEventType::Blur, self.global().upcast(), None, can_gc);
1449        }
1450
1451        self.focused.set(new_focused.as_deref());
1452        self.has_focus.set(new_focus_state);
1453
1454        if old_focus_state != new_focus_state && new_focus_state {
1455            self.fire_focus_event(FocusEventType::Focus, self.global().upcast(), None, can_gc);
1456        }
1457
1458        if old_focused_filtered != new_focused_filtered {
1459            if let Some(elem) = &new_focused_filtered {
1460                elem.set_focus_state(true);
1461                let node = elem.upcast::<Node>();
1462                // FIXME: pass appropriate relatedTarget
1463                self.fire_focus_event(FocusEventType::Focus, node.upcast(), None, can_gc);
1464
1465                // Scroll operation to happen after element gets focus. This is needed to ensure that the
1466                // focused element is visible. Only scroll if preventScroll was not specified.
1467                if !prevent_scroll {
1468                    // We are following the firefox implementation where we are only scrolling to the element
1469                    // if the element itself it not visible.
1470                    let scroll_axis = ScrollAxisState {
1471                        position: ScrollLogicalPosition::Center,
1472                        requirement: ScrollRequirement::IfNotVisible,
1473                    };
1474
1475                    // TODO(stevennovaryo): we doesn't differentiate focus operation from script and from user
1476                    //                      for a scroll yet.
1477                    // TODO(#40474): Implement specific ScrollIntoView for a selection of text control element.
1478                    elem.scroll_into_view_with_options(
1479                        ScrollBehavior::Smooth,
1480                        scroll_axis,
1481                        scroll_axis,
1482                        None,
1483                        None,
1484                    );
1485                }
1486            }
1487        }
1488
1489        if focus_initiator != FocusInitiator::Local {
1490            return;
1491        }
1492
1493        // We are the initiator of the focus operation, so we must broadcast
1494        // the change we intend to make.
1495        match (old_focus_state, new_focus_state) {
1496            (_, true) => {
1497                // Advertise the change in the focus chain.
1498                // <https://html.spec.whatwg.org/multipage/#focus-chain>
1499                // <https://html.spec.whatwg.org/multipage/#focusing-steps>
1500                //
1501                // If the top-level BC doesn't have system focus, this won't
1502                // have an immediate effect, but it will when we gain system
1503                // focus again. Therefore we still have to send `ScriptMsg::
1504                // Focus`.
1505                //
1506                // When a container with a non-null nested browsing context is
1507                // focused, its active document becomes the focused area of the
1508                // top-level browsing context instead. Therefore we need to let
1509                // the constellation know if such a container is focused.
1510                //
1511                // > The focusing steps for an object `new focus target` [...]
1512                // >
1513                // >  3. If `new focus target` is a browsing context container
1514                // >     with non-null nested browsing context, then set
1515                // >     `new focus target` to the nested browsing context's
1516                // >     active document.
1517                let child_browsing_context_id = new_focused
1518                    .as_ref()
1519                    .and_then(|elem| elem.downcast::<HTMLIFrameElement>())
1520                    .and_then(|iframe| iframe.browsing_context_id());
1521
1522                let sequence = self.increment_fetch_focus_sequence();
1523
1524                debug!(
1525                    "Advertising the focus request to the constellation \
1526                        with sequence number {} and child BC ID {}",
1527                    sequence,
1528                    child_browsing_context_id
1529                        .as_ref()
1530                        .map(|id| id as &dyn std::fmt::Display)
1531                        .unwrap_or(&"(none)"),
1532                );
1533
1534                self.window()
1535                    .send_to_constellation(ScriptToConstellationMessage::Focus(
1536                        child_browsing_context_id,
1537                        sequence,
1538                    ));
1539            },
1540            (false, false) => {
1541                // Our `Document` doesn't have focus, and we intend to keep it
1542                // this way.
1543            },
1544            (true, false) => {
1545                unreachable!(
1546                    "Can't lose the document's focus without specifying \
1547                    another one to focus"
1548                );
1549            },
1550        }
1551    }

1627    /// <https://drafts.csswg.org/cssom-view/#document-run-the-scroll-steps>
1628    pub(crate) fn run_the_scroll_steps(&self, can_gc: CanGc) {
1629        // Step 1.
1630        // > Run the steps to dispatch pending scrollsnapchanging events for doc.
1631        // TODO(#7673): Implement scroll snapping
1632
1633        // Step 2
1634        // > For each item target in docs pending scroll event targets, in the order they
1635        // > were added to the list, run these substeps:
1636        // Step 3.
1637        // > Empty docs pending scroll event targets.
1638        // Since the scroll event callback could trigger another scroll event, we are taking all of the
1639        // current scroll event to avoid borrow checking error.
1640        rooted_vec!(let notify_list <- self.pending_scroll_event_targets.take().into_iter());
1641        for target in notify_list.iter() {
1642            if target.downcast::<Document>().is_some() {

1645                target.fire_bubbling_event(Atom::from("scroll"), can_gc);
1646            } else if target.downcast::<Element>().is_some() {
1647                // Step 2.2
1648                // > Otherwise, fire an event named scroll at target.
1649                target.fire_event(Atom::from("scroll"), can_gc);
1650            }
1651        }
1652
1653        // Step 4.
1654        // > Run the steps to dispatch pending scrollsnapchange events for doc.
1655        // TODO(#7673): Implement scroll snapping
1656    }

1658    /// Whenever a viewport gets scrolled (whether in response to user interaction or by an
1659    /// API), the user agent must run these steps:
1660    /// <https://drafts.csswg.org/cssom-view/#scrolling-events>
1661    pub(crate) fn handle_viewport_scroll_event(&self) {
1662        // Step 2.
1663        // > If doc is a snap container, run the steps to update scrollsnapchanging targets
1664        // > for doc with docs eventual snap target in the block axis as newBlockTarget and
1665        // > docs eventual snap target in the inline axis as newInlineTarget.
1666        // TODO(#7673): Implement scroll snapping
1667
1668        // Step 3.
1669        // > If doc is already in docs pending scroll event targets, abort these steps.
1670        let target = self.upcast::<EventTarget>();
1671        if self
1672            .pending_scroll_event_targets
1673            .borrow()
1674            .iter()
1675            .any(|other_target| *other_target == target)
1676        {
1677            return;
1678        }

1680        // Step 4.
1681        // > Append doc to docs pending scroll event targets.
1682        self.pending_scroll_event_targets
1683            .borrow_mut()
1684            .push(Dom::from_ref(target));
1685    }

2667    /// <https://html.spec.whatwg.org/multipage/#abort-a-document>
2668    pub(crate) fn abort(&self, can_gc: CanGc) {
2669        // We need to inhibit the loader before anything else.
2670        self.loader.borrow_mut().inhibit_events();
2671
2672        // Step 1.
2673        for iframe in self.iframes().iter() {
2674            if let Some(document) = iframe.GetContentDocument() {
2675                document.abort(can_gc);
2676            }
2677        }
2678
2679        // Step 2. Cancel any instances of the fetch algorithm in the context of document,
2680        // discarding any tasks queued for them, and discarding any further data received
2681        // from the network for them. If this resulted in any instances of the fetch algorithm
2682        // being canceled or any queued tasks or any network data getting discarded,
2683        // then make document unsalvageable given document and "fetch".
2684        self.script_blocking_stylesheets_count.set(0);
2685        *self.pending_parsing_blocking_script.borrow_mut() = None;
2686        *self.asap_scripts_set.borrow_mut() = vec![];
2687        self.asap_in_order_scripts_list.clear();
2688        self.deferred_scripts.clear();
2689        let loads_cancelled = self.terminate_fetch_group();
2690        let event_sources_canceled = self.window.as_global_scope().close_event_sources();
2691        if loads_cancelled || event_sources_canceled {
2692            // If any loads were canceled.
2693            self.salvageable.set(false);
2694        };
2695
2696        // Also Step 2.
2697        // Note: the spec says to discard any tasks queued for fetch.
2698        // This cancels all tasks on the networking task source, which might be too broad.
2699        // See https://github.com/whatwg/html/issues/3837
2700        self.owner_global()
2701            .task_manager()
2702            .cancel_pending_tasks_for_source(TaskSourceName::Networking);
2703
2704        // Step 3. If document's during-loading navigation ID for WebDriver BiDi is non-null, then:
2705        // TODO
2706
2707        // Step 4. If document has an active parser, then:
2708        if let Some(parser) = self.get_current_parser() {
2709            // Step 4.1. Set document's active parser was aborted to true.
2710            self.active_parser_was_aborted.set(true);
2711            // Step 4.2. Abort that parser.
2712            parser.abort(can_gc);
2713            // Step 4.3. Make document unsalvageable given document and "parser-aborted".
2714            self.salvageable.set(false);
2715        }
2716    }

3422            Some(ref parser) if parser.can_write() => DomRoot::from_ref(&**parser),
3423            // Step 9: If the insertion point is undefined, then:
3424            _ => {
3425                // Step 9.1: If document's unload counter is greater than 0 or
3426                // document's ignore-destructive-writes counter is greater than 0, then return.
3427                if self.is_prompting_or_unloading() ||
3428                    self.ignore_destructive_writes_counter.get() > 0
3429                {
3430                    return Ok(());
3431                }
3432                // Step 9.2: Run the document open steps with document.
3433                self.Open(None, None, can_gc)?;
3434                self.get_current_parser().unwrap()
3435            },
3436        };
3437
3438        // Steps 10-11.
3439        parser.write(string.into(), can_gc);
3440
3441        Ok(())
3442    }

3511// https://html.spec.whatwg.org/multipage/#is-a-registrable-domain-suffix-of-or-is-equal-to
3512// The spec says to return a bool, we actually return an Option<Host> containing
3513// the parsed host in the successful case, to avoid having to re-parse the host.
3514fn get_registrable_domain_suffix_of_or_is_equal_to(
3515    host_suffix_string: &DOMString,
3516    original_host: Host,
3517) -> Option<Host> {
3518    // Step 1
3519    if host_suffix_string.is_empty() {
3520        return None;
3521    }
3522
3523    // Step 2-3.
3524    let host = match Host::parse(&host_suffix_string.str()) {
3525        Ok(host) => host,
3526        Err(_) => return None,
3527    };
3528
3529    // Step 4.
3530    if host != original_host {
3531        // Step 4.1
3532        let host = match host {
3533            Host::Domain(ref host) => host,
3534            _ => return None,
3535        };
3536        let original_host = match original_host {
3537            Host::Domain(ref original_host) => original_host,
3538            _ => return None,
3539        };
3540
3541        // Step 4.2
3542        let index = original_host.len().checked_sub(host.len())?;
3543        let (prefix, suffix) = original_host.split_at(index);
3544
3545        if !prefix.ends_with('.') {
3546            return None;
3547        }
3548        if suffix != host {
3549            return None;
3550        }
3551
3552        // Step 4.3
3553        if is_pub_domain(host) {
3554            return None;
3555        }
3556    }
3557
3558    // Step 5
3559    Some(host)
3560}

4574        // > as the timestamp.

4601    pub(crate) fn shared_declarative_refresh_steps(&self, content: &[u8]) {
4602        // 1. If document's will declaratively refresh is true, then return.
4603        if self.will_declaratively_refresh() {
4604            return;
4605        }
4606
4607        // 2-11 Parsing
4608        static REFRESH_REGEX: LazyLock<Regex> = LazyLock::new(|| {
4609            // s flag is used to match . on newlines since the only places we use . in the
4610            // regex is to go "to end of the string"
4611            // (?s-u:.) is used to consume invalid unicode bytes
4612            Regex::new(
4613                r#"(?xs)
4614                    ^
4615                    \s* # 3
4616                    ((?<time>[0-9]+)|\.) # 5-6
4617                    [0-9.]* # 8
4618                    (
4619                        (
4620                            (\s*;|\s*,|\s) # 10.3
4621                            \s* # 10.4
4622                        )
4623                        (
4624                            (
4625                                (U|u)(R|r)(L|l) # 11.2-11.4
4626                                \s*=\s* # 11.5-11.7
4627                            )?
4628                        ('(?<url1>[^']*)'(?s-u:.)*|"(?<url2>[^"]*)"(?s-u:.)*|['"]?(?<url3>(?s-u:.)*)) # 11.8 - 11.10
4629                        |
4630                        (?<url4>(?s-u:.)*)
4631                    )
4632                )?
4633                $
4634            "#,
4635            )
4636            .unwrap()
4637        });
4638
4639        // 9. Let urlRecord be document's URL.
4640        let mut url_record = self.url();
4641        let captures = if let Some(captures) = REFRESH_REGEX.captures(content) {
4642            captures
4643        } else {
4644            return;
4645        };
4646        let time = if let Some(time_string) = captures.name("time") {
4647            u64::from_str(&String::from_utf8_lossy(time_string.as_bytes())).unwrap_or(0)
4648        } else {
4649            0
4650        };
4651        let captured_url = captures.name("url1").or(captures
4652            .name("url2")
4653            .or(captures.name("url3").or(captures.name("url4"))));

4655        // 11.11 Parse: Set urlRecord to the result of encoding-parsing a URL given urlString, relative to document.
4656        if let Some(url_match) = captured_url {

4658                Some(&url_record),
4659                &String::from_utf8_lossy(url_match.as_bytes()),
4660            ) {
4661                info!("Refresh to {}", url.debug_compact());
4662                url
4663            } else {
4664                // 11.12 If urlRecord is failure, then return.
4665                return;
4666            }
4667        }
4668        // 12. Set document's will declaratively refresh to true.
4669        if self.completely_loaded() {
4670            // TODO: handle active sandboxing flag
4671            self.window.as_global_scope().schedule_callback(
4672                OneshotTimerCallback::RefreshRedirectDue(RefreshRedirectDue {
4673                    window: DomRoot::from_ref(self.window()),
4674                    url: url_record,
4675                }),
4676                Duration::from_secs(time),
4677            );
4678            self.set_declarative_refresh(DeclarativeRefresh::CreatedAfterLoad);
4679        } else {
4680            self.set_declarative_refresh(DeclarativeRefresh::PendingLoad {
4681                url: url_record,
4682                time,
4683            });
4684        }
4685    }

5856    /// <https://html.spec.whatwg.org/multipage/#dom-tree-accessors:dom-document-nameditem-filter>
5857    fn NamedGetter(&self, name: DOMString, can_gc: CanGc) -> Option<NamedPropertyValue> {
5858        if name.is_empty() {
5859            return None;
5860        }
5861        let name = Atom::from(name);
5862
5863        // Step 1. Let elements be the list of named elements with the name name that are in a document tree
5864        // with the Document as their root.
5865        let elements_with_name = self.get_elements_with_name(&name);
5866        let name_iter = elements_with_name
5867            .iter()
5868            .filter(|elem| is_named_element_with_name_attribute(elem));
5869        let elements_with_id = self.get_elements_with_id(&name);
5870        let id_iter = elements_with_id
5871            .iter()
5872            .filter(|elem| is_named_element_with_id_attribute(elem));
5873        let mut elements = name_iter.chain(id_iter);
5874
5875        // Step 2. If elements has only one element, and that element is an iframe element,
5876        // and that iframe element's content navigable is not null, then return the active
5877        // WindowProxy of the element's content navigable.
5878
5879        // NOTE: We have to check if all remaining elements are equal to the first, since
5880        // the same element may appear in both lists.
5881        let first = elements.next()?;
5882        if elements.all(|other| first == other) {
5883            if let Some(nested_window_proxy) = first
5884                .downcast::<HTMLIFrameElement>()
5885                .and_then(|iframe| iframe.GetContentWindow())
5886            {
5887                return Some(NamedPropertyValue::WindowProxy(nested_window_proxy));
5888            }
5889
5890            // Step 3. Otherwise, if elements has only one element, return that element.
5891            return Some(NamedPropertyValue::Element(DomRoot::from_ref(first)));
5892        }
5893
5894        // Step 4. Otherwise, return an HTMLCollection rooted at the Document node,
5895        // whose filter matches only named elements with the name name.
5896        #[derive(JSTraceable, MallocSizeOf)]
5897        struct DocumentNamedGetter {
5898            #[no_trace]
5899            name: Atom,
5900        }
5901        impl CollectionFilter for DocumentNamedGetter {
5902            fn filter(&self, elem: &Element, _root: &Node) -> bool {
5903                let type_ = match elem.upcast::<Node>().type_id() {
5904                    NodeTypeId::Element(ElementTypeId::HTMLElement(type_)) => type_,
5905                    _ => return false,
5906                };
5907                match type_ {
5908                    HTMLElementTypeId::HTMLFormElement | HTMLElementTypeId::HTMLIFrameElement => {
5909                        elem.get_name().as_ref() == Some(&self.name)
5910                    },
5911                    HTMLElementTypeId::HTMLImageElement => elem.get_name().is_some_and(|name| {
5912                        name == *self.name ||
5913                            !name.is_empty() && elem.get_id().as_ref() == Some(&self.name)
5914                    }),
5915                    // TODO handle <embed> and <object>; these depend on whether the element is
5916                    // exposed, a concept that doesnt fully make sense until embed/object
5917                    // behaviour is actually implemented
5918                    _ => false,
5919                }
5920            }
5921        }
5922        let collection = HTMLCollection::create(
5923            self.window(),
5924            self.upcast(),
5925            Box::new(DocumentNamedGetter { name }),
5926            can_gc,
5927        );
5928        Some(NamedPropertyValue::HTMLCollection(collection))
5929    }

5931    /// <https://html.spec.whatwg.org/multipage/#dom-tree-accessors:supported-property-names>
5932    fn SupportedPropertyNames(&self) -> Vec<DOMString> {
5933        let mut names_with_first_named_element_map: HashMap<&Atom, &Element> = HashMap::new();
5934
5935        let name_map = self.name_map.borrow();
5936        for (name, elements) in &(name_map).0 {
5937            if name.is_empty() {
5938                continue;
5939            }
5940            let mut name_iter = elements
5941                .iter()
5942                .filter(|elem| is_named_element_with_name_attribute(elem));
5943            if let Some(first) = name_iter.next() {
5944                names_with_first_named_element_map.insert(name, first);
5945            }
5946        }
5947        let id_map = self.id_map.borrow();
5948        for (id, elements) in &(id_map).0 {
5949            if id.is_empty() {
5950                continue;
5951            }
5952            let mut id_iter = elements
5953                .iter()
5954                .filter(|elem| is_named_element_with_id_attribute(elem));
5955            if let Some(first) = id_iter.next() {
5956                match names_with_first_named_element_map.entry(id) {
5957                    Vacant(entry) => drop(entry.insert(first)),
5958                    Occupied(mut entry) => {
5959                        if first.upcast::<Node>().is_before(entry.get().upcast()) {
5960                            *entry.get_mut() = first;
5961                        }
5962                    },
5963                }
5964            }
5965        }
5966
5967        let mut names_with_first_named_element_vec: Vec<(&Atom, &Element)> =
5968            names_with_first_named_element_map
5969                .iter()
5970                .map(|(k, v)| (*k, *v))
5971                .collect();
5972        names_with_first_named_element_vec.sort_unstable_by(|a, b| {
5973            if a.1 == b.1 {
5974                // This can happen if an img has an id different from its name,
5975                // spec does not say which string to put first.
5976                a.0.cmp(b.0)
5977            } else if a.1.upcast::<Node>().is_before(b.1.upcast::<Node>()) {
5978                Ordering::Less
5979            } else {
5980                Ordering::Greater
5981            }
5982        });
5983
5984        names_with_first_named_element_vec
5985            .iter()
5986            .map(|(k, _v)| DOMString::from(&***k))
5987            .collect()
5988    }

6059    /// <https://html.spec.whatwg.org/multipage/#dom-document-open>
6060    fn Open(
6061        &self,
6062        _unused1: Option<DOMString>,
6063        _unused2: Option<DOMString>,
6064        can_gc: CanGc,
6065    ) -> Fallible<DomRoot<Document>> {
6066        // Step 1
6067        if !self.is_html_document() {
6068            return Err(Error::InvalidState(None));
6069        }
6070
6071        // Step 2
6072        if self.throw_on_dynamic_markup_insertion_counter.get() > 0 {
6073            return Err(Error::InvalidState(None));
6074        }
6075
6076        // Step 3
6077        let entry_responsible_document = GlobalScope::entry().as_window().Document();
6078
6079        // Step 4
6080        if !self.origin.same_origin(&entry_responsible_document.origin) {
6081            return Err(Error::Security(None));
6082        }
6083
6084        // Step 5
6085        if self
6086            .get_current_parser()
6087            .is_some_and(|parser| parser.is_active())
6088        {
6089            return Ok(DomRoot::from_ref(self));
6090        }
6091
6092        // Step 6
6093        if self.is_prompting_or_unloading() {
6094            return Ok(DomRoot::from_ref(self));
6095        }
6096
6097        // Step 7
6098        if self.active_parser_was_aborted.get() {
6099            return Ok(DomRoot::from_ref(self));
6100        }
6101
6102        // TODO: prompt to unload.
6103        // TODO: set unload_event_start and unload_event_end
6104
6105        self.window().set_navigation_start();
6106
6107        // Step 8
6108        // TODO: https://github.com/servo/servo/issues/21937
6109        if self.has_browsing_context() {
6110            // spec says "stop document loading",
6111            // which is a process that does more than just abort
6112            self.abort(can_gc);
6113        }
6114
6115        // Step 9
6116        for node in self
6117            .upcast::<Node>()
6118            .traverse_preorder(ShadowIncluding::Yes)
6119        {
6120            node.upcast::<EventTarget>().remove_all_listeners();
6121        }
6122
6123        // Step 10
6124        if self.window.Document() == DomRoot::from_ref(self) {
6125            self.window.upcast::<EventTarget>().remove_all_listeners();
6126        }
6127
6128        // Step 11. Replace all with null within document.
6129        Node::replace_all(None, self.upcast::<Node>(), can_gc);
6130
6131        // Specs and tests are in a state of flux about whether
6132        // we want to clear the selection when we remove the contents;
6133        // WPT selection/Document-open.html wants us to not clear it
6134        // as of Feb 1 2020
6135
6136        // Step 12. If document is fully active, then:
6137        if self.is_fully_active() {
6138            // Step 12.1. Let newURL be a copy of entryDocument's URL.
6139            let mut new_url = entry_responsible_document.url();
6140
6141            // Step 12.2. If entryDocument is not document, then set newURL's fragment to null.
6142            if entry_responsible_document != DomRoot::from_ref(self) {
6143                new_url.set_fragment(None);
6144            }
6145
6146            // Step 12.3. Run the URL and history update steps with document and newURL.
6147            // TODO: https://github.com/servo/servo/issues/21939
6148            self.set_url(new_url);
6149        }
6150
6151        // Step 13. Set document's is initial about:blank to false.
6152        self.is_initial_about_blank.set(false);
6153
6154        // Step 14. If document's iframe load in progress flag is set, then set document's mute
6155        // iframe load flag.
6156        // TODO: https://github.com/servo/servo/issues/21938
6157
6158        // Step 15: Set document to no-quirks mode.
6159        self.set_quirks_mode(QuirksMode::NoQuirks);
6160
6161        // Step 16. Create a new HTML parser and associate it with document. This is a
6162        // script-created parser (meaning that it can be closed by the document.open() and
6163        // document.close() methods, and that the tokenizer will wait for an explicit call to
6164        // document.close() before emitting an end-of-file token). The encoding confidence is
6165        // irrelevant.
6166        let resource_threads = self.window.as_global_scope().resource_threads().clone();
6167        *self.loader.borrow_mut() =
6168            DocumentLoader::new_with_threads(resource_threads, Some(self.url()));
6169        ServoParser::parse_html_script_input(self, self.url());
6170
6171        // Step 17. Set the insertion point to point at just before the end of the input stream
6172        // (which at this point will be empty).
6173        // Handled when creating the parser in step 16
6174
6175        // Step 18. Update the current document readiness of document to "loading".
6176        self.ready_state.set(DocumentReadyState::Loading);
6177
6178        // Step 19. Return document.
6179        Ok(DomRoot::from_ref(self))
6180    }
