68/// A data structure used for tracking the current click count. This can be
69/// reset to 0 if a mouse button event happens at a sufficient distance or time
70/// from the previous one.
71///
72/// From <https://w3c.github.io/uievents/#current-click-count>:
73/// > Implementations MUST maintain the current click count when generating mouse
74/// > events. This MUST be a non-negative integer indicating the number of consecutive
75/// > clicks of a pointing device button within a specific time. The delay after which
76/// > the count resets is specific to the environment configuration.
77#[derive(Default, JSTraceable, MallocSizeOf)]
78struct ClickCountingInfo {
79    time: Option<Instant>,
80    #[no_trace]
81    point: Option<Point2D<f32, CSSPixel>>,
82    #[no_trace]
83    button: Option<MouseButton>,
84    count: usize,
85}
86
87impl ClickCountingInfo {
88    fn reset_click_count_if_necessary(
89        &mut self,
90        button: MouseButton,
91        point_in_frame: Point2D<f32, CSSPixel>,
92    ) {
93        let (Some(previous_button), Some(previous_point), Some(previous_time)) =
94            (self.button, self.point, self.time)
95        else {
96            assert_eq!(self.count, 0);
97            return;
98        };
99
100        let double_click_timeout =
101            Duration::from_millis(pref!(dom_document_dblclick_timeout) as u64);
102        let double_click_distance_threshold = pref!(dom_document_dblclick_dist) as u64;
103
104        // Calculate distance between this click and the previous click.
105        let line = point_in_frame - previous_point;
106        let distance = (line.dot(line) as f64).sqrt();
107        if previous_button != button ||
108            Instant::now().duration_since(previous_time) > double_click_timeout ||
109            distance > double_click_distance_threshold as f64
110        {
111            self.count = 0;
112            self.time = None;
113            self.point = None;
114        }
115    }
116
117    fn increment_click_count(
118        &mut self,
119        button: MouseButton,
120        point: Point2D<f32, CSSPixel>,
121    ) -> usize {
122        self.time = Some(Instant::now());
123        self.point = Some(point);
124        self.button = Some(button);
125        self.count += 1;
126        self.count
127    }
128}

183    /// Note a pending input event, to be processed at the next `update_the_rendering` task.
184    pub(crate) fn note_pending_input_event(&self, event: ConstellationInputEvent) {
185        let mut pending_input_events = self.pending_input_events.borrow_mut();
186        if matches!(event.event.event, InputEvent::MouseMove(..)) {
187            // First try to replace any existing mouse move event.
188            if let Some(mouse_move_event) = self
189                .mouse_move_event_index
190                .borrow()
191                .and_then(|index| pending_input_events.get_mut(index))
192            {
193                *mouse_move_event = event;
194                return;
195            }
196
197            *self.mouse_move_event_index.borrow_mut() = Some(pending_input_events.len());
198        }
199
200        pending_input_events.push(event);
201    }

224    pub(crate) fn handle_pending_input_events(&self, can_gc: CanGc) {
225        let _realm = enter_realm(&*self.window);
226
227        // Reset the mouse event index.
228        *self.mouse_move_event_index.borrow_mut() = None;
229        let pending_input_events = mem::take(&mut *self.pending_input_events.borrow_mut());
230
231        for event in pending_input_events {
232            self.active_keyboard_modifiers
233                .set(event.active_keyboard_modifiers);
234
235            // TODO: For some of these we still aren't properly calculating whether or not
236            // the event was handled or if `preventDefault()` was called on it. Each of
237            // these cases needs to be examined and some of them either fire more than one
238            // event or fire events later. We have to make a good decision about what to
239            // return to the embedder when that happens.
240            let result = match event.event.event.clone() {
241                InputEvent::MouseButton(mouse_button_event) => {
242                    self.handle_native_mouse_button_event(mouse_button_event, &event, can_gc);
243                    InputEventResult::default()
244                },
245                InputEvent::MouseMove(_) => {
246                    self.handle_native_mouse_move_event(&event, can_gc);
247                    InputEventResult::default()
248                },
249                InputEvent::MouseLeftViewport(mouse_leave_event) => {
250                    self.handle_mouse_left_viewport_event(&event, &mouse_leave_event, can_gc);
251                    InputEventResult::default()
252                },
253                InputEvent::Touch(touch_event) => {
254                    self.handle_touch_event(touch_event, &event, can_gc)
255                },
256                InputEvent::Wheel(wheel_event) => {
257                    self.handle_wheel_event(wheel_event, &event, can_gc)
258                },
259                InputEvent::Keyboard(keyboard_event) => {
260                    self.handle_keyboard_event(keyboard_event, can_gc)
261                },
262                InputEvent::Ime(ime_event) => self.handle_ime_event(ime_event, can_gc),
263                #[cfg(feature = "gamepad")]
264                InputEvent::Gamepad(gamepad_event) => {
265                    self.handle_gamepad_event(gamepad_event);
266                    InputEventResult::default()
267                },
268                InputEvent::EditingAction(editing_action_event) => {
269                    self.handle_editing_action(None, editing_action_event, can_gc)
270                },
271                InputEvent::Scroll(scroll_event) => {
272                    self.handle_embedder_scroll_event(scroll_event);
273                    InputEventResult::default()
274                },
275            };
276
277            self.notify_embedder_that_event_was_handled(event.event, result);
278        }
279    }

282        &self,
283        event: InputEventAndId,
284        result: InputEventResult,
285    ) {
286        // Wait to to notify the embedder that the vent was handled until all pending DOM

296                window.send_to_embedder(

312    fn handle_mouse_left_viewport_event(
313        &self,
314        input_event: &ConstellationInputEvent,
315        mouse_leave_event: &MouseLeftViewportEvent,
316        can_gc: CanGc,
317    ) {
318        if let Some(current_hover_target) = self.current_hover_target.get() {
319            let current_hover_target = current_hover_target.upcast::<Node>();
320            for element in current_hover_target
321                .inclusive_ancestors(ShadowIncluding::No)
322                .filter_map(DomRoot::downcast::<Element>)
323            {
324                element.set_hover_state(false);
325                element.set_active_state(false);
326            }
327
328            if let Some(hit_test_result) = self
329                .most_recent_mousemove_point
330                .get()
331                .and_then(|point| self.window.hit_test_from_point_in_viewport(point))
332            {
333                MouseEvent::new_simple(
334                    &self.window,
335                    FireMouseEventType::Out,
336                    EventBubbles::Bubbles,
337                    EventCancelable::Cancelable,
338                    &hit_test_result,
339                    input_event,
340                    can_gc,
341                )
342                .upcast::<Event>()
343                .fire(current_hover_target.upcast(), can_gc);
344                self.handle_mouse_enter_leave_event(
345                    DomRoot::from_ref(current_hover_target),
346                    None,
347                    FireMouseEventType::Leave,
348                    &hit_test_result,
349                    input_event,
350                    can_gc,
351                );
352            }
353        }
354
355        // We do not want to always inform the embedder that cursor has been set to the
356        // default cursor, in order to avoid a timing issue when moving between `<iframe>`
357        // elements. There is currently no way to control which `SetCursor` message will
358        // reach the embedder first. This is safer when leaving the `WebView` entirely.
359        if !mouse_leave_event.focus_moving_to_another_iframe {
360            // If focus is moving to another frame, it will decide what the new status
361            // text is, but if this mouse leave event is leaving the WebView entirely,
362            // then clear it.
363            self.window
364                .send_to_embedder(EmbedderMsg::Status(self.window.webview_id(), None));
365            self.set_cursor(None);
366        } else {
367            self.current_cursor.set(None);
368        }
369
370        self.current_hover_target.set(None);
371        self.most_recent_mousemove_point.set(None);
372    }

374    fn handle_mouse_enter_leave_event(
375        &self,
376        event_target: DomRoot<Node>,
377        related_target: Option<DomRoot<Node>>,
378        event_type: FireMouseEventType,
379        hit_test_result: &HitTestResult,
380        input_event: &ConstellationInputEvent,
381        can_gc: CanGc,
382    ) {
383        assert!(matches!(
384            event_type,
385            FireMouseEventType::Enter | FireMouseEventType::Leave
386        ));
387
388        let common_ancestor = match related_target.as_ref() {
389            Some(related_target) => event_target
390                .common_ancestor(related_target, ShadowIncluding::No)
391                .unwrap_or_else(|| DomRoot::from_ref(&*event_target)),
392            None => DomRoot::from_ref(&*event_target),
393        };
394
395        // We need to create a target chain in case the event target shares
396        // its boundaries with its ancestors.
397        let mut targets = vec![];
398        let mut current = Some(event_target);
399        while let Some(node) = current {
400            if node == common_ancestor {
401                break;
402            }
403            current = node.GetParentNode();
404            targets.push(node);
405        }
406
407        // The order for dispatching mouseenter events starts from the topmost
408        // common ancestor of the event target and the related target.
409        if event_type == FireMouseEventType::Enter {
410            targets = targets.into_iter().rev().collect();
411        }
412
413        for target in targets {
414            MouseEvent::new_simple(
415                &self.window,
416                event_type,
417                EventBubbles::DoesNotBubble,
418                EventCancelable::NotCancelable,
419                hit_test_result,
420                input_event,
421                can_gc,
422            )
423            .upcast::<Event>()
424            .fire(target.upcast(), can_gc);
425        }
426    }

428    /// <https://w3c.github.io/uievents/#handle-native-mouse-move>
429    fn handle_native_mouse_move_event(&self, input_event: &ConstellationInputEvent, can_gc: CanGc) {
430        // Ignore all incoming events without a hit test.
431        let Some(hit_test_result) = self.window.hit_test_from_input_event(input_event) else {
432            return;
433        };
434
435        // Update the cursor when the mouse moves, if it has changed.
436        self.set_cursor(Some(hit_test_result.cursor));
437
438        let Some(new_target) = hit_test_result
439            .node
440            .inclusive_ancestors(ShadowIncluding::No)
441            .find_map(DomRoot::downcast::<Element>)
442        else {
443            return;
444        };
445
446        let target_has_changed = self
447            .current_hover_target
448            .get()
449            .is_none_or(|old_target| old_target != new_target);
450
451        // Here we know the target has changed, so we must update the state,
452        // dispatch mouseout to the previous one, mouseover to the new one.
453        if target_has_changed {
454            // Dispatch mouseout and mouseleave to previous target.
455            if let Some(old_target) = self.current_hover_target.get() {
456                let old_target_is_ancestor_of_new_target = old_target
457                    .upcast::<Node>()
458                    .is_ancestor_of(new_target.upcast::<Node>());
459
460                // If the old target is an ancestor of the new target, this can be skipped
461                // completely, since the node's hover state will be reset below.
462                if !old_target_is_ancestor_of_new_target {
463                    for element in old_target
464                        .upcast::<Node>()
465                        .inclusive_ancestors(ShadowIncluding::No)
466                        .filter_map(DomRoot::downcast::<Element>)
467                    {
468                        element.set_hover_state(false);
469                        element.set_active_state(false);
470                    }
471                }
472
473                MouseEvent::new_simple(
474                    &self.window,
475                    FireMouseEventType::Out,
476                    EventBubbles::Bubbles,
477                    EventCancelable::Cancelable,
478                    &hit_test_result,
479                    input_event,
480                    can_gc,
481                )
482                .upcast::<Event>()
483                .fire(old_target.upcast(), can_gc);
484
485                if !old_target_is_ancestor_of_new_target {
486                    let event_target = DomRoot::from_ref(old_target.upcast::<Node>());
487                    let moving_into = Some(DomRoot::from_ref(new_target.upcast::<Node>()));
488                    self.handle_mouse_enter_leave_event(
489                        event_target,
490                        moving_into,
491                        FireMouseEventType::Leave,
492                        &hit_test_result,
493                        input_event,
494                        can_gc,
495                    );
496                }
497            }
498
499            // Dispatch mouseover and mouseenter to new target.
500            for element in new_target
501                .upcast::<Node>()
502                .inclusive_ancestors(ShadowIncluding::No)
503                .filter_map(DomRoot::downcast::<Element>)
504            {
505                element.set_hover_state(true);
506            }
507
508            MouseEvent::new_simple(
509                &self.window,
510                FireMouseEventType::Over,
511                EventBubbles::Bubbles,
512                EventCancelable::Cancelable,
513                &hit_test_result,
514                input_event,
515                can_gc,
516            )
517            .upcast::<Event>()
518            .fire(new_target.upcast(), can_gc);
519
520            let moving_from = self
521                .current_hover_target
522                .get()
523                .map(|old_target| DomRoot::from_ref(old_target.upcast::<Node>()));
524            let event_target = DomRoot::from_ref(new_target.upcast::<Node>());
525            self.handle_mouse_enter_leave_event(
526                event_target,
527                moving_from,
528                FireMouseEventType::Enter,
529                &hit_test_result,
530                input_event,
531                can_gc,
532            );
533        }
534
535        // Send mousemove event to topmost target, unless it's an iframe, in which case
536        // `Paint` should have also sent an event to the inner document.
537        MouseEvent::new_simple(
538            &self.window,
539            FireMouseEventType::Move,
540            EventBubbles::Bubbles,
541            EventCancelable::Cancelable,
542            &hit_test_result,
543            input_event,
544            can_gc,
545        )
546        .upcast::<Event>()
547        .fire(new_target.upcast(), can_gc);
548
549        self.update_current_hover_target_and_status(Some(new_target));
550        self.most_recent_mousemove_point
551            .set(Some(hit_test_result.point_in_frame));
552    }

554    fn update_current_hover_target_and_status(&self, new_hover_target: Option<DomRoot<Element>>) {
555        let current_hover_target = self.current_hover_target.get();
556        if current_hover_target == new_hover_target {
557            return;
558        }
559
560        let previous_hover_target = self.current_hover_target.get();
561        self.current_hover_target.set(new_hover_target.as_deref());
562
563        // If the new hover target is an anchor with a status value, inform the embedder
564        // of the new value.
565        if let Some(target) = self.current_hover_target.get() {
566            if let Some(anchor) = target
567                .upcast::<Node>()
568                .inclusive_ancestors(ShadowIncluding::No)
569                .find_map(DomRoot::downcast::<HTMLAnchorElement>)
570            {
571                let status = anchor
572                    .full_href_url_for_user_interface()
573                    .map(|url| url.to_string());
574                self.window
575                    .send_to_embedder(EmbedderMsg::Status(self.window.webview_id(), status));
576                return;
577            }
578        }
579
580        // No state was set above, which means that the new value of the status in the embedder
581        // should be `None`. Set that now. If `previous_hover_target` is `None` that means this
582        // is the first mouse move event we are seeing after getting the cursor. In that case,
583        // we also clear the status.
584        if previous_hover_target.is_none_or(|previous_hover_target| {
585            previous_hover_target
586                .upcast::<Node>()
587                .inclusive_ancestors(ShadowIncluding::No)
588                .any(|node| node.is::<HTMLAnchorElement>())
589        }) {
590            self.window
591                .send_to_embedder(EmbedderMsg::Status(self.window.webview_id(), None));
592        }
593    }

610    /// <https://w3c.github.io/uievents/#mouseevent-algorithms>

612    fn handle_native_mouse_button_event(
613        &self,

615        input_event: &ConstellationInputEvent,
616        can_gc: CanGc,
617    ) {
618        // Ignore all incoming events without a hit test.
619        let Some(hit_test_result) = self.window.hit_test_from_input_event(input_event) else {
620            return;
621        };
622
623        debug!(
624            "{:?}: at {:?}",
625            event.action, hit_test_result.point_in_frame
626        );
627
628        let Some(element) = hit_test_result
629            .node
630            .inclusive_ancestors(ShadowIncluding::Yes)
631            .find_map(DomRoot::downcast::<Element>)
632        else {
633            return;
634        };
635
636        let node = element.upcast::<Node>();
637        debug!("{:?} on {:?}", event.action, node.debug_str());
638
639        // https://w3c.github.io/uievents/#hit-test
640        // Prevent mouse event if element is disabled.
641        // TODO: also inert.
642        if element.is_actually_disabled() {
643            return;
644        }
645
646        let mouse_event_type_string = match event.action {
647            embedder_traits::MouseButtonAction::Up => "mouseup",
648            embedder_traits::MouseButtonAction::Down => "mousedown",
649        };
650
651        // From <https://w3c.github.io/uievents/#event-type-mousedown>
652        // and <https://w3c.github.io/uievents/#event-type-mouseup>:
653        //
654        // UIEvent.detail: indicates the current click count incremented by one. For
655        // example, if no click happened before the mousedown, detail will contain
656        // the value 1
657        if event.action == MouseButtonAction::Down {
658            self.click_counting_info
659                .borrow_mut()
660                .reset_click_count_if_necessary(event.button, hit_test_result.point_in_frame);
661        }
662
663        let dom_event = DomRoot::upcast::<Event>(MouseEvent::for_platform_mouse_event(
664            mouse_event_type_string,
665            event,
666            input_event.pressed_mouse_buttons,
667            &self.window,
668            &hit_test_result,
669            input_event.active_keyboard_modifiers,
670            self.click_counting_info.borrow().count + 1,
671            can_gc,
672        ));
673
674        let activatable = element.as_maybe_activatable();
675        match event.action {
676            MouseButtonAction::Down => {
677                self.last_mouse_button_down_point
678                    .set(Some(hit_test_result.point_in_frame));
679
680                if let Some(a) = activatable {
681                    a.enter_formal_activation_state();
682                }
683
684                // (TODO) Step 6. Maybe send pointerdown event with `dom_event`.
685
686                // For a node within a text input UA shadow DOM,
687                // delegate the focus target into its shadow host.
688                // TODO: This focus delegation should be done
689                // with shadow DOM delegateFocus attribute.
690                let target_el = element.find_focusable_shadow_host_if_necessary();
691
692                let document = self.window.Document();
693                document.begin_focus_transaction();
694
695                // Try to focus `el`. If it's not focusable, focus the document instead.
696                document.request_focus(None, FocusInitiator::Local, can_gc);
697                document.request_focus(target_el.as_deref(), FocusInitiator::Local, can_gc);
698
699                // Step 7. Let result = dispatch event at target
700                let result = dom_event.dispatch(node.upcast(), false, can_gc);
701
702                // Step 8. If result is true and target is a focusable area
703                // that is click focusable, then Run the focusing steps at target.
704                if result && document.has_focus_transaction() {
705                    document.commit_focus_transaction(FocusInitiator::Local, can_gc);
706                }
707
708                // Step 9. If mbutton is the secondary mouse button, then
709                // Maybe show context menu with native, target.
710                if let MouseButton::Right = event.button {
711                    self.maybe_show_context_menu(
712                        node.upcast(),
713                        &hit_test_result,
714                        input_event,
715                        can_gc,
716                    );
717                }
718            },
719            // https://w3c.github.io/uievents/#handle-native-mouse-up
720            MouseButtonAction::Up => {
721                if let Some(a) = activatable {
722                    a.exit_formal_activation_state();
723                }
724
725                // (TODO) Step 6. Maybe send pointerup event with `dom_event``.
726
727                // Step 7. dispatch event at target.
728                dom_event.dispatch(node.upcast(), false, can_gc);
729
730                // Click counts should still work for other buttons even though they
731                // do not trigger "click" and "dblclick" events, so we increment
732                // even when those events are not fired.
733                self.click_counting_info
734                    .borrow_mut()
735                    .increment_click_count(event.button, hit_test_result.point_in_frame);
736
737                self.maybe_trigger_click_for_mouse_button_down_event(
738                    event,
739                    input_event,
740                    &hit_test_result,
741                    &element,
742                    can_gc,
743                );
744            },
745        }
746    }

748    /// <https://w3c.github.io/uievents/#handle-native-mouse-click>
749    /// <https://w3c.github.io/uievents/#event-type-dblclick>
750    fn maybe_trigger_click_for_mouse_button_down_event(
751        &self,
752        event: MouseButtonEvent,
753        input_event: &ConstellationInputEvent,
754        hit_test_result: &HitTestResult,
755        element: &Element,
756        can_gc: CanGc,
757    ) {
758        if event.button != MouseButton::Left {
759            return;
760        }
761
762        let Some(last_mouse_button_down_point) = self.last_mouse_button_down_point.take() else {
763            return;
764        };
765
766        let distance = last_mouse_button_down_point.distance_to(hit_test_result.point_in_frame);
767        let maximum_click_distance = 10.0 * self.window.device_pixel_ratio().get();
768        if distance > maximum_click_distance {
769            return;
770        }
771
772        // From <https://w3c.github.io/uievents/#event-type-click>
773        // > The click event type MUST be dispatched on the topmost event target indicated by the
774        // > pointer, when the user presses down and releases the primary pointer button.
775        // For nodes inside a text input UA shadow DOM, dispatch dblclick at the shadow host.
776        let delegated = element.find_focusable_shadow_host_if_necessary();
777        let element = delegated.as_deref().unwrap_or(element);
778        self.most_recently_clicked_element.set(Some(element));
779
780        let click_count = self.click_counting_info.borrow().count;
781        element.set_click_in_progress(true);
782        MouseEvent::for_platform_mouse_event(
783            "click",
784            event,
785            input_event.pressed_mouse_buttons,
786            &self.window,
787            hit_test_result,
788            input_event.active_keyboard_modifiers,
789            click_count,
790            can_gc,
791        )
792        .upcast::<Event>()
793        .dispatch(element.upcast(), false, can_gc);
794        element.set_click_in_progress(false);
795
796        // The firing of "dbclick" events is dependent on the platform, so we have
797        // some flexibility here. Some browsers on some platforms only fire a
798        // "dbclick" when the click count is 2 and others essentially fire one for
799        // every 2 clicks in a sequence. In all cases, browsers set the click count
800        // `detail` property to 2.
801        //
802        // We follow the latter approach here, considering that every sequence of
803        // even numbered clicks is a series of double clicks.
804        if click_count % 2 == 0 {
805            MouseEvent::for_platform_mouse_event(
806                "dblclick",
807                event,
808                input_event.pressed_mouse_buttons,
809                &self.window,
810                hit_test_result,
811                input_event.active_keyboard_modifiers,
812                2,
813                can_gc,
814            )
815            .upcast::<Event>()
816            .dispatch(element.upcast(), false, can_gc);
817        }
818    }

875    fn handle_touch_event(
876        &self,
877        event: EmbedderTouchEvent,
878        input_event: &ConstellationInputEvent,
879        can_gc: CanGc,
880    ) -> InputEventResult {
881        // Ignore all incoming events without a hit test.
882        let Some(hit_test_result) = self.window.hit_test_from_input_event(input_event) else {
883            self.update_active_touch_points_when_early_return(event);
884            return Default::default();
885        };
886
887        let TouchId(identifier) = event.id;
888        let event_name = match event.event_type {
889            TouchEventType::Down => "touchstart",
890            TouchEventType::Move => "touchmove",
891            TouchEventType::Up => "touchend",
892            TouchEventType::Cancel => "touchcancel",
893        };
894
895        let Some(el) = hit_test_result
896            .node
897            .inclusive_ancestors(ShadowIncluding::No)
898            .find_map(DomRoot::downcast::<Element>)
899        else {
900            self.update_active_touch_points_when_early_return(event);
901            return Default::default();
902        };
903
904        let target = DomRoot::upcast::<EventTarget>(el);
905        let window = &*self.window;
906
907        let client_x = Finite::wrap(hit_test_result.point_in_frame.x as f64);
908        let client_y = Finite::wrap(hit_test_result.point_in_frame.y as f64);
909        let page_x =
910            Finite::wrap(hit_test_result.point_in_frame.x as f64 + window.PageXOffset() as f64);
911        let page_y =
912            Finite::wrap(hit_test_result.point_in_frame.y as f64 + window.PageYOffset() as f64);
913
914        let touch = Touch::new(
915            window, identifier, &target, client_x,
916            client_y, // TODO: Get real screen coordinates?
917            client_x, client_y, page_x, page_y, can_gc,
918        );
919
920        match event.event_type {
921            TouchEventType::Down => {
922                // Add a new touch point
923                self.active_touch_points
924                    .borrow_mut()
925                    .push(Dom::from_ref(&*touch));
926            },
927            TouchEventType::Move => {
928                // Replace an existing touch point
929                let mut active_touch_points = self.active_touch_points.borrow_mut();
930                match active_touch_points
931                    .iter_mut()
932                    .find(|t| t.Identifier() == identifier)
933                {
934                    Some(t) => *t = Dom::from_ref(&*touch),
935                    None => warn!("Got a touchmove event for a non-active touch point"),
936                }
937            },
938            TouchEventType::Up | TouchEventType::Cancel => {
939                // Remove an existing touch point
940                let mut active_touch_points = self.active_touch_points.borrow_mut();
941                match active_touch_points
942                    .iter()
943                    .position(|t| t.Identifier() == identifier)
944                {
945                    Some(i) => {
946                        active_touch_points.swap_remove(i);
947                    },
948                    None => warn!("Got a touchend event for a non-active touch point"),
949                }
950            },
951        }
952
953        rooted_vec!(let mut target_touches);
954        let touches = {
955            let touches = self.active_touch_points.borrow();
956            target_touches.extend(touches.iter().filter(|t| t.Target() == target).cloned());
957            TouchList::new(window, touches.r(), can_gc)
958        };
959
960        let touch_event = TouchEvent::new(
961            window,
962            DOMString::from(event_name),
963            EventBubbles::Bubbles,
964            EventCancelable::from(event.is_cancelable()),
965            EventComposed::Composed,
966            Some(window),
967            0i32,
968            &touches,
969            &TouchList::new(window, from_ref(&&*touch), can_gc),
970            &TouchList::new(window, target_touches.r(), can_gc),
971            // FIXME: modifier keys
972            false,
973            false,
974            false,
975            false,
976            can_gc,
977        );
978
979        let event = touch_event.upcast::<Event>();
980        event.fire(&target, can_gc);
981        event.flags().into()
982    }

984    // If hittest fails, we still need to update the active point information.
985    fn update_active_touch_points_when_early_return(&self, event: EmbedderTouchEvent) {
986        match event.event_type {
987            TouchEventType::Down => {
988                // If the touchdown fails, we don't need to do anything.
989                // When a touchmove or touchdown occurs at that touch point,
990                // a warning is triggered: Got a touchmove/touchend event for a non-active touch point
991            },
992            TouchEventType::Move => {
993                // The failure of touchmove does not affect the number of active points.
994                // Since there is no position information when it fails, we do not need to update.
995            },
996            TouchEventType::Up | TouchEventType::Cancel => {
997                // Remove an existing touch point
998                let mut active_touch_points = self.active_touch_points.borrow_mut();
999                match active_touch_points
1000                    .iter()
1001                    .position(|t| t.Identifier() == event.id.0)
1002                {
1003                    Some(i) => {
1004                        active_touch_points.swap_remove(i);
1005                    },
1006                    None => warn!("Got a touchend event for a non-active touch point"),
1007                }
1008            },
1009        }
1010    }

1012    /// The entry point for all key processing for web content
1013    fn handle_keyboard_event(
1014        &self,
1015        keyboard_event: EmbedderKeyboardEvent,
1016        can_gc: CanGc,
1017    ) -> InputEventResult {
1018        let document = self.window.Document();
1019        let focused = document.get_focused_element();
1020        let body = document.GetBody();
1021
1022        let target = match (&focused, &body) {
1023            (Some(focused), _) => focused.upcast(),
1024            (&None, Some(body)) => body.upcast(),
1025            (&None, &None) => self.window.upcast(),
1026        };
1027
1028        let keyevent = KeyboardEvent::new(
1029            &self.window,
1030            DOMString::from(keyboard_event.event.state.event_type()),
1031            true,
1032            true,
1033            Some(&self.window),
1034            0,
1035            keyboard_event.event.key.clone(),
1036            DOMString::from(keyboard_event.event.code.to_string()),
1037            keyboard_event.event.location as u32,
1038            keyboard_event.event.repeat,
1039            keyboard_event.event.is_composing,
1040            keyboard_event.event.modifiers,
1041            0,
1042            keyboard_event.event.key.legacy_keycode(),
1043            can_gc,
1044        );
1045
1046        let event = keyevent.upcast::<Event>();
1047        event.fire(target, can_gc);
1048
1049        let mut flags = event.flags();
1050        if flags.contains(EventFlags::Canceled) {
1051            return flags.into();
1052        }
1053
1054        // https://w3c.github.io/uievents/#keys-cancelable-keys
1055        // it MUST prevent the respective beforeinput and input
1056        // (and keypress if supported) events from being generated
1057        // TODO: keypress should be deprecated and superceded by beforeinput
1058
1059        let is_character_value_key = matches!(
1060            keyboard_event.event.key,
1061            Key::Character(_) | Key::Named(NamedKey::Enter)
1062        );
1063        if keyboard_event.event.state == KeyState::Down &&
1064            is_character_value_key &&
1065            !keyboard_event.event.is_composing
1066        {
1067            // https://w3c.github.io/uievents/#keypress-event-order
1068            let keypress_event = KeyboardEvent::new(
1069                &self.window,
1070                DOMString::from("keypress"),
1071                true,
1072                true,
1073                Some(&self.window),
1074                0,
1075                keyboard_event.event.key.clone(),
1076                DOMString::from(keyboard_event.event.code.to_string()),
1077                keyboard_event.event.location as u32,
1078                keyboard_event.event.repeat,
1079                keyboard_event.event.is_composing,
1080                keyboard_event.event.modifiers,
1081                keyboard_event.event.key.legacy_charcode(),
1082                0,
1083                can_gc,
1084            );
1085            let event = keypress_event.upcast::<Event>();
1086            event.fire(target, can_gc);
1087            flags = event.flags();
1088        }
1089
1090        if flags.contains(EventFlags::Canceled) {
1091            return flags.into();
1092        }
1093
1094        // This behavior is unspecced
1095        // We are supposed to dispatch synthetic click activation for Space and/or Return,
1096        // however *when* we do it is up to us.
1097        // Here, we're dispatching it after the key event so the script has a chance to cancel it
1098        // https://www.w3.org/Bugs/Public/show_bug.cgi?id=27337
1099        if (keyboard_event.event.key == Key::Named(NamedKey::Enter) ||
1100            keyboard_event.event.code == Code::Space) &&
1101            keyboard_event.event.state == KeyState::Up
1102        {
1103            if let Some(elem) = target.downcast::<Element>() {
1104                elem.upcast::<Node>()
1105                    .fire_synthetic_pointer_event_not_trusted(DOMString::from("click"), can_gc);
1106            }
1107        }
1108
1109        flags.into()
1110    }

1113        let document = self.window.Document();
1114        let composition_event = match event {
1115            ImeEvent::Dismissed => {

1128        // spec: https://w3c.github.io/uievents/#compositionend

1140            &self.window,
1141            DOMString::from(composition_event.state.event_type()),
1142            true,
1143            cancelable,
1144            Some(&self.window),
1145            0,
1146            DOMString::from(composition_event.data),

1151        event.fire(target, can_gc);

1153    }

1345    /// <https://www.w3.org/TR/clipboard-apis/#clipboard-actions>
1346    pub(crate) fn handle_editing_action(
1347        &self,
1348        element: Option<DomRoot<Element>>,
1349        action: EditingActionEvent,
1350        can_gc: CanGc,
1351    ) -> InputEventResult {
1352        let clipboard_event_type = match action {
1353            EditingActionEvent::Copy => ClipboardEventType::Copy,
1354            EditingActionEvent::Cut => ClipboardEventType::Cut,
1355            EditingActionEvent::Paste => ClipboardEventType::Paste,
1356        };
1357
1358        // The script_triggered flag is set if the action runs because of a script, e.g. document.execCommand()
1359        let script_triggered = false;
1360
1361        // The script_may_access_clipboard flag is set
1362        // if action is paste and the script thread is allowed to read from clipboard or
1363        // if action is copy or cut and the script thread is allowed to modify the clipboard
1364        let script_may_access_clipboard = false;
1365
1366        // Step 1 If the script-triggered flag is set and the script-may-access-clipboard flag is unset
1367        if script_triggered && !script_may_access_clipboard {
1368            return InputEventResult::empty();
1369        }
1370
1371        // Step 2 Fire a clipboard event
1372        let clipboard_event =
1373            self.fire_clipboard_event(element.clone(), clipboard_event_type, can_gc);
1374
1375        // Step 3 If a script doesn't call preventDefault()
1376        // the event will be handled inside target's VirtualMethods::handle_event
1377        let event = clipboard_event.upcast::<Event>();
1378        if !event.IsTrusted() {
1379            return event.flags().into();
1380        }
1381
1382        // Step 4 If the event was canceled, then
1383        if event.DefaultPrevented() {
1384            let event_type = event.Type();
1385            match_domstring_ascii!(event_type,
1386
1387                "copy" => {
1388                    // Step 4.1 Call the write content to the clipboard algorithm,
1389                    // passing on the DataTransferItemList items, a clear-was-called flag and a types-to-clear list.
1390                    if let Some(clipboard_data) = clipboard_event.get_clipboard_data() {
1391                        let drag_data_store =
1392                            clipboard_data.data_store().expect("This shouldn't fail");
1393                        self.write_content_to_the_clipboard(&drag_data_store);
1394                    }
1395                },
1396                "cut" => {
1397                    // Step 4.1 Call the write content to the clipboard algorithm,
1398                    // passing on the DataTransferItemList items, a clear-was-called flag and a types-to-clear list.
1399                    if let Some(clipboard_data) = clipboard_event.get_clipboard_data() {
1400                        let drag_data_store =
1401                            clipboard_data.data_store().expect("This shouldn't fail");
1402                        self.write_content_to_the_clipboard(&drag_data_store);
1403                    }
1404
1405                    // Step 4.2 Fire a clipboard event named clipboardchange
1406                    self.fire_clipboard_event(element, ClipboardEventType::Change, can_gc);
1407                },
1408                // Step 4.1 Return false.
1409                // Note: This function deviates from the specification a bit by returning
1410                // the `InputEventResult` below.
1411                "paste" => (),
1412                _ => (),
1413            )
1414        }
1415
1416        // Step 5: Return true from the action.
1417        // In this case we are returning the `InputEventResult` instead of true or false.
1418        event.flags().into()
1419    }

1421    /// <https://www.w3.org/TR/clipboard-apis/#fire-a-clipboard-event>
1422    pub(crate) fn fire_clipboard_event(
1423        &self,
1424        target: Option<DomRoot<Element>>,
1425        clipboard_event_type: ClipboardEventType,
1426        can_gc: CanGc,
1427    ) -> DomRoot<ClipboardEvent> {
1428        let clipboard_event = ClipboardEvent::new(
1429            &self.window,
1430            None,
1431            DOMString::from(clipboard_event_type.as_str()),
1432            EventBubbles::Bubbles,
1433            EventCancelable::Cancelable,
1434            None,
1435            can_gc,
1436        );
1437
1438        // Step 1 Let clear_was_called be false
1439        // Step 2 Let types_to_clear an empty list
1440        let mut drag_data_store = DragDataStore::new();
1441
1442        // Step 4 let clipboard-entry be the sequence number of clipboard content, null if the OS doesn't support it.
1443
1444        // Step 5 let trusted be true if the event is generated by the user agent, false otherwise
1445        let trusted = true;
1446
1447        // Step 6 if the context is editable:
1448        let document = self.window.Document();
1449        let target = target.or(document.get_focused_element());
1450        let target = target
1451            .map(|target| DomRoot::from_ref(target.upcast()))
1452            .or_else(|| {
1453                document
1454                    .GetBody()
1455                    .map(|body| DomRoot::from_ref(body.upcast()))
1456            })
1457            .unwrap_or_else(|| DomRoot::from_ref(self.window.upcast()));
1458
1459        // Step 6.2 else TODO require Selection see https://github.com/w3c/clipboard-apis/issues/70
1460        // Step 7
1461        match clipboard_event_type {
1462            ClipboardEventType::Copy | ClipboardEventType::Cut => {
1463                // Step 7.2.1
1464                drag_data_store.set_mode(Mode::ReadWrite);
1465            },
1466            ClipboardEventType::Paste => {
1467                let (callback, receiver) =
1468                    GenericCallback::new_blocking().expect("Could not create callback");
1469                self.window.send_to_embedder(EmbedderMsg::GetClipboardText(
1470                    self.window.webview_id(),
1471                    callback,
1472                ));
1473                let text_contents = receiver
1474                    .recv()
1475                    .map(Result::unwrap_or_default)
1476                    .unwrap_or_default();
1477
1478                // Step 7.1.1
1479                drag_data_store.set_mode(Mode::ReadOnly);
1480                // Step 7.1.2 If trusted or the implementation gives script-generated events access to the clipboard
1481                if trusted {
1482                    // Step 7.1.2.1 For each clipboard-part on the OS clipboard:
1483
1484                    // Step 7.1.2.1.1 If clipboard-part contains plain text, then
1485                    let data = DOMString::from(text_contents.to_string());
1486                    let type_ = DOMString::from("text/plain");
1487                    let _ = drag_data_store.add(Kind::Text { data, type_ });
1488
1489                    // Step 7.1.2.1.2 TODO If clipboard-part represents file references, then for each file reference
1490                    // Step 7.1.2.1.3 TODO If clipboard-part contains HTML- or XHTML-formatted text then
1491
1492                    // Step 7.1.3 Update clipboard-event-datas files to match clipboard-event-datas items
1493                    // Step 7.1.4 Update clipboard-event-datas types to match clipboard-event-datas items
1494                }
1495            },
1496            ClipboardEventType::Change => (),
1497        }
1498
1499        // Step 3
1500        let clipboard_event_data = DataTransfer::new(
1501            &self.window,
1502            Rc::new(RefCell::new(Some(drag_data_store))),
1503            can_gc,
1504        );
1505
1506        // Step 8
1507        clipboard_event.set_clipboard_data(Some(&clipboard_event_data));
1508
1509        // Step 9
1510        let event = clipboard_event.upcast::<Event>();
1511        event.set_trusted(trusted);
1512
1513        // Step 10 Set events composed to true.
1514        event.set_composed(true);
1515
1516        // Step 11
1517        event.dispatch(&target, false, can_gc);
1518
1519        DomRoot::from(clipboard_event)
1520    }

1587    pub(crate) fn run_default_keyboard_event_handler(&self, event: &KeyboardEvent) {
1588        if event.upcast::<Event>().type_() != atom!("keydown") {
1589            return;
1590        }
1591        if !event.modifiers().is_empty() {
1592            return;
1593        }
1594        let scroll = match event.key() {
1595            Key::Named(NamedKey::ArrowDown) => KeyboardScroll::Down,
1596            Key::Named(NamedKey::ArrowLeft) => KeyboardScroll::Left,
1597            Key::Named(NamedKey::ArrowRight) => KeyboardScroll::Right,
1598            Key::Named(NamedKey::ArrowUp) => KeyboardScroll::Up,
1599            Key::Named(NamedKey::End) => KeyboardScroll::End,
1600            Key::Named(NamedKey::Home) => KeyboardScroll::Home,
1601            Key::Named(NamedKey::PageDown) => KeyboardScroll::PageDown,
1602            Key::Named(NamedKey::PageUp) => KeyboardScroll::PageUp,
1603            _ => return,
1604        };
1605        self.do_keyboard_scroll(scroll);
1606    }

1608    pub(crate) fn do_keyboard_scroll(&self, scroll: KeyboardScroll) {
1609        let scroll_axis = match scroll {
1610            KeyboardScroll::Left | KeyboardScroll::Right => ScrollingBoxAxis::X,
1611            _ => ScrollingBoxAxis::Y,
1612        };
1613
1614        let document = self.window.Document();
1615        let mut scrolling_box = document
1616            .get_focused_element()
1617            .or(self.most_recently_clicked_element.get())
1618            .and_then(|element| element.scrolling_box(ScrollContainerQueryFlags::Inclusive))
1619            .unwrap_or_else(|| {
1620                document.viewport_scrolling_box(ScrollContainerQueryFlags::Inclusive)
1621            });
1622
1623        while !scrolling_box.can_keyboard_scroll_in_axis(scroll_axis) {
1624            // Always fall back to trying to scroll the entire document.
1625            if scrolling_box.is_viewport() {
1626                break;
1627            }
1628            let parent = scrolling_box.parent().unwrap_or_else(|| {
1629                document.viewport_scrolling_box(ScrollContainerQueryFlags::Inclusive)
1630            });
1631            scrolling_box = parent;
1632        }
1633
1634        let calculate_current_scroll_offset_and_delta = || {
1635            const LINE_HEIGHT: f32 = 76.0;
1636            const LINE_WIDTH: f32 = 76.0;
1637
1638            let current_scroll_offset = scrolling_box.scroll_position();
1639            (
1640                current_scroll_offset,
1641                match scroll {
1642                    KeyboardScroll::Home => Vector2D::new(0.0, -current_scroll_offset.y),
1643                    KeyboardScroll::End => Vector2D::new(
1644                        0.0,
1645                        -current_scroll_offset.y + scrolling_box.content_size().height -
1646                            scrolling_box.size().height,
1647                    ),
1648                    KeyboardScroll::PageDown => {
1649                        Vector2D::new(0.0, scrolling_box.size().height - 2.0 * LINE_HEIGHT)
1650                    },
1651                    KeyboardScroll::PageUp => {
1652                        Vector2D::new(0.0, 2.0 * LINE_HEIGHT - scrolling_box.size().height)
1653                    },
1654                    KeyboardScroll::Up => Vector2D::new(0.0, -LINE_HEIGHT),
1655                    KeyboardScroll::Down => Vector2D::new(0.0, LINE_HEIGHT),
1656                    KeyboardScroll::Left => Vector2D::new(-LINE_WIDTH, 0.0),
1657                    KeyboardScroll::Right => Vector2D::new(LINE_WIDTH, 0.0),
1658                },
1659            )
1660        };
1661
1662        // If trying to scroll the viewport of this `Window` and this is the root `Document`
1663        // of the `WebView`, then send the srolling operation to the renderer, so that it
1664        // can properly pan any pinch zoom viewport.
1665        let parent_pipeline = self.window.parent_info();
1666        if scrolling_box.is_viewport() && parent_pipeline.is_none() {
1667            let (_, delta) = calculate_current_scroll_offset_and_delta();
1668            self.window
1669                .paint_api()
1670                .scroll_viewport_by_delta(self.window.webview_id(), delta);
1671        }
1672
1673        // If this is the viewport and we cannot scroll, try to ask a parent viewport to scroll,
1674        // if we are inside an `<iframe>`.
1675        if !scrolling_box.can_keyboard_scroll_in_axis(scroll_axis) {
1676            assert!(scrolling_box.is_viewport());
1677
1678            let window_proxy = document.window().window_proxy();
1679            if let Some(iframe) = window_proxy.frame_element() {
1680                // When the `<iframe>` is local (in this ScriptThread), we can
1681                // synchronously chain up the keyboard scrolling event.
1682                let cx = GlobalScope::get_cx();
1683                let iframe_window = iframe.owner_window();
1684                let _ac = JSAutoRealm::new(*cx, iframe_window.reflector().get_jsobject().get());
1685                iframe_window
1686                    .Document()
1687                    .event_handler()
1688                    .do_keyboard_scroll(scroll);
1689            } else if let Some(parent_pipeline) = parent_pipeline {
1690                // Otherwise, if we have a parent (presumably from a different origin)
1691                // asynchronously ask the Constellation to forward the event to the parent
1692                // pipeline, if we have one.
1693                document.window().send_to_constellation(
1694                    ScriptToConstellationMessage::ForwardKeyboardScroll(parent_pipeline, scroll),
1695                );
1696            };
1697            return;
1698        }
1699
1700        let (current_scroll_offset, delta) = calculate_current_scroll_offset_and_delta();
1701        scrolling_box.scroll_to(delta + current_scroll_offset, ScrollBehavior::Auto);
1702    }
