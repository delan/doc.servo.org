120    /// Calls into the layout engine to generate a plain text representation
121    /// of a [`HTMLElement`] as specified when getting the `.innerText` or
122    /// `.outerText` in JavaScript.`
123    ///
124    /// <https://html.spec.whatwg.org/multipage/#get-the-text-steps>
125    pub(crate) fn get_inner_outer_text(&self) -> DOMString {
126        let node = self.upcast::<Node>();
127        let window = node.owner_window();
128        let element = self.as_element();
129
130        // Step 1.
131        let element_not_rendered = !node.is_connected() || !element.has_css_layout_box();
132        if element_not_rendered {
133            return node.GetTextContent().unwrap();
134        }
135
136        window.layout_reflow(QueryMsg::ElementInnerOuterTextQuery);
137        let text = window
138            .layout()
139            .query_element_inner_outer_text(node.to_trusted_node_address());
140
141        DOMString::from(text)
142    }

144    /// <https://html.spec.whatwg.org/multipage/#set-the-inner-text-steps>
145    pub(crate) fn set_inner_text(&self, input: DOMString, can_gc: CanGc) {
146        // Step 1: Let fragment be the rendered text fragment for value given element's node
147        // document.
148        let fragment = self.rendered_text_fragment(input, can_gc);
149
150        // Step 2: Replace all with fragment within element.
151        Node::replace_all(Some(fragment.upcast()), self.upcast::<Node>(), can_gc);
152    }

376        let atoms = self

386        let mut item_attr_values = HashSet::new();

389        }

391        Some(item_attr_values.into_iter().collect())
392    }

546    /// <https://html.spec.whatwg.org/multipage/#the-innertext-idl-attribute:dom-outertext-2>
547    fn SetOuterText(&self, input: DOMString, can_gc: CanGc) -> Fallible<()> {
548        // Step 1: If this's parent is null, then throw a "NoModificationAllowedError" DOMException.
549        let Some(parent) = self.upcast::<Node>().GetParentNode() else {
550            return Err(Error::NoModificationAllowed(None));
551        };
552
553        let node = self.upcast::<Node>();
554        let document = self.owner_document();
555
556        // Step 2: Let next be this's next sibling.
557        let next = node.GetNextSibling();
558
559        // Step 3: Let previous be this's previous sibling.
560        let previous = node.GetPreviousSibling();
561
562        // Step 4: Let fragment be the rendered text fragment for the given value given this's node
563        // document.
564        let fragment = self.rendered_text_fragment(input, can_gc);
565
566        // Step 5: If fragment has no children, then append a new Text node whose data is the empty
567        // string and node document is this's node document to fragment.
568        if fragment.upcast::<Node>().children_count() == 0 {
569            let text_node = Text::new(DOMString::from("".to_owned()), &document, can_gc);
570
571            fragment
572                .upcast::<Node>()
573                .AppendChild(text_node.upcast(), can_gc)?;
574        }
575
576        // Step 6: Replace this with fragment within this's parent.
577        parent.ReplaceChild(fragment.upcast(), node, can_gc)?;
578
579        // Step 7: If next is non-null and next's previous sibling is a Text node, then merge with
580        // the next text node given next's previous sibling.
581        if let Some(next_sibling) = next {
582            if let Some(node) = next_sibling.GetPreviousSibling() {
583                Self::merge_with_the_next_text_node(node, can_gc);
584            }
585        }
586
587        // Step 8: If previous is a Text node, then merge with the next text node given previous.
588        if let Some(previous) = previous {
589            Self::merge_with_the_next_text_node(previous, can_gc)
590        }
591
592        Ok(())
593    }

633    /// <https://html.spec.whatwg.org/multipage#dom-attachinternals>
634    fn AttachInternals(&self, can_gc: CanGc) -> Fallible<DomRoot<ElementInternals>> {
635        let element = self.as_element();
636        // Step 1: If this's is value is not null, then throw a "NotSupportedError" DOMException
637        if element.get_is().is_some() {
638            return Err(Error::NotSupported(None));
639        }
640
641        // Step 2: Let definition be the result of looking up a custom element definition
642        // Note: the element can pass this check without yet being a custom
643        // element, as long as there is a registered definition
644        // that could upgrade it to one later.
645        let registry = self.owner_window().CustomElements();
646        let definition = registry.lookup_definition(self.as_element().local_name(), None);
647
648        // Step 3: If definition is null, then throw an "NotSupportedError" DOMException
649        let definition = match definition {
650            Some(definition) => definition,
651            None => return Err(Error::NotSupported(None)),
652        };
653
654        // Step 4: If definition's disable internals is true, then throw a "NotSupportedError" DOMException
655        if definition.disable_internals {
656            return Err(Error::NotSupported(None));
657        }
658
659        // Step 5: If this's attached internals is non-null, then throw an "NotSupportedError" DOMException
660        let internals = element.ensure_element_internals(can_gc);
661        if internals.attached() {
662            return Err(Error::NotSupported(None));
663        }
664
665        // Step 6: If this's custom element state is not "precustomized" or "custom",
666        // then throw a "NotSupportedError" DOMException.
667        if !matches!(
668            element.get_custom_element_state(),
669            CustomElementState::Precustomized | CustomElementState::Custom
670        ) {
671            return Err(Error::NotSupported(None));
672        }
673
674        if self.is_form_associated_custom_element() {
675            element.init_state_for_internals();
676        }
677
678        // Step 6-7: Set this's attached internals to a new ElementInternals instance
679        internals.set_attached();
680        Ok(internals)
681    }

719// https://html.spec.whatwg.org/multipage/#attr-data-*
720
721static DATA_PREFIX: &str = "data-";
722static DATA_HYPHEN_SEPARATOR: char = '\x2d';
723
724fn to_snake_case(name: DOMString) -> DOMString {
725    let mut attr_name = String::with_capacity(name.len() + DATA_PREFIX.len());
726    attr_name.push_str(DATA_PREFIX);
727    for ch in name.str().chars() {
728        if ch.is_ascii_uppercase() {
729            attr_name.push(DATA_HYPHEN_SEPARATOR);
730            attr_name.push(ch.to_ascii_lowercase());
731        } else {
732            attr_name.push(ch);
733        }
734    }
735    DOMString::from(attr_name)
736}
737
738// https://html.spec.whatwg.org/multipage/#attr-data-*
739// if this attribute is in snake case with a data- prefix,
740// this function returns a name converted to camel case
741// without the data prefix.
742
743fn to_camel_case(name: &str) -> Option<DOMString> {
744    if !name.starts_with(DATA_PREFIX) {
745        return None;
746    }
747    let name = &name[5..];
748    let has_uppercase = name.chars().any(|curr_char| curr_char.is_ascii_uppercase());
749    if has_uppercase {
750        return None;
751    }
752    let mut result = String::with_capacity(name.len().saturating_sub(DATA_PREFIX.len()));
753    let mut name_chars = name.chars();
754    while let Some(curr_char) = name_chars.next() {
755        // check for hyphen followed by character
756        if curr_char == DATA_HYPHEN_SEPARATOR {
757            if let Some(next_char) = name_chars.next() {
758                if next_char.is_ascii_lowercase() {
759                    result.push(next_char.to_ascii_uppercase());
760                } else {
761                    result.push(curr_char);
762                    result.push(next_char);
763                }
764            } else {
765                result.push(curr_char);
766            }
767        } else {
768            result.push(curr_char);
769        }
770    }
771    Some(DOMString::from(result))
772}

896    // This gets the nth label in tree order.
897    pub(crate) fn label_at(&self, index: u32) -> Option<DomRoot<Node>> {
898        let element = self.as_element();
899
900        // Traverse entire tree for <label> elements that have
901        // this as their control.
902        // There is room for performance optimization, as we don't need
903        // the actual result of GetControl, only whether the result
904        // would match self.
905        // (Even more room for performance optimization: do what
906        // nodelist ChildrenList does and keep a mutation-aware cursor
907        // around; this may be hard since labels need to keep working
908        // even as they get detached into a subtree and reattached to

912        root_node
913            .traverse_preorder(ShadowIncluding::No)
914            .filter_map(DomRoot::downcast::<HTMLLabelElement>)
915            .filter(|elem| match elem.GetControl() {
916                Some(control) => &*control == self,
917                _ => false,
918            })
919            .nth(index as usize)
920            .map(|n| DomRoot::from_ref(n.upcast::<Node>()))
921    }

982    // https://html.spec.whatwg.org/multipage/#the-summary-element:activation-behaviour
983    pub(crate) fn summary_activation_behavior(&self) {
984        debug_assert!(self.as_element().local_name() == &local_name!("summary"));
985
986        // Step 1. If this summary element is not the summary for its parent details, then return.
987        if !self.is_a_summary_for_its_parent_details() {
988            return;
989        }
990
991        // Step 2. Let parent be this summary element's parent.
992        let parent = if self.is_implicit_summary_element() {
993            DomRoot::downcast::<HTMLDetailsElement>(self.containing_shadow_root().unwrap().Host())
994                .unwrap()
995        } else {
996            self.upcast::<Node>()
997                .GetParentNode()
998                .and_then(DomRoot::downcast::<HTMLDetailsElement>)
999                .unwrap()
1000        };
1001
1002        // Step 3. If the open attribute is present on parent, then remove it.
1003        // Otherwise, set parent's open attribute to the empty string.
1004        parent.toggle();
1005    }

1007    /// <https://html.spec.whatwg.org/multipage/#summary-for-its-parent-details>
1008    fn is_a_summary_for_its_parent_details(&self) -> bool {
1009        if self.is_implicit_summary_element() {
1010            return true;
1011        }
1012
1013        // Step 1. If this summary element has no parent, then return false.
1014        // Step 2. Let parent be this summary element's parent.
1015        let Some(parent) = self.upcast::<Node>().GetParentNode() else {
1016            return false;
1017        };
1018
1019        // Step 3. If parent is not a details element, then return false.
1020        let Some(details) = parent.downcast::<HTMLDetailsElement>() else {
1021            return false;
1022        };
1023
1024        // Step 4. If parent's first summary element child is not this summary
1025        // element, then return false.
1026        // Step 5. Return true.
1027        details
1028            .find_corresponding_summary_element()
1029            .is_some_and(|summary| &*summary == self.upcast())
1030    }

1044    /// <https://html.spec.whatwg.org/multipage/#rendered-text-fragment>
1045    fn rendered_text_fragment(&self, input: DOMString, can_gc: CanGc) -> DomRoot<DocumentFragment> {
1046        // Step 1: Let fragment be a new DocumentFragment whose node document is document.
1047        let document = self.owner_document();
1048        let fragment = DocumentFragment::new(&document, can_gc);
1049
1050        // Step 2: Let position be a position variable for input, initially pointing at the start
1051        // of input.
1052        let input = input.str();
1053        let mut position = input.chars().peekable();
1054
1055        // Step 3: Let text be the empty string.
1056        let mut text = String::new();
1057
1058        // Step 4
1059        while let Some(ch) = position.next() {
1060            match ch {
1061                // While position is not past the end of input, and the code point at position is
1062                // either U+000A LF or U+000D CR:
1063                '\u{000A}' | '\u{000D}' => {
1064                    if ch == '\u{000D}' && position.peek() == Some(&'\u{000A}') {
1065                        // a \r\n pair should only generate one <br>,
1066                        // so just skip the \r.
1067                        position.next();
1068                    }
1069
1070                    if !text.is_empty() {
1071                        append_text_node_to_fragment(&document, &fragment, text, can_gc);
1072                        text = String::new();
1073                    }
1074
1075                    let br = Element::create(
1076                        QualName::new(None, ns!(html), local_name!("br")),
1077                        None,
1078                        &document,
1079                        ElementCreator::ScriptCreated,
1080                        CustomElementCreationMode::Asynchronous,
1081                        None,
1082                        can_gc,
1083                    );
1084                    fragment
1085                        .upcast::<Node>()
1086                        .AppendChild(br.upcast(), can_gc)
1087                        .unwrap();
1088                },
1089                _ => {
1090                    // Collect a sequence of code points that are not U+000A LF or U+000D CR from
1091                    // input given position, and set text to the result.
1092                    text.push(ch);
1093                },
1094            }
1095        }
1096
1097        // If text is not the empty string, then append a new Text node whose data is text and node
1098        // document is document to fragment.
1099        if !text.is_empty() {
1100            append_text_node_to_fragment(&document, &fragment, text, can_gc);
1101        }
1102
1103        fragment
1104    }

1106    /// Checks whether a given [`DomRoot<Node>`] and its next sibling are
1107    /// of type [`Text`], and if so merges them into a single [`Text`]
1108    /// node.
1109    ///
1110    /// <https://html.spec.whatwg.org/multipage/#merge-with-the-next-text-node>
1111    fn merge_with_the_next_text_node(node: DomRoot<Node>, can_gc: CanGc) {
1112        // Make sure node is a Text node
1113        if !node.is::<Text>() {
1114            return;
1115        }
1116
1117        // Step 1: Let next be node's next sibling.
1118        let next = match node.GetNextSibling() {
1119            Some(next) => next,
1120            None => return,
1121        };
1122
1123        // Step 2: If next is not a Text node, then return.
1124        if !next.is::<Text>() {
1125            return;
1126        }
1127        // Step 3: Replace data with node, node's data's length, 0, and next's data.
1128        let node_chars = node.downcast::<CharacterData>().expect("Node is Text");
1129        let next_chars = next.downcast::<CharacterData>().expect("Next node is Text");
1130        node_chars
1131            .ReplaceData(node_chars.Length(), 0, next_chars.Data())
1132            .expect("Got chars from Text");
1133
1134        // Step 4:Remove next.
1135        next.remove_self(can_gc);
1136    }

1144    fn attribute_mutated(&self, attr: &Attr, mutation: AttributeMutation, can_gc: CanGc) {
1145        self.super_type()
1146            .unwrap()
1147            .attribute_mutated(attr, mutation, can_gc);
1148        let element = self.as_element();
1149        match (attr.local_name(), mutation) {
1150            // https://html.spec.whatwg.org/multipage/#event-handler-attributes:event-handler-content-attributes-3
1151            (name, mutation)
1152                if name.starts_with("on") && EventTarget::is_content_event_handler(name) =>
1153            {
1154                let evtarget = self.upcast::<EventTarget>();
1155                let event_name = &name[2..];
1156                match mutation {
1157                    // https://html.spec.whatwg.org/multipage/#activate-an-event-handler
1158                    AttributeMutation::Set(..) => {
1159                        let source = &**attr.value();
1160                        let source_line = 1; // TODO(#9604) get current JS execution line
1161                        evtarget.set_event_handler_uncompiled(
1162                            self.owner_window().get_url(),
1163                            source_line,
1164                            event_name,
1165                            source,
1166                        );
1167                    },
1168                    // https://html.spec.whatwg.org/multipage/#deactivate-an-event-handler
1169                    AttributeMutation::Removed => {
1170                        evtarget.set_event_handler_common::<EventHandlerNonNull>(event_name, None);
1171                    },
1172                }
1173            },
1174
1175            (&local_name!("form"), mutation) if self.is_form_associated_custom_element() => {
1176                self.form_attribute_mutated(mutation, can_gc);
1177            },
1178            // Adding a "disabled" attribute disables an enabled form element.
1179            (&local_name!("disabled"), AttributeMutation::Set(..))
1180                if self.is_form_associated_custom_element() && element.enabled_state() =>
1181            {
1182                element.set_disabled_state(true);
1183                element.set_enabled_state(false);
1184                ScriptThread::enqueue_callback_reaction(
1185                    element,
1186                    CallbackReaction::FormDisabled(true),
1187                    None,
1188                );
1189            },
1190            // Removing the "disabled" attribute may enable a disabled
1191            // form element, but a fieldset ancestor may keep it disabled.
1192            (&local_name!("disabled"), AttributeMutation::Removed)
1193                if self.is_form_associated_custom_element() && element.disabled_state() =>
1194            {
1195                element.set_disabled_state(false);
1196                element.set_enabled_state(true);
1197                element.check_ancestors_disabled_state_for_form_control();
1198                if element.enabled_state() {
1199                    ScriptThread::enqueue_callback_reaction(
1200                        element,
1201                        CallbackReaction::FormDisabled(false),
1202                        None,
1203                    );
1204                }
1205            },
1206            (&local_name!("readonly"), mutation) if self.is_form_associated_custom_element() => {
1207                match mutation {
1208                    AttributeMutation::Set(..) => {
1209                        element.set_read_write_state(true);
1210                    },
1211                    AttributeMutation::Removed => {
1212                        element.set_read_write_state(false);
1213                    },
1214                }
1215            },
1216            (&local_name!("nonce"), mutation) => match mutation {
1217                AttributeMutation::Set(..) => {
1218                    let nonce = &**attr.value();
1219                    element.update_nonce_internal_slot(nonce.to_owned());
1220                },
1221                AttributeMutation::Removed => {
1222                    element.update_nonce_internal_slot("".to_owned());
1223                },
1224            },
1225            _ => {},
1226        }
1227    }
1228
1229    fn bind_to_tree(&self, context: &BindContext, can_gc: CanGc) {
1230        if let Some(super_type) = self.super_type() {
1231            super_type.bind_to_tree(context, can_gc);
1232        }
1233        let element = self.as_element();
1234        element.update_sequentially_focusable_status(can_gc);
1235
1236        // Binding to a tree can disable a form control if one of the new
1237        // ancestors is a fieldset.
1238        if self.is_form_associated_custom_element() && element.enabled_state() {
1239            element.check_ancestors_disabled_state_for_form_control();
1240            if element.disabled_state() {
1241                ScriptThread::enqueue_callback_reaction(
1242                    element,
1243                    CallbackReaction::FormDisabled(true),
1244                    None,
1245                );
1246            }
1247        }
1248    }
1249
1250    fn unbind_from_tree(&self, context: &UnbindContext, can_gc: CanGc) {
1251        if let Some(super_type) = self.super_type() {
1252            super_type.unbind_from_tree(context, can_gc);
1253        }
1254
1255        // Unbinding from a tree might enable a form control, if a
1256        // fieldset ancestor is the only reason it was disabled.
1257        // (The fact that it's enabled doesn't do much while it's
1258        // disconnected, but it is an observable fact to keep track of.)
1259        let element = self.as_element();
1260        if self.is_form_associated_custom_element() && element.disabled_state() {
1261            element.check_disabled_attribute();
1262            element.check_ancestors_disabled_state_for_form_control();
1263            if element.enabled_state() {
1264                ScriptThread::enqueue_callback_reaction(
1265                    element,
1266                    CallbackReaction::FormDisabled(false),
1267                    None,
1268                );
1269            }
1270        }
1271    }
1272
1273    fn attribute_affects_presentational_hints(&self, attr: &Attr) -> bool {
1274        let element = self.upcast::<Element>();
1275        if is_element_affected_by_legacy_background_presentational_hint(
1276            element.namespace(),
1277            element.local_name(),
1278        ) && attr.local_name() == &local_name!("background")
1279        {
1280            return true;
1281        }
1282
1283        self.super_type()
1284            .unwrap()
1285            .attribute_affects_presentational_hints(attr)
1286    }
1287
1288    fn parse_plain_attribute(&self, name: &LocalName, value: DOMString) -> AttrValue {
1289        let element = self.upcast::<Element>();
1290        match *name {
1291            local_name!("itemprop") => AttrValue::from_serialized_tokenlist(value.into()),
1292            local_name!("itemtype") => AttrValue::from_serialized_tokenlist(value.into()),
1293            local_name!("background")
1294                if is_element_affected_by_legacy_background_presentational_hint(
1295                    element.namespace(),
1296                    element.local_name(),
1297                ) =>
1298            {
1299                AttrValue::from_resolved_url(
1300                    &self.owner_document().base_url().get_arc(),
1301                    value.into(),
1302                )
1303            },
1304            _ => self
1305                .super_type()
1306                .unwrap()
1307                .parse_plain_attribute(name, value),
1308        }
1309    }

1327// Form-associated custom elements are the same interface type as
1328// normal HTMLElements, so HTMLElement needs to have the FormControl trait
1329// even though it's usually more specific trait implementations, like the
1330// HTMLInputElement one, that we really want. (Alternately we could put
1331// the FormControl trait on ElementInternals, but that raises lifetime issues.)
1332impl FormControl for HTMLElement {
1333    fn form_owner(&self) -> Option<DomRoot<HTMLFormElement>> {
1334        debug_assert!(self.is_form_associated_custom_element());
1335        self.as_element()
1336            .get_element_internals()
1337            .and_then(|e| e.form_owner())
1338    }
1339
1340    fn set_form_owner(&self, form: Option<&HTMLFormElement>) {
1341        debug_assert!(self.is_form_associated_custom_element());
1342        self.as_element()
1343            .ensure_element_internals(CanGc::note())
1344            .set_form_owner(form);
1345    }
1346
1347    fn to_element(&self) -> &Element {
1348        self.as_element()
1349    }
1350
1351    fn is_listed(&self) -> bool {
1352        debug_assert!(self.is_form_associated_custom_element());
1353        true
1354    }
