194    // https://html.spec.whatwg.org/multipage/#check-the-usability-of-the-image-argument
195    pub(crate) fn is_usable(&self) -> Fallible<bool> {
196        // If image has an intrinsic width or intrinsic height (or both) equal to zero, then return bad.
197        if let Some(image) = &self.current_request.borrow().image {
198            let intrinsic_size = image.metadata();
199            if intrinsic_size.width == 0 || intrinsic_size.height == 0 {
200                return Ok(false);
201            }
202        }
203
204        match self.current_request.borrow().state {
205            // If image's current request's state is broken, then throw an "InvalidStateError" DOMException.
206            State::Broken => Err(Error::InvalidState(None)),
207            State::CompletelyAvailable => Ok(true),
208            // If image is not fully decodable, then return bad.
209            State::PartiallyAvailable | State::Unavailable => Ok(false),
210        }
211    }

214        self.current_request.borrow().image.clone()
215    }

243impl FetchResponseListener for ImageContext {
244    fn should_invoke(&self) -> bool {
245        !self.aborted
246    }
247
248    fn process_request_body(&mut self, _: RequestId) {}
249    fn process_request_eof(&mut self, _: RequestId) {}
250
251    fn process_response(
252        &mut self,
253        request_id: RequestId,
254        metadata: Result<FetchMetadata, NetworkError>,
255    ) {
256        debug!("got {:?} for {:?}", metadata.as_ref().map(|_| ()), self.url);
257        self.image_cache.notify_pending_response(
258            self.id,
259            FetchResponseMsg::ProcessResponse(request_id, metadata.clone()),
260        );
261
262        let metadata = metadata.ok().map(|meta| match meta {
263            FetchMetadata::Unfiltered(m) => m,
264            FetchMetadata::Filtered { unsafe_, .. } => unsafe_,
265        });
266
267        // Step 14.5 of https://html.spec.whatwg.org/multipage/#img-environment-changes
268        if let Some(metadata) = metadata.as_ref() {
269            if let Some(ref content_type) = metadata.content_type {
270                let mime: Mime = content_type.clone().into_inner().into();
271                if mime.type_() == mime::MULTIPART && mime.subtype().as_str() == "x-mixed-replace" {
272                    self.aborted = true;
273                }
274            }
275        }
276
277        let status = metadata
278            .as_ref()
279            .map(|m| m.status.clone())
280            .unwrap_or_else(HttpStatus::new_error);
281
282        self.status = {
283            if status.is_error() {
284                Err(NetworkError::ResourceLoadError(
285                    "No http status code received".to_owned(),
286                ))
287            } else if status.is_success() {
288                Ok(())
289            } else {
290                Err(NetworkError::ResourceLoadError(format!(
291                    "HTTP error code {}",
292                    status.code()
293                )))
294            }
295        };
296    }
297
298    fn process_response_chunk(&mut self, request_id: RequestId, payload: Vec<u8>) {
299        if self.status.is_ok() {
300            self.image_cache.notify_pending_response(
301                self.id,
302                FetchResponseMsg::ProcessResponseChunk(request_id, payload.into()),
303            );
304        }
305    }
306
307    fn process_response_eof(
308        self,
309        request_id: RequestId,
310        response: Result<ResourceFetchTiming, NetworkError>,
311    ) {
312        self.image_cache.notify_pending_response(
313            self.id,
314            FetchResponseMsg::ProcessResponseEOF(request_id, response.clone()),
315        );
316        if let Ok(response) = response {
317            network_listener::submit_timing(&self, &response, CanGc::note());
318        }
319    }
320
321    fn process_csp_violations(&mut self, _request_id: RequestId, violations: Vec<Violation>) {
322        let global = &self.resource_timing_global();
323        let elem = self.element.root();
324        let source_position = elem
325            .upcast::<Element>()
326            .compute_source_position(elem.line_number as u32);
327        global.report_csp_violations(violations, None, Some(source_position));
328    }

346    /// Update the current image with a valid URL.
347    fn fetch_image(&self, img_url: &ServoUrl, can_gc: CanGc) {
348        let window = self.owner_window();
349
350        let cache_result = window.image_cache().get_cached_image_status(
351            img_url.clone(),
352            window.origin().immutable().clone(),
353            cors_setting_for_element(self.upcast()),
354        );
355
356        match cache_result {
357            ImageCacheResult::Available(ImageOrMetadataAvailable::ImageAvailable {
358                image,
359                url,
360            }) => self.process_image_response(ImageResponse::Loaded(image, url), can_gc),
361            ImageCacheResult::Available(ImageOrMetadataAvailable::MetadataAvailable(
362                metadata,
363                id,
364            )) => {
365                self.process_image_response(ImageResponse::MetadataLoaded(metadata), can_gc);
366                self.register_image_cache_callback(id, ChangeType::Element);
367            },
368            ImageCacheResult::Pending(id) => {
369                self.register_image_cache_callback(id, ChangeType::Element);
370            },
371            ImageCacheResult::ReadyForRequest(id) => {
372                self.fetch_request(img_url, id);
373                self.register_image_cache_callback(id, ChangeType::Element);
374            },
375            ImageCacheResult::FailedToLoadOrDecode => {
376                self.process_image_response(ImageResponse::FailedToLoadOrDecode, can_gc)
377            },
378        };
379    }

381    fn register_image_cache_callback(&self, id: PendingImageId, change_type: ChangeType) {
382        let trusted_node = Trusted::new(self);
383        let generation = self.generation_id();
384        let window = self.owner_window();
385        let callback = window.register_image_cache_listener(id, move |response| {
386            let trusted_node = trusted_node.clone();
387            let window = trusted_node.root().owner_window();
388            let callback_type = change_type.clone();
389
390            window
391                .as_global_scope()
392                .task_manager()
393                .networking_task_source()
394                .queue(task!(process_image_response: move || {
395                let element = trusted_node.root();
396
397                // Ignore any image response for a previous request that has been discarded.
398                if generation != element.generation_id() {
399                    return;
400                }
401
402                match callback_type {
403                    ChangeType::Element => {
404                        element.process_image_response(response.response, CanGc::note());
405                    }
406                    ChangeType::Environment { selected_source, selected_pixel_density } => {
407                        element.process_image_response_for_environment_change(
408                            response.response, selected_source, generation, selected_pixel_density, CanGc::note()
409                        );
410                    }
411                }
412            }));
413        });
414
415        window.image_cache().add_listener(ImageLoadListener::new(
416            callback,
417            window.pipeline_id(),
418            id,
419        ));
420    }

422    fn fetch_request(&self, img_url: &ServoUrl, id: PendingImageId) {
423        let document = self.owner_document();
424        let window = self.owner_window();
425
426        let context = ImageContext {
427            image_cache: window.image_cache(),
428            status: Ok(()),
429            id,
430            aborted: false,
431            doc: Trusted::new(&document),
432            element: Trusted::new(self),
433            url: img_url.clone(),
434        };
435
436        // https://html.spec.whatwg.org/multipage/#update-the-image-data steps 17-20
437        // This function is also used to prefetch an image in `script::dom::servoparser::prefetch`.
438        let global = document.global();
439        let mut request = create_a_potential_cors_request(
440            Some(window.webview_id()),
441            img_url.clone(),
442            Destination::Image,
443            cors_setting_for_element(self.upcast()),
444            None,
445            global.get_referrer(),
446        )
447        .with_global_scope(&global)
448        .referrer_policy(referrer_policy_for_element(self.upcast()));
449
450        if self.uses_srcset_or_picture() {
451            request = request.initiator(Initiator::ImageSet);
452        }
453
454        // This is a background load because the load blocker already fulfills the
455        // purpose of delaying the document's load event.
456        document.fetch_background(request, context);
457    }

459    // Steps common to when an image has been loaded.
460    fn handle_loaded_image(&self, image: Image, url: ServoUrl, can_gc: CanGc) {
461        self.current_request.borrow_mut().metadata = Some(image.metadata());
462        self.current_request.borrow_mut().final_url = Some(url);
463        self.current_request.borrow_mut().image = Some(image);

471    /// <https://html.spec.whatwg.org/multipage/#update-the-image-data>
472    fn process_image_response(&self, image: ImageResponse, can_gc: CanGc) {
473        // Step 27. As soon as possible, jump to the first applicable entry from the following list:
474
475        // TODO => "If the resource type is multipart/x-mixed-replace"
476
477        // => "If the resource type and data corresponds to a supported image format ...""
478        let (trigger_image_load, trigger_image_error) = match (image, self.image_request.get()) {
479            (ImageResponse::Loaded(image, url), ImageRequestPhase::Current) => {
480                self.handle_loaded_image(image, url, can_gc);
481                (true, false)
482            },
483            (ImageResponse::Loaded(image, url), ImageRequestPhase::Pending) => {
484                self.abort_request(State::Unavailable, ImageRequestPhase::Pending, can_gc);
485                self.image_request.set(ImageRequestPhase::Current);
486                self.handle_loaded_image(image, url, can_gc);
487                (true, false)
488            },
489            (ImageResponse::MetadataLoaded(meta), ImageRequestPhase::Current) => {
490                // Otherwise, if the user agent is able to determine image request's image's width
491                // and height, and image request is the current request, prepare image request for
492                // presentation given the img element and set image request's state to partially
493                // available.
494                self.current_request.borrow_mut().state = State::PartiallyAvailable;
495                self.current_request.borrow_mut().metadata = Some(meta);
496                (false, false)
497            },
498            (ImageResponse::MetadataLoaded(_), ImageRequestPhase::Pending) => {
499                // If the user agent is able to determine image request's image's width and height,
500                // and image request is the pending request, set image request's state to partially
501                // available.
502                self.pending_request.borrow_mut().state = State::PartiallyAvailable;
503                (false, false)
504            },
505            (ImageResponse::FailedToLoadOrDecode, ImageRequestPhase::Current) => {
506                // Otherwise, if the user agent is able to determine that image request's image is
507                // corrupted in some fatal way such that the image dimensions cannot be obtained,
508                // and image request is the current request:
509
510                // Step 1. Abort the image request for image request.
511                self.abort_request(State::Broken, ImageRequestPhase::Current, can_gc);
512
513                self.load_broken_image_icon();
514
515                // Step 2. If maybe omit events is not set or previousURL is not equal to urlString,
516                // then fire an event named error at the img element.
517                // TODO: Add missing `maybe omit events` flag and previousURL.
518                (false, true)
519            },
520            (ImageResponse::FailedToLoadOrDecode, ImageRequestPhase::Pending) => {
521                // Otherwise, if the user agent is able to determine that image request's image is
522                // corrupted in some fatal way such that the image dimensions cannot be obtained,
523                // and image request is the pending request:
524
525                // Step 1. Abort the image request for the current request and the pending request.
526                self.abort_request(State::Broken, ImageRequestPhase::Current, can_gc);
527                self.abort_request(State::Unavailable, ImageRequestPhase::Pending, can_gc);
528
529                // Step 2. Upgrade the pending request to the current request.
530                mem::swap(
531                    &mut *self.current_request.borrow_mut(),
532                    &mut *self.pending_request.borrow_mut(),
533                );
534                self.image_request.set(ImageRequestPhase::Current);
535
536                // Step 3. Set the current request's state to broken.
537                self.current_request.borrow_mut().state = State::Broken;
538
539                self.load_broken_image_icon();
540
541                // Step 4. Fire an event named error at the img element.
542                (false, true)
543            },
544        };
545
546        // Fire image.onload and loadend
547        if trigger_image_load {
548            // TODO: https://html.spec.whatwg.org/multipage/#fire-a-progress-event-or-event
549            self.upcast::<EventTarget>()
550                .fire_event(atom!("load"), can_gc);
551            self.upcast::<EventTarget>()
552                .fire_event(atom!("loadend"), can_gc);
553        }
554
555        // Fire image.onerror
556        if trigger_image_error {
557            self.upcast::<EventTarget>()
558                .fire_event(atom!("error"), can_gc);
559            self.upcast::<EventTarget>()
560                .fire_event(atom!("loadend"), can_gc);
561        }
562    }

564    /// The response part of

566    fn process_image_response_for_environment_change(

577                self.pending_request.borrow_mut().final_url = Some(url);

580                    selected_source,
581                    generation,
582                    selected_pixel_density,
583                );
584            },
585            ImageResponse::FailedToLoadOrDecode => {
586                // > Step 15.6: If response's unsafe response is a network error or if the
587                // > image format is unsupported (as determined by applying the image
588                // > sniffing rules, again as mentioned earlier), or if the user agent is
589                // > able to determine that image request's image is corrupted in some fatal
590                // > way such that the image dimensions cannot be obtained, or if the
591                // > resource type is multipart/x-mixed-replace, then set the pending
592                // > request to null and abort these steps.
593                self.abort_request(State::Unavailable, ImageRequestPhase::Pending, can_gc);
594            },
595            ImageResponse::MetadataLoaded(meta) => {
596                self.pending_request.borrow_mut().metadata = Some(meta);
597            },
598        };
599    }

601    /// <https://html.spec.whatwg.org/multipage/#abort-the-image-request>
602    fn abort_request(&self, state: State, request: ImageRequestPhase, can_gc: CanGc) {
603        let mut request = match request {
604            ImageRequestPhase::Current => self.current_request.borrow_mut(),
605            ImageRequestPhase::Pending => self.pending_request.borrow_mut(),
606        };
607        LoadBlocker::terminate(&request.blocker, can_gc);
608        request.state = state;
609        request.image = None;
610        request.metadata = None;
611        request.current_pixel_density = None;
612
613        if matches!(state, State::Broken) {
614            self.reject_image_decode_promises();
615        } else if matches!(state, State::CompletelyAvailable) {
616            self.resolve_image_decode_promises();
617        }
618    }

620    /// <https://html.spec.whatwg.org/multipage/#create-a-source-set>
621    fn create_source_set(&self) -> SourceSet {
622        let element = self.upcast::<Element>();
623
624        // Step 1. Let source set be an empty source set.
625        let mut source_set = SourceSet::new();
626
627        // Step 2. If srcset is not an empty string, then set source set to the result of parsing
628        // srcset.
629        if let Some(srcset) = element.get_attribute(&ns!(), &local_name!("srcset")) {
630            source_set.image_sources = parse_a_srcset_attribute(&srcset.value());
631        }
632
633        // Step 3. Set source set's source size to the result of parsing sizes with img.
634        if let Some(sizes) = element.get_attribute(&ns!(), &local_name!("sizes")) {
635            source_set.source_size = parse_a_sizes_attribute(&sizes.value());
636        }
637
638        // Step 4. If default source is not the empty string and source set does not contain an
639        // image source with a pixel density descriptor value of 1, and no image source with a width
640        // descriptor, append default source to source set.
641        let src = element.get_string_attribute(&local_name!("src"));
642        let no_density_source_of_1 = source_set
643            .image_sources
644            .iter()
645            .all(|source| source.descriptor.density != Some(1.));
646        let no_width_descriptor = source_set
647            .image_sources
648            .iter()
649            .all(|source| source.descriptor.width.is_none());
650        if !src.is_empty() && no_density_source_of_1 && no_width_descriptor {
651            source_set.image_sources.push(ImageSource {
652                url: src.to_string(),
653                descriptor: Descriptor {
654                    width: None,
655                    density: None,
656                },
657            })
658        }
659
660        // Step 5. Normalize the source densities of source set.
661        self.normalise_source_densities(&mut source_set);
662
663        // Step 6. Return source set.
664        source_set
665    }

667    /// <https://html.spec.whatwg.org/multipage/#update-the-source-set>
668    fn update_source_set(&self) {
669        // Step 1. Set el's source set to an empty source set.
670        *self.source_set.borrow_mut() = SourceSet::new();
671
672        // Step 2. Let elements be  el .
673        // Step 3. If el is an img element whose parent node is a picture element, then replace the
674        // contents of elements with el's parent node's child elements, retaining relative order.
675        // Step 4. Let img be el if el is an img element, otherwise null.
676        let elem = self.upcast::<Element>();
677        let parent = elem.upcast::<Node>().GetParentElement();
678        let elements = match parent.as_ref() {
679            Some(p) => {
680                if p.is::<HTMLPictureElement>() {
681                    p.upcast::<Node>()
682                        .children()
683                        .filter_map(DomRoot::downcast::<Element>)
684                        .map(|n| DomRoot::from_ref(&*n))
685                        .collect()
686                } else {
687                    vec![DomRoot::from_ref(elem)]
688                }
689            },
690            None => vec![DomRoot::from_ref(elem)],
691        };
692
693        // Step 5. For each child in elements:
694        for element in &elements {
695            // Step 5.1. If child is el:
696            if *element == DomRoot::from_ref(elem) {
697                // Step 5.1.10. Set el's source set to the result of creating a source set given
698                // default source, srcset, sizes, and img.
699                *self.source_set.borrow_mut() = self.create_source_set();
700
701                // Step 5.1.11. Return.
702                return;
703            }
704            // Step 5.2. If child is not a source element, then continue.
705            if !element.is::<HTMLSourceElement>() {
706                continue;
707            }
708
709            let mut source_set = SourceSet::new();
710
711            // Step 5.3. If child does not have a srcset attribute, continue to the next child.
712            // Step 5.4. Parse child's srcset attribute and let source set be the returned source
713            // set.
714            match element.get_attribute(&ns!(), &local_name!("srcset")) {
715                Some(srcset) => {
716                    source_set.image_sources = parse_a_srcset_attribute(&srcset.value());
717                },
718                _ => continue,
719            }
720
721            // Step 5.5. If source set has zero image sources, continue to the next child.
722            if source_set.image_sources.is_empty() {
723                continue;
724            }
725
726            // Step 5.6. If child has a media attribute, and its value does not match the
727            // environment, continue to the next child.
728            if let Some(media) = element.get_attribute(&ns!(), &local_name!("media")) {
729                if !MediaList::matches_environment(&element.owner_document(), &media.value()) {
730                    continue;
731                }
732            }
733
734            // Step 5.7. Parse child's sizes attribute with img, and let source set's source size be
735            // the returned value.
736            if let Some(sizes) = element.get_attribute(&ns!(), &local_name!("sizes")) {
737                source_set.source_size = parse_a_sizes_attribute(&sizes.value());
738            }
739
740            // Step 5.8. If child has a type attribute, and its value is an unknown or unsupported
741            // MIME type, continue to the next child.
742            if let Some(type_) = element.get_attribute(&ns!(), &local_name!("type")) {
743                if !is_supported_image_mime_type(&type_.value()) {
744                    continue;
745                }
746            }
747
748            // Step 5.9. If child has width or height attributes, set el's dimension attribute
749            // source to child. Otherwise, set el's dimension attribute source to el.
750            if element
751                .get_attribute(&ns!(), &local_name!("width"))
752                .is_some() ||
753                element
754                    .get_attribute(&ns!(), &local_name!("height"))
755                    .is_some()
756            {
757                self.dimension_attribute_source.set(Some(element));
758            } else {
759                self.dimension_attribute_source.set(Some(elem));
760            }
761
762            // Step 5.10. Normalize the source densities of source set.
763            self.normalise_source_densities(&mut source_set);
764
765            // Step 5.11. Set el's source set to source set.
766            *self.source_set.borrow_mut() = source_set;
767
768            // Step 5.12. Return.
769            return;
770        }
771    }

779    /// <https://html.spec.whatwg.org/multipage/#normalise-the-source-densities>
780    fn normalise_source_densities(&self, source_set: &mut SourceSet) {
781        // Step 1. Let source size be source set's source size.
782        let source_size = self.evaluate_source_size_list(&source_set.source_size);
783
784        // Step 2. For each image source in source set:
785        for image_source in &mut source_set.image_sources {
786            // Step 2.1. If the image source has a pixel density descriptor, continue to the next
787            // image source.
788            if image_source.descriptor.density.is_some() {
789                continue;
790            }
791
792            // Step 2.2. Otherwise, if the image source has a width descriptor, replace the width
793            // descriptor with a pixel density descriptor with a value of the width descriptor value
794            // divided by source size and a unit of x.
795            if image_source.descriptor.width.is_some() {
796                let width = image_source.descriptor.width.unwrap();
797                image_source.descriptor.density = Some(width as f64 / source_size.to_f64_px());
798            } else {
799                // Step 2.3. Otherwise, give the image source a pixel density descriptor of 1x.
800                image_source.descriptor.density = Some(1_f64);
801            }
802        }
803    }

805    /// <https://html.spec.whatwg.org/multipage/#select-an-image-source>
806    fn select_image_source(&self) -> Option<(USVString, f64)> {
807        // Step 1. Update the source set for el.
808        self.update_source_set();
809
810        // Step 2. If el's source set is empty, return null as the URL and undefined as the pixel
811        // density.
812        if self.source_set.borrow().image_sources.is_empty() {
813            return None;
814        }
815
816        // Step 3. Return the result of selecting an image from el's source set.
817        self.select_image_source_from_source_set()

820    /// <https://html.spec.whatwg.org/multipage/#select-an-image-source-from-a-source-set>
821    fn select_image_source_from_source_set(&self) -> Option<(USVString, f64)> {
822        // Step 1. If an entry b in sourceSet has the same associated pixel density descriptor as an
823        // earlier entry a in sourceSet, then remove entry b. Repeat this step until none of the
824        // entries in sourceSet have the same associated pixel density descriptor as an earlier
825        // entry.
826        let source_set = self.source_set.borrow();
827        let len = source_set.image_sources.len();
828
829        // Using FxHash is ok here as the indices are just 0..len
830        let mut repeat_indices = FxHashSet::default();
831        for outer_index in 0..len {
832            if repeat_indices.contains(&outer_index) {
833                continue;
834            }
835            let imgsource = &source_set.image_sources[outer_index];
836            let pixel_density = imgsource.descriptor.density.unwrap();
837            for inner_index in (outer_index + 1)..len {
838                let imgsource2 = &source_set.image_sources[inner_index];
839                if pixel_density == imgsource2.descriptor.density.unwrap() {
840                    repeat_indices.insert(inner_index);
841                }
842            }
843        }
844
845        let mut max = (0f64, 0);
846        let img_sources = &mut vec![];
847        for (index, image_source) in source_set.image_sources.iter().enumerate() {
848            if repeat_indices.contains(&index) {
849                continue;
850            }
851            let den = image_source.descriptor.density.unwrap();
852            if max.0 < den {
853                max = (den, img_sources.len());
854            }
855            img_sources.push(image_source);
856        }
857
858        // Step 2. In an implementation-defined manner, choose one image source from sourceSet. Let
859        // selectedSource be this choice.
860        let mut best_candidate = max;
861        let device_pixel_ratio = self
862            .owner_document()
863            .window()
864            .viewport_details()
865            .hidpi_scale_factor
866            .get() as f64;
867        for (index, image_source) in img_sources.iter().enumerate() {
868            let current_den = image_source.descriptor.density.unwrap();
869            if current_den < best_candidate.0 && current_den >= device_pixel_ratio {
870                best_candidate = (current_den, index);
871            }
872        }
873        let selected_source = img_sources.remove(best_candidate.1).clone();
874
875        // Step 3. Return selectedSource and its associated pixel density.
876        Some((
877            USVString(selected_source.url),
878            selected_source.descriptor.density.unwrap(),
879        ))
880    }

899    /// <https://html.spec.whatwg.org/multipage/#update-the-image-data>
900    fn prepare_image_request(
901        &self,
902        selected_source: &USVString,
903        selected_pixel_density: f64,
904        image_url: &ServoUrl,
905        can_gc: CanGc,
906    ) {
907        match self.image_request.get() {
908            ImageRequestPhase::Pending => {
909                // Step 14. If the pending request is not null and urlString is the same as the
910                // pending request's current URL, then return.
911                if self
912                    .pending_request
913                    .borrow()
914                    .parsed_url
915                    .as_ref()
916                    .is_some_and(|parsed_url| *parsed_url == *image_url)
917                {
918                    return;
919                }
920            },
921            ImageRequestPhase::Current => {
922                // Step 16. Abort the image request for the pending request.
923                self.abort_request(State::Unavailable, ImageRequestPhase::Pending, can_gc);
924
925                // Step 17. Set image request to a new image request whose current URL is urlString.
926
927                let mut current_request = self.current_request.borrow_mut();
928                let mut pending_request = self.pending_request.borrow_mut();
929
930                match (current_request.parsed_url.as_ref(), current_request.state) {
931                    (Some(parsed_url), State::PartiallyAvailable) => {
932                        // Step 15. If urlString is the same as the current request's current URL
933                        // and the current request's state is partially available, then abort the
934                        // image request for the pending request, queue an element task on the DOM
935                        // manipulation task source given the img element to restart the animation
936                        // if restart animation is set, and return.
937                        if *parsed_url == *image_url {
938                            // TODO: queue a task to restart animation, if restart-animation is set
939                            return;
940                        }
941
942                        // Step 18. If the current request's state is unavailable or broken, then
943                        // set the current request to image request. Otherwise, set the pending
944                        // request to image request.
945                        self.image_request.set(ImageRequestPhase::Pending);
946                        self.init_image_request(
947                            &mut pending_request,
948                            image_url,
949                            selected_source,
950                            can_gc,
951                        );
952                        pending_request.current_pixel_density = Some(selected_pixel_density);
953                    },
954                    (_, State::Broken) | (_, State::Unavailable) => {
955                        // Step 18. If the current request's state is unavailable or broken, then
956                        // set the current request to image request. Otherwise, set the pending
957                        // request to image request.
958                        self.init_image_request(
959                            &mut current_request,
960                            image_url,
961                            selected_source,
962                            can_gc,
963                        );
964                        current_request.current_pixel_density = Some(selected_pixel_density);
965                        self.reject_image_decode_promises();
966                    },
967                    (_, _) => {
968                        // Step 18. If the current request's state is unavailable or broken, then
969                        // set the current request to image request. Otherwise, set the pending
970                        // request to image request.
971                        self.image_request.set(ImageRequestPhase::Pending);
972                        self.init_image_request(
973                            &mut pending_request,
974                            image_url,
975                            selected_source,
976                            can_gc,
977                        );
978                        pending_request.current_pixel_density = Some(selected_pixel_density);
979                    },
980                }
981            },
982        }
983
984        self.fetch_image(image_url, can_gc);
985    }

1199                }

1207            generation: self.generation.get(),

1224    /// <https://html.spec.whatwg.org/multipage/#img-environment-changes>
1225    fn react_to_environment_changes_sync_steps(&self, generation: u32, can_gc: CanGc) {
1226        let document = self.owner_document();
1227        let has_pending_request = matches!(self.image_request.get(), ImageRequestPhase::Pending);
1228
1229        // Step 2. If the img element does not use srcset or picture, its node document is not fully
1230        // active, it has image data whose resource type is multipart/x-mixed-replace, or its
1231        // pending request is not null, then return.
1232        if !document.is_active() || !self.uses_srcset_or_picture() || has_pending_request {
1233            return;
1234        }
1235
1236        // Step 3. Let selected source and selected pixel density be the URL and pixel density that
1237        // results from selecting an image source, respectively.
1238        let Some((selected_source, selected_pixel_density)) = self.select_image_source() else {
1239            // Step 4. If selected source is null, then return.
1240            return;
1241        };
1242
1243        // Step 5. If selected source and selected pixel density are the same as the element's last
1244        // selected source and current pixel density, then return.
1245        let mut same_selected_source = self
1246            .last_selected_source
1247            .borrow()
1248            .as_ref()
1249            .is_some_and(|source| *source == selected_source);
1250
1251        // There are missing steps for the element's last selected source in specification so let's
1252        // check the current request's current URL as well.
1253        // <https://github.com/whatwg/html/issues/5060>
1254        same_selected_source = same_selected_source ||
1255            self.current_request
1256                .borrow()
1257                .source_url
1258                .as_ref()
1259                .is_some_and(|source| *source == selected_source);
1260
1261        let same_selected_pixel_density = self
1262            .current_request
1263            .borrow()
1264            .current_pixel_density
1265            .is_some_and(|pixel_density| pixel_density == selected_pixel_density);
1266
1267        if same_selected_source && same_selected_pixel_density {
1268            return;
1269        }
1270
1271        // Step 6. Let urlString be the result of encoding-parsing-and-serializing a URL given
1272        // selected source, relative to the element's node document.
1273        // Step 7. If urlString is failure, then return.
1274        let Ok(image_url) = document.base_url().join(&selected_source) else {
1275            return;
1276        };
1277
1278        // Step 13. Set the element's pending request to image request.
1279        self.image_request.set(ImageRequestPhase::Pending);
1280        self.init_image_request(
1281            &mut self.pending_request.borrow_mut(),
1282            &image_url,
1283            &selected_source,
1284            can_gc,
1285        );
1286
1287        // Step 15. If the list of available images contains an entry for key, then set image
1288        // request's image data to that of the entry. Continue to the next step.
1289        let window = self.owner_window();
1290        let cache_result = window.image_cache().get_cached_image_status(
1291            image_url.clone(),
1292            window.origin().immutable().clone(),
1293            cors_setting_for_element(self.upcast()),
1294        );
1295
1296        let change_type = ChangeType::Environment {
1297            selected_source: selected_source.clone(),
1298            selected_pixel_density,
1299        };
1300
1301        match cache_result {
1302            ImageCacheResult::Available(ImageOrMetadataAvailable::ImageAvailable { .. }) => {
1303                self.finish_reacting_to_environment_change(
1304                    selected_source,
1305                    generation,
1306                    selected_pixel_density,
1307                );
1308            },
1309            ImageCacheResult::Available(ImageOrMetadataAvailable::MetadataAvailable(m, id)) => {
1310                self.process_image_response_for_environment_change(
1311                    ImageResponse::MetadataLoaded(m),
1312                    selected_source,
1313                    generation,
1314                    selected_pixel_density,
1315                    can_gc,
1316                );
1317                self.register_image_cache_callback(id, change_type);
1318            },
1319            ImageCacheResult::FailedToLoadOrDecode => {
1320                self.process_image_response_for_environment_change(
1321                    ImageResponse::FailedToLoadOrDecode,
1322                    selected_source,
1323                    generation,
1324                    selected_pixel_density,
1325                    can_gc,
1326                );
1327            },
1328            ImageCacheResult::ReadyForRequest(id) => {
1329                self.fetch_request(&image_url, id);
1330                self.register_image_cache_callback(id, change_type);
1331            },
1332            ImageCacheResult::Pending(id) => {
1333                self.register_image_cache_callback(id, change_type);
1334            },
1335        }
1336    }

1338    /// <https://html.spec.whatwg.org/multipage/#dom-img-decode>
1339    fn react_to_decode_image_sync_steps(&self, promise: Rc<Promise>, can_gc: CanGc) {
1340        // Step 2.2. If any of the following are true: this's node document is not fully active; or
1341        // this's current request's state is broken, then reject promise with an "EncodingError"
1342        // DOMException.
1343        if !self.owner_document().is_fully_active() ||
1344            matches!(self.current_request.borrow().state, State::Broken)
1345        {
1346            promise.reject_error(Error::Encoding(None), can_gc);
1347        } else if matches!(
1348            self.current_request.borrow().state,
1349            State::CompletelyAvailable
1350        ) {
1351            // this doesn't follow the spec, but it's been discussed in <https://github.com/whatwg/html/issues/4217>
1352            promise.resolve_native(&(), can_gc);
1353        } else if matches!(self.current_request.borrow().state, State::Unavailable) &&
1354            self.current_request.borrow().source_url.is_none()
1355        {
1356            // Note: Despite being not explicitly stated in the specification but if current
1357            // request's state is unavailable and current URL is empty string (<img> without "src"
1358            // and "srcset" attributes) then reject promise with an "EncodingError" DOMException.
1359            // <https://github.com/whatwg/html/issues/11769>
1360            promise.reject_error(Error::Encoding(None), can_gc);
1361        } else {
1362            self.image_decode_promises.borrow_mut().push(promise);
1363        }
1364    }

1366    /// <https://html.spec.whatwg.org/multipage/#dom-img-decode>
1367    fn resolve_image_decode_promises(&self) {
1368        if self.image_decode_promises.borrow().is_empty() {
1369            return;
1370        }
1371
1372        // Step 2.3. If the decoding process completes successfully, then queue a global task on the
1373        // DOM manipulation task source with global to resolve promise with undefined.
1374        let trusted_image_decode_promises: Vec<TrustedPromise> = self
1375            .image_decode_promises
1376            .borrow()
1377            .iter()
1378            .map(|promise| TrustedPromise::new(promise.clone()))
1379            .collect();
1380
1381        self.image_decode_promises.borrow_mut().clear();
1382
1383        self.owner_global()
1384            .task_manager()
1385            .dom_manipulation_task_source()
1386            .queue(task!(fulfill_image_decode_promises: move || {
1387                for trusted_promise in trusted_image_decode_promises {
1388                    trusted_promise.root().resolve_native(&(), CanGc::note());
1389                }
1390            }));
1391    }

1393    /// <https://html.spec.whatwg.org/multipage/#dom-img-decode>
1394    fn reject_image_decode_promises(&self) {
1395        if self.image_decode_promises.borrow().is_empty() {
1396            return;
1397        }
1398
1399        // Step 2.3. Queue a global task on the DOM manipulation task source with global to reject
1400        // promise with an "EncodingError" DOMException.
1401        let trusted_image_decode_promises: Vec<TrustedPromise> = self
1402            .image_decode_promises
1403            .borrow()
1404            .iter()
1405            .map(|promise| TrustedPromise::new(promise.clone()))
1406            .collect();
1407
1408        self.image_decode_promises.borrow_mut().clear();
1409
1410        self.owner_global()
1411            .task_manager()
1412            .dom_manipulation_task_source()
1413            .queue(task!(reject_image_decode_promises: move || {
1414                for trusted_promise in trusted_image_decode_promises {
1415                    trusted_promise.root().reject_error(Error::Encoding(None), CanGc::note());
1416                }
1417            }));
1418    }

1420    /// <https://html.spec.whatwg.org/multipage/#img-environment-changes>
1421    fn finish_reacting_to_environment_change(
1422        &self,
1423        selected_source: USVString,
1424        generation: u32,
1425        selected_pixel_density: f64,
1426    ) {
1427        // Step 16. Queue an element task on the DOM manipulation task source given the img element
1428        // and the following steps:
1429        let this = Trusted::new(self);
1430
1431        self.owner_global().task_manager().dom_manipulation_task_source().queue(
1432            task!(image_load_event: move || {
1433                let this = this.root();
1434
1435                // Step 16.1. If the img element has experienced relevant mutations since this
1436                // algorithm started, then set the pending request to null and abort these steps.
1437                if this.generation.get() != generation {
1438                    this.abort_request(State::Unavailable, ImageRequestPhase::Pending, CanGc::note());
1439                    this.image_request.set(ImageRequestPhase::Current);
1440                    return;
1441                }
1442
1443                // Step 16.2. Set the img element's last selected source to selected source and the
1444                // img element's current pixel density to selected pixel density.
1445                *this.last_selected_source.borrow_mut() = Some(selected_source);
1446
1447                {
1448                    let mut pending_request = this.pending_request.borrow_mut();
1449
1450                    // Step 16.3. Set the image request's state to completely available.
1451                    pending_request.state = State::CompletelyAvailable;
1452
1453                    pending_request.current_pixel_density = Some(selected_pixel_density);
1454
1455                    // Step 16.4. Add the image to the list of available images using the key key,
1456                    // with the ignore higher-layer caching flag set.
1457                    // Already a part of the list of available images due to Step 15.
1458
1459                    // Step 16.5. Upgrade the pending request to the current request.
1460                    mem::swap(&mut *this.current_request.borrow_mut(), &mut *pending_request);
1461                }
1462
1463                this.abort_request(State::Unavailable, ImageRequestPhase::Pending, CanGc::note());
1464                this.image_request.set(ImageRequestPhase::Current);
1465
1466                // TODO Step 16.6. Prepare image request for presentation given the img element.
1467                this.upcast::<Node>().dirty(NodeDamage::Other);
1468
1469                // Step 16.7. Fire an event named load at the img element.
1470                this.upcast::<EventTarget>().fire_event(atom!("load"), CanGc::note());
1471            })
1472        );
1473    }

1632impl MicrotaskRunnable for ImageElementMicrotask {
1633    fn handler(&self, can_gc: CanGc) {
1634        match *self {
1635            ImageElementMicrotask::UpdateImageData {
1636                ref elem,
1637                ref generation,
1638            } => {
1639                // <https://html.spec.whatwg.org/multipage/#update-the-image-data>
1640                // Step 9. If another instance of this algorithm for this img element was started
1641                // after this instance (even if it aborted and is no longer running), then return.

1643                    elem.update_the_image_data_sync_steps(can_gc);
1644                }
1645            },
1646            ImageElementMicrotask::EnvironmentChanges {
1647                ref elem,
1648                ref generation,

1651            },
1652            ImageElementMicrotask::Decode {
1653                ref elem,
1654                ref promise,
1655            } => {
1656                elem.react_to_decode_image_sync_steps(promise.clone(), can_gc);
1657            },
1658        }
1659    }

2187/// <https://html.spec.whatwg.org/multipage/#parsing-a-srcset-attribute>.
2188pub fn parse_a_srcset_attribute(input: &str) -> Vec<ImageSource> {
2189    // > 1. Let input be the value passed to this algorithm.
2190    // > 2. Let position be a pointer into input, initially pointing at the start of the string.
2191    let mut current_index = 0;
2192
2193    // > 3. Let candidates be an initially empty source set.
2194    let mut candidates = vec![];
2195    while current_index < input.len() {
2196        let remaining_string = &input[current_index..];
2197
2198        // > 4. Splitting loop: Collect a sequence of code points that are ASCII whitespace or
2199        // > U+002C COMMA characters from input given position. If any U+002C COMMA
2200        // > characters were collected, that is a parse error.
2201        // NOTE: A parse error indicating a non-fatal mismatch between the input and the
2202        // requirements will be silently ignored to match the behavior of other browsers.
2203        // <https://html.spec.whatwg.org/multipage/#concept-microsyntax-parse-error>
2204        let (collected_characters, string_after_whitespace) =
2205            collect_sequence_characters(remaining_string, |character| {
2206                *character == ',' || character.is_ascii_whitespace()
2207            });
2208
2209        // Add the length of collected whitespace, to find the start of the URL we are going
2210        // to parse.
2211        current_index += collected_characters.len();
2212

2214        if string_after_whitespace.is_empty() {
2215            return candidates;
2216        }
2217
2218        // 6. Collect a sequence of code points that are not ASCII whitespace from input
2219        // given position, and let that be url.
2220        let (url, _) =
2221            collect_sequence_characters(string_after_whitespace, |c| !char::is_ascii_whitespace(c));

2224        // of the string to prase.
2225        current_index += url.len();
2226
2227        // 7. Let descriptors be a new empty list.
2228        let mut descriptors = Vec::new();
2229
2230        // > 8. If url ends with U+002C (,), then:
2231        // >    1. Remove all trailing U+002C COMMA characters from url. If this removed
2232        // >       more than one character, that is a parse error.

2235                url: url.trim_end_matches(',').into(),

2237                    width: None,
2238                    density: None,

2247        let descriptors_string = &input[current_index..];

2250                character.is_ascii_whitespace()
2251            });

2255        let mut current_descriptor = String::new();

2258        let mut state = ParseState::InDescriptor;

2261        // > state. For the purpose of this step, "EOF" is a special character representing

2274                    // > state to after descriptor.

2276                        descriptors.push(current_descriptor);
2277                        current_descriptor = String::new();

2286                        descriptors.push(current_descriptor);
2287                    }
2288                    break;
2289                },
2290                (ParseState::InDescriptor, Some('(')) => {
2291                    // > Append c to current descriptor. Set state to in parens.
2292                    current_descriptor.push('(');
2293                    state = ParseState::InParens;
2294                },
2295                (ParseState::InDescriptor, Some(character)) => {
2296                    // > Append c to current descriptor.
2297                    current_descriptor.push(character);
2298                },

2366                // > 2. If width and density are not both absent, then let error be yes.
2367                // > 3. Apply the rules for parsing non-negative integers to the descriptor.
2368                // >    If the result is 0, let error be yes. Otherwise, let width be the result.
2369                'w' if is_valid_non_negative_integer_string(first_part_of_string) &&
2370                    density.is_none() &&
2371                    width.is_none() =>
2372                {
2373                    match parse_unsigned_integer(first_part_of_string.chars()) {
2374                        Ok(number) if number > 0 => {
2375                            width = Some(number);
2376                            continue;
2377                        },
2378                        _ => error = true,
2379                    }
2380                },
2381
2382                // > If the descriptor consists of a valid floating-point number followed by a
2383                // > U+0078 LATIN SMALL LETTER X character
2384                // > 1. If width, density and future-compat-h are not all absent, then let
2385                // >    error be yes.
2386                // > 2. Apply the rules for parsing floating-point number values to the
2387                // >    descriptor. If the result is less than 0, let error be yes. Otherwise, let
2388                // >    density be the result.
2389                //
2390                // The HTML specification has a procedure for parsing floats that is different enough from
2391                // the one that stylo uses, that it's better to use Rust's float parser here. This is
2392                // what Gecko does, but it also checks to see if the number is a valid HTML-spec compliant
2393                // number first. Not doing that means that we might be parsing numbers that otherwise
2394                // wouldn't parse.
2395                'x' if is_valid_floating_point_number_string(first_part_of_string) &&
2396                    width.is_none() &&
2397                    density.is_none() &&
2398                    future_compat_h.is_none() =>
2399                {
2400                    match first_part_of_string.parse::<f64>() {
2401                        Ok(number) if number.is_finite() && number >= 0. => {
2402                            density = Some(number);
2403                            continue;
2404                        },
2405                        _ => error = true,
2406                    }
2407                },
2408
2409                // > If the descriptor consists of a valid non-negative integer followed by a
2410                // > U+0068 LATIN SMALL LETTER H character
2411                // >   This is a parse error.
2412                // > 1. If future-compat-h and density are not both absent, then let error be
2413                // >    yes.
2414                // > 2. Apply the rules for parsing non-negative integers to the descriptor.
2415                // >    If the result is 0, let error be yes. Otherwise, let future-compat-h be the
2416                // >    result.
2417                'h' if is_valid_non_negative_integer_string(first_part_of_string) &&
2418                    future_compat_h.is_none() &&
2419                    density.is_none() =>
2420                {
2421                    match parse_unsigned_integer(first_part_of_string.chars()) {
2422                        Ok(number) if number > 0 => {
2423                            future_compat_h = Some(number);
2424                            continue;
2425                        },
2426                        _ => error = true,
2427                    }
2428                },
2429
2430                // > Anything else
2431                // >  Let error be yes.
2432                _ => error = true,
2433            }
2434
2435            if error {
2436                break;
2437            }
2438        }
2439
2440        // > 14. If future-compat-h is not absent and width is absent, let error be yes.
2441        if future_compat_h.is_some() && width.is_none() {
2442            error = true;
2443        }
2444
2445        // Step 15. If error is still no, then append a new image source to candidates whose URL is
2446        // url, associated with a width width if not absent and a pixel density density if not
2447        // absent. Otherwise, there is a parse error.
2448        if !error {
2449            let image_source = ImageSource {
2450                url: url.into(),
2451                descriptor: Descriptor { width, density },
2452            };
2453            candidates.push(image_source);
2454        }
2455
2456        // Step 16. Return to the step labeled splitting loop.
2457    }
2458    candidates
2459}

2470/// Returns true if the given image MIME type is supported.
2471fn is_supported_image_mime_type(input: &str) -> bool {
2472    // Remove any leading and trailing HTTP whitespace from input.
2473    let mime_type = input.trim();
2474
2475    // <https://mimesniff.spec.whatwg.org/#mime-type-essence>
2476    let mime_type_essence = match mime_type.find(';') {
2477        Some(semi) => &mime_type[..semi],
2478        _ => mime_type,
2479    };
2480
2481    // The HTML specification says the type attribute may be present and if present, the value
2482    // must be a valid MIME type string. However an empty type attribute is implicitly supported
2483    // to match the behavior of other browsers.
2484    // <https://html.spec.whatwg.org/multipage/#attr-source-type>
2485    if mime_type_essence.is_empty() {
2486        return true;
2487    }
2488
2489    SUPPORTED_IMAGE_MIME_TYPES.contains(&mime_type_essence)
