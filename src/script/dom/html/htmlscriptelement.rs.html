261/// Final steps of <https://html.spec.whatwg.org/multipage/#prepare-the-script-element>
262fn finish_fetching_a_classic_script(
263    elem: &HTMLScriptElement,
264    script_kind: ExternalScriptKind,
265    url: ServoUrl,
266    load: ScriptResult,
267    can_gc: CanGc,
268) {
269    // Step 33. The "steps to run when the result is ready" for each type of script in 33.2-33.5.
270    // of https://html.spec.whatwg.org/multipage/#prepare-the-script-element
271    let document;
272
273    match script_kind {
274        ExternalScriptKind::Asap => {
275            document = elem.preparation_time_document.get().unwrap();
276            document.asap_script_loaded(elem, load, can_gc)
277        },
278        ExternalScriptKind::AsapInOrder => {
279            document = elem.preparation_time_document.get().unwrap();
280            document.asap_in_order_script_loaded(elem, load, can_gc)
281        },
282        ExternalScriptKind::Deferred => {
283            document = elem.parser_document.as_rooted();
284            document.deferred_script_loaded(elem, load, can_gc);
285        },
286        ExternalScriptKind::ParsingBlocking => {
287            document = elem.parser_document.as_rooted();
288            document.pending_parsing_blocking_script_loaded(elem, load, can_gc);
289        },
290    }
291
292    document.finish_load(LoadType::Script(url), can_gc);
293}

336    fn process_response(&mut self, _: RequestId, metadata: Result<FetchMetadata, NetworkError>) {
337        self.metadata = metadata.ok().map(|meta| {
338            self.response_was_cors_cross_origin = meta.is_cors_cross_origin();
339            match meta {
340                FetchMetadata::Unfiltered(m) => m,
341                FetchMetadata::Filtered { unsafe_, .. } => unsafe_,
342            }
343        });
344
345        let status = self
346            .metadata
347            .as_ref()
348            .map(|m| m.status.clone())
349            .unwrap_or_else(HttpStatus::new_error);
350
351        self.status = {
352            if status.is_error() {
353                Err(NetworkError::ResourceLoadError(
354                    "No http status code received".to_owned(),
355                ))
356            } else if status.is_success() {
357                Ok(())
358            } else {
359                Err(NetworkError::ResourceLoadError(format!(
360                    "HTTP error code {}",
361                    status.code()
362                )))
363            }
364        };
365    }

368        if self.status.is_ok() {

532/// <https://html.spec.whatwg.org/multipage/#fetch-a-classic-script>
533fn fetch_a_classic_script(
534    script: &HTMLScriptElement,
535    kind: ExternalScriptKind,
536    url: ServoUrl,
537    cors_setting: Option<CorsSettings>,
538    options: ScriptFetchOptions,
539    character_encoding: &'static Encoding,
540) {
541    // Step 1, 2.
542    let doc = script.owner_document();
543    let global = script.global();
544    let request =
545        script_fetch_request(doc.webview_id(), url.clone(), cors_setting, options.clone())
546            .with_global_scope(&global);
547
548    // TODO: Step 3, Add custom steps to perform fetch
549
550    let context = ClassicContext {
551        elem: Trusted::new(script),
552        kind,
553        character_encoding,
554        data: vec![],
555        metadata: None,
556        url: url.clone(),
557        status: Ok(()),
558        fetch_options: options,
559        response_was_cors_cross_origin: false,
560    };
561    doc.fetch(LoadType::Script(url), request, context);
562}

570    /// <https://w3c.github.io/trusted-types/dist/spec/#abstract-opdef-prepare-the-script-text>
571    fn prepare_the_script_text(&self, can_gc: CanGc) -> Fallible<()> {
572        // Step 1. If scripts script text value is not equal to its child text content,
573        // set scripts script text to the result of executing
574        // Get Trusted Type compliant string, with the following arguments:
575        if self.script_text.borrow().clone() != self.text() {
576            *self.script_text.borrow_mut() = TrustedScript::get_trusted_script_compliant_string(
577                &self.owner_global(),
578                self.Text(),
579                "HTMLScriptElement text",
580                can_gc,
581            )?;
582        }
583
584        Ok(())
585    }

587    /// <https://html.spec.whatwg.org/multipage/#prepare-the-script-element>
588    pub(crate) fn prepare(&self, introduction_type_override: Option<&'static CStr>, can_gc: CanGc) {
589        self.introduction_type_override
590            .set(introduction_type_override);
591
592        // Step 1. If el's already started is true, then return.
593        if self.already_started.get() {
594            return;
595        }
596
597        // Step 2. Let parser document be el's parser document.
598        // TODO
599
600        // Step 3. Set el's parser document to null.
601        let was_parser_inserted = self.parser_inserted.get();
602        self.parser_inserted.set(false);
603
604        // Step 4.
605        // If parser document is non-null and el does not have an async attribute, then set el's force async to true.
606        let element = self.upcast::<Element>();
607        let asynch = element.has_attribute(&local_name!("async"));
608        // Note: confusingly, this is done if the element does *not* have an "async" attribute.
609        if was_parser_inserted && !asynch {
610            self.non_blocking.set(true);
611        }
612
613        // Step 5. Execute the Prepare the script text algorithm on el.
614        // If that algorithm threw an error, then return.
615        if self.prepare_the_script_text(can_gc).is_err() {
616            return;
617        }
618        // Step 5a. Let source text be els script text value.
619        let text = self.script_text.borrow().clone();
620        // Step 6. If el has no src attribute, and source text is the empty string, then return.
621        if text.is_empty() && !element.has_attribute(&local_name!("src")) {
622            return;
623        }
624
625        // Step 7. If el is not connected, then return.
626        if !self.upcast::<Node>().is_connected() {
627            return;
628        }
629
630        let script_type = if let Some(ty) = self.get_script_type() {
631            // Step 9-11.
632            ty
633        } else {
634            // Step 12. Otherwise, return. (No script is executed, and el's type is left as null.)
635            return;
636        };
637
638        // Step 13.
639        // If parser document is non-null, then set el's parser document back to parser document and set el's force
640        // async to false.
641        if was_parser_inserted {
642            self.parser_inserted.set(true);
643            self.non_blocking.set(false);
644        }
645
646        // Step 14. Set el's already started to true.
647        self.already_started.set(true);
648
649        // Step 15. Set el's preparation-time document to its node document.
650        let doc = self.owner_document();
651        self.preparation_time_document.set(Some(&doc));
652
653        // Step 16.
654        // If parser document is non-null, and parser document is not equal to el's preparation-time document, then
655        // return.
656        if self.parser_inserted.get() && *self.parser_document != *doc {
657            return;
658        }
659
660        // Step 17. If scripting is disabled for el, then return.
661        if !doc.scripting_enabled() {
662            return;
663        }
664
665        // Step 18. If el has a nomodule content attribute and its type is "classic", then return.
666        if element.has_attribute(&local_name!("nomodule")) && script_type == ScriptType::Classic {
667            return;
668        }
669
670        let global = &doc.global();
671
672        // Step 19. CSP.
673        if !element.has_attribute(&local_name!("src")) &&
674            global
675                .get_csp_list()
676                .should_elements_inline_type_behavior_be_blocked(
677                    global,
678                    element,
679                    InlineCheckType::Script,
680                    &text.str(),
681                    self.line_number as u32,
682                )
683        {
684            warn!("Blocking inline script due to CSP");
685            return;
686        }
687
688        // Step 20. If el has an event attribute and a for attribute, and el's type is "classic", then:
689        if script_type == ScriptType::Classic {
690            let for_attribute = element.get_attribute(&ns!(), &local_name!("for"));
691            let event_attribute = element.get_attribute(&ns!(), &local_name!("event"));
692            if let (Some(ref for_attribute), Some(ref event_attribute)) =
693                (for_attribute, event_attribute)
694            {
695                let for_value = for_attribute.value().to_ascii_lowercase();
696                let for_value = for_value.trim_matches(HTML_SPACE_CHARACTERS);
697                if for_value != "window" {
698                    return;
699                }
700
701                let event_value = event_attribute.value().to_ascii_lowercase();
702                let event_value = event_value.trim_matches(HTML_SPACE_CHARACTERS);
703                if event_value != "onload" && event_value != "onload()" {
704                    return;
705                }
706            }
707        }
708
709        // Step 21. If el has a charset attribute, then let encoding be the result of getting
710        // an encoding from the value of the charset attribute.
711        // If el does not have a charset attribute, or if getting an encoding failed,
712        // then let encoding be el's node document's the encoding.
713        let encoding = element
714            .get_attribute(&ns!(), &local_name!("charset"))
715            .and_then(|charset| Encoding::for_label(charset.value().as_bytes()))
716            .unwrap_or_else(|| doc.encoding());
717
718        // Step 22. CORS setting.
719        let cors_setting = cors_setting_for_element(element);
720
721        // Step 23. Module script credentials mode.
722        let module_credentials_mode = match script_type {
723            ScriptType::Classic => CredentialsMode::CredentialsSameOrigin,
724            ScriptType::Module | ScriptType::ImportMap => reflect_cross_origin_attribute(element)
725                .map_or(
726                    CredentialsMode::CredentialsSameOrigin,
727                    |attr| match &*attr.str() {
728                        "use-credentials" => CredentialsMode::Include,
729                        "anonymous" => CredentialsMode::CredentialsSameOrigin,
730                        _ => CredentialsMode::CredentialsSameOrigin,
731                    },
732                ),
733        };
734
735        // Step 24. Let cryptographic nonce be el's [[CryptographicNonce]] internal slot's value.
736        let cryptographic_nonce = self.upcast::<Element>().nonce_value();
737
738        // Step 25. If el has an integrity attribute, then let integrity metadata be that attribute's value.
739        // Otherwise, let integrity metadata be the empty string.
740        let im_attribute = element.get_attribute(&ns!(), &local_name!("integrity"));
741        let integrity_val = im_attribute.as_ref().map(|a| a.value());
742        let integrity_metadata = match integrity_val {
743            Some(ref value) => &***value,
744            None => "",
745        };
746
747        // Step 26. Let referrer policy be the current state of el's referrerpolicy content attribute.
748        let referrer_policy = referrer_policy_for_element(self.upcast::<Element>());
749
750        // TODO: Step 27. Fetch priority.
751
752        // Step 28. Let parser metadata be "parser-inserted" if el is parser-inserted,
753        // and "not-parser-inserted" otherwise.
754        let parser_metadata = if self.parser_inserted.get() {
755            ParserMetadata::ParserInserted
756        } else {
757            ParserMetadata::NotParserInserted
758        };
759
760        // Step 29. Fetch options.
761        let options = ScriptFetchOptions {
762            cryptographic_nonce,
763            integrity_metadata: integrity_metadata.to_owned(),
764            parser_metadata,
765            referrer: self.global().get_referrer(),
766            referrer_policy,
767            credentials_mode: module_credentials_mode,
768        };
769
770        // Step 30. Let settings object be el's node document's relevant settings object.
771        // This is done by passing ModuleOwner in step 31.11 and step 32.2.
772        // What we actually need is global's import map eventually.
773
774        let base_url = doc.base_url();
775        if let Some(src) = element.get_attribute(&ns!(), &local_name!("src")) {
776            // Step 31. If el has a src content attribute, then:
777
778            // Step 31.1. If el's type is "importmap".
779            if script_type == ScriptType::ImportMap {
780                // then queue an element task on the DOM manipulation task source
781                // given el to fire an event named error at el, and return.
782                self.queue_error_event();
783                return;
784            }
785
786            // Step 31.2. Let src be the value of el's src attribute.
787            let src = src.value();
788
789            // Step 31.3. If src is the empty string.
790            if src.is_empty() {
791                self.queue_error_event();
792                return;
793            }
794
795            // Step 31.4. Set el's from an external file to true.
796            self.from_an_external_file.set(true);
797
798            // Step 31.5-31.6. Parse URL.
799            let url = match base_url.join(&src) {
800                Ok(url) => url,
801                Err(_) => {
802                    warn!("error parsing URL for script {}", &**src);
803                    self.queue_error_event();
804                    return;
805                },
806            };
807
808            // TODO:
809            // Step 31.7. If el is potentially render-blocking, then block rendering on el.
810            // Step 31.8. Set el's delaying the load event to true.
811            // Step 31.9. If el is currently render-blocking, then set options's render-blocking to true.
812
813            // Step 31.11. Switch on el's type:
814            match script_type {
815                ScriptType::Classic => {
816                    let kind = if element.has_attribute(&local_name!("defer")) &&
817                        was_parser_inserted &&
818                        !asynch
819                    {
820                        // Step 33.4: classic, has src, has defer, was parser-inserted, is not async.
821                        ExternalScriptKind::Deferred
822                    } else if was_parser_inserted && !asynch {
823                        // Step 33.5: classic, has src, was parser-inserted, is not async.
824                        ExternalScriptKind::ParsingBlocking
825                    } else if !asynch && !self.non_blocking.get() {
826                        // Step 33.3: classic, has src, is not async, is not non-blocking.
827                        ExternalScriptKind::AsapInOrder
828                    } else {
829                        // Step 33.2: classic, has src.
830                        ExternalScriptKind::Asap
831                    };
832
833                    // Step 31.11. Fetch a classic script.
834                    fetch_a_classic_script(self, kind, url, cors_setting, options, encoding);
835
836                    // Step 33.2/33.3/33.4/33.5, substeps 1-2. Add el to the corresponding script list.
837                    match kind {
838                        ExternalScriptKind::Deferred => doc.add_deferred_script(self),
839                        ExternalScriptKind::ParsingBlocking => {
840                            doc.set_pending_parsing_blocking_script(self, None)
841                        },
842                        ExternalScriptKind::AsapInOrder => doc.push_asap_in_order_script(self),
843                        ExternalScriptKind::Asap => doc.add_asap_script(self),
844                    }
845                },
846                ScriptType::Module => {
847                    // Step 31.11. Fetch an external module script graph.
848                    fetch_external_module_script(
849                        ModuleOwner::Window(Trusted::new(self)),
850                        url.clone(),
851                        Destination::Script,
852                        options,
853                        can_gc,
854                    );
855
856                    if !asynch && was_parser_inserted {
857                        // 33.4: module, not async, parser-inserted
858                        doc.add_deferred_script(self);
859                    } else if !asynch && !self.non_blocking.get() {
860                        // 33.3: module, not parser-inserted
861                        doc.push_asap_in_order_script(self);
862                    } else {
863                        // 33.2: module, async
864                        doc.add_asap_script(self);
865                    };
866                },
867                ScriptType::ImportMap => (),
868            }
869        } else {
870            // Step 32. If el does not have a src content attribute:
871
872            assert!(!text.is_empty());
873
874            let text_rc = Rc::new(text.clone());
875
876            // Step 32.2: Switch on el's type:
877            match script_type {
878                ScriptType::Classic => {
879                    // Step 32.2.1 Let script be the result of creating a classic script
880                    // using source text, settings object, base URL, and options.
881                    let script = self.global().create_a_classic_script(
882                        std::borrow::Cow::Borrowed(&text.str()),
883                        base_url,
884                        options,
885                        ErrorReporting::Unmuted,
886                        introduction_type_override.or(Some(IntroductionType::INLINE_SCRIPT)),
887                        self.line_number as u32,
888                        false,
889                    );
890                    let result = Ok(Script::Classic(script));
891
892                    if was_parser_inserted &&
893                        doc.get_current_parser()
894                            .is_some_and(|parser| parser.script_nesting_level() <= 1) &&
895                        doc.get_script_blocking_stylesheets_count() > 0
896                    {
897                        // Step 34.2: classic, has no src, was parser-inserted, is blocked on stylesheet.
898                        doc.set_pending_parsing_blocking_script(self, Some(result));
899                    } else {
900                        // Step 34.3: otherwise.
901                        self.execute(result, can_gc);
902                    }
903                },
904                ScriptType::Module => {
905                    // We should add inline module script elements
906                    // into those vectors in case that there's no
907                    // descendants in the inline module script.
908                    if !asynch && was_parser_inserted {
909                        doc.add_deferred_script(self);
910                    } else if !asynch && !self.non_blocking.get() {
911                        doc.push_asap_in_order_script(self);
912                    } else {
913                        doc.add_asap_script(self);
914                    };
915
916                    fetch_inline_module_script(
917                        ModuleOwner::Window(Trusted::new(self)),
918                        text_rc,
919                        base_url.clone(),
920                        self.id,
921                        options,
922                        self.line_number,
923                        can_gc,
924                    );
925                },
926                ScriptType::ImportMap => {
927                    // Step 32.1 Let result be the result of creating an import map
928                    // parse result given source text and base URL.
929                    let import_map_result = parse_an_import_map_string(
930                        ModuleOwner::Window(Trusted::new(self)),
931                        Rc::clone(&text_rc),
932                        base_url.clone(),
933                        can_gc,
934                    );
935                    let script = Script::ImportMap(ScriptOrigin::internal(
936                        text_rc,
937                        base_url,
938                        options,
939                        script_type,
940                        self.global().unminified_js_dir(),
941                        import_map_result,
942                    ));
943
944                    // Step 34.3
945                    self.execute(Ok(script), can_gc);
946                },
947            }
948        }
949    }

951    /// <https://html.spec.whatwg.org/multipage/#execute-the-script-element>
952    pub(crate) fn execute(&self, result: ScriptResult, can_gc: CanGc) {
953        // Step 1. Let document be el's node document.
954        let doc = self.owner_document();
955
956        // Step 2. If el's preparation-time document is not equal to document, then return.
957        if *doc != *self.preparation_time_document.get().unwrap() {
958            return;
959        }
960
961        // TODO: Step 3. Unblock rendering on el.
962        let script = match result {
963            // Step 4. If el's result is null, then fire an event named error at el, and return.
964            Err(e) => {
965                warn!("error loading script {:?}", e);
966                self.dispatch_error_event(can_gc);
967                return;
968            },
969
970            Ok(script) => script,
971        };
972
973        // Step 5.
974        // If el's from an external file is true, or el's type is "module", then increment document's
975        // ignore-destructive-writes counter.
976        let neutralized_doc =
977            if self.from_an_external_file.get() || matches!(script, Script::Module(_)) {
978                let doc = self.owner_document();
979                doc.incr_ignore_destructive_writes_counter();
980                Some(doc)
981            } else {
982                None
983            };
984
985        let document = self.owner_document();
986
987        match script {
988            Script::Classic(script) => {
989                // Step 6."classic".1. Let oldCurrentScript be the value to which document's currentScript object was most recently set.
990                let old_script = document.GetCurrentScript();
991
992                // Step 6."classic".2. If el's root is not a shadow root,
993                // then set document's currentScript attribute to el. Otherwise, set it to null.
994                if self.upcast::<Node>().is_in_a_shadow_tree() {
995                    document.set_current_script(None)
996                } else {
997                    document.set_current_script(Some(self))
998                }
999
1000                // Step 6."classic".3. Run the classic script given by el's result.
1001                _ = self.owner_window().as_global_scope().run_a_classic_script(
1002                    script,
1003                    RethrowErrors::No,
1004                    can_gc,
1005                );
1006
1007                // Step 6."classic".4. Set document's currentScript attribute to oldCurrentScript.
1008                document.set_current_script(old_script.as_deref());
1009            },
1010            Script::Module(module_tree) => {
1011                // TODO Step 6."module".1. Assert: document's currentScript attribute is null.
1012                document.set_current_script(None);
1013
1014                // Step 6."module".2. Run the module script given by el's result.
1015                self.run_a_module_script(module_tree, false, can_gc);
1016            },
1017            Script::ImportMap(script) => {
1018                // Step 6."importmap".1. Register an import map given el's relevant global object and el's result.
1019                register_import_map(&self.owner_global(), script.import_map, can_gc);
1020            },
1021        }
1022
1023        // Step 7.
1024        // Decrement the ignore-destructive-writes counter of document, if it was incremented in the earlier step.
1025        if let Some(doc) = neutralized_doc {
1026            doc.decr_ignore_destructive_writes_counter();
1027        }
1028
1029        // Step 8. If el's from an external file is true, then fire an event named load at el.
1030        if self.from_an_external_file.get() {
1031            self.dispatch_load_event(can_gc);
1032        }
1033    }

1035    /// <https://html.spec.whatwg.org/multipage/#run-a-module-script>
1036    pub(crate) fn run_a_module_script(
1037        &self,
1038        module_tree: Rc<ModuleTree>,
1039        _rethrow_errors: bool,
1040        can_gc: CanGc,
1041    ) {
1042        // TODO use a settings object rather than this element's document/window
1043        // Step 2
1044        let document = self.owner_document();
1045        if !document.is_fully_active() || !document.scripting_enabled() {
1046            return;
1047        }
1048
1049        // Step 4
1050        let window = self.owner_window();
1051        let global = window.as_global_scope();
1052        let _aes = AutoEntryScript::new(global);
1053
1054        // Step 6.
1055        {
1056            let module_error = module_tree.get_rethrow_error().borrow();
1057            let network_error = module_tree.get_network_error().borrow();
1058            if module_error.is_some() && network_error.is_none() {
1059                module_tree.report_error(global, can_gc);
1060                return;
1061            }
1062        }
1063
1064        let record = module_tree
1065            .get_record()
1066            .borrow()
1067            .as_ref()
1068            .map(|record| record.handle());
1069
1070        if let Some(record) = record {
1071            rooted!(in(*GlobalScope::get_cx()) let mut rval = UndefinedValue());
1072            let evaluated =
1073                module_tree.execute_module(global, record, rval.handle_mut().into(), can_gc);
1074
1075            if let Err(exception) = evaluated {
1076                module_tree.set_rethrow_error(exception);
1077                module_tree.report_error(global, can_gc);
1078            }
1079        }
1080    }

1107    // https://html.spec.whatwg.org/multipage/#prepare-a-script Step 7.
1108    pub(crate) fn get_script_type(&self) -> Option<ScriptType> {
1109        let element = self.upcast::<Element>();
1110
1111        let type_attr = element.get_attribute(&ns!(), &local_name!("type"));
1112        let language_attr = element.get_attribute(&ns!(), &local_name!("language"));
1113
1114        match (
1115            type_attr.as_ref().map(|t| t.value()),
1116            language_attr.as_ref().map(|l| l.value()),
1117        ) {
1118            (Some(ref ty), _) if ty.is_empty() => {
1119                debug!("script type empty, inferring js");
1120                Some(ScriptType::Classic)
1121            },
1122            (None, Some(ref lang)) if lang.is_empty() => {
1123                debug!("script type empty, inferring js");
1124                Some(ScriptType::Classic)
1125            },
1126            (None, None) => {
1127                debug!("script type empty, inferring js");
1128                Some(ScriptType::Classic)
1129            },
1130            (None, Some(ref lang)) => {
1131                debug!("script language={}", &***lang);
1132                let language = format!("text/{}", &***lang);
1133
1134                if SCRIPT_JS_MIMES.contains(&language.to_ascii_lowercase().as_str()) {
1135                    Some(ScriptType::Classic)
1136                } else {
1137                    None
1138                }
1139            },
1140            (Some(ref ty), _) => {
1141                debug!("script type={}", &***ty);
1142
1143                if ty.to_ascii_lowercase().trim_matches(HTML_SPACE_CHARACTERS) == "module" {
1144                    return Some(ScriptType::Module);
1145                }
1146
1147                if ty.to_ascii_lowercase().trim_matches(HTML_SPACE_CHARACTERS) == "importmap" {
1148                    return Some(ScriptType::ImportMap);
1149                }
1150
1151                if SCRIPT_JS_MIMES
1152                    .contains(&ty.to_ascii_lowercase().trim_matches(HTML_SPACE_CHARACTERS))
1153                {
1154                    Some(ScriptType::Classic)
1155                } else {
1156                    None
1157                }
1158            },
1159        }
1160    }

1439pub(crate) fn substitute_with_local_script(
1440    window: &Window,
1441    script: &mut Cow<'_, str>,
1442    url: ServoUrl,
1443) {
1444    if window.local_script_source().is_none() {
1445        return;
1446    }
1447    let mut path = PathBuf::from(window.local_script_source().clone().unwrap());
1448    path = path.join(&url[url::Position::BeforeHost..]);
1449    debug!("Attempting to read script stored at: {:?}", path);
1450    match read_to_string(path.clone()) {
1451        Ok(local_script) => {
1452            debug!("Found script stored at: {:?}", path);
1453            *script = Cow::Owned(local_script);
1454        },
1455        Err(why) => warn!("Could not restore script from file {:?}", why),
1456    }
1457}
