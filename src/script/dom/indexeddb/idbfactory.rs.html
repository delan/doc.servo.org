31/// A non-jstraceable string wrapper for use in `HashMapTracedValues`.
32#[derive(Clone, Eq, Hash, MallocSizeOf, PartialEq)]
33pub(crate) struct DBName(pub(crate) String);
34
35#[dom_struct]
36pub struct IDBFactory {
37    reflector_: Reflector,
38    /// <https://www.w3.org/TR/IndexedDB-2/#connection>
39    /// The connections pending #open-a-database-connection.
40    /// TODO: remove names when connections close.
41    /// TODO: track not db names but open requests,
42    /// because by tracking db names, if any global at an
43    /// origin `abort_pending_upgrades`, it will abort open requests
44    /// for other globals within the same origin.
45    #[no_trace]
46    db_with_connections: DomRefCell<HashSet<DBName>>,
47}
48
49impl IDBFactory {
50    pub fn new_inherited() -> IDBFactory {
51        IDBFactory {
52            reflector_: Reflector::new(),
53            db_with_connections: Default::default(),
54        }
55    }
56
57    pub fn new(global: &GlobalScope, can_gc: CanGc) -> DomRoot<IDBFactory> {
58        reflect_dom_object(Box::new(IDBFactory::new_inherited()), global, can_gc)
59    }
60
61    fn note_start_of_open(&self, name: DBName) {
62        self.db_with_connections.borrow_mut().insert(name);
63    }
64
65    pub(crate) fn abort_pending_upgrades(&self) {
66        let global = self.global();
67        let names = self
68            .db_with_connections
69            .borrow_mut()
70            .drain()
71            .map(|name| name.0)
72            .collect();
73        let origin = global.origin().immutable().clone();
74        if global
75            .storage_threads()
76            .send(IndexedDBThreadMsg::Sync(
77                SyncOperation::AbortPendingUpgrades { names, origin },
78            ))
79            .is_err()
80        {
81            error!("Failed to send SyncOperation::AbortPendingUpgrade");
82        }
83    }
84}

86impl IDBFactoryMethods<crate::DomTypeHolder> for IDBFactory {
87    /// <https://w3c.github.io/IndexedDB/#dom-idbfactory-open>
88    fn Open(&self, name: DOMString, version: Option<u64>) -> Fallible<DomRoot<IDBOpenDBRequest>> {
89        // Step 1: If version is 0 (zero), throw a TypeError.
90        if version == Some(0) {
91            return Err(Error::Type(
92                "The version must be an integer >= 1".to_owned(),
93            ));
94        };
95
96        // Step 2: Let origin be the origin of the global scope used to
97        // access this IDBFactory.
98        // TODO: update to 3.0 spec.
99        // Let environment be this’s relevant settings object.
100        let global = self.global();
101        let origin = global.origin();
102
103        // Step 3: if origin is an opaque origin,
104        // throw a "SecurityError" DOMException and abort these steps.
105        // TODO: update to 3.0 spec.
106        // Let storageKey be the result of running obtain a storage key given environment.
107        if let ImmutableOrigin::Opaque(_) = origin.immutable() {
108            return Err(Error::Security(None));
109        }
110
111        // Step 4: Let request be a new open request.
112        let request = IDBOpenDBRequest::new(&self.global(), CanGc::note());
113
114        // Step 5: Runs in parallel
115        if request.open_database(name.clone(), version).is_err() {
116            return Err(Error::Operation(None));
117        }
118
119        self.note_start_of_open(DBName(name.to_string()));
120
121        // Step 6: Return a new IDBOpenDBRequest object for request.
122        Ok(request)
123    }
124
125    /// <https://www.w3.org/TR/IndexedDB/#dom-idbfactory-deletedatabase>
126    fn DeleteDatabase(&self, name: DOMString) -> Fallible<DomRoot<IDBOpenDBRequest>> {
127        // Step 1: Let environment be this’s relevant settings object.
128        let global = self.global();
129
130        // Step 2: Let storageKey be the result of running obtain a storage key given environment.
131        // If failure is returned, then throw a "SecurityError" DOMException and abort these steps.
132        // TODO: use a storage key.
133        let origin = global.origin();
134
135        // Legacy step 2: if origin is an opaque origin,
136        // throw a "SecurityError" DOMException and abort these steps.
137        // TODO: remove when a storage key is used.
138        if let ImmutableOrigin::Opaque(_) = origin.immutable() {
139            return Err(Error::Security(None));
140        }
141
142        // Step 3: Let request be a new open request
143        let request = IDBOpenDBRequest::new(&self.global(), CanGc::note());
144
145        // Step 4: Runs in parallel
146        if request.delete_database(name.to_string()).is_err() {
147            return Err(Error::Operation(None));
148        }
149
150        // Step 5: Return request
151        Ok(request)
152    }
153
154    /// <https://www.w3.org/TR/IndexedDB/#dom-idbfactory-databases>
155    fn Databases(&self, can_gc: CanGc) -> Rc<Promise> {
156        // Step 1: Let environment be this’s relevant settings object
157        let global = self.global();
158
159        // Step 2: Let storageKey be the result of running obtain a storage key given environment.
160        // If failure is returned, then return a promise rejected with a "SecurityError" DOMException
161        // TODO: implement storage keys.
162
163        // Step 3: Let p be a new promise.
164        let p = Promise::new(&global, can_gc);
165
166        // Note: the option is required to pass the promise to a task from within the generic callback,
167        // see #41356
168        let mut trusted_promise: Option<TrustedPromise> = Some(TrustedPromise::new(p.clone()));
169
170        // Step 4: Run these steps in parallel:
171        // Note implementing by communicating with the backend.
172        let task_source = global
173            .task_manager()
174            .database_access_task_source()
175            .to_sendable();
176        let callback = GenericCallback::new(global.time_profiler_chan().clone(), move |message| {
177            let result: BackendResult<Vec<DatabaseInfo>> = message.unwrap();
178            let Some(trusted_promise) = trusted_promise.take() else {
179                return error!("Callback for `DataBases` called twice.");
180            };
181
182            // Step 3.5: Queue a database task to resolve p with result.
183            task_source.queue(task!(set_request_result_to_database: move || {
184                let can_gc = CanGc::note();
185                let promise = trusted_promise.root();
186                match result {
187                    Err(err) => {
188                        let error = map_backend_error_to_dom_error(err);
189                        let cx = GlobalScope::get_cx();
190                        rooted!(in(*cx) let mut rval = UndefinedValue());
191                        error
192                            .clone()
193                            .to_jsval(cx, &promise.global(), rval.handle_mut(), can_gc);
194                        promise.reject_native(&rval.handle(), can_gc);
195                    },
196                    Ok(info_list) => {
197                        let info_list: Vec<IDBDatabaseInfo> = info_list
198                            .into_iter()
199                            .map(|info| IDBDatabaseInfo {
200                                name: Some(DOMString::from(info.name)),
201                                version: Some(info.version),
202                        })
203                        .collect();
204                        promise.resolve_native(&info_list, can_gc);
205                },
206            }
207            }));
208        })
209        .expect("Could not create delete database callback");
210
211        let get_operation =
212            SyncOperation::GetDatabases(callback, global.origin().immutable().clone());
213        if global
214            .storage_threads()
215            .send(IndexedDBThreadMsg::Sync(get_operation))
216            .is_err()
217        {
218            error!("Failed to send SyncOperation::GetDatabases");
219        }
220
221        // Step 5: Return p.
222        p
223    }

225    /// <https://www.w3.org/TR/IndexedDB-2/#dom-idbfactory-cmp>
226    fn Cmp(&self, cx: SafeJSContext, first: HandleValue, second: HandleValue) -> Fallible<i16> {
227        let first_key = convert_value_to_key(cx, first, None)?.into_result()?;
228        let second_key = convert_value_to_key(cx, second, None)?.into_result()?;
229        let cmp = first_key.partial_cmp(&second_key);
230        if let Some(cmp) = cmp {
231            match cmp {
232                std::cmp::Ordering::Less => Ok(-1),
233                std::cmp::Ordering::Equal => Ok(0),
234                std::cmp::Ordering::Greater => Ok(1),
235            }
236        } else {
237            Ok(i16::MAX)
238        }
239    }
