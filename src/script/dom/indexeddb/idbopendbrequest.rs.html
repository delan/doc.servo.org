40impl OpenRequestListener {
41    /// <https://w3c.github.io/IndexedDB/#open-a-database-connection>
42    /// The steps that continue on the script-thread.
43    fn handle_open_db(&self, name: String, response: OpenDatabaseResult, can_gc: CanGc) {
44        let request = self.open_request.root();
45        let global = request.global();
46        let name = DBName(name);
47        match response {
48            OpenDatabaseResult::Connection { version, upgraded } => {
49                // Step 2.2: Otherwise,
50                // set request’s result to result,
51                // set request’s done flag,
52                // and fire an event named success at request.
53                let connection = request.pending_connection.or_init(|| {
54                    debug_assert!(!upgraded, "A connection should exist for the upgraded db.");
55                    IDBDatabase::new(
56                        &global,
57                        DOMString::from_string(name.0.clone()),
58                        version,
59                        can_gc,
60                    )
61                });
62                request.dispatch_success(&connection);
63            },
64            OpenDatabaseResult::Upgrade {
65                version,
66                old_version,
67                transaction,
68            } => {
69                // TODO: link with backend connection concept.
70                let connection = IDBDatabase::new(
71                    &global,
72                    DOMString::from_string(name.0.clone()),
73                    version,
74                    can_gc,
75                );
76                request.pending_connection.set(Some(&connection));
77                request.upgrade_db_version(&connection, old_version, version, transaction, can_gc);
78            },
79            OpenDatabaseResult::VersionError => {
80                // Step 2.1 If result is an error, see dispatch_error().
81                self.dispatch_error(Error::Version(None), can_gc);
82            },
83            OpenDatabaseResult::AbortError => {
84                // Step 2.1 If result is an error, see dispatch_error().
85                self.dispatch_error(Error::Abort(None), can_gc);
86            },
87        }
88    }
89
90    fn handle_backend_error(&self, backend_error: BackendError, can_gc: CanGc) {
91        self.dispatch_error(map_backend_error_to_dom_error(backend_error), can_gc);
92    }
93
94    // Step 2.1 If result is an error,
95    // set request’s result to undefined,
96    // set request’s error to result,
97    // set request’s done flag,
98    // and fire an event named error at request
99    // with its bubbles and cancelable attributes initialized to true.
100    fn dispatch_error(&self, dom_exception: Error, can_gc: CanGc) {
101        let request = self.open_request.root();
102        let global = request.global();
103        request.set_result(HandleValue::undefined());
104        request.set_error(Some(dom_exception), can_gc);
105        let event = Event::new(
106            &global,
107            Atom::from("error"),
108            EventBubbles::Bubbles,
109            EventCancelable::Cancelable,
110            can_gc,
111        );
112        event.fire(request.upcast(), can_gc);
113    }
114
115    /// The contionuation of the parallel steps of
116    /// <https://www.w3.org/TR/IndexedDB/#dom-idbfactory-deletedatabase>
117    fn handle_delete_db(&self, result: BackendResult<u64>, can_gc: CanGc) {
118        // Step 4.1: Let result be the result of deleting a database, with storageKey, name, and request.
119        // Note: done with the `result` argument.
120
121        // Step 4.2: Set request’s processed flag to true.
122        // TODO: implemen the flag.
123        // Note: the flag may be need to be set on the backend(as well as here?).
124
125        // Step 3: Queue a database task to run these steps:
126        // Note: we are in the queued task.
127
128        let open_request = self.open_request.root();
129        let global = open_request.global();
130
131        // Note: setting the done flag here as it is done in both branches below.
132        open_request.idbrequest.set_ready_state_done();
133
134        let cx = GlobalScope::get_cx();
135        rooted!(in(*cx) let mut rval = UndefinedValue());
136
137        let _ac = enter_realm(&*open_request);
138
139        match result {
140            Ok(version) => {
141                // Step 4.3.2: Otherwise,
142                // set request’s result to undefined,
143                // set request’s done flag to true,
144                // and fire a version change event named success at request with result and null.
145                open_request.set_result(rval.handle());
146                let event = IDBVersionChangeEvent::new(
147                    &global,
148                    Atom::from("success"),
149                    EventBubbles::DoesNotBubble,
150                    EventCancelable::NotCancelable,
151                    version,
152                    None,
153                    CanGc::note(),
154                );
155                event.upcast::<Event>().fire(open_request.upcast(), can_gc);
156            },
157            Err(err) => {
158                // Step 4.3.1:
159                // If result is an error,
160                // set request’s error to result,
161                // set request’s done flag to true,
162                // and fire an event named error at request
163                // with its bubbles and cancelable attributes initialized to true.
164
165                // TODO: transform backend error into jsval.
166                let error = map_backend_error_to_dom_error(err);
167                let cx = GlobalScope::get_cx();
168                rooted!(in(*cx) let mut rval = UndefinedValue());
169                error
170                    .clone()
171                    .to_jsval(cx, &global, rval.handle_mut(), can_gc);
172                open_request.set_result(rval.handle());
173                let event = Event::new(
174                    &global,
175                    Atom::from("error"),
176                    EventBubbles::Bubbles,
177                    EventCancelable::Cancelable,
178                    can_gc,
179                );
180                event.fire(open_request.upcast(), can_gc);
181            },
182        }
183    }
184}

204    /// <https://w3c.github.io/IndexedDB/#upgrade-a-database>
205    /// Step 10: Queue a database task to run these steps:
206    /// The below are the steps in the task.
207    fn upgrade_db_version(
208        &self,
209        connection: &IDBDatabase,
210        old_version: u64,
211        version: u64,
212        transaction: u64,
213        can_gc: CanGc,
214    ) {
215        let global = self.global();
216        let cx = GlobalScope::get_cx();
217
218        // Note: the transaction should link wiht one created on the backend.
219        // Steps here are meant to create the corresponding webidl object.
220        let transaction = IDBTransaction::new_with_id(
221            &global,
222            connection,
223            IDBTransactionMode::Versionchange,
224            &connection.object_stores(),
225            transaction,
226            can_gc,
227        );
228        connection.set_transaction(&transaction);
229
230        rooted!(in(*cx) let mut connection_val = UndefinedValue());
231        connection.safe_to_jsval(cx, connection_val.handle_mut(), can_gc);
232
233        // Step 10.1: Set request’s result to connection.
234        self.idbrequest.set_result(connection_val.handle());
235
236        // Step 10.2: Set request’s transaction to transaction.
237        self.idbrequest.set_transaction(&transaction);
238
239        // Step 10.3: Set request’s done flag to true.
240        self.idbrequest.set_ready_state_done();
241
242        // Step 10.4: Set transaction’s state to active.
243        // TODO: message to set state of backend transaction.
244        transaction.set_active_flag(true);
245
246        // Step 10.5: Let didThrow be the result of
247        // firing a version change event named upgradeneeded
248        // at request with old version and version.
249        let event = IDBVersionChangeEvent::new(
250            &global,
251            Atom::from("upgradeneeded"),
252            EventBubbles::DoesNotBubble,
253            EventCancelable::NotCancelable,
254            old_version,
255            Some(version),
256            CanGc::note(),
257        );
258
259        // TODO: use as part of step 10.6.2
260        let _did_throw = event.upcast::<Event>().fire(self.upcast(), can_gc);
261
262        // Step 10.6: If transaction’s state is active, then:
263        if transaction.is_active() {
264            // Step 10.6.1: Set transaction’s state to inactive.
265            transaction.set_active_flag(false);
266
267            // Step 10.6.2: If didThrow is true,
268            // run abort a transaction with transaction
269            // and a newly created "AbortError" DOMException.
270            // TODO: implement.
271
272            // Note: only sending this if active,
273            // because if the transaction was aborted
274            // we don't want to create a connection.
275            if global
276                .storage_threads()
277                .send(IndexedDBThreadMsg::OpenTransactionInactive {
278                    name: connection.get_name().to_string(),
279                    origin: global.origin().immutable().clone(),
280                })
281                .is_err()
282            {
283                error!("Failed to send OpenTransactionInactive.");
284            }
285        }
286    }

296    /// <https://w3c.github.io/IndexedDB/#open-a-database-connection>
297    pub fn open_database(&self, name: DOMString, version: Option<u64>) -> Result<(), ()> {
298        let global = self.global();
299
300        let response_listener = OpenRequestListener {
301            open_request: Trusted::new(self),
302        };
303
304        let task_source = global
305            .task_manager()
306            .database_access_task_source()
307            .to_sendable();
308        let name = name.to_string();
309        let name_copy = name.clone();
310        let callback = GenericCallback::new(global.time_profiler_chan().clone(), move |message| {
311            let response_listener = response_listener.clone();
312            let name = name_copy.clone();
313            let backend_result = match message {
314                Ok(inner) => inner,
315                Err(err) => Err(BackendError::DbErr(format!("{err:?}"))),
316            };
317            task_source.queue(task!(set_request_result_to_database: move || {
318                match backend_result {
319                    Ok(response) => {
320                        response_listener.handle_open_db(name, response, CanGc::note())
321                    }
322                    Err(error) => response_listener.handle_backend_error(error, CanGc::note()),
323                }
324            }));
325        })
326        .expect("Could not create open database callback");
327
328        let open_operation = SyncOperation::OpenDatabase(
329            callback,
330            global.origin().immutable().clone(),
331            name.to_string(),
332            version,
333        );
334
335        // Note: algo continues in parallel.
336        if global
337            .storage_threads()
338            .send(IndexedDBThreadMsg::Sync(open_operation))
339            .is_err()
340        {
341            return Err(());
342        }
343        Ok(())
344    }
345
346    pub fn delete_database(&self, name: String) -> Result<(), ()> {
347        let global = self.global();
348
349        let task_source = global
350            .task_manager()
351            .database_access_task_source()
352            .to_sendable();
353        let response_listener = OpenRequestListener {
354            open_request: Trusted::new(self),
355        };
356        let callback = GenericCallback::new(global.time_profiler_chan().clone(), move |message| {
357            let response_listener = response_listener.clone();
358            task_source.queue(task!(request_callback: move || {
359                response_listener.handle_delete_db(message.unwrap(), CanGc::note());
360            }))
361        })
362        .expect("Could not create delete database callback");
363
364        let delete_operation =
365            SyncOperation::DeleteDatabase(callback, global.origin().immutable().clone(), name);
366
367        if global
368            .storage_threads()
369            .send(IndexedDBThreadMsg::Sync(delete_operation))
370            .is_err()
371        {
372            return Err(());
373        }
374        Ok(())
375    }

377    pub fn dispatch_success(&self, result: &IDBDatabase) {
378        self.idbrequest.set_ready_state_done();
379        let global = self.global();
380        let cx = GlobalScope::get_cx();
381
382        let _ac = enter_realm(result);
383        rooted!(in(*cx) let mut result_val = UndefinedValue());
384        result.safe_to_jsval(cx, result_val.handle_mut(), CanGc::note());
385        self.set_result(result_val.handle());
386
387        let event = Event::new(
388            &global,
389            Atom::from("success"),
390            EventBubbles::DoesNotBubble,
391            EventCancelable::NotCancelable,
392            CanGc::note(),
393        );
394        event.fire(self.upcast(), CanGc::note());
395    }
