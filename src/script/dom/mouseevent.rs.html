35/// <https://w3c.github.io/uievents/#interface-mouseevent>
36#[dom_struct]
37pub(crate) struct MouseEvent {
38    uievent: UIEvent,
39
40    /// The point on the screen of where this [`MouseEvent`] was originally triggered,
41    /// to use during the dispatch phase.
42    ///
43    /// See:
44    /// <https://w3c.github.io/uievents/#dom-mouseevent-screenx>
45    /// <https://w3c.github.io/uievents/#dom-mouseevent-screeny>
46    #[no_trace]
47    screen_point: Cell<Point2D<i32, CSSPixel>>,
48
49    /// The point in the viewport of where this [`MouseEvent`] was originally triggered,
50    /// to use during the dispatch phase.
51    ///
52    /// See:
53    /// <https://w3c.github.io/uievents/#dom-mouseevent-clientx>
54    /// <https://w3c.github.io/uievents/#dom-mouseevent-clienty>
55    #[no_trace]
56    client_point: Cell<Point2D<i32, CSSPixel>>,
57
58    /// The point in the initial containing block of where this [`MouseEvent`] was
59    /// originally triggered to use during the dispatch phase.
60    ///
61    /// See:
62    /// <https://w3c.github.io/uievents/#dom-mouseevent-pagex>
63    /// <https://w3c.github.io/uievents/#dom-mouseevent-pagey>
64    #[no_trace]
65    page_point: Cell<Point2D<i32, CSSPixel>>,
66
67    /// The keyboard modifiers that were active when this mouse event was triggered.
68    #[no_trace]
69    modifiers: Cell<Modifiers>,
70
71    /// <https://w3c.github.io/uievents/#dom-mouseevent-button>
72    button: Cell<i16>,
73
74    /// <https://w3c.github.io/uievents/#dom-mouseevent-buttons>
75    buttons: Cell<u16>,
76
77    #[no_trace]
78    point_in_target: Cell<Option<Point2D<f32, CSSPixel>>>,
79}

213    /// <https://w3c.github.io/uievents/#initialize-a-mouseevent>
214    #[expect(clippy::too_many_arguments)]
215    pub(crate) fn initialize_mouse_event(
216        &self,
217        type_: DOMString,
218        can_bubble: EventBubbles,
219        cancelable: EventCancelable,
220        view: Option<&Window>,
221        detail: i32,
222        screen_point: Point2D<i32, CSSPixel>,
223        client_point: Point2D<i32, CSSPixel>,
224        page_point: Point2D<i32, CSSPixel>,
225        modifiers: Modifiers,
226        button: i16,
227        buttons: u16,
228        related_target: Option<&EventTarget>,
229        point_in_target: Option<Point2D<f32, CSSPixel>>,
230    ) {
231        self.uievent.initialize_ui_event(
232            type_,
233            view.map(|window| window.upcast::<EventTarget>()),
234            can_bubble,
235            cancelable,
236        );
237
238        self.uievent.set_detail(detail);
239        self.screen_point.set(screen_point);
240        self.client_point.set(client_point);
241        self.page_point.set(page_point);
242        self.modifiers.set(modifiers);
243        self.button.set(button);
244        self.buttons.set(buttons);
245        self.upcast::<Event>().set_related_target(related_target);
246        self.point_in_target.set(point_in_target);
247        // Legacy mapping per spec: left/middle/right => 1/2/3 (button + 1), else 0.
248        let w = if button >= 0 { (button as u32) + 1 } else { 0 };
249        self.uievent.set_which(w);
250    }

256    /// Create a [MouseEvent] triggered by the embedder
257    /// <https://w3c.github.io/uievents/#create-a-cancelable-mouseevent-id>
258    #[expect(clippy::too_many_arguments)]
259    pub(crate) fn for_platform_mouse_event(
260        event_type_string: &'static str,
261        event: embedder_traits::MouseButtonEvent,
262        pressed_mouse_buttons: u16,
263        window: &Window,
264        hit_test_result: &HitTestResult,
265        modifiers: Modifiers,
266        click_count: usize,
267        can_gc: CanGc,
268    ) -> DomRoot<Self> {
269        let client_point = hit_test_result.point_in_frame.to_i32();
270        let page_point = hit_test_result
271            .point_relative_to_initial_containing_block
272            .to_i32();
273
274        let mouse_event = MouseEvent::new(
275            window,
276            event_type_string.into(),
277            EventBubbles::Bubbles,
278            EventCancelable::Cancelable,
279            Some(window),
280            click_count as i32,
281            client_point, // TODO: Get real screen coordinates?
282            client_point,
283            page_point,
284            modifiers,
285            event.button.into(),
286            pressed_mouse_buttons,
287            None,
288            Some(hit_test_result.point_in_node),
289            can_gc,
290        );
291
292        mouse_event.upcast::<Event>().set_trusted(true);
293        mouse_event.upcast::<Event>().set_composed(true);
294
295        mouse_event
296    }

300    /// <https://w3c.github.io/uievents/#dom-mouseevent-mouseevent>
301    fn Constructor(
302        window: &Window,
303        proto: Option<HandleObject>,
304        can_gc: CanGc,
305        type_: DOMString,
306        init: &MouseEventBinding::MouseEventInit,
307    ) -> Fallible<DomRoot<MouseEvent>> {
308        let bubbles = EventBubbles::from(init.parent.parent.parent.bubbles);
309        let cancelable = EventCancelable::from(init.parent.parent.parent.cancelable);
310        let scroll_offset = window.scroll_offset();
311        let page_point = Point2D::new(
312            scroll_offset.x as i32 + init.clientX,
313            scroll_offset.y as i32 + init.clientY,
314        );
315        let event = MouseEvent::new_with_proto(
316            window,
317            proto,
318            type_,
319            bubbles,
320            cancelable,
321            init.parent.parent.view.as_deref(),
322            init.parent.parent.detail,
323            Point2D::new(init.screenX, init.screenY),
324            Point2D::new(init.clientX, init.clientY),
325            page_point,
326            init.parent.modifiers(),
327            init.button,
328            init.buttons,
329            init.relatedTarget.as_deref(),
330            None,
331            can_gc,
332        );
333        event
334            .upcast::<Event>()
335            .set_composed(init.parent.parent.parent.composed);
336        Ok(event)
337    }

359    /// <https://drafts.csswg.org/cssom-view/#dom-mouseevent-pagex>
360    fn PageX(&self) -> i32 {
361        // The pageX attribute must follow these steps:
362        // > 1. If the event’s dispatch flag is set, return the horizontal coordinate of the
363        // > position where the event occurred relative to the origin of the initial containing
364        // > block and terminate these steps.
365        if self.upcast::<Event>().dispatching() {
366            return self.page_point.get().x;
367        }
368
369        // > 2. Let offset be the value of the scrollX attribute of the event’s associated
370        // > Window object, if there is one, or zero otherwise.
371        // > 3. Return the sum of offset and the value of the event’s clientX attribute.
372        self.global().as_window().ScrollX() + self.ClientX()
373    }
374
375    /// <https://drafts.csswg.org/cssom-view/#dom-mouseevent-pagey>
376    fn PageY(&self) -> i32 {
377        // The pageY attribute must follow these steps:
378        // > 1. If the event’s dispatch flag is set, return the vertical coordinate of the
379        // > position where the event occurred relative to the origin of the initial
380        // > containing block and terminate these steps.
381        if self.upcast::<Event>().dispatching() {
382            return self.page_point.get().y;
383        }
384
385        // > 2. Let offset be the value of the scrollY attribute of the event’s associated
386        // > Window object, if there is one, or zero otherwise.
387        // > 3. Return the sum of offset and the value of the event’s clientY attribute.
388        self.global().as_window().ScrollY() + self.ClientY()
389    }

401    /// <https://drafts.csswg.org/cssom-view/#dom-mouseevent-offsetx>
402    fn OffsetX(&self) -> i32 {
403        // > The offsetX attribute must follow these steps:
404        // > 1. If the event’s dispatch flag is set, return the x-coordinate of the position
405        // >    where the event occurred relative to the origin of the padding edge of the
406        // >    target node, ignoring the transforms that apply to the element and its
407        // >    ancestors, and terminate these steps.
408        let event = self.upcast::<Event>();
409        if event.dispatching() {
410            let Some(target) = event.GetTarget() else {
411                return 0;
412            };
413            let Some(node) = target.downcast::<Node>() else {
414                return 0;
415            };
416            return self.ClientX() - node.client_rect().origin.x;
417        }
418
419        // > 2. Return the value of the event’s pageX attribute.
420        self.PageX()
421    }
422
423    /// <https://drafts.csswg.org/cssom-view/#dom-mouseevent-offsety>
424    fn OffsetY(&self) -> i32 {
425        // > The offsetY attribute must follow these steps:
426        // > 1. If the event’s dispatch flag is set, return the y-coordinate of the
427        // >    position where the event occurred relative to the origin of the padding edge of
428        // >    the target node, ignoring the transforms that apply to the element and its
429        // >    ancestors, and terminate these steps.
430        let event = self.upcast::<Event>();
431        if event.dispatching() {
432            let Some(target) = event.GetTarget() else {
433                return 0;
434            };
435            let Some(node) = target.downcast::<Node>() else {
436                return 0;
437            };
438            return self.ClientY() - node.client_rect().origin.y;
439        }
440
441        // 2. Return the value of the event’s pageY attribute.
442        self.PageY()
443    }

480    /// <https://w3c.github.io/uievents/#widl-MouseEvent-initMouseEvent>
481    fn InitMouseEvent(
482        &self,
483        type_arg: DOMString,
484        can_bubble_arg: bool,
485        cancelable_arg: bool,
486        view_arg: Option<&Window>,
487        detail_arg: i32,
488        screen_x_arg: i32,
489        screen_y_arg: i32,
490        client_x_arg: i32,
491        client_y_arg: i32,
492        ctrl_key_arg: bool,
493        alt_key_arg: bool,
494        shift_key_arg: bool,
495        meta_key_arg: bool,
496        button_arg: i16,
497        related_target_arg: Option<&EventTarget>,
498    ) {
499        if self.upcast::<Event>().dispatching() {
500            return;
501        }
502
503        self.upcast::<UIEvent>().InitUIEvent(
504            type_arg,
505            can_bubble_arg,
506            cancelable_arg,
507            view_arg,
508            detail_arg,
509        );
510        self.screen_point
511            .set(Point2D::new(screen_x_arg, screen_y_arg));
512        self.client_point
513            .set(Point2D::new(client_x_arg, client_y_arg));
514
515        let global = self.global();
516        let scroll_offset = global.as_window().scroll_offset();
517        self.page_point.set(Point2D::new(
518            scroll_offset.x as i32 + client_x_arg,
519            scroll_offset.y as i32 + client_y_arg,
520        ));
521
522        let mut modifiers = Modifiers::empty();
523        if ctrl_key_arg {
524            modifiers.insert(Modifiers::CONTROL);
525        }
526        if alt_key_arg {
527            modifiers.insert(Modifiers::ALT);
528        }
529        if shift_key_arg {
530            modifiers.insert(Modifiers::SHIFT);
531        }
532        if meta_key_arg {
533            modifiers.insert(Modifiers::META);
534        }
535        self.modifiers.set(modifiers);
536
537        self.button.set(button_arg);
538        self.upcast::<Event>()
539            .set_related_target(related_target_arg);
540
541        // Keep UIEvent.which in sync for legacy init path too.
542        let w = if button_arg >= 0 {
543            (button_arg as u32) + 1
544        } else {
545            0
546        };
547        self.uievent.set_which(w);
548    }

555    /// <https://w3c.github.io/uievents/#dom-mouseevent-getmodifierstate>
556    fn GetModifierState(&self, key_arg: DOMString) -> bool {
557        self.modifiers
558            .get()
559            .contains(match_domstring_ascii!(key_arg,
560                "Alt" => Modifiers::ALT,
561                "AltGraph" => Modifiers::ALT_GRAPH,
562                "CapsLock" => Modifiers::CAPS_LOCK,
563                "Control" => Modifiers::CONTROL,
564                "Fn" => Modifiers::FN,
565                "FnLock" => Modifiers::FN_LOCK,
566                "Meta" => Modifiers::META,
567                "NumLock" => Modifiers::NUM_LOCK,
568                "ScrollLock" => Modifiers::SCROLL_LOCK,
569                "Shift" => Modifiers::SHIFT,
570                "Symbol" => Modifiers::SYMBOL,
571                "SymbolLock" => Modifiers::SYMBOL_LOCK,
572                    _ => { return false; },
573            ))
574    }
