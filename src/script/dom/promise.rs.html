5//! Native representation of JS Promise values.
6//!
7//! This implementation differs from the traditional Rust DOM object, because the reflector
8//! is provided by SpiderMonkey and has no knowledge of an associated native representation
9//! (ie. dom::Promise). This means that native instances use native reference counting (Rc)
10//! to ensure that no memory is leaked, which means that there can be multiple instances of
11//! native Promise values that refer to the same JS value yet are distinct native objects
12//! (ie. address equality for the native objects is meaningless).
13

50#[dom_struct]
51#[cfg_attr(crown, crown::unrooted_must_root_lint::allow_unrooted_in_rc)]
52pub(crate) struct Promise {
53    reflector: Reflector,
54    /// Since Promise values are natively reference counted without the knowledge of
55    /// the SpiderMonkey GC, an explicit root for the reflector is stored while any
56    /// native instance exists. This ensures that the reflector will never be GCed
57    /// while native code could still interact with its native representation.
58    #[ignore_malloc_size_of = "SM handles JS values"]
59    permanent_js_root: Heap<JSVal>,
60}
61
62/// Private helper to enable adding new methods to `Rc<Promise>`.
63trait PromiseHelper {
64    fn initialize(&self, cx: SafeJSContext);
65}
66
67impl PromiseHelper for Rc<Promise> {
68    #[expect(unsafe_code)]
69    fn initialize(&self, cx: SafeJSContext) {
70        let obj = self.reflector().get_jsobject();
71        self.permanent_js_root.set(ObjectValue(*obj));
72        unsafe {
73            assert!(AddRawValueRoot(
74                *cx,
75                self.permanent_js_root.get_unsafe(),
76                c"Promise::root".as_ptr(),
77            ));
78        }
79    }
80}

82// Promise objects are stored inside Rc values, so Drop is run when the last Rc is dropped,
83// rather than when SpiderMonkey runs a GC. This makes it safe to interact with the JS engine unlike
84// Drop implementations for other DOM types.
85impl Drop for Promise {
86    #[expect(unsafe_code)]
87    fn drop(&mut self) {
88        unsafe {
89            let object = self.permanent_js_root.get().to_object();
90            assert!(!object.is_null());
91            if let Some(cx) = Runtime::get() {
92                RemoveRawValueRoot(cx.as_ptr(), self.permanent_js_root.get_unsafe());
93            }
94        }
95    }
96}

149    #[expect(unsafe_code)]
150    // The apparently-unused CanGc parameter reflects the fact that the JS API calls
151    // like JS_NewFunction can trigger a GC.
152    fn create_js_promise(cx: SafeJSContext, mut obj: MutableHandleObject, _can_gc: CanGc) {
153        unsafe {
154            let do_nothing_func = JS_NewFunction(
155                *cx,
156                Some(do_nothing_promise_executor),
157                /* nargs = */ 2,
158                /* flags = */ 0,
159                ptr::null(),
160            );
161            assert!(!do_nothing_func.is_null());
162            rooted!(in(*cx) let do_nothing_obj = JS_GetFunctionObject(do_nothing_func));
163            assert!(!do_nothing_obj.is_null());
164            obj.set(NewPromiseObject(*cx, do_nothing_obj.handle()));
165            assert!(!obj.is_null());
166            let is_user_interacting = if ScriptThread::is_user_interacting() {
167                PromiseUserInputEventHandlingState::HadUserInteractionAtCreation
168            } else {
169                PromiseUserInputEventHandlingState::DidntHaveUserInteractionAtCreation
170            };
171            SetPromiseUserInputEventHandlingState(obj.handle(), is_user_interacting);
172        }
173    }

286        &self,
287        handler: &PromiseNativeHandler,
288        realm: InRealm,
289        can_gc: CanGc,
290    ) {

313        }

339const SLOT_NATIVEHANDLER: usize = 0;
340const SLOT_NATIVEHANDLER_TASK: usize = 1;
341
342#[derive(PartialEq)]
343enum NativeHandlerTask {
344    Resolve = 0,
345    Reject = 1,
346}
347
348#[expect(unsafe_code)]
349unsafe extern "C" fn native_handler_callback(
350    cx: *mut RawJSContext,
351    argc: u32,
352    vp: *mut JSVal,
353) -> bool {
354    // Safety: it is safe to construct a JSContext from engine hook.
355    let mut cx = unsafe { JSContext::from_ptr(ptr::NonNull::new(cx).unwrap()) };
356    let mut cx = CurrentRealm::assert(&mut cx);
357    let cx = &mut cx;
358
359    let args = unsafe { CallArgs::from_vp(vp, argc) };
360    let native_handler_value =
361        unsafe { *GetFunctionNativeReserved(args.callee(), SLOT_NATIVEHANDLER) };
362    rooted!(&in(cx) let native_handler_value = native_handler_value);
363    assert!(native_handler_value.get().is_object());
364
365    let handler = unsafe {
366        root_from_object::<PromiseNativeHandler>(native_handler_value.to_object(), cx.raw_cx())
367    }
368    .expect("unexpected value for native handler in promise native handler callback");
369
370    let native_handler_task_value =
371        unsafe { *GetFunctionNativeReserved(args.callee(), SLOT_NATIVEHANDLER_TASK) };
372    rooted!(&in(cx) let native_handler_task_value = native_handler_task_value);
373    match native_handler_task_value.to_int32() {
374        native_handler_task_value
375            if native_handler_task_value == NativeHandlerTask::Resolve as i32 =>
376        {
377            handler.resolved_callback(cx, unsafe { HandleValue::from_raw(args.get(0)) })
378        },
379        native_handler_task_value
380            if native_handler_task_value == NativeHandlerTask::Reject as i32 =>
381        {
382            handler.rejected_callback(cx, unsafe { HandleValue::from_raw(args.get(0)) })
383        },
384        _ => panic!("unexpected native handler task value"),
385    };
386
387    true
388}

390#[expect(unsafe_code)]
391// The apparently-unused CanGc argument reflects the fact that the JS API calls
392// like NewFunctionWithReserved can trigger a GC.
393fn create_native_handler_function(
394    cx: *mut RawJSContext,
395    holder: HandleObject,
396    task: NativeHandlerTask,
397    _can_gc: CanGc,
398) -> *mut JSObject {
399    unsafe {
400        let func = NewFunctionWithReserved(cx, Some(native_handler_callback), 1, 0, ptr::null());
401        assert!(!func.is_null());
402
403        rooted!(in(cx) let obj = JS_GetFunctionObject(func));
404        assert!(!obj.is_null());
405        SetFunctionNativeReserved(obj.get(), SLOT_NATIVEHANDLER, &ObjectValue(*holder));
406        SetFunctionNativeReserved(obj.get(), SLOT_NATIVEHANDLER_TASK, &Int32Value(task as i32));
407        obj.get()
408    }
409}

411impl FromJSValConvertibleRc for Promise {
412    #[expect(unsafe_code)]
413    unsafe fn from_jsval(
414        cx: *mut RawJSContext,
415        value: HandleValue,
416    ) -> Result<ConversionResult<Rc<Promise>>, ()> {
417        if value.get().is_null() {
418            return Ok(ConversionResult::Failure("null not allowed".into()));
419        }
420
421        let cx = unsafe { SafeJSContext::from_ptr(cx) };
422        let in_realm_proof = AlreadyInRealm::assert_for_cx(cx);
423        let global_scope =
424            unsafe { GlobalScope::from_context(*cx, InRealm::Already(&in_realm_proof)) };
425
426        let promise = Promise::new_resolved(&global_scope, cx, value, CanGc::note());
427        Ok(ConversionResult::Success(promise))
428    }
429}

431/// The success steps of <https://webidl.spec.whatwg.org/#wait-for-all>
432type WaitForAllSuccessSteps = Rc<dyn Fn(Vec<HandleValue>)>;
433
434/// The failure steps of <https://webidl.spec.whatwg.org/#wait-for-all>
435type WaitForAllFailureSteps = Rc<dyn Fn(HandleValue)>;
436
437/// The fulfillment handler for the list of promises in
438/// <https://webidl.spec.whatwg.org/#wait-for-all>.
439#[derive(JSTraceable, MallocSizeOf)]
440#[cfg_attr(crown, crown::unrooted_must_root_lint::must_root)]
441struct WaitForAllFulfillmentHandler {
442    /// The steps to call when all promises are resolved.
443    #[ignore_malloc_size_of = "callbacks are hard"]
444    #[no_trace]
445    success_steps: WaitForAllSuccessSteps,
446
447    /// The results of the promises.
448    #[ignore_malloc_size_of = "mozjs"]
449    #[expect(clippy::vec_box)]
450    result: Rc<RefCell<Vec<Box<Heap<JSVal>>>>>,
451
452    /// The index identifying which promise this handler is attached to.
453    promise_index: usize,
454
455    /// A count of fulfilled promises.
456    #[conditional_malloc_size_of]
457    fulfilled_count: Rc<RefCell<usize>>,
458}
459
460impl Callback for WaitForAllFulfillmentHandler {
461    fn callback(&self, _cx: &mut CurrentRealm, v: HandleValue) {
462        // Let fulfillmentHandler be the following steps given arg:
463
464        let equals_total = {
465            // Set result[promiseIndex] to arg.
466            let result = self.result.borrow_mut();
467            result[self.promise_index].set(v.get());
468
469            // Set fulfilledCount to fulfilledCount + 1.
470            let mut fulfilled_count = self.fulfilled_count.borrow_mut();
471            *fulfilled_count += 1;
472
473            *fulfilled_count == result.len()
474        };
475
476        // If fulfilledCount equals total, then perform successSteps given result.
477        if equals_total {
478            let result_ref = self.result.borrow();
479            let result_handles: Vec<HandleValue> =
480                result_ref.iter().map(|v| v.as_handle_value()).collect();
481
482            (self.success_steps)(result_handles);
483        }
484    }
485}

487/// The rejection handler for the list of promises in
488/// <https://webidl.spec.whatwg.org/#wait-for-all>.
489#[derive(Clone, JSTraceable, MallocSizeOf)]
490struct WaitForAllRejectionHandler {
491    /// The steps to call if any promise rejects.
492    #[ignore_malloc_size_of = "callbacks are hard"]
493    #[no_trace]
494    failure_steps: WaitForAllFailureSteps,
495
496    /// Whether any promises have been rejected already.
497    rejected: Cell<bool>,
498}
499
500impl Callback for WaitForAllRejectionHandler {
501    fn callback(&self, _cx: &mut CurrentRealm, v: HandleValue) {
502        // Let rejectionHandlerSteps be the following steps given arg:
503
504        if self.rejected.replace(true) {
505            // If rejected is true, abort these steps.
506            return;
507        }
508
509        // Set rejected to true.
510        // Done above with `replace`.
511        (self.failure_steps)(v);
512    }
513}

536/// <https://webidl.spec.whatwg.org/#wait-for-all>
537#[cfg_attr(crown, expect(crown::unrooted_must_root))]
538pub(crate) fn wait_for_all(
539    cx: SafeJSContext,
540    global: &GlobalScope,
541    promises: Vec<Rc<Promise>>,
542    success_steps: WaitForAllSuccessSteps,
543    failure_steps: WaitForAllFailureSteps,
544    realm: InRealm,
545    can_gc: CanGc,
546) {
547    // Let fulfilledCount be 0.
548    let fulfilled_count: Rc<RefCell<usize>> = Default::default();
549
550    // Let rejected be false.
551    // Note: done below when constructing a rejection handler.
552
553    // Let rejectionHandlerSteps be the following steps given arg:
554    // Note: implemented with the `WaitForAllRejectionHandler`.
555
556    // Let rejectionHandler be CreateBuiltinFunction(rejectionHandlerSteps,  ):
557    // Note: done as part of attaching the `WaitForAllRejectionHandler` as native rejection handler.
558    let rejection_handler = WaitForAllRejectionHandler {
559        failure_steps,
560        rejected: Default::default(),
561    };
562
563    // Let total be promisess size.
564    // Note: done using the len of result.
565
566    // If total is 0, then:
567    if promises.is_empty() {
568        // Queue a microtask to perform successSteps given  .
569        global.enqueue_microtask(Microtask::WaitForAllSuccessSteps(
570            WaitForAllSuccessStepsMicrotask {
571                global: DomRoot::from_ref(global),
572                success_steps,
573            },
574        ));
575
576        // Return.
577        return;
578    }
579
580    // Let index be 0.
581    // Note: done with `enumerate` below.
582
583    // Let result be a list containing total null values.
584    let result: Rc<RefCell<Vec<Box<Heap<JSVal>>>>> = Default::default();
585
586    // For each promise of promises:
587    for (promise_index, promise) in promises.into_iter().enumerate() {
588        let result = result.clone();
589
590        {
591            // Note: adding a null value for this promise result.
592            let mut result_list = result.borrow_mut();
593            rooted!(in(*cx) let null_value = NullValue());
594            result_list.push(Heap::boxed(null_value.get()));
595        }
596
597        // Let promiseIndex be index.
598        // Note: done with `enumerate` above.
599
600        // Let fulfillmentHandler be the following steps given arg:
601        // Note: implemented with the `WaitForAllFulFillmentHandler`.
602
603        // Let fulfillmentHandler be CreateBuiltinFunction(fulfillmentHandler,  ):
604        // Note: passed below to avoid the need to root it.
605
606        // Perform PerformPromiseThen(promise, fulfillmentHandler, rejectionHandler).
607        let handler = PromiseNativeHandler::new(
608            global,
609            Some(Box::new(WaitForAllFulfillmentHandler {
610                success_steps: success_steps.clone(),
611                result,
612                promise_index,
613                fulfilled_count: fulfilled_count.clone(),
614            })),
615            Some(Box::new(rejection_handler.clone())),
616            can_gc,
617        );
618        promise.append_native_handler(&handler, realm, can_gc);
619
620        // Set index to index + 1.
621        // Note: done above with `enumerate`.
622    }
623}

625/// <https://webidl.spec.whatwg.org/#waiting-for-all-promise>
626pub(crate) fn wait_for_all_promise(
627    cx: SafeJSContext,
628    global: &GlobalScope,
629    promises: Vec<Rc<Promise>>,
630    realm: InRealm,
631    can_gc: CanGc,
632) -> Rc<Promise> {
633    // Let promise be a new promise of type Promise<sequence<T>> in realm.
634    let promise = Promise::new(global, can_gc);
635    let success_promise = promise.clone();
636    let failure_promise = promise.clone();
637
638    // Let successSteps be the following steps, given results:
639    let success_steps = Rc::new(move |results: Vec<HandleValue>| {
640        // Resolve promise with results.
641        success_promise.resolve_native(&results, can_gc);
642    });
643
644    // Let failureSteps be the following steps, given reason:
645    let failure_steps = Rc::new(move |reason: HandleValue| {
646        // Reject promise with reason.
647        failure_promise.reject_native(&reason, can_gc);
648    });
649
650    // Wait for all with promises, given successSteps and failureSteps.
651    wait_for_all(
652        cx,
653        global,
654        promises,
655        success_steps,
656        failure_steps,
657        realm,
658        can_gc,
659    );
660
661    // Return promise.
662    promise
663}
