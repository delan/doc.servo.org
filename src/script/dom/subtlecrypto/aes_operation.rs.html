46/// <https://w3c.github.io/webcrypto/#aes-cbc-operations-encrypt>
47pub(crate) fn encrypt_aes_cbc(
48    normalized_algorithm: &SubtleAesCbcParams,
49    key: &CryptoKey,
50    plaintext: &[u8],
51) -> Result<Vec<u8>, Error> {
52    // Step 1. If the iv member of normalizedAlgorithm does not have a length of 16 bytes, then
53    // throw an OperationError.
54    if normalized_algorithm.iv.len() != 16 {
55        return Err(Error::Operation(Some(
56            "The initialization vector length is not 16 bytes".into(),
57        )));
58    }
59
60    // Step 2. Let paddedPlaintext be the result of adding padding octets to plaintext according to
61    // the procedure defined in Section 10.3 of [RFC2315], step 2, with a value of k of 16.
62    // Step 3. Let ciphertext be the result of performing the CBC Encryption operation described in
63    // Section 6.2 of [NIST-SP800-38A] using AES as the block cipher, the iv member of
64    // normalizedAlgorithm as the IV input parameter and paddedPlaintext as the input plaintext.
65    let plaintext = Vec::from(plaintext);
66    let iv = GenericArray::from_slice(&normalized_algorithm.iv);
67    let ciphertext = match key.handle() {
68        Handle::Aes128(data) => {
69            let key_data = GenericArray::from_slice(data);
70            Aes128CbcEnc::new(key_data, iv).encrypt_padded_vec_mut::<Pkcs7>(&plaintext)
71        },
72        Handle::Aes192(data) => {
73            let key_data = GenericArray::from_slice(data);
74            Aes192CbcEnc::new(key_data, iv).encrypt_padded_vec_mut::<Pkcs7>(&plaintext)
75        },
76        Handle::Aes256(data) => {
77            let key_data = GenericArray::from_slice(data);
78            Aes256CbcEnc::new(key_data, iv).encrypt_padded_vec_mut::<Pkcs7>(&plaintext)
79        },
80        _ => return Err(Error::Data(Some("The key is not an AES key".into()))),
81    };
82
83    // Step 4. Return ciphertext.
84    Ok(ciphertext)
85}

87/// <https://w3c.github.io/webcrypto/#aes-cbc-operations-decrypt>
88pub(crate) fn decrypt_aes_cbc(
89    normalized_algorithm: &SubtleAesCbcParams,
90    key: &CryptoKey,
91    ciphertext: &[u8],
92) -> Result<Vec<u8>, Error> {
93    // Step 1. If the iv member of normalizedAlgorithm does not have a length of 16 bytes, then
94    // throw an OperationError.
95    if normalized_algorithm.iv.len() != 16 {
96        return Err(Error::Operation(Some(
97            "The initialization vector length is not 16 bytes".into(),
98        )));
99    }
100
101    // Step 2. If the length of ciphertext is zero or is not a multiple of 16 bytes, then throw an
102    // OperationError.
103    if ciphertext.is_empty() {
104        return Err(Error::Operation(Some("The ciphertext is empty".into())));
105    }
106    if ciphertext.len() % 16 != 0 {
107        return Err(Error::Operation(Some(
108            "The ciphertext length is not a multiple of 16 bytes".into(),
109        )));
110    }
111
112    // Step 3. Let paddedPlaintext be the result of performing the CBC Decryption operation
113    // described in Section 6.2 of [NIST-SP800-38A] using AES as the block cipher, the iv member of
114    // normalizedAlgorithm as the IV input parameter and ciphertext as the input ciphertext.
115    // Step 4. Let p be the value of the last octet of paddedPlaintext.
116    // Step 5. If p is zero or greater than 16, or if any of the last p octets of paddedPlaintext
117    // have a value which is not p, then throw an OperationError.
118    // Step 6. Let plaintext be the result of removing p octets from the end of paddedPlaintext.
119    let mut ciphertext = Vec::from(ciphertext);
120    let iv = GenericArray::from_slice(&normalized_algorithm.iv);
121    let plaintext = match key.handle() {
122        Handle::Aes128(data) => {
123            let key_data = GenericArray::from_slice(data);
124            Aes128CbcDec::new(key_data, iv)
125                .decrypt_padded_mut::<Pkcs7>(ciphertext.as_mut_slice())
126                .map_err(|_| Error::Operation(Some("Failed to perform CBC decryption".into())))?
127        },
128        Handle::Aes192(data) => {
129            let key_data = GenericArray::from_slice(data);
130            Aes192CbcDec::new(key_data, iv)
131                .decrypt_padded_mut::<Pkcs7>(ciphertext.as_mut_slice())
132                .map_err(|_| Error::Operation(Some("Failed to perform CBC decryption".into())))?
133        },
134        Handle::Aes256(data) => {
135            let key_data = GenericArray::from_slice(data);
136            Aes256CbcDec::new(key_data, iv)
137                .decrypt_padded_mut::<Pkcs7>(ciphertext.as_mut_slice())
138                .map_err(|_| Error::Operation(Some("Failed to perform CBC decryption".into())))?
139        },
140        _ => return Err(Error::Data(Some("The key is not an AES key".into()))),
141    };
142
143    // Step 7. Return plaintext.
144    Ok(plaintext.to_vec())
145}

203/// <https://w3c.github.io/webcrypto/#aes-gcm-operations-encrypt>
204pub(crate) fn encrypt_aes_gcm(
205    normalized_algorithm: &SubtleAesGcmParams,
206    key: &CryptoKey,
207    plaintext: &[u8],
208) -> Result<Vec<u8>, Error> {
209    // Step 1. If plaintext has a length greater than 2^39 - 256 bytes, then throw an OperationError.
210    if plaintext.len() as u64 > (2 << 39) - 256 {
211        return Err(Error::Operation(Some("The plaintext is too long".into())));
212    }
213
214    // Step 2. If the iv member of normalizedAlgorithm has a length greater than 2^64 - 1 bytes,
215    // then throw an OperationError.
216    // NOTE: servo does not currently support 128-bit platforms, so this can never happen
217
218    // Step 3. If the additionalData member of normalizedAlgorithm is present and has a length
219    // greater than 2^64 - 1 bytes, then throw an OperationError.
220    if normalized_algorithm
221        .additional_data
222        .as_ref()
223        .is_some_and(|data| data.len() > u64::MAX as usize)
224    {
225        return Err(Error::Operation(Some(
226            "The additional authentication data is too long".into(),
227        )));
228    }
229
230    // Step 4.
231    // If the tagLength member of normalizedAlgorithm is not present:
232    //     Let tagLength be 128.
233    // If the tagLength member of normalizedAlgorithm is one of 32, 64, 96, 104, 112, 120 or 128:
234    //     Let tagLength be equal to the tagLength member of normalizedAlgorithm
235    // Otherwise:
236    //     throw an OperationError.
237    let tag_length = match normalized_algorithm.tag_length {
238        None => 128,
239        Some(length) if matches!(length, 32 | 64 | 96 | 104 | 112 | 120 | 128) => length,
240        _ => {
241            return Err(Error::Operation(Some("The tag length is invalid".into())));
242        },
243    };
244
245    // Step 5. Let additionalData be the additionalData member of normalizedAlgorithm if present or
246    // an empty byte sequence otherwise.
247    let additional_data = normalized_algorithm
248        .additional_data
249        .as_deref()
250        .unwrap_or_default();
251
252    // Step 6. Let C and T be the outputs that result from performing the Authenticated Encryption
253    // Function described in Section 7.1 of [NIST-SP800-38D] using AES as the block cipher, the
254    // contents of the iv member of normalizedAlgorithm as the IV input parameter, the contents of
255    // additionalData as the A input parameter, tagLength as the t pre-requisite and the contents
256    // of plaintext as the input plaintext.
257    let key_length = key.handle().as_bytes().len();
258    let iv_length = normalized_algorithm.iv.len();
259    let mut ciphertext = plaintext.to_vec();
260    let key_bytes = key.handle().as_bytes();
261    let tag = match (key_length, iv_length) {
262        (16, 12) => {
263            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
264            <Aes128Gcm96Iv>::new_from_slice(key_bytes)
265                .expect("key length did not match")
266                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
267        },
268        (16, 16) => {
269            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
270            <Aes128Gcm128Iv>::new_from_slice(key_bytes)
271                .expect("key length did not match")
272                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
273        },
274        (24, 12) => {
275            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
276            <Aes192Gcm96Iv>::new_from_slice(key_bytes)
277                .expect("key length did not match")
278                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
279        },
280        (32, 12) => {
281            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
282            <Aes256Gcm96Iv>::new_from_slice(key_bytes)
283                .expect("key length did not match")
284                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
285        },
286        (16, 32) => {
287            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
288            <Aes128Gcm256Iv>::new_from_slice(key_bytes)
289                .expect("key length did not match")
290                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
291        },
292        (24, 32) => {
293            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
294            <Aes192Gcm256Iv>::new_from_slice(key_bytes)
295                .expect("key length did not match")
296                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
297        },
298        (32, 32) => {
299            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
300            <Aes256Gcm256Iv>::new_from_slice(key_bytes)
301                .expect("key length did not match")
302                .encrypt_in_place_detached(nonce, additional_data, &mut ciphertext)
303        },
304        _ => {
305            log::warn!(
306                "Missing AES-GCM encryption implementation with {key_length}-byte key and {iv_length}-byte IV"
307            );
308            return Err(Error::NotSupported(Some(format!(
309                "AES-GCM encryption with {key_length}-byte key and {iv_length}-byte IV is unsupported"
310            ))));
311        },
312    };
313
314    // Step 7. Let ciphertext be equal to C | T, where '|' denotes concatenation.
315    ciphertext.extend_from_slice(&tag.unwrap()[..tag_length as usize / 8]);
316
317    // Step 8. Return ciphertext.
318    Ok(ciphertext)
319}

321/// <https://w3c.github.io/webcrypto/#aes-gcm-operations-decrypt>
322pub(crate) fn decrypt_aes_gcm(
323    normalized_algorithm: &SubtleAesGcmParams,
324    key: &CryptoKey,
325    ciphertext: &[u8],
326) -> Result<Vec<u8>, Error> {
327    // Step 1.
328    // If the tagLength member of normalizedAlgorithm is not present:
329    //     Let tagLength be 128.
330    // If the tagLength member of normalizedAlgorithm is one of 32, 64, 96, 104, 112, 120 or 128:
331    //     Let tagLength be equal to the tagLength member of normalizedAlgorithm
332    // Otherwise:
333    //     throw an OperationError.
334    let tag_length = match normalized_algorithm.tag_length {
335        None => 128,
336        Some(length) if matches!(length, 32 | 64 | 96 | 104 | 112 | 120 | 128) => length as usize,
337        _ => {
338            return Err(Error::Operation(Some("The tag length is invalid".into())));
339        },
340    };
341
342    // Step 2. If ciphertext has a length in bits less than tagLength, then throw an
343    // OperationError.
344    if ciphertext.len() * 8 < tag_length {
345        return Err(Error::Operation(Some(
346            "The ciphertext is shorter than the tag".into(),
347        )));
348    }
349
350    // Step 3. If the iv member of normalizedAlgorithm has a length greater than 2^64 - 1 bytes,
351    // then throw an OperationError.
352    // NOTE: servo does not currently support 128-bit platforms, so this can never happen
353
354    // Step 4. If the additionalData member of normalizedAlgorithm is present and has a length
355    // greater than 2^64 - 1 bytes, then throw an OperationError.
356    // NOTE: servo does not currently support 128-bit platforms, so this can never happen
357
358    // Step 5. Let tag be the last tagLength bits of ciphertext.
359    // Step 6. Let actualCiphertext be the result of removing the last tagLength bits from
360    // ciphertext.
361    // NOTE: aes_gcm splits the ciphertext for us
362
363    // Step 7. Let additionalData be the additionalData member of normalizedAlgorithm if present or
364    // an empty byte sequence otherwise.
365    let additional_data = normalized_algorithm
366        .additional_data
367        .as_deref()
368        .unwrap_or_default();
369
370    // Step 8. Perform the Authenticated Decryption Function described in Section 7.2 of
371    // [NIST-SP800-38D] using AES as the block cipher, the iv member of normalizedAlgorithm as the
372    // IV input parameter, additionalData as the A input parameter, tagLength as the t
373    // pre-requisite, actualCiphertext as the input ciphertext, C and tag as the authentication
374    // tag, T.
375    // If the result of the algorithm is the indication of inauthenticity, "FAIL":
376    //     throw an OperationError
377    // Otherwise:
378    //     Let plaintext be the output P of the Authenticated Decryption Function.
379    let mut plaintext = ciphertext.to_vec();
380    let key_length = key.handle().as_bytes().len();
381    let iv_length = normalized_algorithm.iv.len();
382    let key_bytes = key.handle().as_bytes();
383    let result = match (key_length, iv_length) {
384        (16, 12) => {
385            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
386            <Aes128Gcm96Iv>::new_from_slice(key_bytes)
387                .expect("key length did not match")
388                .decrypt_in_place(nonce, additional_data, &mut plaintext)
389        },
390        (16, 16) => {
391            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
392            <Aes128Gcm128Iv>::new_from_slice(key_bytes)
393                .expect("key length did not match")
394                .decrypt_in_place(nonce, additional_data, &mut plaintext)
395        },
396        (24, 12) => {
397            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
398            <Aes192Gcm96Iv>::new_from_slice(key_bytes)
399                .expect("key length did not match")
400                .decrypt_in_place(nonce, additional_data, &mut plaintext)
401        },
402        (32, 12) => {
403            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
404            <Aes256Gcm96Iv>::new_from_slice(key_bytes)
405                .expect("key length did not match")
406                .decrypt_in_place(nonce, additional_data, &mut plaintext)
407        },
408        (16, 32) => {
409            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
410            <Aes128Gcm256Iv>::new_from_slice(key_bytes)
411                .expect("key length did not match")
412                .decrypt_in_place(nonce, additional_data, &mut plaintext)
413        },
414        (24, 32) => {
415            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
416            <Aes192Gcm256Iv>::new_from_slice(key_bytes)
417                .expect("key length did not match")
418                .decrypt_in_place(nonce, additional_data, &mut plaintext)
419        },
420        (32, 32) => {
421            let nonce = GenericArray::from_slice(&normalized_algorithm.iv);
422            <Aes256Gcm256Iv>::new_from_slice(key_bytes)
423                .expect("key length did not match")
424                .decrypt_in_place(nonce, additional_data, &mut plaintext)
425        },
426        _ => {
427            log::warn!(
428                "Missing AES-GCM decryption implementation with {key_length}-byte key and {iv_length}-byte IV"
429            );
430            return Err(Error::NotSupported(Some(format!(
431                "AES-GCM decryption with {key_length}-byte key and {iv_length}-byte IV is unsupported"
432            ))));
433        },
434    };
435    if result.is_err() {
436        return Err(Error::Operation(Some(
437            "Failed to perform GCM decryption".into(),
438        )));
439    }
440
441    Ok(plaintext)
442}

500/// <https://w3c.github.io/webcrypto/#aes-kw-operations-wrap-key>
501pub(crate) fn wrap_key_aes_kw(key: &CryptoKey, plaintext: &[u8]) -> Result<Vec<u8>, Error> {
502    // Step 1. If plaintext is not a multiple of 64 bits in length, then throw an OperationError.
503    if plaintext.len() % 8 != 0 {
504        return Err(Error::Operation(Some(
505            "The plaintext bit-length is not a multiple of 64".into(),
506        )));
507    }
508
509    // Step 2. Let ciphertext be the result of performing the Key Wrap operation described in
510    // Section 2.2.1 of [RFC3394] with plaintext as the plaintext to be wrapped and using the
511    // default Initial Value defined in Section 2.2.3.1 of the same document.
512    let key_data = key.handle().as_bytes();
513    let ciphertext = match key_data.len() {
514        16 => {
515            let key_array = GenericArray::from_slice(key_data);
516            let kek = KekAes128::new(key_array);
517            match kek.wrap_vec(plaintext) {
518                Ok(key) => key,
519                Err(_) => {
520                    return Err(Error::Operation(Some(
521                        "Plaintext key wrapping failed".into(),
522                    )));
523                },
524            }
525        },
526        24 => {
527            let key_array = GenericArray::from_slice(key_data);
528            let kek = KekAes192::new(key_array);
529            match kek.wrap_vec(plaintext) {
530                Ok(key) => key,
531                Err(_) => {
532                    return Err(Error::Operation(Some(
533                        "Plaintext key wrapping failed".into(),
534                    )));
535                },
536            }
537        },
538        32 => {
539            let key_array = GenericArray::from_slice(key_data);
540            let kek = KekAes256::new(key_array);
541            match kek.wrap_vec(plaintext) {
542                Ok(key) => key,
543                Err(_) => {
544                    return Err(Error::Operation(Some(
545                        "Plaintext key wrapping failed".into(),
546                    )));
547                },
548            }
549        },
550        _ => return Err(Error::Operation(Some("Key length is invalid".into()))),
551    };
552
553    // Step 3. Return ciphertext.
554    Ok(ciphertext)
555}

557/// <https://w3c.github.io/webcrypto/#aes-kw-operations-unwrap-key>
558pub(crate) fn unwrap_key_aes_kw(key: &CryptoKey, ciphertext: &[u8]) -> Result<Vec<u8>, Error> {
559    // Step 1. Let plaintext be the result of performing the Key Unwrap operation described in
560    // Section 2.2.2 of [RFC3394] with ciphertext as the input ciphertext and using the default
561    // Initial Value defined in Section 2.2.3.1 of the same document.
562    // Step 2. If the Key Unwrap operation returns an error, then throw an OperationError.
563    let key_data = key.handle().as_bytes();
564    let plaintext = match key_data.len() {
565        16 => {
566            let key_array = GenericArray::from_slice(key_data);
567            let kek = KekAes128::new(key_array);
568            match kek.unwrap_vec(ciphertext) {
569                Ok(key) => key,
570                Err(_) => {
571                    return Err(Error::Operation(Some(
572                        "Ciphertext key unwrapping failed".into(),
573                    )));
574                },
575            }
576        },
577        24 => {
578            let key_array = GenericArray::from_slice(key_data);
579            let kek = KekAes192::new(key_array);
580            match kek.unwrap_vec(ciphertext) {
581                Ok(key) => key,
582                Err(_) => {
583                    return Err(Error::Operation(Some(
584                        "Ciphertext key unwrapping failed".into(),
585                    )));
586                },
587            }
588        },
589        32 => {
590            let key_array = GenericArray::from_slice(key_data);
591            let kek = KekAes256::new(key_array);
592            match kek.unwrap_vec(ciphertext) {
593                Ok(key) => key,
594                Err(_) => {
595                    return Err(Error::Operation(Some(
596                        "Ciphertext key unwrapping failed".into(),
597                    )));
598                },
599            }
600        },
601        _ => {
602            return Err(Error::Operation(Some(
603                "Key length is not 128, 192, or 256 bits".into(),
604            )));
605        },
606    };
607
608    // Step 3. Return plaintext.
609    Ok(plaintext)
610}

663/// Helper function for
664/// <https://w3c.github.io/webcrypto/#aes-cbc-operations-generate-key>
665/// <https://w3c.github.io/webcrypto/#aes-gcm-operations-generate-key>
666/// <https://w3c.github.io/webcrypto/#aes-kw-operations-generate-key>
667#[allow(clippy::too_many_arguments)]
668fn generate_key_aes(
669    global: &GlobalScope,
670    normalized_algorithm: &SubtleAesKeyGenParams,
671    extractable: bool,
672    usages: Vec<KeyUsage>,
673    alg_name: &str,
674    allowed_usages: &[KeyUsage],
675    can_gc: CanGc,
676) -> Result<DomRoot<CryptoKey>, Error> {
677    // Step 1. If usages contains any entry which is not one of allowed_usages, then throw a SyntaxError.
678    if usages.iter().any(|usage| !allowed_usages.contains(usage)) {
679        return Err(Error::Syntax(Some(
680            "Key generation usage is not allowed".into(),
681        )));
682    }
683
684    // Step 2. If the length member of normalizedAlgorithm is not equal to one of 128, 192 or 256,
685    // then throw an OperationError.
686    if !matches!(normalized_algorithm.length, 128 | 192 | 256) {
687        return Err(Error::Operation(Some(
688            "Key length is not 128, 192, or 256 bits".into(),
689        )));
690    }
691
692    // Step 3. Generate an AES key of length equal to the length member of normalizedAlgorithm.
693    // Step 4. If the key generation step fails, then throw an OperationError.
694    let mut rand = vec![0; normalized_algorithm.length as usize / 8];
695    if OsRng.try_fill_bytes(&mut rand).is_err() {
696        return Err(Error::JSFailed);
697    }
698
699    let handle = match normalized_algorithm.length {
700        128 => Handle::Aes128(rand),
701        192 => Handle::Aes192(rand),
702        256 => Handle::Aes256(rand),
703        _ => {
704            return Err(Error::Operation(Some(
705                "Key length is not 128, 192, or 256 bits".into(),
706            )));
707        },
708    };
709
710    // Step 6. Let algorithm be a new AesKeyAlgorithm.
711    // Step 7. Set the name attribute of algorithm to alg_name.
712    // Step 8. Set the length attribute of algorithm to equal the length member of normalizedAlgorithm.
713    let algorithm = SubtleAesKeyAlgorithm {
714        name: alg_name.to_string(),
715        length: normalized_algorithm.length,
716    };
717
718    // Step 5. Let key be a new CryptoKey object representing the generated AES key.
719    // Step 9. Set the [[type]] internal slot of key to "secret".
720    // Step 10. Set the [[algorithm]] internal slot of key to algorithm.
721    // Step 11. Set the [[extractable]] internal slot of key to be extractable.
722    // Step 12. Set the [[usages]] internal slot of key to be usages.
723    let crypto_key = CryptoKey::new(
724        global,
725        KeyType::Secret,
726        extractable,
727        KeyAlgorithmAndDerivatives::AesKeyAlgorithm(algorithm),
728        usages,
729        handle,
730        can_gc,
731    );
732
733    // Step 13. Return key.
734    Ok(crypto_key)
735}

737/// Helper function for
738/// <https://w3c.github.io/webcrypto/#aes-cbc-operations-import-key>
739/// <https://w3c.github.io/webcrypto/#aes-gcm-operations-import-key>
740/// <https://w3c.github.io/webcrypto/#aes-kw-operations-import-key>
741fn import_key_aes(
742    global: &GlobalScope,
743    format: KeyFormat,
744    key_data: &[u8],
745    extractable: bool,
746    usages: Vec<KeyUsage>,
747    alg_name: &str,
748    can_gc: CanGc,
749) -> Result<DomRoot<CryptoKey>, Error> {
750    // Step 1. If usages contains an entry which is not one of "encrypt", "decrypt", "wrapKey" or
751    // "unwrapKey", then throw a SyntaxError.
752    if usages.iter().any(|usage| {
753        !matches!(
754            usage,
755            KeyUsage::Encrypt | KeyUsage::Decrypt | KeyUsage::WrapKey | KeyUsage::UnwrapKey
756        )
757    }) || usages.is_empty()
758    {
759        return Err(Error::Syntax(Some(
760            "Key generation usage is not allowed".into(),
761        )));
762    }
763
764    // Step 2.
765    let data;
766    match format {
767        // If format is "raw":
768        KeyFormat::Raw | KeyFormat::Raw_secret => {
769            // Step 2.1. Let data be keyData.
770            data = key_data.to_vec();
771
772            // Step 2.2. If the length in bits of data is not 128, 192 or 256 then throw a DataError.
773            if !matches!(data.len() * 8, 128 | 192 | 256) {
774                return Err(Error::Data(Some(
775                    "Key length is not 128, 192, or 256 bits".into(),
776                )));
777            }
778        },
779        // If format is "jwk":
780        KeyFormat::Jwk => {
781            // Step 2.1. If keyData is a JsonWebKey dictionary: Let jwk equal keyData.
782            // Otherwise: Throw a DataError.
783            // NOTE: Deserialize keyData to JsonWebKey dictionary by calling JsonWebKey::parse
784            let jwk = JsonWebKey::parse(GlobalScope::get_cx(), key_data)?;
785
786            // Step 2.2. If the kty field of jwk is not "oct", then throw a DataError.
787            if jwk.kty.as_ref().is_none_or(|kty| kty != "oct") {
788                return Err(Error::Data(Some("JWK `kty` field is not \"oct\"".into())));
789            }
790
791            // Step 2.3. If jwk does not meet the requirements of Section 6.4 of JSON Web
792            // Algorithms [JWA], then throw a DataError.
793            // NOTE: Done by Step 2.4 and 2.5.
794
795            // Step 2.4. Let data be the byte sequence obtained by decoding the k field of jwk.
796            data = jwk.decode_required_string_field(JwkStringField::K)?;
797
798            // NOTE: This function is shared by AES-CBC, AES-GCM and AES-KW.
799            // Different static texts are used in different AES types, in the following step.
800            let alg_matching = match alg_name {
801                ALG_AES_CBC => ["A128CBC", "A192CBC", "A256CBC"],
802                ALG_AES_GCM => ["A128GCM", "A192GCM", "A256GCM"],
803                ALG_AES_KW => ["A128KW", "A192KW", "A256KW"],
804                _ => unreachable!(),
805            };
806
807            // Step 2.5.
808            match data.len() * 8 {
809                // If the length in bits of data is 128:
810                128 => {
811                    // If the alg field of jwk is present, and is not "A128CBC", then throw a DataError.
812                    // If the alg field of jwk is present, and is not "A128GCM", then throw a DataError.
813                    // If the alg field of jwk is present, and is not "A128KW", then throw a DataError.
814                    // NOTE: Only perform the step of the corresponding AES type.
815                    if jwk.alg.as_ref().is_some_and(|alg| alg != alg_matching[0]) {
816                        return Err(Error::Data(Some(
817                            "JWK algorithm and key length do not match".into(),
818                        )));
819                    }
820                },
821                // If the length in bits of data is 192:
822                192 => {
823                    // If the alg field of jwk is present, and is not "A192CBC", then throw a DataError.
824                    // If the alg field of jwk is present, and is not "A192GCM", then throw a DataError.
825                    // If the alg field of jwk is present, and is not "A192KW", then throw a DataError.
826                    // NOTE: Only perform the step of the corresponding AES type.
827                    if jwk.alg.as_ref().is_some_and(|alg| alg != alg_matching[1]) {
828                        return Err(Error::Data(Some(
829                            "JWK algorithm and key length do not match".into(),
830                        )));
831                    }
832                },
833                // If the length in bits of data is 256:
834                256 => {
835                    // If the alg field of jwk is present, and is not "A256CBC", then throw a DataError.
836                    // If the alg field of jwk is present, and is not "A256GCM", then throw a DataError.
837                    // If the alg field of jwk is present, and is not "A256KW", then throw a DataError.
838                    // NOTE: Only perform the step of the corresponding AES type.
839                    if jwk.alg.as_ref().is_some_and(|alg| alg != alg_matching[2]) {
840                        return Err(Error::Data(Some(
841                            "JWK algorithm and key length do not match".into(),
842                        )));
843                    }
844                },
845                // Otherwise:
846                _ => {
847                    // throw a DataError.
848                    return Err(Error::Data(Some(
849                        "Key length is not 128, 192, or 256 bits".into(),
850                    )));
851                },
852            }
853
854            // Step 2.6. If usages is non-empty and the use field of jwk is present and is not
855            // "enc", then throw a DataError.
856            if !usages.is_empty() && jwk.use_.as_ref().is_some_and(|use_| use_ != "enc") {
857                return Err(Error::Data(Some("JWK usage is not encryption".into())));
858            }
859
860            // Step 2.7. If the key_ops field of jwk is present, and is invalid according to the
861            // requirements of JSON Web Key [JWK] or does not contain all of the specified usages
862            // values, then throw a DataError.
863            jwk.check_key_ops(&usages)?;
864
865            // Step 2.8. If the ext field of jwk is present and has the value false and extractable
866            // is true, then throw a DataError.
867            if jwk.ext.is_some_and(|ext| !ext) && extractable {
868                return Err(Error::Data(Some("JWK is not extractable".into())));
869            }
870        },
871        // Otherwise:
872        _ => {
873            // throw a NotSupportedError
874            return Err(Error::NotSupported(Some(
875                "Key format is unsupported".into(),
876            )));
877        },
878    };
879
880    // Step 5. Let algorithm be a new AesKeyAlgorithm.
881    // Step 6. Set the name attribute of algorithm to alg_name.
882    // Step 7. Set the length attribute of algorithm to the length, in bits, of data.
883    let algorithm = SubtleAesKeyAlgorithm {
884        name: alg_name.to_string(),
885        length: (data.len() * 8) as u16,
886    };
887
888    // Step 3. Let key be a new CryptoKey object representing an AES key with value data.
889    // Step 4. Set the [[type]] internal slot of key to "secret".
890    // Step 8. Set the [[algorithm]] internal slot of key to algorithm.
891    let handle = match data.len() * 8 {
892        128 => Handle::Aes128(data.to_vec()),
893        192 => Handle::Aes192(data.to_vec()),
894        256 => Handle::Aes256(data.to_vec()),
895        _ => {
896            return Err(Error::Data(Some(
897                "Key length is not 128, 192, or 256 bits".into(),
898            )));
899        },
900    };
901    let key = CryptoKey::new(
902        global,
903        KeyType::Secret,
904        extractable,
905        KeyAlgorithmAndDerivatives::AesKeyAlgorithm(algorithm),
906        usages,
907        handle,
908        can_gc,
909    );
910
911    // Step 9. Return key.
912    Ok(key)
913}

915/// Helper function for
916/// <https://w3c.github.io/webcrypto/#aes-cbc-operations-export-key>
917/// <https://w3c.github.io/webcrypto/#aes-gcm-operations-export-key>
918/// <https://w3c.github.io/webcrypto/#aes-kw-operations-export-key>
919fn export_key_aes(format: KeyFormat, key: &CryptoKey) -> Result<ExportedKey, Error> {
920    // Step 1. If the underlying cryptographic key material represented by the [[handle]]
921    // internal slot of key cannot be accessed, then throw an OperationError.
922    // NOTE: key.handle() guarantees access.
923
924    // Step 2.
925    let result;
926    match format {
927        // If format is "raw":
928        KeyFormat::Raw | KeyFormat::Raw_secret => match key.handle() {
929            // Step 2.1. Let data be a byte sequence containing the raw octets of the key
930            // represented by the [[handle]] internal slot of key.
931            // Step 2.2. Let result be data.
932            Handle::Aes128(key_data) => {
933                result = ExportedKey::Bytes(key_data.clone());
934            },
935            Handle::Aes192(key_data) => {
936                result = ExportedKey::Bytes(key_data.clone());
937            },
938            Handle::Aes256(key_data) => {
939                result = ExportedKey::Bytes(key_data.clone());
940            },
941            _ => unreachable!(),
942        },
943        // If format is "jwk":
944        KeyFormat::Jwk => {
945            // Step 2.1. Let jwk be a new JsonWebKey dictionary.
946            // Step 2.2. Set the kty attribute of jwk to the string "oct".
947            let mut jwk = JsonWebKey {
948                kty: Some(DOMString::from("oct")),
949                ..Default::default()
950            };
951
952            // Step 2.3. Set the k attribute of jwk to be a string containing the raw octets of
953            // the key represented by the [[handle]] internal slot of key, encoded according to
954            // Section 6.4 of JSON Web Algorithms [JWA].
955            match key.handle() {
956                Handle::Aes128(key) => jwk.encode_string_field(JwkStringField::K, key),
957                Handle::Aes192(key) => jwk.encode_string_field(JwkStringField::K, key),
958                Handle::Aes256(key) => jwk.encode_string_field(JwkStringField::K, key),
959                _ => unreachable!(),
960            };
961
962            // Step 2.4.
963            // If the length attribute of key is 128: Set the alg attribute of jwk to the string "A128CBC".
964            // If the length attribute of key is 192: Set the alg attribute of jwk to the string "A192CBC".
965            // If the length attribute of key is 256: Set the alg attribute of jwk to the string "A256CBC".
966            //
967            // If the length attribute of key is 128: Set the alg attribute of jwk to the string "A128GCM".
968            // If the length attribute of key is 192: Set the alg attribute of jwk to the string "A192GCM".
969            // If the length attribute of key is 256: Set the alg attribute of jwk to the string "A256GCM".
970            //
971            // If the length attribute of key is 128: Set the alg attribute of jwk to the string "A128KW".
972            // If the length attribute of key is 192: Set the alg attribute of jwk to the string "A192KW".
973            // If the length attribute of key is 256: Set the alg attribute of jwk to the string "A256KW".
974            //
975            // NOTE: Check key length via key.handle()
976            jwk.alg = Some(
977                match (key.handle(), key.algorithm().name()) {
978                    (Handle::Aes128(_), ALG_AES_CBC) => "A128CBC",
979                    (Handle::Aes192(_), ALG_AES_CBC) => "A192CBC",
980                    (Handle::Aes256(_), ALG_AES_CBC) => "A256CBC",
981                    (Handle::Aes128(_), ALG_AES_GCM) => "A128GCM",
982                    (Handle::Aes192(_), ALG_AES_GCM) => "A192GCM",
983                    (Handle::Aes256(_), ALG_AES_GCM) => "A256GCM",
984                    (Handle::Aes128(_), ALG_AES_KW) => "A128KW",
985                    (Handle::Aes192(_), ALG_AES_KW) => "A192KW",
986                    (Handle::Aes256(_), ALG_AES_KW) => "A256KW",
987                    _ => unreachable!(),
988                }
989                .into(),
990            );
991
992            // Step 2.5. Set the key_ops attribute of jwk to equal the [[usages]] internal slot of key.
993            jwk.set_key_ops(key.usages());
994
995            // Step 2.6. Set the ext attribute of jwk to equal the [[extractable]] internal slot of key.
996            jwk.ext = Some(key.Extractable());
997
998            // Step 2.7. Let result be jwk.
999            result = ExportedKey::Jwk(Box::new(jwk));
1000        },
1001        // Otherwise:
1002        _ => {
1003            // throw a NotSupportedError.
1004            return Err(Error::NotSupported(Some(
1005                "Key format is unsupported".into(),
1006            )));
1007        },
1008    };
1009
1010    // Step 3. Return result.
1011    Ok(result)
1012}
