33/// <https://w3c.github.io/webcrypto/#ecdh-operations-generate-key>
34pub(crate) fn generate_key(
35    global: &GlobalScope,
36    normalized_algorithm: &SubtleEcKeyGenParams,
37    extractable: bool,
38    usages: Vec<KeyUsage>,
39    can_gc: CanGc,
40) -> Result<CryptoKeyPair, Error> {
41    // Step 1. If usages contains an entry which is not "deriveKey" or "deriveBits" then throw a
42    // SyntaxError.
43    if usages
44        .iter()
45        .any(|usage| !matches!(usage, KeyUsage::DeriveKey | KeyUsage::DeriveBits))
46    {
47        return Err(Error::Syntax(None));
48    }
49
50    // Step 2.
51    // If the namedCurve member of normalizedAlgorithm is "P-256", "P-384" or "P-521":
52    //     Generate an Elliptic Curve key pair, as defined in [RFC6090] with domain parameters for
53    //     the curve identified by the namedCurve member of normalizedAlgorithm.
54    // If the namedCurve member of normalizedAlgorithm is a value specified in an applicable
55    // specification that specifies the use of that value with ECDH:
56    //     Perform the ECDH generation steps specified in that specification, passing in
57    //     normalizedAlgorithm and resulting in an elliptic curve key pair.
58    // Otherwise:
59    //     throw a NotSupportedError
60    // Step 3. If performing the operation results in an error, then throw a OperationError.
61    let (private_key_handle, public_key_handle) = match normalized_algorithm.named_curve.as_str() {
62        NAMED_CURVE_P256 => {
63            let private_key = SecretKey::<NistP256>::random(&mut OsRng);
64            let public_key = private_key.public_key();
65            (
66                Handle::P256PrivateKey(private_key),
67                Handle::P256PublicKey(public_key),
68            )
69        },
70        NAMED_CURVE_P384 => {
71            let private_key = SecretKey::<NistP384>::random(&mut OsRng);
72            let public_key = private_key.public_key();
73            (
74                Handle::P384PrivateKey(private_key),
75                Handle::P384PublicKey(public_key),
76            )
77        },
78        NAMED_CURVE_P521 => {
79            let private_key = SecretKey::<NistP521>::random(&mut OsRng);
80            let public_key = private_key.public_key();
81            (
82                Handle::P521PrivateKey(private_key),
83                Handle::P521PublicKey(public_key),
84            )
85        },
86        _ => return Err(Error::NotSupported(None)),
87    };
88
89    // Step 4. Let algorithm be a new EcKeyAlgorithm object.
90    // Step 5. Set the name member of algorithm to "ECDH".
91    // Step 6. Set the namedCurve attribute of algorithm to equal the namedCurve member of
92    // normalizedAlgorithm.
93    let algorithm = SubtleEcKeyAlgorithm {
94        name: ALG_ECDH.to_string(),
95        named_curve: normalized_algorithm.named_curve.clone(),
96    };
97
98    // Step 7. Let publicKey be a new CryptoKey representing the public key of the generated key pair.
99    // Step 8. Set the [[type]] internal slot of publicKey to "public"
100    // Step 9. Set the [[algorithm]] internal slot of publicKey to algorithm.
101    // Step 10. Set the [[extractable]] internal slot of publicKey to true.
102    // Step 11. Set the [[usages]] internal slot of publicKey to be the empty list.
103    let public_key = CryptoKey::new(
104        global,
105        KeyType::Public,
106        true,
107        KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm.clone()),
108        Vec::new(),
109        public_key_handle,
110        can_gc,
111    );
112
113    // Step 12. Let privateKey be a new CryptoKey representing the private key of the generated key pair.
114    // Step 13. Set the [[type]] internal slot of privateKey to "private"
115    // Step 14. Set the [[algorithm]] internal slot of privateKey to algorithm.
116    // Step 15. Set the [[extractable]] internal slot of privateKey to extractable.
117    // Step 16. Set the [[usages]] internal slot of privateKey to be the usage intersection of
118    // usages and [ "deriveKey", "deriveBits" ].
119    let private_key = CryptoKey::new(
120        global,
121        KeyType::Private,
122        extractable,
123        KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm),
124        usages
125            .iter()
126            .filter(|usage| matches!(usage, KeyUsage::DeriveKey | KeyUsage::DeriveBits))
127            .cloned()
128            .collect(),
129        private_key_handle,
130        can_gc,
131    );
132
133    // Step 17. Let result be a new CryptoKeyPair dictionary.
134    // Step 18. Set the publicKey attribute of result to be publicKey.
135    // Step 19. Set the privateKey attribute of result to be privateKey.
136    let result = CryptoKeyPair {
137        publicKey: Some(public_key),
138        privateKey: Some(private_key),
139    };
140
141    // Step 20. Return result.
142    Ok(result)
143}

145/// <https://w3c.github.io/webcrypto/#ecdh-operations-derive-bits>
146pub(crate) fn derive_bits(
147    normalized_algorithm: &SubtleEcdhKeyDeriveParams,
148    key: &CryptoKey,
149    length: Option<u32>,
150) -> Result<Vec<u8>, Error> {
151    // Step 1. If the [[type]] internal slot of key is not "private", then throw an
152    // InvalidAccessError.
153    if key.Type() != KeyType::Private {
154        return Err(Error::InvalidAccess(None));
155    }
156
157    // Step 2. Let publicKey be the public member of normalizedAlgorithm.
158    let public_key = normalized_algorithm.public.root();
159
160    // Step 3. If the [[type]] internal slot of publicKey is not "public", then throw an
161    // InvalidAccessError.
162    if public_key.Type() != KeyType::Public {
163        return Err(Error::InvalidAccess(None));
164    }
165
166    // Step 4. If the name attribute of the [[algorithm]] internal slot of publicKey is not equal
167    // to the name property of the [[algorithm]] internal slot of key, then throw an
168    // InvalidAccessError.
169    if public_key.algorithm().name() != key.algorithm().name() {
170        return Err(Error::InvalidAccess(None));
171    }
172
173    // Step 5. If the namedCurve attribute of the [[algorithm]] internal slot of publicKey is not
174    // equal to the namedCurve property of the [[algorithm]] internal slot of key, then throw an
175    // InvalidAccessError.
176    let (
177        KeyAlgorithmAndDerivatives::EcKeyAlgorithm(public_key_algorithm),
178        KeyAlgorithmAndDerivatives::EcKeyAlgorithm(key_algorithm),
179    ) = (public_key.algorithm(), key.algorithm())
180    else {
181        return Err(Error::Operation(None));
182    };
183    if public_key_algorithm.named_curve != key_algorithm.named_curve {
184        return Err(Error::InvalidAccess(None));
185    }
186
187    // Step 6.
188    // If the namedCurve property of the [[algorithm]] internal slot of key is "P-256", "P-384" or "P-521":
189    //     Step 6.1. Perform the ECDH primitive specified in [RFC6090] Section 4 with key as the EC
190    //     private key d and the EC public key represented by the [[handle]] internal slot of
191    //     publicKey as the EC public key.
192    //
193    //     Step 6.2. Let secret be a byte sequence containing the result of applying the field
194    //     element to octet string conversion defined in Section 6.2 of [RFC6090] to the output of
195    //     the ECDH primitive.
196    //
197    // If the namedCurve property of the [[algorithm]] internal slot of key is a value specified in
198    // an applicable specification that specifies the use of that value with ECDH:
199    //     Perform the ECDH derivation steps specified in that specification, passing in key and
200    //     publicKey and resulting in secret.
201    //
202    // Otherwise:
203    //     throw a NotSupportedError
204    //
205    // Step 7. If performing the operation results in an error, then throw a OperationError.
206    let secret = match key_algorithm.named_curve.as_str() {
207        NAMED_CURVE_P256 => {
208            let Handle::P256PrivateKey(private_key) = key.handle() else {
209                return Err(Error::Operation(None));
210            };
211            let Handle::P256PublicKey(public_key) = public_key.handle() else {
212                return Err(Error::Operation(None));
213            };
214            p256::ecdh::diffie_hellman(private_key.to_nonzero_scalar(), public_key.as_affine())
215                .raw_secret_bytes()
216                .to_vec()
217        },
218        NAMED_CURVE_P384 => {
219            let Handle::P384PrivateKey(private_key) = key.handle() else {
220                return Err(Error::Operation(None));
221            };
222            let Handle::P384PublicKey(public_key) = public_key.handle() else {
223                return Err(Error::Operation(None));
224            };
225            p384::ecdh::diffie_hellman(private_key.to_nonzero_scalar(), public_key.as_affine())
226                .raw_secret_bytes()
227                .to_vec()
228        },
229        NAMED_CURVE_P521 => {
230            let Handle::P521PrivateKey(private_key) = key.handle() else {
231                return Err(Error::Operation(None));
232            };
233            let Handle::P521PublicKey(public_key) = public_key.handle() else {
234                return Err(Error::Operation(None));
235            };
236            p521::ecdh::diffie_hellman(private_key.to_nonzero_scalar(), public_key.as_affine())
237                .raw_secret_bytes()
238                .to_vec()
239        },
240        _ => return Err(Error::NotSupported(None)),
241    };
242
243    // Step 8.
244    // If length is null:
245    //     Return secret
246    // Otherwise:
247    //     If the length in bits of secret is less than length:
248    //         throw an OperationError.
249    //     Otherwise:
250    //         Return a byte sequence containing the first length bits of secret.
251    match length {
252        None => Ok(secret),
253        Some(length) => {
254            if secret.len() * 8 < length as usize {
255                Err(Error::Operation(None))
256            } else {
257                let mut secret = secret[..length.div_ceil(8) as usize].to_vec();
258                if length % 8 != 0 {
259                    // Clean excess bits in last byte of secret.
260                    let mask = u8::MAX << (8 - length % 8);
261                    if let Some(last_byte) = secret.last_mut() {
262                        *last_byte &= mask;
263                    }
264                }
265                Ok(secret)
266            }
267        },
268    }
269}

271/// <https://w3c.github.io/webcrypto/#ecdh-operations-import-key>
272pub(crate) fn import_key(
273    global: &GlobalScope,
274    normalized_algorithm: &SubtleEcKeyImportParams,
275    format: KeyFormat,
276    key_data: &[u8],
277    extractable: bool,
278    usages: Vec<KeyUsage>,
279    can_gc: CanGc,
280) -> Result<DomRoot<CryptoKey>, Error> {
281    // Step 1. Let keyData be the key data to be imported.
282
283    // Step 2.
284    let key = match format {
285        KeyFormat::Spki => {
286            // Step 2.1. If usages is not empty then throw a SyntaxError.
287            if !usages.is_empty() {
288                return Err(Error::Syntax(None));
289            }
290
291            // Step 2.2. Let spki be the result of running the parse a subjectPublicKeyInfo
292            // algorithm over keyData
293            // Step 2.3. If an error occurred while parsing, then throw a DataError.
294            let spki = SubjectPublicKeyInfo::<_, BitString>::from_der(key_data)
295                .map_err(|_| Error::Data(None))?;
296
297            // Step 2.4. If the algorithm object identifier field of the algorithm
298            // AlgorithmIdentifier field of spki is not equal to the id-ecPublicKey object
299            // identifier defined in [RFC5480], then throw a DataError.
300            if spki.algorithm.oid != elliptic_curve::ALGORITHM_OID {
301                return Err(Error::Data(None));
302            }
303
304            // Step 2.5. If the parameters field of the algorithm AlgorithmIdentifier field of spki
305            // is absent, then throw a DataError.
306            // Step 2.6. Let params be the parameters field of the algorithm AlgorithmIdentifier
307            // field of spki.
308            // Step 2.7. If params is not an instance of the ECParameters ASN.1 type defined in
309            // [RFC5480] that specifies a namedCurve, then throw a DataError.
310            let Some(params): Option<EcParameters> = spki.algorithm.parameters else {
311                return Err(Error::Data(None));
312            };
313
314            // Step 2.8. Let namedCurve be a string whose initial value is undefined.
315            // Step 2.9.
316            // If params is equivalent to the secp256r1 object identifier defined in [RFC5480]:
317            //     Set namedCurve "P-256".
318            // If params is equivalent to the secp384r1 object identifier defined in [RFC5480]:
319            //     Set namedCurve "P-384".
320            // If params is equivalent to the secp521r1 object identifier defined in [RFC5480]:
321            //     Set namedCurve "P-521".
322            let named_curve = match params {
323                EcParameters::NamedCurve(NistP256::OID) => Some(NAMED_CURVE_P256),
324                EcParameters::NamedCurve(NistP384::OID) => Some(NAMED_CURVE_P384),
325                EcParameters::NamedCurve(NistP521::OID) => Some(NAMED_CURVE_P521),
326                _ => None,
327            };
328
329            // Step 2.10.
330            let handle = match named_curve {
331                // If namedCurve is not undefined:
332                Some(curve) => {
333                    // Step 2.10.1. Let publicKey be the Elliptic Curve public key identified by
334                    // performing the conversion steps defined in Section 2.3.4 of [SEC1] to the
335                    // subjectPublicKey field of spki.
336                    // Step 2.10.2. The uncompressed point format MUST be supported.
337                    // Step 2.10.3. If the implementation does not support the compressed point
338                    // format and a compressed point is provided, throw a DataError.
339                    // Step 2.10.4. If a decode error occurs or an identity point is found, throw a
340                    // DataError.
341                    let sec1_bytes = spki
342                        .subject_public_key
343                        .as_bytes()
344                        .ok_or(Error::Data(None))?;
345                    match curve {
346                        NAMED_CURVE_P256 => {
347                            let public_key = p256::PublicKey::from_sec1_bytes(sec1_bytes)
348                                .map_err(|_| Error::Data(None))?;
349                            Handle::P256PublicKey(public_key)
350                        },
351                        NAMED_CURVE_P384 => {
352                            let public_key = p384::PublicKey::from_sec1_bytes(sec1_bytes)
353                                .map_err(|_| Error::Data(None))?;
354                            Handle::P384PublicKey(public_key)
355                        },
356                        NAMED_CURVE_P521 => {
357                            let public_key = p521::PublicKey::from_sec1_bytes(sec1_bytes)
358                                .map_err(|_| Error::Data(None))?;
359                            Handle::P521PublicKey(public_key)
360                        },
361                        _ => unreachable!(),
362                    }
363
364                    // Step 2.10.5. Let key be a new CryptoKey that represents publicKey.
365                    // NOTE: CryptoKey is created in Step 2.13 - 2.17.
366                },
367                // Otherwise:
368                None => {
369                    // Step 2.10.1. Perform any key import steps defined by other applicable
370                    // specifications, passing format, spki and obtaining namedCurve and key.
371                    // Step 2.10.2. If an error occurred or there are no applicable specifications,
372                    // throw a DataError.
373                    // NOTE: We currently do not support applicable specifications.
374                    return Err(Error::NotSupported(None));
375                },
376            };
377
378            // Step 2.11. If namedCurve is defined, and not equal to the namedCurve member of
379            // normalizedAlgorithm, throw a DataError.
380            if named_curve.is_some_and(|curve| curve != normalized_algorithm.named_curve) {
381                return Err(Error::Data(None));
382            }
383
384            // Step 2.12. If the key value is not a valid point on the Elliptic Curve identified by
385            // the namedCurve member of normalizedAlgorithm throw a DataError.
386            // NOTE: Done in Step 2.10.
387
388            // Step 2.13. Set the [[type]] internal slot of key to "public"
389            // Step 2.14. Let algorithm be a new EcKeyAlgorithm.
390            // Step 2.15. Set the name attribute of algorithm to "ECDH".
391            // Step 2.16. Set the namedCurve attribute of algorithm to namedCurve.
392            // Step 2.17. Set the [[algorithm]] internal slot of key to algorithm.
393            let algorithm = SubtleEcKeyAlgorithm {
394                name: ALG_ECDH.to_string(),
395                named_curve: named_curve
396                    .expect("named_curve must exist here")
397                    .to_string(),
398            };
399            CryptoKey::new(
400                global,
401                KeyType::Public,
402                extractable,
403                KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm),
404                usages,
405                handle,
406                can_gc,
407            )
408        },
409        KeyFormat::Pkcs8 => {
410            // Step 2.1. If usages contains an entry which is not "deriveKey" or "deriveBits" then
411            // throw a SyntaxError.
412            if usages
413                .iter()
414                .any(|usage| !matches!(usage, KeyUsage::DeriveKey | KeyUsage::DeriveBits))
415            {
416                return Err(Error::Syntax(None));
417            }
418
419            // Step 2.2. Let privateKeyInfo be the result of running the parse a privateKeyInfo
420            // algorithm over keyData.
421            // Step 2.3. If an error occurs while parsing, throw a DataError.
422            let private_key_info =
423                PrivateKeyInfo::from_der(key_data).map_err(|_| Error::Data(None))?;
424
425            // Step 2.4. If the algorithm object identifier field of the privateKeyAlgorithm
426            // PrivateKeyAlgorithm field of privateKeyInfo is not equal to the id-ecPublicKey
427            // object identifier defined in [RFC5480], throw a DataError.
428            if private_key_info.algorithm.oid != elliptic_curve::ALGORITHM_OID {
429                return Err(Error::Data(None));
430            }
431
432            // Step 2.5. If the parameters field of the privateKeyAlgorithm
433            // PrivateKeyAlgorithmIdentifier field of privateKeyInfo is not present, throw a
434            // DataError.
435            // Step 2.6. Let params be the parameters field of the privateKeyAlgorithm
436            // PrivateKeyAlgorithmIdentifier field of privateKeyInfo.
437            // Step 2.7. If params is not an instance of the ECParameters ASN.1 type defined in
438            // [RFC5480] that specifies a namedCurve, then throw a DataError.
439            let params: EcParameters = if let Some(params) = private_key_info.algorithm.parameters {
440                params.decode_as().map_err(|_| Error::Data(None))?
441            } else {
442                return Err(Error::Data(None));
443            };
444
445            // Step 2.8. Let namedCurve be a string whose initial value is undefined.
446            // Step 2.9.
447            // If params is equivalent to the secp256r1 object identifier defined in [RFC5480]:
448            //     Set namedCurve to "P-256".
449            // If params is equivalent to the secp384r1 object identifier defined in [RFC5480]:
450            //     Set namedCurve to "P-384".
451            // If params is equivalent to the secp521r1 object identifier defined in [RFC5480]:
452            //     Set namedCurve to "P-521".
453            let named_curve = match params {
454                EcParameters::NamedCurve(NistP256::OID) => Some(NAMED_CURVE_P256),
455                EcParameters::NamedCurve(NistP384::OID) => Some(NAMED_CURVE_P384),
456                EcParameters::NamedCurve(NistP521::OID) => Some(NAMED_CURVE_P521),
457                _ => None,
458            };
459
460            // Step 2.10.
461            let handle = match named_curve {
462                // If namedCurve is not undefined:
463                Some(curve) => {
464                    // Step 2.10.1. Let ecPrivateKey be the result of performing the parse an ASN.1
465                    // structure algorithm, with data as the privateKey field of privateKeyInfo,
466                    // structure as the ASN.1 ECPrivateKey structure specified in Section 3 of
467                    // [RFC5915], and exactData set to true.
468                    // Step 2.10.2. If an error occurred while parsing, then throw a DataError.
469                    let ec_private_key = EcPrivateKey::try_from(private_key_info.private_key)
470                        .map_err(|_| Error::Data(None))?;
471
472                    // Step 2.10.3. If the parameters field of ecPrivateKey is present, and is not
473                    // an instance of the namedCurve ASN.1 type defined in [RFC5480], or does not
474                    // contain the same object identifier as the parameters field of the
475                    // privateKeyAlgorithm PrivateKeyAlgorithmIdentifier field of privateKeyInfo,
476                    // throw a DataError.
477                    if ec_private_key
478                        .parameters
479                        .is_some_and(|parameters| parameters != params)
480                    {
481                        return Err(Error::Data(None));
482                    }
483
484                    // Step 2.10.4. Let key be a new CryptoKey that represents the Elliptic Curve
485                    // private key identified by performing the conversion steps defined in Section
486                    // 3 of [RFC5915] using ecPrivateKey.
487                    // NOTE: CryptoKey is created in Step 2.13 - 2.17.
488                    match curve {
489                        NAMED_CURVE_P256 => {
490                            let private_key = p256::SecretKey::try_from(ec_private_key)
491                                .map_err(|_| Error::Data(None))?;
492                            Handle::P256PrivateKey(private_key)
493                        },
494                        NAMED_CURVE_P384 => {
495                            let private_key = p384::SecretKey::try_from(ec_private_key)
496                                .map_err(|_| Error::Data(None))?;
497                            Handle::P384PrivateKey(private_key)
498                        },
499                        NAMED_CURVE_P521 => {
500                            let private_key = p521::SecretKey::try_from(ec_private_key)
501                                .map_err(|_| Error::Data(None))?;
502                            Handle::P521PrivateKey(private_key)
503                        },
504                        _ => unreachable!(),
505                    }
506                },
507                // Otherwise:
508                None => {
509                    // Step 2.10.1. Perform any key import steps defined by other applicable
510                    // specifications, passing format, privateKeyInfo and obtaining namedCurve and
511                    // key.
512                    // Step 2.10.2. If an error occurred or there are no applicable specifications,
513                    // throw a DataError.
514                    // NOTE: We currently do not support applicable specifications.
515                    return Err(Error::NotSupported(None));
516                },
517            };
518
519            // Step 2.11. If namedCurve is defined, and not equal to the namedCurve member of
520            // normalizedAlgorithm, throw a DataError.
521            if named_curve.is_some_and(|curve| curve != normalized_algorithm.named_curve) {
522                return Err(Error::Data(None));
523            }
524
525            // Step 2.12. If the key value is not a valid point on the Elliptic Curve identified by
526            // the namedCurve member of normalizedAlgorithm throw a DataError.
527            // NOTE: Done in Step 2.10.
528
529            // Step 2.13. Set the [[type]] internal slot of key to "private".
530            // Step 2.14. Let algorithm be a new EcKeyAlgorithm.
531            // Step 2.15. Set the name attribute of algorithm to "ECDH".
532            // Step 2.16. Set the namedCurve attribute of algorithm to namedCurve.
533            // Step 2.17. Set the [[algorithm]] internal slot of key to algorithm.
534            let algorithm = SubtleEcKeyAlgorithm {
535                name: ALG_ECDH.to_string(),
536                named_curve: named_curve
537                    .expect("named_curve must exist here")
538                    .to_string(),
539            };
540            CryptoKey::new(
541                global,
542                KeyType::Private,
543                extractable,
544                KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm),
545                usages,
546                handle,
547                can_gc,
548            )
549        },
550        KeyFormat::Jwk => {
551            // Step 2.1.
552            // If keyData is a JsonWebKey dictionary:
553            //     Let jwk equal keyData.
554            // Otherwise:
555            //     Throw a DataError.
556            let jwk = JsonWebKey::parse(GlobalScope::get_cx(), key_data)?;
557
558            // Step 2.2. If the d field is present and if usages contains an entry which is not
559            // "deriveKey" or "deriveBits" then throw a SyntaxError.
560            if jwk.d.as_ref().is_some() &&
561                usages
562                    .iter()
563                    .any(|usage| !matches!(usage, KeyUsage::DeriveKey | KeyUsage::DeriveBits))
564            {
565                return Err(Error::Syntax(None));
566            }
567
568            // Step 2.3. If the d field is not present and if usages is not empty then throw a
569            // SyntaxError.
570            if jwk.d.as_ref().is_none() && !usages.is_empty() {
571                return Err(Error::Syntax(None));
572            }
573
574            // Step 2.4. If the kty field of jwk is not "EC", then throw a DataError.
575            if jwk.kty.as_ref().is_none_or(|kty| kty != "EC") {
576                return Err(Error::Data(None));
577            }
578
579            // Step 2.5. If usages is non-empty and the use field of jwk is present and is not
580            // equal to "enc" then throw a DataError.
581            if !usages.is_empty() && jwk.use_.as_ref().is_some_and(|use_| use_ != "enc") {
582                return Err(Error::Data(None));
583            }
584
585            // Step 2.6. If the key_ops field of jwk is present, and is invalid according to the
586            // requirements of JSON Web Key [JWK], or it does not contain all of the specified
587            // usages values, then throw a DataError.
588            jwk.check_key_ops(&usages)?;
589
590            // Step 2.7. If the ext field of jwk is present and has the value false and extractable
591            // is true, then throw a DataError.
592            if jwk.ext.is_some_and(|ext| !ext) && extractable {
593                return Err(Error::Data(None));
594            }
595
596            // Step 2.8. Let namedCurve be a string whose value is equal to the crv field of jwk.
597            // Step 2.9. If namedCurve is not equal to the namedCurve member of
598            // normalizedAlgorithm, throw a DataError.
599            let named_curve = jwk
600                .crv
601                .as_ref()
602                .filter(|crv| **crv == normalized_algorithm.named_curve)
603                .map(|crv| crv.to_string())
604                .ok_or(Error::Data(None))?;
605
606            // Step 2.10.
607            // If namedCurve is "P-256", "P-384" or "P-521":
608            let (handle, key_type) = if matches!(
609                named_curve.as_str(),
610                NAMED_CURVE_P256 | NAMED_CURVE_P384 | NAMED_CURVE_P521
611            ) {
612                // If the d field is present:
613                if jwk.d.is_some() {
614                    // Step 2.10.1. If jwk does not meet the requirements of Section 6.2.2 of
615                    // JSON Web Algorithms [JWA], then throw a DataError.
616                    let x = jwk.decode_required_string_field(JwkStringField::X)?;
617                    let y = jwk.decode_required_string_field(JwkStringField::Y)?;
618                    let d = jwk.decode_required_string_field(JwkStringField::D)?;
619
620                    // Step 2.10.2. Let key be a new CryptoKey object that represents the
621                    // Elliptic Curve private key identified by interpreting jwk according to
622                    // Section 6.2.2 of JSON Web Algorithms [JWA].
623                    // NOTE: CryptoKey is created in Step 2.12 - 2.15.
624                    let handle = match named_curve.as_str() {
625                        NAMED_CURVE_P256 => {
626                            let private_key =
627                                p256::SecretKey::from_slice(&d).map_err(|_| Error::Data(None))?;
628                            let mut sec1_bytes = vec![4u8];
629                            sec1_bytes.extend_from_slice(&x);
630                            sec1_bytes.extend_from_slice(&y);
631                            let encoded_point = EncodedPoint::from_bytes(&sec1_bytes)
632                                .map_err(|_| Error::Data(None))?;
633                            NistP256::validate_public_key(&private_key, &encoded_point)
634                                .map_err(|_| Error::Data(None))?;
635                            Handle::P256PrivateKey(private_key)
636                        },
637                        NAMED_CURVE_P384 => {
638                            let private_key =
639                                p384::SecretKey::from_slice(&d).map_err(|_| Error::Data(None))?;
640                            let mut sec1_bytes = vec![4u8];
641                            sec1_bytes.extend_from_slice(&x);
642                            sec1_bytes.extend_from_slice(&y);
643                            let encoded_point = EncodedPoint::from_bytes(&sec1_bytes)
644                                .map_err(|_| Error::Data(None))?;
645                            NistP384::validate_public_key(&private_key, &encoded_point)
646                                .map_err(|_| Error::Data(None))?;
647                            Handle::P384PrivateKey(private_key)
648                        },
649                        NAMED_CURVE_P521 => {
650                            let private_key =
651                                p521::SecretKey::from_slice(&d).map_err(|_| Error::Data(None))?;
652                            let mut sec1_bytes = vec![4u8];
653                            sec1_bytes.extend_from_slice(&x);
654                            sec1_bytes.extend_from_slice(&y);
655                            let encoded_point = EncodedPoint::from_bytes(&sec1_bytes)
656                                .map_err(|_| Error::Data(None))?;
657                            NistP521::validate_public_key(&private_key, &encoded_point)
658                                .map_err(|_| Error::Data(None))?;
659                            Handle::P521PrivateKey(private_key)
660                        },
661                        _ => unreachable!(),
662                    };
663
664                    // Step 2.10.3. Set the [[type]] internal slot of Key to "private".
665                    // NOTE: CryptoKey is created in Step 2.12 - 2.15.
666                    let key_type = KeyType::Private;
667
668                    (handle, key_type)
669                }
670                // Otherwise:
671                else {
672                    // Step 2.10.1. If jwk does not meet the requirements of Section 6.2.1 of
673                    // JSON Web Algorithms [JWA], then throw a DataError.
674                    let x = jwk.decode_required_string_field(JwkStringField::X)?;
675                    let y = jwk.decode_required_string_field(JwkStringField::Y)?;
676
677                    // Step 2.10.2. Let key be a new CryptoKey object that represents the
678                    // Elliptic Curve public key identified by interpreting jwk according to
679                    // Section 6.2.1 of JSON Web Algorithms [JWA].
680                    // NOTE: CryptoKey is created in Step 2.12 - 2.15.
681                    let handle = match named_curve.as_str() {
682                        NAMED_CURVE_P256 => {
683                            let mut sec1_bytes = vec![4u8];
684                            sec1_bytes.extend_from_slice(&x);
685                            sec1_bytes.extend_from_slice(&y);
686                            let encoded_point = EncodedPoint::from_bytes(&sec1_bytes)
687                                .map_err(|_| Error::Data(None))?;
688                            let public_key = p256::PublicKey::from_encoded_point(&encoded_point)
689                                .into_option()
690                                .ok_or(Error::Data(None))?;
691                            Handle::P256PublicKey(public_key)
692                        },
693                        NAMED_CURVE_P384 => {
694                            let mut sec1_bytes = vec![4u8];
695                            sec1_bytes.extend_from_slice(&x);
696                            sec1_bytes.extend_from_slice(&y);
697                            let encoded_point = EncodedPoint::from_bytes(&sec1_bytes)
698                                .map_err(|_| Error::Data(None))?;
699                            let public_key = p384::PublicKey::from_encoded_point(&encoded_point)
700                                .into_option()
701                                .ok_or(Error::Data(None))?;
702                            Handle::P384PublicKey(public_key)
703                        },
704                        NAMED_CURVE_P521 => {
705                            let mut sec1_bytes = vec![4u8];
706                            sec1_bytes.extend_from_slice(&x);
707                            sec1_bytes.extend_from_slice(&y);
708                            let encoded_point = EncodedPoint::from_bytes(&sec1_bytes)
709                                .map_err(|_| Error::Data(None))?;
710                            let public_key = p521::PublicKey::from_encoded_point(&encoded_point)
711                                .into_option()
712                                .ok_or(Error::Data(None))?;
713                            Handle::P521PublicKey(public_key)
714                        },
715                        _ => unreachable!(),
716                    };
717
718                    // Step 2.10.3. Set the [[type]] internal slot of Key to "public".
719                    // NOTE: CryptoKey is created in Step 2.12 - 2.15.
720                    let key_type = KeyType::Public;
721
722                    (handle, key_type)
723                }
724            }
725            // Otherwise
726            else {
727                // Step 2.10.1. Perform any key import steps defined by other applicable
728                // specifications, passing format, jwk and obtaining key.
729                // Step 2.10.2. If an error occurred or there are no applicable specifications,
730                // throw a DataError.
731                // NOTE: We currently do not support applicable specifications.
732                return Err(Error::NotSupported(None));
733            };
734
735            // Step 2.11. If the key value is not a valid point on the Elliptic Curve identified by
736            // the namedCurve member of normalizedAlgorithm throw a DataError.
737            // NOTE: Done in Step 2.10.
738
739            // Step 2.12. Let algorithm be a new instance of an EcKeyAlgorithm object.
740            // Step 2.13. Set the name attribute of algorithm to "ECDH".
741            // Step 2.14. Set the namedCurve attribute of algorithm to namedCurve.
742            // Step 2.15. Set the [[algorithm]] internal slot of key to algorithm.
743            let algorithm = SubtleEcKeyAlgorithm {
744                name: ALG_ECDH.to_string(),
745                named_curve,
746            };
747            CryptoKey::new(
748                global,
749                key_type,
750                extractable,
751                KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm),
752                usages,
753                handle,
754                can_gc,
755            )
756        },
757        KeyFormat::Raw | KeyFormat::Raw_public => {
758            // Step 2.1. If the namedCurve member of normalizedAlgorithm is not a named curve, then
759            // throw a DataError.
760            if !SUPPORTED_CURVES
761                .iter()
762                .any(|&supported_curve| supported_curve == normalized_algorithm.named_curve)
763            {
764                return Err(Error::Data(None));
765            }
766
767            // Step 2.2. If usages is not the empty list, then throw a SyntaxError.
768            if !usages.is_empty() {
769                return Err(Error::Syntax(None));
770            }
771
772            // Step 2.3.
773            // If namedCurve is "P-256", "P-384" or "P-521":
774            let handle = if matches!(
775                normalized_algorithm.named_curve.as_str(),
776                NAMED_CURVE_P256 | NAMED_CURVE_P384 | NAMED_CURVE_P521
777            ) {
778                // Step 2.3.1. Let Q be the Elliptic Curve public key on the curve identified by
779                // the namedCurve member of normalizedAlgorithm identified by performing the
780                // conversion steps defined in Section 2.3.4 of [SEC1] to keyData.
781                // Step 2.3.1. The uncompressed point format MUST be supported.
782                // Step 2.3.1. If the implementation does not support the compressed point format
783                // and a compressed point is provided, throw a DataError.
784                // Step 2.3.1. If a decode error occurs or an identity point is found, throw a
785                // DataError.
786                match normalized_algorithm.named_curve.as_str() {
787                    NAMED_CURVE_P256 => {
788                        let q = p256::PublicKey::from_sec1_bytes(key_data)
789                            .map_err(|_| Error::Data(None))?;
790                        Handle::P256PublicKey(q)
791                    },
792                    NAMED_CURVE_P384 => {
793                        let q = p384::PublicKey::from_sec1_bytes(key_data)
794                            .map_err(|_| Error::Data(None))?;
795                        Handle::P384PublicKey(q)
796                    },
797                    NAMED_CURVE_P521 => {
798                        let q = p521::PublicKey::from_sec1_bytes(key_data)
799                            .map_err(|_| Error::Data(None))?;
800                        Handle::P521PublicKey(q)
801                    },
802                    _ => unreachable!(),
803                }
804
805                // Step 2.3.1. Let key be a new CryptoKey that represents Q.
806                // NOTE: CryptoKey is created in Step 2.7 - 2.8.
807            }
808            // Otherwise:
809            else {
810                // Step. 2.3.1. Perform any key import steps defined by other applicable
811                // specifications, passing format, keyData and obtaining key.
812                // Step. 2.3.2. If an error occurred or there are no applicable specifications,
813                // throw a DataError.
814                // NOTE: We currently do not support applicable specifications.
815                return Err(Error::NotSupported(None));
816            };
817
818            // Step 2.4. Let algorithm be a new EcKeyAlgorithm object.
819            // Step 2.5. Set the name attribute of algorithm to "ECDH".
820            // Step 2.6. Set the namedCurve attribute of algorithm to equal the namedCurve member
821            // of normalizedAlgorithm.
822            let algorithm = SubtleEcKeyAlgorithm {
823                name: ALG_ECDH.to_string(),
824                named_curve: normalized_algorithm.named_curve.clone(),
825            };
826
827            // Step 2.7. Set the [[type]] internal slot of key to "public"
828            // Step 2.8. Set the [[algorithm]] internal slot of key to algorithm.
829            CryptoKey::new(
830                global,
831                KeyType::Public,
832                extractable,
833                KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm),
834                usages,
835                handle,
836                can_gc,
837            )
838        },
839        // Otherwise:
840        _ => {
841            // throw a NotSupportedError.
842            return Err(Error::NotSupported(None));
843        },
844    };
845
846    // Step 3. Return key.
847    Ok(key)
848}

850/// <https://w3c.github.io/webcrypto/#ecdh-operations-export-key>
851pub(crate) fn export_key(format: KeyFormat, key: &CryptoKey) -> Result<ExportedKey, Error> {
852    // Step 1. Let key be the CryptoKey to be exported.
853
854    // Step 2. If the underlying cryptographic key material represented by the [[handle]] internal
855    // slot of key cannot be accessed, then throw an OperationError.
856    // NOTE: Done in Step 3.
857
858    // Step 3.
859    let result = match format {
860        KeyFormat::Spki => {
861            // Step 3.1. If the [[type]] internal slot of key is not "public", then throw an
862            // InvalidAccessError.
863            if key.Type() != KeyType::Public {
864                return Err(Error::InvalidAccess(None));
865            }
866
867            // Step 3.2.
868            // Let data be an instance of the SubjectPublicKeyInfo ASN.1 structure defined in
869            // [RFC5280] with the following properties:
870            //     * Set the algorithm field to an AlgorithmIdentifier ASN.1 type with the
871            //     following properties:
872            //         * Set the algorithm field to the OID id-ecPublicKey defined in [RFC5480].
873            //         * Set the parameters field to an instance of the ECParameters ASN.1 type
874            //         defined in [RFC5480] as follows:
875            //             If the namedCurve attribute of the [[algorithm]] internal slot of key is
876            //             "P-256", "P-384" or "P-521":
877            //                 Let keyData be the byte sequence that represents the Elliptic Curve
878            //                 public key represented by the [[handle]] internal slot of key
879            //                 according to the encoding rules specified in Section 2.3.3 of [SEC1]
880            //                 and using the uncompressed form.
881            //                     If the namedCurve attribute of the [[algorithm]] internal slot
882            //                     of key is "P-256":
883            //                         Set parameters to the namedCurve choice with value equal to
884            //                         the object identifier secp256r1 defined in [RFC5480]
885            //                     If the namedCurve attribute of the [[algorithm]] internal slot
886            //                     of key is "P-384":
887            //                         Set parameters to the namedCurve choice with value equal to
888            //                         the object identifier secp384r1 defined in [RFC5480]
889            //                     If the namedCurve attribute of the [[algorithm]] internal slot
890            //                     of key is "P-521":
891            //                         Set parameters to the namedCurve choice with value equal to
892            //                         the object identifier secp521r1 defined in [RFC5480]
893            //             Otherwise:
894            //                 1. Perform any key export steps defined by other applicable
895            //                    specifications, passing format and the namedCurve attribute of
896            //                    the [[algorithm]] internal slot of key and obtaining
897            //                    namedCurveOid and keyData.
898            //                 2. Set parameters to the namedCurve choice with value equal to the
899            //                    object identifier namedCurveOid.
900            //     * Set the subjectPublicKey field to keyData
901            let data = match key.handle() {
902                Handle::P256PublicKey(public_key) => public_key.to_public_key_der(),
903                Handle::P384PublicKey(public_key) => public_key.to_public_key_der(),
904                Handle::P521PublicKey(public_key) => public_key.to_public_key_der(),
905                _ => return Err(Error::Operation(None)),
906            }
907            .map_err(|_| Error::Operation(None))?;
908
909            ExportedKey::Bytes(data.to_vec())
910        },
911        KeyFormat::Pkcs8 => {
912            // Step 3.1. If the [[type]] internal slot of key is not "private", then throw an
913            // InvalidAccessError.
914            if key.Type() != KeyType::Private {
915                return Err(Error::InvalidAccess(None));
916            }
917
918            // Step 3.2.
919            // Let data be an instance of the PrivateKeyInfo ASN.1 structure defined in [RFC5208]
920            // with the following properties:
921            //     * Set the version field to 0.
922            //     * Set the privateKeyAlgorithm field to a PrivateKeyAlgorithmIdentifier ASN.1
923            //     type with the following properties:
924            //         * Set the algorithm field to the OID id-ecPublicKey defined in [RFC5480].
925            //         * Set the parameters field to an instance of the ECParameters ASN.1 type
926            //         defined in [RFC5480] as follows:
927            //             If the namedCurve attribute of the [[algorithm]] internal slot of key is
928            //             "P-256", "P-384" or "P-521":
929            //                 Let keyData be the result of DER-encoding an instance of the
930            //                 ECPrivateKey structure defined in Section 3 of [RFC5915] for the
931            //                 Elliptic Curve private key represented by the [[handle]] internal
932            //                 slot of key and that conforms to the following:
933            //                     * The parameters field is present, and is equivalent to the
934            //                     parameters field of the privateKeyAlgorithm field of this
935            //                     PrivateKeyInfo ASN.1 structure.
936            //                     * The publicKey field is present and represents the Elliptic
937            //                     Curve public key associated with the Elliptic Curve private key
938            //                     represented by the [[handle]] internal slot of key.
939            //                     * If the namedCurve attribute of the [[algorithm]] internal slot
940            //                     of key is "P-256":
941            //                         Set parameters to the namedCurve choice with value equal to
942            //                         the object identifier secp256r1 defined in [RFC5480]
943            //                     * If the namedCurve attribute of the [[algorithm]] internal slot
944            //                     of key is "P-384":
945            //                         Set parameters to the namedCurve choice with value equal to
946            //                         the object identifier secp384r1 defined in [RFC5480]
947            //                     * If the namedCurve attribute of the [[algorithm]] internal slot
948            //                     of key is "P-521":
949            //                         Set parameters to the namedCurve choice with value equal to
950            //                         the object identifier secp521r1 defined in [RFC5480]
951            //             Otherwise:
952            //                 1. Perform any key export steps defined by other applicable
953            //                    specifications, passing format and the namedCurve attribute of
954            //                    the [[algorithm]] internal slot of key and obtaining
955            //                    namedCurveOid and keyData.
956            //                 2. Set parameters to the namedCurve choice with value equal to the
957            //                    object identifier namedCurveOid.
958            //     * Set the privateKey field to keyData.
959            let data = match key.handle() {
960                Handle::P256PrivateKey(private_key) => private_key.to_pkcs8_der(),
961                Handle::P384PrivateKey(private_key) => private_key.to_pkcs8_der(),
962                Handle::P521PrivateKey(private_key) => private_key.to_pkcs8_der(),
963                _ => return Err(Error::Operation(None)),
964            }
965            .map_err(|_| Error::Operation(None))?;
966
967            ExportedKey::Bytes(data.as_bytes().to_vec())
968        },
969        KeyFormat::Jwk => {
970            // Step 3.1. Let jwk be a new JsonWebKey dictionary.
971            // Step 3.2. Set the kty attribute of jwk to "EC".
972            let mut jwk = JsonWebKey {
973                kty: Some(DOMString::from("EC")),
974                ..Default::default()
975            };
976
977            // Step 3.3.
978            let named_curve =
979                if let KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm) = key.algorithm() {
980                    algorithm.named_curve.as_str()
981                } else {
982                    return Err(Error::Operation(None));
983                };
984            // If the namedCurve attribute of the [[algorithm]] internal slot of key is "P-256",
985            // "P-384" or "P-521":
986            if matches!(
987                named_curve,
988                NAMED_CURVE_P256 | NAMED_CURVE_P384 | NAMED_CURVE_P521
989            ) {
990                // Step 3.3.1.
991                // If the namedCurve attribute of the [[algorithm]] internal slot of key is
992                // "P-256":
993                //     Set the crv attribute of jwk to "P-256"
994                // If the namedCurve attribute of the [[algorithm]] internal slot of key is
995                // "P-384":
996                //     Set the crv attribute of jwk to "P-384"
997                // If the namedCurve attribute of the [[algorithm]] internal slot of key is
998                // "P-521":
999                //     Set the crv attribute of jwk to "P-521"
1000                jwk.crv = Some(DOMString::from(named_curve));
1001
1002                // Step 3.3.2. Set the x attribute of jwk according to the definition in Section
1003                // 6.2.1.2 of JSON Web Algorithms [JWA].
1004                // Step 3.3.3. Set the y attribute of jwk according to the definition in Section
1005                // 6.2.1.3 of JSON Web Algorithms [JWA].
1006                let (x, y) = match key.handle() {
1007                    Handle::P256PublicKey(public_key) => {
1008                        let encoded_point = public_key.to_encoded_point(false);
1009                        (
1010                            encoded_point.x().ok_or(Error::Operation(None))?.to_vec(),
1011                            encoded_point.y().ok_or(Error::Operation(None))?.to_vec(),
1012                        )
1013                    },
1014                    Handle::P384PublicKey(public_key) => {
1015                        let encoded_point = public_key.to_encoded_point(false);
1016                        (
1017                            encoded_point.x().ok_or(Error::Operation(None))?.to_vec(),
1018                            encoded_point.y().ok_or(Error::Operation(None))?.to_vec(),
1019                        )
1020                    },
1021                    Handle::P521PublicKey(public_key) => {
1022                        let encoded_point = public_key.to_encoded_point(false);
1023                        (
1024                            encoded_point.x().ok_or(Error::Operation(None))?.to_vec(),
1025                            encoded_point.y().ok_or(Error::Operation(None))?.to_vec(),
1026                        )
1027                    },
1028                    Handle::P256PrivateKey(private_key) => {
1029                        let public_key = private_key.public_key();
1030                        let encoded_point = public_key.to_encoded_point(false);
1031                        (
1032                            encoded_point.x().ok_or(Error::Operation(None))?.to_vec(),
1033                            encoded_point.y().ok_or(Error::Operation(None))?.to_vec(),
1034                        )
1035                    },
1036                    Handle::P384PrivateKey(private_key) => {
1037                        let public_key = private_key.public_key();
1038                        let encoded_point = public_key.to_encoded_point(false);
1039                        (
1040                            encoded_point.x().ok_or(Error::Operation(None))?.to_vec(),
1041                            encoded_point.y().ok_or(Error::Operation(None))?.to_vec(),
1042                        )
1043                    },
1044                    Handle::P521PrivateKey(private_key) => {
1045                        let public_key = private_key.public_key();
1046                        let encoded_point = public_key.to_encoded_point(false);
1047                        (
1048                            encoded_point.x().ok_or(Error::Operation(None))?.to_vec(),
1049                            encoded_point.y().ok_or(Error::Operation(None))?.to_vec(),
1050                        )
1051                    },
1052                    _ => return Err(Error::Operation(None)),
1053                };
1054                jwk.encode_string_field(JwkStringField::X, &x);
1055                jwk.encode_string_field(JwkStringField::Y, &y);
1056
1057                // Step 3.3.4.
1058                // If the [[type]] internal slot of key is "private"
1059                //     Set the d attribute of jwk according to the definition in Section 6.2.2.1 of
1060                //     JSON Web Algorithms [JWA].
1061                if key.Type() == KeyType::Private {
1062                    let d = match key.handle() {
1063                        Handle::P256PrivateKey(private_key) => private_key.to_bytes().to_vec(),
1064                        Handle::P384PrivateKey(private_key) => private_key.to_bytes().to_vec(),
1065                        Handle::P521PrivateKey(private_key) => private_key.to_bytes().to_vec(),
1066                        _ => return Err(Error::NotSupported(None)),
1067                    };
1068                    jwk.encode_string_field(JwkStringField::D, &d);
1069                }
1070            }
1071            // Otherwise:
1072            else {
1073                // Step 3.3.1. Perform any key export steps defined by other applicable
1074                // specifications, passing format and the namedCurve attribute of the [[algorithm]]
1075                // internal slot of key and obtaining namedCurve and a new value of jwk.
1076                // Step 3.3.2. Set the crv attribute of jwk to namedCurve.
1077                // NOTE: We currently do not support applicable specifications.
1078            }
1079
1080            // Step 3.4. Set the key_ops attribute of jwk to the usages attribute of key.
1081            jwk.set_key_ops(key.usages());
1082
1083            // Step 3.4. Set the ext attribute of jwk to the [[extractable]] internal slot of key.
1084            jwk.ext = Some(key.Extractable());
1085
1086            // Step 3.4. Let result be jwk.
1087            ExportedKey::Jwk(Box::new(jwk))
1088        },
1089        KeyFormat::Raw | KeyFormat::Raw_public => {
1090            // Step 3.1. If the [[type]] internal slot of key is not "public", then throw an
1091            // InvalidAccessError.
1092            if key.Type() != KeyType::Public {
1093                return Err(Error::InvalidAccess(None));
1094            }
1095
1096            // Step 3.2.
1097            // If the namedCurve attribute of the [[algorithm]] internal slot of key is "P-256",
1098            // "P-384" or "P-521":
1099            //     Let data be the byte sequence that represents the Elliptic Curve public key
1100            //     represented by the [[handle]] internal slot of key according to the encoding
1101            //     rules specified in Section 2.3.3 of [SEC1] and using the uncompressed form.
1102            // Otherwise:
1103            //     Perform any key export steps defined by other applicable specifications, passing
1104            //     format and the namedCurve attribute of the [[algorithm]] internal slot of key
1105            //     and obtaining namedCurve and data.
1106            //     NOTE: We currently do not support applicable specifications.
1107            let named_curve =
1108                if let KeyAlgorithmAndDerivatives::EcKeyAlgorithm(algorithm) = key.algorithm() {
1109                    algorithm.named_curve.as_str()
1110                } else {
1111                    return Err(Error::Operation(None));
1112                };
1113            let data = if matches!(
1114                named_curve,
1115                NAMED_CURVE_P256 | NAMED_CURVE_P384 | NAMED_CURVE_P521
1116            ) {
1117                match key.handle() {
1118                    Handle::P256PublicKey(public_key) => public_key.to_sec1_bytes().to_vec(),
1119                    Handle::P384PublicKey(public_key) => public_key.to_sec1_bytes().to_vec(),
1120                    Handle::P521PublicKey(public_key) => public_key.to_sec1_bytes().to_vec(),
1121                    _ => return Err(Error::Operation(None)),
1122                }
1123            } else {
1124                return Err(Error::NotSupported(None));
1125            };
1126
1127            // Step 3.3. Let result be data.
1128            ExportedKey::Bytes(data)
1129        },
1130        // Otherwise:
1131        _ => {
1132            // throw a NotSupportedError.
1133            return Err(Error::NotSupported(None));
1134        },
1135    };
1136
1137    // Step 4. Return result.
1138    Ok(result)
1139}
