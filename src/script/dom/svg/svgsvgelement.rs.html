29#[dom_struct]
30pub(crate) struct SVGSVGElement {
31    svggraphicselement: SVGGraphicsElement,
32    // The XML source of subtree rooted at this SVG element, serialized into
33    // a base64 encoded `data:` url. This is cached to avoid recomputation
34    // on each layout and must be invalidated when the subtree changes.
35    #[no_trace]
36    cached_serialized_data_url: DomRefCell<Option<Result<ServoUrl, ()>>>,
37}
38
39impl SVGSVGElement {
40    fn new_inherited(
41        local_name: LocalName,
42        prefix: Option<Prefix>,
43        document: &Document,
44    ) -> SVGSVGElement {
45        SVGSVGElement {
46            svggraphicselement: SVGGraphicsElement::new_inherited(local_name, prefix, document),
47            cached_serialized_data_url: Default::default(),
48        }
49    }
50
51    pub(crate) fn new(
52        local_name: LocalName,
53        prefix: Option<Prefix>,
54        document: &Document,
55        proto: Option<HandleObject>,
56        can_gc: CanGc,
57    ) -> DomRoot<SVGSVGElement> {
58        Node::reflect_node_with_proto(
59            Box::new(SVGSVGElement::new_inherited(local_name, prefix, document)),
60            document,
61            proto,
62            can_gc,
63        )
64    }
65
66    pub(crate) fn serialize_and_cache_subtree(&self) {
67        let can_gc = CanGc::note();
68        let cloned_nodes = self.process_use_elements(can_gc);
69
70        let serialize_result = self
71            .upcast::<Node>()
72            .xml_serialize(TraversalScope::IncludeNode);
73
74        self.cleanup_cloned_nodes(&cloned_nodes, can_gc);
75
76        let Ok(xml_source) = serialize_result else {
77            *self.cached_serialized_data_url.borrow_mut() = Some(Err(()));
78            return;
79        };
80
81        let xml_source: String = xml_source.into();
82        let base64_encoded_source = base64::engine::general_purpose::STANDARD.encode(xml_source);
83        let data_url = format!("data:image/svg+xml;base64,{}", base64_encoded_source);
84        match ServoUrl::parse(&data_url) {
85            Ok(url) => *self.cached_serialized_data_url.borrow_mut() = Some(Ok(url)),
86            Err(error) => error!("Unable to parse serialized SVG data url: {error}"),
87        };
88    }
89
90    fn process_use_elements(&self, can_gc: CanGc) -> Vec<DomRoot<Node>> {
91        let mut cloned_nodes = Vec::new();
92        let root_node = self.upcast::<Node>();
93
94        for node in root_node.traverse_preorder(ShadowIncluding::No) {
95            if let Some(element) = node.downcast::<Element>() {
96                if element.local_name() == &local_name!("use") {
97                    if let Some(cloned) = self.process_single_use_element(element, can_gc) {
98                        cloned_nodes.push(cloned);
99                    }
100                }
101            }
102        }
103
104        cloned_nodes
105    }
106
107    fn process_single_use_element(
108        &self,
109        use_element: &Element,
110        can_gc: CanGc,
111    ) -> Option<DomRoot<Node>> {
112        let href = use_element.get_string_attribute(&local_name!("href"));
113        let href_view = href.str();
114        let id_str = href_view.strip_prefix("#")?;
115        let id = DOMString::from(id_str);
116        let document = self.upcast::<Node>().owner_doc();
117        let referenced_element = document.GetElementById(id)?;
118        let referenced_node = referenced_element.upcast::<Node>();
119        let has_svg_ancestor = referenced_node
120            .inclusive_ancestors(ShadowIncluding::No)
121            .any(|ancestor| ancestor.is::<SVGSVGElement>());
122        if !has_svg_ancestor {
123            return None;
124        }
125        let cloned_node = Node::clone(
126            referenced_node,
127            None,
128            CloneChildrenFlag::CloneChildren,
129            None,
130            can_gc,
131        );
132        let root_node = self.upcast::<Node>();
133        let _ = root_node.AppendChild(&cloned_node, can_gc);
134
135        Some(cloned_node)
136    }
137
138    fn cleanup_cloned_nodes(&self, cloned_nodes: &[DomRoot<Node>], can_gc: CanGc) {
139        if cloned_nodes.is_empty() {
140            return;
141        }
142        let root_node = self.upcast::<Node>();
143
144        for cloned_node in cloned_nodes {
145            let _ = root_node.RemoveChild(cloned_node, can_gc);
146        }
147    }
148
149    fn invalidate_cached_serialized_subtree(&self) {
150        *self.cached_serialized_data_url.borrow_mut() = None;
151        self.upcast::<Node>().dirty(NodeDamage::Other);
152    }
153}

157}

159fn ratio_from_view_box(view_box: &AttrValue) -> Option<f32> {
160    let mut iter = view_box.chars();
161    let _min_x = parse_integer(&mut iter).ok()?;
162    let _min_y = parse_integer(&mut iter).ok()?;
163    let width = parse_unsigned_integer(&mut iter).ok()?;
164    if width == 0 {
165        return None;
166    }
167    let height = parse_unsigned_integer(&mut iter).ok()?;
168    if height == 0 {
169        return None;
170    }
171    let mut iter = iter.skip_while(|c| char_is_whitespace(*c));
172    iter.next().is_none().then(|| width as f32 / height as f32)
173}
174
175impl LayoutSVGSVGElementHelpers for LayoutDom<'_, SVGSVGElement> {
176    #[expect(unsafe_code)]
177    fn data(self) -> SVGElementData {
178        let element = self.upcast::<Element>();
179        let get_size = |attr| {
180            element
181                .get_attr_for_layout(&ns!(), &attr)
182                .map(|val| val.as_int())
183                .filter(|val| *val >= 0)
184        };
185        let width = get_size(local_name!("width"));
186        let height = get_size(local_name!("height"));
187        let ratio = match (width, height) {
188            (Some(width), Some(height)) if width != 0 && height != 0 => {
189                Some(width as f32 / height as f32)
190            },
191            _ => element
192                .get_attr_for_layout(&ns!(), &local_name!("viewBox"))
193                .and_then(ratio_from_view_box),
194        };
195        SVGElementData {
196            source: unsafe {
197                self.unsafe_get()
198                    .cached_serialized_data_url
199                    .borrow_for_layout()
200                    .clone()
201            },
202            width,
203            height,
204            ratio,
205        }
206    }
207}

209impl VirtualMethods for SVGSVGElement {
210    fn super_type(&self) -> Option<&dyn VirtualMethods> {
211        Some(self.upcast::<SVGGraphicsElement>() as &dyn VirtualMethods)
212    }
213
214    fn attribute_mutated(&self, attr: &Attr, mutation: AttributeMutation, can_gc: CanGc) {
215        self.super_type()
216            .unwrap()
217            .attribute_mutated(attr, mutation, can_gc);
218
219        self.invalidate_cached_serialized_subtree();
220    }
221
222    fn attribute_affects_presentational_hints(&self, attr: &Attr) -> bool {
223        match attr.local_name() {
224            &local_name!("width") | &local_name!("height") => true,
225            _ => self
226                .super_type()
227                .unwrap()
228                .attribute_affects_presentational_hints(attr),
229        }
230    }
231
232    fn parse_plain_attribute(&self, name: &LocalName, value: DOMString) -> AttrValue {
233        match *name {
234            // TODO: This should accept lengths in arbitrary units instead of assuming px.
235            local_name!("width") | local_name!("height") => AttrValue::from_i32(value.into(), -1),
236            _ => self
237                .super_type()
238                .unwrap()
239                .parse_plain_attribute(name, value),
240        }
241    }
242
243    fn children_changed(&self, mutation: &ChildrenMutation, can_gc: CanGc) {
244        if let Some(super_type) = self.super_type() {
245            super_type.children_changed(mutation, can_gc);
246        }
247
248        self.invalidate_cached_serialized_subtree();
249    }
