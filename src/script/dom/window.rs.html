217/// How long we should wait before performing the initial reflow after `<body>` is parsed,
218/// assuming that `<body>` take this long to parse.
219const INITIAL_REFLOW_DELAY: Duration = Duration::from_millis(200);
220
221/// During loading and parsing, layouts are suppressed to avoid flashing incomplete page
222/// contents.
223///
224/// Exceptions:
225///  - Parsing the body takes so long, that layouts are no longer suppressed in order
226///    to show the user that the page is loading.
227///  - Script triggers a layout query or scroll event in which case, we want to layout
228///    but not display the contents.
229///
230/// For more information see: <https://github.com/servo/servo/pull/6028>.
231#[derive(Clone, Copy, MallocSizeOf)]
232enum LayoutBlocker {
233    /// The first load event hasn't been fired and we have not started to parse the `<body>` yet.
234    WaitingForParse,
235    /// The body is being parsed the `<body>` starting at the `Instant` specified.
236    Parsing(Instant),
237    /// The body finished parsing and the `load` event has been fired or parsing took so
238    /// long, that we are going to do layout anyway. Note that subsequent changes to the body
239    /// can trigger parsing again, but the `Window` stays in this state.
240    FiredLoadEventOrParsingTimerExpired,
241}
242
243impl LayoutBlocker {
244    fn layout_blocked(&self) -> bool {
245        !matches!(self, Self::FiredLoadEventOrParsingTimerExpired)
246    }
247}

999// https://html.spec.whatwg.org/multipage/#atob
1000pub(crate) fn base64_btoa(input: DOMString) -> Fallible<DOMString> {
1001    // "The btoa() method must throw an InvalidCharacterError exception if
1002    //  the method's first argument contains any character whose code point
1003    //  is greater than U+00FF."
1004    if input.str().chars().any(|c: char| c > '\u{FF}') {
1005        Err(Error::InvalidCharacter(None))
1006    } else {
1007        // "Otherwise, the user agent must convert that argument to a
1008        //  sequence of octets whose nth octet is the eight-bit
1009        //  representation of the code point of the nth character of
1010        //  the argument,"
1011        let octets = input
1012            .str()
1013            .chars()
1014            .map(|c: char| c as u8)
1015            .collect::<Vec<u8>>();
1016
1017        // "and then must apply the base64 algorithm to that sequence of
1018        //  octets, and return the result. [RFC4648]"
1019        let config =
1020            base64::engine::general_purpose::GeneralPurposeConfig::new().with_encode_padding(true);
1021        let engine = base64::engine::GeneralPurpose::new(&base64::alphabet::STANDARD, config);
1022        Ok(DOMString::from(engine.encode(octets)))
1023    }
1024}

1026// https://html.spec.whatwg.org/multipage/#atob
1027pub(crate) fn base64_atob(input: DOMString) -> Fallible<DOMString> {
1028    // "Remove all space characters from input."
1029    fn is_html_space(c: char) -> bool {
1030        HTML_SPACE_CHARACTERS.contains(&c)
1031    }
1032    let without_spaces = input
1033        .str()
1034        .chars()
1035        .filter(|&c| !is_html_space(c))
1036        .collect::<String>();
1037    let mut input = &*without_spaces;
1038
1039    // "If the length of input divides by 4 leaving no remainder, then:
1040    //  if input ends with one or two U+003D EQUALS SIGN (=) characters,
1041    //  remove them from input."
1042    if input.len() % 4 == 0 {
1043        if input.ends_with("==") {
1044            input = &input[..input.len() - 2]
1045        } else if input.ends_with('=') {
1046            input = &input[..input.len() - 1]
1047        }
1048    }
1049
1050    // "If the length of input divides by 4 leaving a remainder of 1,
1051    //  throw an InvalidCharacterError exception and abort these steps."
1052    if input.len() % 4 == 1 {
1053        return Err(Error::InvalidCharacter(None));
1054    }
1055
1056    // "If input contains a character that is not in the following list of
1057    //  characters and character ranges, throw an InvalidCharacterError
1058    //  exception and abort these steps:
1059    //
1060    //  U+002B PLUS SIGN (+)
1061    //  U+002F SOLIDUS (/)
1062    //  Alphanumeric ASCII characters"
1063    if input
1064        .chars()
1065        .any(|c| c != '+' && c != '/' && !c.is_alphanumeric())
1066    {
1067        return Err(Error::InvalidCharacter(None));
1068    }
1069
1070    let config = base64::engine::general_purpose::GeneralPurposeConfig::new()
1071        .with_decode_padding_mode(base64::engine::DecodePaddingMode::RequireNone)
1072        .with_decode_allow_trailing_bits(true);
1073    let engine = base64::engine::GeneralPurpose::new(&base64::alphabet::STANDARD, config);
1074
1075    let data = engine
1076        .decode(input)
1077        .map_err(|_| Error::InvalidCharacter(None))?;
1078    Ok(data.iter().map(|&b| b as char).collect::<String>().into())
1079}

1203        // We let the embedder handle these steps.

1219        // "prompt", and message.
1220        // TODO: Add support for WebDriver BiDi.
1221        //
1222        // Step 7: Let result be null.
1223        //
1224        // Step 8: If userPromptHandler is "none", then:
1225        //  1. Pause while waiting for the user's response.
1226        //  2. If the user did not abort, then set result to the string that the user responded with.
1227        //
1228        // Step 9: Otherwise, if userPromptHandler is "accept", then set result to the empty string.
1229        // TODO: Implement this.
1230        //
1231        // Step 10: Invoke WebDriver BiDi user prompt closed with this, "prompt", false if

1239            Err(_) => {
1240                warn!("Prompt dialog was cancelled or failed to show.");

1350    fn Close(&self) {

1352        let window_proxy = match self.window_proxy.get() {
1353            Some(proxy) => proxy,
1354            // Step 2. If thisTraversable is not a top-level traversable, then return.
1355            None => return,
1356        };
1357        // Step 3. If thisTraversable's is closing is true, then return.
1358        if window_proxy.is_closing() {
1359            return;
1360        }
1361        // Note: check the length of the "session history", as opposed to the joint session history?
1362        // see https://github.com/whatwg/html/issues/3734
1363        if let Ok(history_length) = self.History().GetLength() {
1364            let is_auxiliary = window_proxy.is_auxiliary();
1365
1366            // https://html.spec.whatwg.org/multipage/#script-closable
1367            let is_script_closable = (self.is_top_level() && history_length == 1) ||
1368                is_auxiliary ||
1369                pref!(dom_allow_scripts_to_close_windows);
1370
1371            // TODO: rest of Step 3:
1372            // Is the incumbent settings object's responsible browsing context familiar with current?
1373            // Is the incumbent settings object's responsible browsing context allowed to navigate current?
1374            if is_script_closable {
1375                // Step 6.1. Set thisTraversable's is closing to true.
1376                window_proxy.close();
1377
1378                // Step 6.2. Queue a task on the DOM manipulation task source to definitely close thisTraversable.
1379                let this = Trusted::new(self);
1380                let task = task!(window_close_browsing_context: move || {
1381                    let window = this.root();
1382                    window.definitely_close(CanGc::note());
1383                });

1386                    .dom_manipulation_task_source()

1388            }
1389        }

1837    /// <https://drafts.csswg.org/cssom/#dom-window-getcomputedstyle>
1838    fn GetComputedStyle(
1839        &self,
1840        element: &Element,
1841        pseudo: Option<DOMString>,
1842    ) -> DomRoot<CSSStyleDeclaration> {
1843        // Step 2: Let obj be elt.
1844        // We don't store CSSStyleOwner directly because it stores a `Dom` which must be
1845        // rooted. This avoids the rooting the value temporarily.
1846        let mut is_null = false;
1847
1848        // Step 3: If pseudoElt is provided, is not the empty string, and starts with a colon, then:
1849        // Step 3.1: Parse pseudoElt as a <pseudo-element-selector>, and let type be the result.
1850        let pseudo = pseudo.map(|mut s| {
1851            s.make_ascii_lowercase();
1852            s
1853        });
1854        let pseudo = match pseudo {
1855            Some(ref pseudo) if pseudo == ":before" || pseudo == "::before" => {
1856                Some(PseudoElement::Before)
1857            },
1858            Some(ref pseudo) if pseudo == ":after" || pseudo == "::after" => {
1859                Some(PseudoElement::After)
1860            },
1861            Some(ref pseudo) if pseudo == "::selection" => Some(PseudoElement::Selection),
1862            Some(ref pseudo) if pseudo == "::marker" => Some(PseudoElement::Marker),
1863            Some(ref pseudo) if pseudo.starts_with(':') => {
1864                // Step 3.2: If type is failure, or is a ::slotted() or ::part()
1865                // pseudo-element, let obj be null.
1866                is_null = true;
1867                None
1868            },
1869            _ => None,
1870        };
1871
1872        // Step 4. Let decls be an empty list of CSS declarations.
1873        // Step 5: If obj is not null, and elt is connected, part of the flat tree, and
1874        // its shadow-including root has a browsing context which either doesnt have a
1875        // browsing context container, or whose browsing context container is being
1876        // rendered, set decls to a list of all longhand properties that are supported CSS
1877        // properties, in lexicographical order, with the value being the resolved value
1878        // computed for obj using the style rules associated with doc.  Additionally,
1879        // append to decls all the custom properties whose computed value for obj is not
1880        // the guaranteed-invalid value.
1881        //
1882        // Note: The specification says to generate the list of declarations beforehand, yet
1883        // also says the list should be alive. This is why we do not do step 4 and 5 here.
1884        // See: https://github.com/w3c/csswg-drafts/issues/6144
1885        //
1886        // Step 6:  Return a live CSSStyleProperties object with the following properties:
1887        CSSStyleDeclaration::new(
1888            self,
1889            if is_null {
1890                CSSStyleOwner::Null
1891            } else {
1892                CSSStyleOwner::Element(Dom::from_ref(element))
1893            },
1894            pseudo,
1895            CSSModificationAccess::Readonly,
1896            CanGc::note(),
1897        )

2445    /// <https://drafts.csswg.org/cssom-view/#dom-window-scroll>
2446    pub(crate) fn scroll(&self, x: f32, y: f32, behavior: ScrollBehavior) {
2447        // Step 3: Normalize non-finite values for x and y.
2448        let xfinite = if x.is_finite() { x } else { 0.0 };
2449        let yfinite = if y.is_finite() { y } else { 0.0 };
2450
2451        // Step 4: If there is no viewport, abort these steps.
2452        // Currently every frame has a viewport in Servo.
2453
2454        // Step 5. Let `viewport width` be the width of the viewport excluding the width
2455        // of the scroll bar, if any.
2456        // Step 6. `Let viewport height` be the height of the viewport excluding the
2457        // height of the scroll bar, if any.
2458        //
2459        // TODO: Servo does not yet support scrollbars.
2460        let viewport = self.viewport_details.get().size;
2461
2462        // Step 7:
2463        // If the viewport has rightward overflow direction
2464        //    Let x be max(0, min(x, viewport scrolling area width - viewport width)).
2465        // If the viewport has leftward overflow direction
2466        //    Let x be min(0, max(x, viewport width - viewport scrolling area width)).
2467        // TODO: Implement this.
2468
2469        // Step 8:
2470        // If the viewport has downward overflow direction
2471        //    Let y be max(0, min(y, viewport scrolling area height - viewport height)).
2472        // If the viewport has upward overflow direction
2473        //    Let y be min(0, max(y, viewport height - viewport scrolling area height)).
2474        // TODO: Implement this.
2475
2476        // Step 9: Let position be the scroll position the viewport would have by aligning
2477        // the x-coordinate x of the viewport scrolling area with the left of the viewport
2478        // and aligning the y-coordinate y of the viewport scrolling area with the top of
2479        // the viewport.
2480        let scrolling_area = self.scrolling_area_query(None).to_f32();
2481        let x = xfinite.clamp(0.0, 0.0f32.max(scrolling_area.width() - viewport.width));
2482        let y = yfinite.clamp(0.0, 0.0f32.max(scrolling_area.height() - viewport.height));
2483
2484        // Step 10: If position is the same as the viewports current scroll position, and
2485        // the viewport does not have an ongoing smooth scroll, abort these steps.
2486        let scroll_offset = self.scroll_offset();
2487        if x == scroll_offset.x && y == scroll_offset.y {
2488            return;
2489        }
2490
2491        // Step 11: Let document be the viewports associated Document.
2492        // Step 12: Perform a scroll of the viewport to position, documents root element
2493        // as the associated element, if there is one, or null otherwise, and the scroll
2494        // behavior being the value of the behavior dictionary member of options.
2495        self.perform_a_scroll(x, y, self.pipeline_id().root_scroll_id(), behavior, None);
2496    }

2549    /// Reflows the page unconditionally if possible and not suppressed. This method will wait for
2550    /// the layout to complete. If there is no window size yet, the page is presumed invisible and
2551    /// no reflow is performed. If reflow is suppressed, no reflow will be performed for ForDisplay
2552    /// goals.
2553    ///
2554    /// NOTE: This method should almost never be called directly! Layout and rendering updates should
2555    /// happen as part of the HTML event loop via *update the rendering*.
2556    pub(crate) fn reflow(&self, reflow_goal: ReflowGoal) -> ReflowPhasesRun {
2557        let document = self.Document();
2558
2559        // Never reflow inactive Documents.
2560        if !document.is_fully_active() {
2561            return ReflowPhasesRun::empty();
2562        }
2563
2564        self.Document().ensure_safe_to_run_script_or_layout();
2565
2566        // If layouts are blocked, we block all layouts that are for display only. Other
2567        // layouts (for queries and scrolling) are not blocked, as they do not display
2568        // anything and script expects the layout to be up-to-date after they run.
2569        let pipeline_id = self.pipeline_id();
2570        if reflow_goal == ReflowGoal::UpdateTheRendering &&
2571            self.layout_blocker.get().layout_blocked()
2572        {
2573            debug!("Suppressing pre-load-event reflow pipeline {pipeline_id}");
2574            return ReflowPhasesRun::empty();
2575        }
2576
2577        debug!("script: performing reflow for goal {reflow_goal:?}");
2578        let marker = if self.need_emit_timeline_marker(TimelineMarkerType::Reflow) {
2579            Some(TimelineMarker::start("Reflow".to_owned()))
2580        } else {
2581            None
2582        };
2583
2584        let restyle_reason = document.restyle_reason();
2585        document.clear_restyle_reasons();
2586        let restyle = if restyle_reason.needs_restyle() {
2587            debug!("Invalidating layout cache due to reflow condition {restyle_reason:?}",);
2588            // Invalidate any existing cached layout values.
2589            self.layout_marker.borrow().set(false);
2590            // Create a new layout caching token.
2591            *self.layout_marker.borrow_mut() = Rc::new(Cell::new(true));
2592
2593            let stylesheets_changed = document.flush_stylesheets_for_reflow();
2594            let pending_restyles = document.drain_pending_restyles();
2595            let dirty_root = document
2596                .take_dirty_root()
2597                .filter(|_| !stylesheets_changed)
2598                .or_else(|| document.GetDocumentElement())
2599                .map(|root| root.upcast::<Node>().to_trusted_node_address());
2600
2601            Some(ReflowRequestRestyle {
2602                reason: restyle_reason,
2603                dirty_root,
2604                stylesheets_changed,
2605                pending_restyles,
2606            })
2607        } else {
2608            None
2609        };
2610
2611        let document_context = self.web_font_context();
2612
2613        // Send new document and relevant styles to layout.
2614        let reflow = ReflowRequest {
2615            document: document.upcast::<Node>().to_trusted_node_address(),
2616            epoch: document.current_rendering_epoch(),
2617            restyle,
2618            viewport_details: self.viewport_details.get(),
2619            origin: self.origin().immutable().clone(),
2620            reflow_goal,
2621            dom_count: document.dom_count(),
2622            animation_timeline_value: document.current_animation_timeline_value(),
2623            animations: document.animations().sets.clone(),
2624            animating_images: document.image_animation_manager().animating_images(),
2625            highlighted_dom_node: document.highlighted_dom_node().map(|node| node.to_opaque()),
2626            document_context,
2627        };
2628
2629        let Some(reflow_result) = self.layout.borrow_mut().reflow(reflow) else {
2630            return ReflowPhasesRun::empty();
2631        };
2632
2633        debug!("script: layout complete");
2634        if let Some(marker) = marker {
2635            self.emit_timeline_marker(marker.end());
2636        }
2637
2638        self.handle_pending_images_post_reflow(
2639            reflow_result.pending_images,
2640            reflow_result.pending_rasterization_images,
2641            reflow_result.pending_svg_elements_for_serialization,
2642        );
2643
2644        if let Some(iframe_sizes) = reflow_result.iframe_sizes {
2645            document
2646                .iframes_mut()
2647                .handle_new_iframe_sizes_after_layout(self, iframe_sizes);
2648        }
2649
2650        document.update_animations_post_reflow();
2651
2652        reflow_result.reflow_phases_run
2653    }

2748    pub(crate) fn allow_layout_if_necessary(&self) {

2750            self.layout_blocker.get(),
2751            LayoutBlocker::FiredLoadEventOrParsingTimerExpired
2752        ) {
2753            return;
2754        }

2756        self.layout_blocker
2757            .set(LayoutBlocker::FiredLoadEventOrParsingTimerExpired);

2760        // happen if parsing is taking a very long time, which means that the
2761        // `ScriptThread` is busy doing the parsing and not doing layouts.
2762        //
2763        // TOOD(mrobinson): It's expected that this is necessary when in the process of
2764        // parsing, as we need to interrupt it to update contents, but why is this

2767        // iframe size updates.
2768        //
2769        // See <https://github.com/servo/servo/issues/14719>
2770        if self.Document().update_the_rendering().needs_frame() {
2771            self.paint_api()
2772                .generate_frame(vec![self.webview_id().into()]);

3728fn should_move_clip_rect(clip_rect: UntypedRect<Au>, new_viewport: UntypedRect<f32>) -> bool {
3729    let clip_rect = UntypedRect::new(
3730        Point2D::new(
3731            clip_rect.origin.x.to_f32_px(),
3732            clip_rect.origin.y.to_f32_px(),
3733        ),
3734        Size2D::new(
3735            clip_rect.size.width.to_f32_px(),
3736            clip_rect.size.height.to_f32_px(),
3737        ),
3738    );
3739
3740    // We only need to move the clip rect if the viewport is getting near the edge of
3741    // our preexisting clip rect. We use half of the size of the viewport as a heuristic
3742    // for "close."
3743    static VIEWPORT_SCROLL_MARGIN_SIZE: f32 = 0.5;
3744    let viewport_scroll_margin = new_viewport.size * VIEWPORT_SCROLL_MARGIN_SIZE;
3745
3746    (clip_rect.origin.x - new_viewport.origin.x).abs() <= viewport_scroll_margin.width ||
3747        (clip_rect.max_x() - new_viewport.max_x()).abs() <= viewport_scroll_margin.width ||
3748        (clip_rect.origin.y - new_viewport.origin.y).abs() <= viewport_scroll_margin.height ||
3749        (clip_rect.max_y() - new_viewport.max_y()).abs() <= viewport_scroll_margin.height
3750}

3764            let this = this.root();

3773            }

3781                // Step 7.6, 7.7

3783                    this.upcast(),

3785                    message_clone.handle(),
3786                    Some(&source_origin.ascii_serialization()),

3788                    ports,
3789                    CanGc::note()

3791            } else {
3792                // Step 4, fire messageerror.
3793                MessageEvent::dispatch_error(
3794                    this.upcast(),
3795                    this.upcast(),
3796                    CanGc::note()
3797                );
3798            }

3801        self.as_global_scope()
3802            .task_manager()
