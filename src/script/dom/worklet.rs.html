5//! An implementation of Houdini worklets.
6//!
7//! The goal of this implementation is to maximize responsiveness of worklets,
8//! and in particular to ensure that the thread performing worklet tasks
9//! is never busy GCing or loading worklet code. We do this by providing a custom
10//! thread pool implementation, which only performs GC or code loading on
11//! a backup thread, not on the primary worklet thread.

77impl Drop for DroppableField {
78    fn drop(&mut self) {
79        let worklet_id = self.worklet_id;
80        if let Some(thread_pool) = self.thread_pool.get_mut() {
81            thread_pool.exit_worklet(worklet_id);
82        }
83    }

131impl WorkletMethods<crate::DomTypeHolder> for Worklet {
132    /// <https://drafts.css-houdini.org/worklets/#dom-worklet-addmodule>
133    fn AddModule(
134        &self,
135        module_url: USVString,
136        options: &WorkletOptions,
137        comp: InRealm,
138        can_gc: CanGc,
139    ) -> Rc<Promise> {
140        // Step 1.
141        let promise = Promise::new_in_current_realm(comp, can_gc);
142
143        // Step 3.
144        let module_url_record = match self.window.Document().base_url().join(&module_url.0) {
145            Ok(url) => url,
146            Err(err) => {
147                // Step 4.
148                debug!("URL {:?} parse error {:?}.", module_url.0, err);
149                promise.reject_error(Error::Syntax(None), can_gc);
150                return promise;
151            },
152        };
153        debug!("Adding Worklet module {}.", module_url_record);
154
155        // Steps 6-12 in parallel.
156        let pending_tasks_struct = PendingTasksStruct::new();
157        let global_scope = self.window.as_global_scope();
158
159        self.droppable_field
160            .thread_pool
161            .get_or_init(|| ScriptThread::worklet_thread_pool(self.global().image_cache()))
162            .fetch_and_invoke_a_worklet_script(
163                self.window.webview_id(),
164                self.window.pipeline_id(),
165                self.droppable_field.worklet_id,
166                self.global_type,
167                self.window.origin().immutable().clone(),
168                global_scope.api_base_url(),
169                module_url_record,
170                global_scope.policy_container(),
171                options.credentials,
172                pending_tasks_struct,
173                &promise,
174                global_scope.inherited_secure_context(),
175            );
176
177        // Step 5.
178        debug!("Returning promise.");
179        promise
180    }

215/// Worklets execute in a dedicated thread pool.
216///
217/// The goal is to ensure that there is a primary worklet thread,
218/// which is able to responsively execute worklet code. In particular,
219/// worklet execution should not be delayed by GC, or by script
220/// loading.
221///
222/// To achieve this, we implement a three-thread pool, with the
223/// threads cycling between three thread roles:
224///
225///  * The primary worklet thread is the one available to execute
226///    worklet code.
227///
228///  * The hot backup thread may peform GC, but otherwise is expected
229///    to take over the primary role.
230///
231///  * The cold backup thread may peform script loading and other
232///    long-running tasks.
233///
234/// In the implementation, we use two kinds of messages:
235///
236///  * Data messages are expected to be processed quickly, and include
237///    the worklet tasks to be performed by the primary thread, as
238///    well as requests to change role or quit execution.
239///
240///  * Control messages are expected to be processed more slowly, and
241///    include script loading.
242///
243/// Data messages are targeted at a role, for example, task execution
244/// is expected to be performed by whichever thread is currently
245/// primary. Control messages are targeted at a thread, for example
246/// adding a module is performed in every thread, even if they change roles
247/// in the middle of module loading.
248///
249/// The thread pool lives in the script thread, and is initialized
250/// when a worklet adds a module. It is dropped when the script thread
251/// is dropped, and asks each of the worklet threads to quit.
252///
253/// Layout can end up blocking on the primary worklet thread
254/// (e.g. when invoking a paint callback), so it is important to avoid
255/// deadlock by making sure the primary worklet thread doesn't end up
256/// blocking waiting on layout. In particular, since the constellation
257/// can block waiting on layout, this means the primary worklet thread
258/// can't block waiting on the constellation. In general, the primary
259/// worklet thread shouldn't perform any blocking operations. If a worklet
260/// thread needs to do anything blocking, it should send a control
261/// message, to make sure that the blocking operation is performed
262/// by a backup thread, not by the primary thread.
263
264#[derive(Clone, JSTraceable)]
265pub(crate) struct WorkletThreadPool {
266    // Channels to send data messages to the three roles.
267    #[no_trace]
268    primary_sender: Sender<WorkletData>,
269    #[no_trace]
270    hot_backup_sender: Sender<WorkletData>,
271    #[no_trace]
272    cold_backup_sender: Sender<WorkletData>,
273    // Channels to send control messages to the three threads.
274    #[no_trace]
275    control_sender_0: Sender<WorkletControl>,
276    #[no_trace]
277    control_sender_1: Sender<WorkletControl>,
278    #[no_trace]
279    control_sender_2: Sender<WorkletControl>,
280}
281
282impl Drop for WorkletThreadPool {
283    fn drop(&mut self) {
284        let _ = self.cold_backup_sender.send(WorkletData::Quit);
285        let _ = self.hot_backup_sender.send(WorkletData::Quit);
286        let _ = self.primary_sender.send(WorkletData::Quit);
287    }
288}

290impl WorkletThreadPool {
291    /// Create a new thread pool and spawn the threads.
292    /// When the thread pool is dropped, the threads will be asked to quit.
293    pub(crate) fn spawn(global_init: WorkletGlobalScopeInit) -> WorkletThreadPool {
294        let primary_role = WorkletThreadRole::new(false, false);
295        let hot_backup_role = WorkletThreadRole::new(true, false);
296        let cold_backup_role = WorkletThreadRole::new(false, true);
297        let primary_sender = primary_role.sender.clone();
298        let hot_backup_sender = hot_backup_role.sender.clone();
299        let cold_backup_sender = cold_backup_role.sender.clone();
300        let init = WorkletThreadInit {
301            primary_sender: primary_sender.clone(),
302            hot_backup_sender: hot_backup_sender.clone(),
303            cold_backup_sender: cold_backup_sender.clone(),
304            global_init,
305        };
306        WorkletThreadPool {
307            primary_sender,
308            hot_backup_sender,
309            cold_backup_sender,
310            control_sender_0: WorkletThread::spawn(primary_role, init.clone(), 0),
311            control_sender_1: WorkletThread::spawn(hot_backup_role, init.clone(), 1),
312            control_sender_2: WorkletThread::spawn(cold_backup_role, init, 2),
313        }
314    }

316    /// Loads a worklet module into every worklet thread.
317    /// If all of the threads load successfully, the promise is resolved.
318    /// If any of the threads fails to load, the promise is rejected.
319    /// <https://drafts.css-houdini.org/worklets/#fetch-and-invoke-a-worklet-script>
320    #[allow(clippy::too_many_arguments)]
321    fn fetch_and_invoke_a_worklet_script(
322        &self,
323        webview_id: WebViewId,
324        pipeline_id: PipelineId,
325        worklet_id: WorkletId,
326        global_type: WorkletGlobalScopeType,
327        origin: ImmutableOrigin,
328        base_url: ServoUrl,
329        script_url: ServoUrl,
330        policy_container: PolicyContainer,
331        credentials: RequestCredentials,
332        pending_tasks_struct: PendingTasksStruct,
333        promise: &Rc<Promise>,
334        inherited_secure_context: Option<bool>,
335    ) {
336        // Send each thread a control message asking it to load the script.
337        for sender in &[
338            &self.control_sender_0,
339            &self.control_sender_1,
340            &self.control_sender_2,
341        ] {
342            let _ = sender.send(WorkletControl::FetchAndInvokeAWorkletScript {
343                webview_id,
344                pipeline_id,
345                worklet_id,
346                global_type,
347                origin: origin.clone(),
348                base_url: base_url.clone(),
349                script_url: script_url.clone(),
350                policy_container: policy_container.clone(),
351                credentials,
352                pending_tasks_struct: pending_tasks_struct.clone(),
353                promise: TrustedPromise::new(promise.clone()),
354                inherited_secure_context,
355            });
356        }
357        self.wake_threads();
358    }

360    pub(crate) fn exit_worklet(&self, worklet_id: WorkletId) {
361        for sender in &[
362            &self.control_sender_0,
363            &self.control_sender_1,
364            &self.control_sender_2,
365        ] {
366            let _ = sender.send(WorkletControl::ExitWorklet(worklet_id));
367        }
368        self.wake_threads();

416/// A role that a worklet thread can be playing.
417///
418/// These roles are used as tokens or capabilities, we track unique
419/// ownership using Rust's types, and use atomic swapping to exchange
420/// them between worklet threads. This ensures that each thread pool has
421/// exactly one primary, one hot backup and one cold backup.
422struct WorkletThreadRole {
423    receiver: Receiver<WorkletData>,
424    sender: Sender<WorkletData>,
425    is_hot_backup: bool,
426    is_cold_backup: bool,
427}
428
429impl WorkletThreadRole {
430    fn new(is_hot_backup: bool, is_cold_backup: bool) -> WorkletThreadRole {
431        let (sender, receiver) = unbounded();
432        WorkletThreadRole {
433            sender,
434            receiver,
435            is_hot_backup,
436            is_cold_backup,
437        }
438    }
439}

496impl WorkletThread {
497    #[allow(unsafe_code)]
498    /// Spawn a new worklet thread, returning the channel to send it control messages.
499    fn spawn(
500        role: WorkletThreadRole,
501        init: WorkletThreadInit,
502        thread_index: u8,
503    ) -> Sender<WorkletControl> {
504        let (control_sender, control_receiver) = unbounded();
505        let _ = thread::Builder::new()
506            .name(format!("Worklet#{thread_index}"))
507            .spawn(move || {
508                // TODO: add a new IN_WORKLET thread state?
509                // TODO: set interrupt handler?
510                // TODO: configure the JS runtime (e.g. discourage GC, encourage agressive JIT)
511                debug!("Initializing worklet thread.");
512                thread_state::initialize(ThreadState::SCRIPT | ThreadState::IN_WORKER);
513                let runtime = Runtime::new(None);
514                let mut cx = unsafe { runtime.cx() };
515                let mut thread = RootedTraceableBox::new(WorkletThread {
516                    role,
517                    control_receiver,
518                    primary_sender: init.primary_sender,
519                    hot_backup_sender: init.hot_backup_sender,
520                    cold_backup_sender: init.cold_backup_sender,
521                    global_init: init.global_init,
522                    global_scopes: FxHashMap::default(),
523                    control_buffer: None,
524                    runtime,
525                    should_gc: false,
526                    gc_threshold: MIN_GC_THRESHOLD,
527                });
528                thread.run(&mut cx);
529            })
530            .expect("Couldn't start worklet thread");
531        control_sender
532    }
533
534    /// The main event loop for a worklet thread
535    fn run(&mut self, cx: &mut js::context::JSContext) {
536        loop {
537            // The handler for data messages
538            let message = self.role.receiver.recv().unwrap();
539            match message {
540                // The whole point of this thread pool is to perform tasks!
541                WorkletData::Task(id, task) => {
542                    self.perform_a_worklet_task(id, task);
543                },
544                // To start swapping roles, get ready to perform an atomic swap,
545                // and block waiting for the other end to finish it.
546                // NOTE: the cold backup can block on the primary or the hot backup;
547                //       the hot backup can block on the primary;
548                //       the primary can block on nothing;
549                //       this total ordering on thread roles is what guarantees deadlock-freedom.
550                WorkletData::StartSwapRoles(sender) => {
551                    let (our_swapper, their_swapper) = swapper();
552                    match sender.send(WorkletData::FinishSwapRoles(their_swapper)) {
553                        Ok(_) => {},
554                        Err(_) => {
555                            // This might happen if the script thread shuts down while
556                            // waiting for the worklet to finish.
557                            return;
558                        },
559                    };
560                    let _ = our_swapper.swap(&mut self.role);
561                },
562                // To finish swapping roles, perform the atomic swap.
563                // The other end should have already started the swap, so this shouldn't block.
564                WorkletData::FinishSwapRoles(swapper) => {
565                    let _ = swapper.swap(&mut self.role);
566                },
567                // Wake up! There may be control messages to process.
568                WorkletData::WakeUp => {},
569                // Quit!
570                WorkletData::Quit => {
571                    return;
572                },
573            }
574            // Only process control messages if we're the cold backup,
575            // otherwise if there are outstanding control messages,
576            // try to become the cold backup.
577            if self.role.is_cold_backup {
578                if let Some(control) = self.control_buffer.take() {
579                    self.process_control(control, cx);
580                }
581                while let Ok(control) = self.control_receiver.try_recv() {
582                    self.process_control(control, cx);
583                }
584                self.gc(cx);
585            } else if self.control_buffer.is_none() {
586                if let Ok(control) = self.control_receiver.try_recv() {
587                    self.control_buffer = Some(control);
588                    let msg = WorkletData::StartSwapRoles(self.role.sender.clone());
589                    let _ = self.cold_backup_sender.send(msg);
590                }
591            }
592            // If we are tight on memory, and we're a backup then perform a gc.
593            // If we are tight on memory, and we're the primary then try to become the hot backup.
594            // Hopefully this happens soon!
595            if self.current_memory_usage() > self.gc_threshold {
596                if self.role.is_hot_backup || self.role.is_cold_backup {
597                    self.should_gc = false;
598                    self.gc(cx);
599                } else if !self.should_gc {
600                    self.should_gc = true;
601                    let msg = WorkletData::StartSwapRoles(self.role.sender.clone());
602                    let _ = self.hot_backup_sender.send(msg);
603                }
604            }
605        }
606    }

608    /// The current memory usage of the thread

614    /// Perform a GC.
615    #[expect(unsafe_code)]
616    fn gc(&mut self, cx: &mut js::context::JSContext) {
617        debug!(
618            "BEGIN GC (usage = {}, threshold = {}).",
619            self.current_memory_usage(),
620            self.gc_threshold
621        );
622        unsafe { JS_GC(cx, GCReason::API) };
623        self.gc_threshold = max(MIN_GC_THRESHOLD, self.current_memory_usage() * 2);
624        debug!(
625            "END GC (usage = {}, threshold = {}).",
626            self.current_memory_usage(),
627            self.gc_threshold
628        );
629    }

642        cx: &mut js::context::JSContext,
643    ) -> DomRoot<WorkletGlobalScope> {
644        match self.global_scopes.entry(worklet_id) {
645            hash_map::Entry::Occupied(entry) => DomRoot::from_ref(entry.get()),
646            hash_map::Entry::Vacant(entry) => {
647                debug!("Creating new worklet global scope.");
648                let executor = WorkletExecutor::new(worklet_id, self.primary_sender.clone());
649                let result = WorkletGlobalScope::new(
650                    global_type,
651                    webview_id,
652                    pipeline_id,
653                    base_url,
654                    inherited_secure_context,
655                    executor,
656                    &self.global_init,
657                    cx,
658                );
659                entry.insert(Dom::from_ref(&*result));
660                result
661            },
662        }
663    }
664
665    /// Fetch and invoke a worklet script.
666    /// <https://html.spec.whatwg.org/multipage/#fetch-a-worklet-script-graph>
667    #[allow(clippy::too_many_arguments)]
668    fn fetch_and_invoke_a_worklet_script(
669        &self,
670        global_scope: &WorkletGlobalScope,
671        pipeline_id: PipelineId,
672        origin: ImmutableOrigin,
673        script_url: ServoUrl,
674        policy_container: PolicyContainer,
675        credentials: RequestCredentials,
676        pending_tasks_struct: PendingTasksStruct,
677        promise: TrustedPromise,
678        cx: &mut js::context::JSContext,
679    ) {
680        debug!("Fetching from {}.", script_url);
681        // Step 1.
682        // TODO: Settings object?
683
684        // Step 2.
685        // TODO: Fetch a module graph, not just a single script.
686        // TODO: Fetch the script asynchronously?
687        // TODO: Caching.
688        let global = global_scope.upcast::<GlobalScope>();
689        let resource_fetcher = self.global_init.resource_threads.sender();
690        let request = RequestBuilder::new(None, script_url, global.get_referrer())
691            .destination(Destination::Script)
692            .mode(RequestMode::CorsMode)
693            .credentials_mode(credentials.convert())
694            .policy_container(policy_container)
695            .origin(origin);
696
697        let script = load_whole_resource(
698            request,
699            &resource_fetcher,
700            global,
701            &WorkletCspProcessor {},
702            CanGc::from_cx(cx),
703        )
704        .ok()
705        .and_then(|(_, bytes, _)| String::from_utf8(bytes).ok());
706
707        // Step 4.
708        // NOTE: the spec parses and executes the script in separate steps,
709        // but our JS API doesn't separate these, so we do the steps out of order.
710        // Also, the spec currently doesn't allow exceptions to be propagated
711        // to the main script thread.
712        // https://github.com/w3c/css-houdini-drafts/issues/407
713        let ok = script.is_some_and(|s| {
714            global_scope
715                .evaluate_js(s.into(), CanGc::from_cx(cx))
716                .is_ok()
717        });
718
719        if !ok {
720            // Step 3.
721            debug!("Failed to load script.");
722            let old_counter = pending_tasks_struct.set_counter_to(-1);
723            if old_counter > 0 {
724                self.run_in_script_thread(promise.reject_task(Error::Abort(None)));
725            }
726        } else {
727            // Step 5.
728            debug!("Finished adding script.");
729            let old_counter = pending_tasks_struct.decrement_counter_by(1);
730            if old_counter == 1 {
731                debug!("Resolving promise.");
732                let msg = MainThreadScriptMsg::WorkletLoaded(pipeline_id);
733                self.global_init
734                    .to_script_thread_sender
735                    .send(msg)
736                    .expect("Worklet thread outlived script thread.");
737                self.run_in_script_thread(promise.resolve_task(()));
738            }
739        }
740    }
741
742    /// Perform a task.
743    fn perform_a_worklet_task(&self, worklet_id: WorkletId, task: WorkletTask) {
744        match self.global_scopes.get(&worklet_id) {
745            Some(global) => global.perform_a_worklet_task(task),
746            None => warn!("No such worklet as {:?}.", worklet_id),
747        }
748    }
749
750    /// Process a control message.
751    fn process_control(&mut self, control: WorkletControl, cx: &mut js::context::JSContext) {
752        match control {
753            WorkletControl::ExitWorklet(worklet_id) => {
754                self.global_scopes.remove(&worklet_id);
755            },
756            WorkletControl::FetchAndInvokeAWorkletScript {
757                webview_id,
758                pipeline_id,
759                worklet_id,
760                global_type,
761                origin,
762                base_url,
763                script_url,
764                policy_container,
765                credentials,
766                pending_tasks_struct,
767                promise,
768                inherited_secure_context,
769            } => {
770                let global = self.get_worklet_global_scope(
771                    webview_id,
772                    pipeline_id,
773                    worklet_id,
774                    inherited_secure_context,
775                    global_type,
776                    base_url,
777                    cx,
778                );
779                self.fetch_and_invoke_a_worklet_script(
780                    &global,
781                    pipeline_id,
782                    origin,
783                    script_url,
784                    policy_container,
785                    credentials,
786                    pending_tasks_struct,
787                    promise,
788                    cx,
789                )
790            },
791        }
792    }
793
794    /// Run a task in the main script thread.
795    fn run_in_script_thread<T>(&self, task: T)
796    where
797        T: TaskBox + 'static,
798    {
799        // NOTE: It's unclear which task source should be used here:
800        // https://drafts.css-houdini.org/worklets/#dom-worklet-addmodule
801        let msg = CommonScriptMsg::Task(
802            ScriptThreadEventCategory::WorkletEvent,
803            Box::new(task),
804            None,
805            TaskSourceName::DOMManipulation,
806        );
807        let msg = MainThreadScriptMsg::Common(msg);
808        self.global_init
809            .to_script_thread_sender
810            .send(msg)
811            .expect("Worklet thread outlived script thread.");
812    }
813}

815/// An executor of worklet tasks
816#[derive(Clone, JSTraceable, MallocSizeOf)]
817pub(crate) struct WorkletExecutor {
818    worklet_id: WorkletId,
819    #[no_trace]
820    primary_sender: Sender<WorkletData>,
821}
822
823impl WorkletExecutor {
824    fn new(worklet_id: WorkletId, primary_sender: Sender<WorkletData>) -> WorkletExecutor {
825        WorkletExecutor {
826            worklet_id,
827            primary_sender,
828        }
829    }

831    /// Schedule a worklet task to be peformed by the worklet thread pool.
832    pub(crate) fn schedule_a_worklet_task(&self, task: WorkletTask) {
833        let _ = self
834            .primary_sender
835            .send(WorkletData::Task(self.worklet_id, task));
836    }
