5use dom_struct::dom_struct;
6use js::rust::HandleObject;
7use xml5ever::serialize::{SerializeOpts, TraversalScope, serialize};

9use crate::dom::bindings::codegen::Bindings::XMLSerializerBinding::XMLSerializerMethods;
10use crate::dom::bindings::error::{Error, Fallible};
11use crate::dom::bindings::reflector::{Reflector, reflect_dom_object_with_proto};
12use crate::dom::bindings::root::{Dom, DomRoot};
13use crate::dom::bindings::str::DOMString;
14use crate::dom::node::Node;
15use crate::dom::servoparser::html::HtmlSerialize;
16use crate::dom::window::Window;
17use crate::script_runtime::CanGc;
18
19#[dom_struct]
20pub(crate) struct XMLSerializer {
21    reflector_: Reflector,
22    window: Dom<Window>,
23}
24
25impl XMLSerializer {
26    fn new_inherited(window: &Window) -> XMLSerializer {
27        XMLSerializer {
28            reflector_: Reflector::new(),
29            window: Dom::from_ref(window),
30        }
31    }
32
33    pub(crate) fn new(
34        window: &Window,
35        proto: Option<HandleObject>,
36        can_gc: CanGc,
37    ) -> DomRoot<XMLSerializer> {
38        reflect_dom_object_with_proto(
39            Box::new(XMLSerializer::new_inherited(window)),
40            window,
41            proto,
42            can_gc,
43        )
44    }
45}
46
47impl XMLSerializerMethods<crate::DomTypeHolder> for XMLSerializer {
48    /// <https://w3c.github.io/DOM-Parsing/#dom-xmlserializer>
49    fn Constructor(
50        window: &Window,
51        proto: Option<HandleObject>,
52        can_gc: CanGc,
53    ) -> Fallible<DomRoot<XMLSerializer>> {
54        Ok(XMLSerializer::new(window, proto, can_gc))
55    }
56
57    /// <https://w3c.github.io/DOM-Parsing/#the-xmlserializer-interface>
58    fn SerializeToString(&self, root: &Node) -> Fallible<DOMString> {
59        let mut writer = vec![];
60        match serialize(
61            &mut writer,
62            &HtmlSerialize::new(root),
63            SerializeOpts {
64                traversal_scope: TraversalScope::IncludeNode,
65            },
66        ) {
67            Ok(_) => Ok(DOMString::from(String::from_utf8(writer).unwrap())),
68            Err(_) => Err(Error::Type(String::from(
69                "root must be a Node or an Attr object",
70            ))),
71        }
72    }
73}
