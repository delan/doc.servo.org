38// https://www.w3.org/TR/IndexedDB-2/#convert-key-to-value
39#[expect(unsafe_code)]
40pub fn key_type_to_jsval(
41    cx: SafeJSContext,
42    key: &IndexedDBKeyType,
43    mut result: MutableHandleValue,
44    can_gc: CanGc,
45) {
46    match key {
47        IndexedDBKeyType::Number(n) => result.set(DoubleValue(*n)),
48        IndexedDBKeyType::String(s) => s.safe_to_jsval(cx, result, can_gc),
49        IndexedDBKeyType::Binary(b) => b.safe_to_jsval(cx, result, can_gc),
50        IndexedDBKeyType::Date(d) => {
51            let time = js::jsapi::ClippedTime { t: *d };
52            let date = unsafe { js::jsapi::NewDateObject(*cx, time) };
53            date.safe_to_jsval(cx, result, can_gc);
54        },
55        IndexedDBKeyType::Array(a) => {
56            rooted_vec!(let mut values);
57            for key in a {
58                rooted!(in(*cx) let mut value: JSVal);
59                key_type_to_jsval(cx, key, value.handle_mut(), can_gc);
60                values.push(Heap::boxed(value.get()));
61            }
62            values.safe_to_jsval(cx, result, can_gc);
63        },
64    }
65}

67/// <https://www.w3.org/TR/IndexedDB-2/#valid-key-path>
68pub(crate) fn is_valid_key_path(key_path: &StrOrStringSequence) -> Result<bool, Error> {
69    // <https://tc39.es/ecma262/#prod-IdentifierName>
70    #[expect(unsafe_code)]
71    let is_identifier_name = |name: &str| -> Result<bool, Error> {
72        let cx = GlobalScope::get_cx();
73        rooted!(in(*cx) let mut value = UndefinedValue());
74        name.safe_to_jsval(cx, value.handle_mut(), CanGc::note());
75        rooted!(in(*cx) let string = value.to_string());
76
77        unsafe {
78            let mut is_identifier = false;
79            if !JS_IsIdentifier(*cx, string.handle(), &mut is_identifier) {
80                return Err(Error::JSFailed);
81            }
82            Ok(is_identifier)
83        }
84    };
85
86    // A valid key path is one of:
87    let is_valid = |path: &DOMString| -> Result<bool, Error> {
88        // An empty string.
89        let is_empty_string = path.is_empty();
90
91        // An identifier, which is a string matching the IdentifierName production from the
92        // ECMAScript Language Specification [ECMA-262].
93        let is_identifier = is_identifier_name(&path.str())?;
94
95        // A string consisting of two or more identifiers separated by periods (U+002E FULL STOP).
96        let is_identifier_list = path
97            .str()
98            .split('.')
99            .map(is_identifier_name)
100            .try_collect::<bool, Vec<bool>, Error>()?
101            .iter()
102            .all(|&value| value);
103
104        Ok(is_empty_string || is_identifier || is_identifier_list)
105    };
106
107    match key_path {
108        StrOrStringSequence::StringSequence(paths) => {
109            // A non-empty list containing only strings conforming to the above requirements.
110            if paths.is_empty() {
111                Ok(false)
112            } else {
113                Ok(paths
114                    .iter()
115                    .map(is_valid)
116                    .try_collect::<bool, Vec<bool>, Error>()?
117                    .iter()
118                    .all(|&value| value))
119            }
120        },
121        StrOrStringSequence::String(path) => is_valid(path),
122    }
123}

139// https://www.w3.org/TR/IndexedDB-2/#convert-value-to-key
140#[expect(unsafe_code)]
141pub fn convert_value_to_key(
142    cx: SafeJSContext,
143    input: HandleValue,
144    seen: Option<Vec<HandleValue>>,
145) -> Result<ConversionResult, Error> {
146    // Step 1: If seen was not given, then let seen be a new empty set.
147    let mut seen = seen.unwrap_or_default();
148
149    // Step 2: If seen contains input, then return invalid.
150    // FIXME:(arihant2math) implement this
151    // Check if we have seen this key
152    // Does not currently work with HandleValue,
153    // as it does not implement PartialEq
154
155    // Step 3
156    // FIXME:(arihant2math) Accept array as well
157    if input.is_number() {
158        if input.to_number().is_nan() {
159            return Ok(ConversionResult::Invalid);
160        }
161        return Ok(ConversionResult::Valid(IndexedDBKeyType::Number(
162            input.to_number(),
163        )));
164    }
165
166    if input.is_string() {
167        let string_ptr = std::ptr::NonNull::new(input.to_string()).unwrap();
168        let key = unsafe { jsstr_to_string(*cx, string_ptr) };
169        return Ok(ConversionResult::Valid(IndexedDBKeyType::String(key)));
170    }
171
172    if input.is_object() {
173        rooted!(in(*cx) let object = input.to_object());
174        unsafe {
175            let mut built_in_class = ESClass::Other;
176
177            if !GetBuiltinClass(*cx, object.handle().into(), &mut built_in_class) {
178                return Err(Error::JSFailed);
179            }
180
181            if let ESClass::Date = built_in_class {
182                let mut f = f64::NAN;
183                if !js::jsapi::DateGetMsecSinceEpoch(*cx, object.handle().into(), &mut f) {
184                    return Err(Error::JSFailed);
185                }
186                if f.is_nan() {
187                    return Err(Error::Data(None));
188                }
189                return Ok(ConversionResult::Valid(IndexedDBKeyType::Date(f)));
190            }
191
192            if IsArrayBufferObject(*object) || JS_IsArrayBufferViewObject(*object) {
193                // FIXME:(arihant2math) implement it the correct way (is this correct?)
194                let key = structuredclone::write(cx, input, None)?;
195                return Ok(ConversionResult::Valid(IndexedDBKeyType::Binary(
196                    key.serialized.clone(),
197                )));
198            }
199
200            if let ESClass::Array = built_in_class {
201                let mut len = 0;
202                if !GetArrayLength(*cx, object.handle().into_handle(), &mut len) {
203                    return Err(Error::JSFailed);
204                }
205                seen.push(input);
206                let mut values = vec![];
207                for i in 0..len {
208                    rooted!(in(*cx) let mut id: PropertyKey);
209                    if !JS_IndexToId(*cx, i, js::jsapi::MutableHandleId::from(id.handle_mut())) {
210                        return Err(Error::JSFailed);
211                    }
212                    let mut has_own = false;
213                    if !JS_HasOwnPropertyById(
214                        *cx,
215                        object.handle().into_handle(),
216                        id.handle().into_handle(),
217                        &mut has_own,
218                    ) {
219                        return Err(Error::JSFailed);
220                    }
221                    if !has_own {
222                        return Ok(ConversionResult::Invalid);
223                    }
224                    rooted!(in(*cx) let mut item = UndefinedValue());
225                    if !js::jsapi::JS_GetPropertyById(
226                        *cx,
227                        object.handle().into_handle(),
228                        id.handle().into_handle(),
229                        item.handle_mut().into_handle_mut(),
230                    ) {
231                        return Err(Error::JSFailed);
232                    }
233                    let key = match convert_value_to_key(cx, item.handle(), Some(seen.clone()))? {
234                        ConversionResult::Valid(key) => key,
235                        ConversionResult::Invalid => return Ok(ConversionResult::Invalid),
236                    };
237                    values.push(key);
238                }
239                return Ok(ConversionResult::Valid(IndexedDBKeyType::Array(values)));
240            }
241        }
242    }
243
244    Ok(ConversionResult::Invalid)
245}

247/// <https://www.w3.org/TR/IndexedDB-2/#convert-a-value-to-a-key-range>
248#[expect(unsafe_code)]
249pub fn convert_value_to_key_range(
250    cx: SafeJSContext,
251    input: HandleValue,
252    null_disallowed: Option<bool>,
253) -> Result<IndexedDBKeyRange, Error> {
254    // Step 1. If value is a key range, return value.
255    if input.is_object() {
256        rooted!(in(*cx) let object = input.to_object());
257        unsafe {
258            if let Ok(obj) = root_from_object::<IDBKeyRange>(object.get(), *cx) {
259                let obj = obj.inner().clone();
260                return Ok(obj);
261            }
262        }
263    }
264
265    // Step 2. If value is undefined or is null, then throw a "DataError" DOMException if null
266    // disallowed flag is set, or return an unbounded key range otherwise.
267    if input.get().is_undefined() || input.get().is_null() {
268        if null_disallowed.is_some_and(|flag| flag) {
269            return Err(Error::Data(None));
270        } else {
271            return Ok(IndexedDBKeyRange {
272                lower: None,
273                upper: None,
274                lower_open: Default::default(),
275                upper_open: Default::default(),
276            });
277        }
278    }
279
280    // Step 3. Let key be the result of running the steps to convert a value to a key with value.
281    // Rethrow any exceptions.
282    let key = convert_value_to_key(cx, input, None)?;
283
284    // Step 4. If key is invalid, throw a "DataError" DOMException.
285    let key = key.into_result()?;
286
287    // Step 5. Return a key range containing only key.
288    Ok(IndexedDBKeyRange::only(key))
289}

304/// The result of steps in
305/// <https://www.w3.org/TR/IndexedDB-2/#evaluate-a-key-path-on-a-value>
306pub(crate) enum EvaluationResult {
307    Success,
308    Failure,
309}

311/// <https://www.w3.org/TR/IndexedDB-2/#evaluate-a-key-path-on-a-value>
312#[expect(unsafe_code)]
313pub(crate) fn evaluate_key_path_on_value(
314    cx: SafeJSContext,
315    value: HandleValue,
316    key_path: &KeyPath,
317    mut return_val: MutableHandleValue,
318) -> Result<EvaluationResult, Error> {
319    match key_path {
320        // Step 1. If keyPath is a list of strings, then:
321        KeyPath::StringSequence(key_path) => {
322            // Step 1.1. Let result be a new Array object created as if by the expression [].
323            rooted!(in(*cx) let mut result = unsafe { JS_NewObject(*cx, ptr::null()) });
324
325            // Step 1.2. Let i be 0.
326            // Step 1.3. For each item in keyPath:
327            for (i, item) in key_path.iter().enumerate() {
328                // Step 1.3.1. Let key be the result of recursively running the steps to evaluate a key
329                // path on a value using item as keyPath and value as value.
330                // Step 1.3.2. Assert: key is not an abrupt completion.
331                // Step 1.3.3. If key is failure, abort the overall algorithm and return failure.
332                rooted!(in(*cx) let mut key = UndefinedValue());
333                if let EvaluationResult::Failure = evaluate_key_path_on_value(
334                    cx,
335                    value,
336                    &KeyPath::String(item.clone()),
337                    key.handle_mut(),
338                )? {
339                    return Ok(EvaluationResult::Failure);
340                };
341
342                // Step 1.3.4. Let p be ! ToString(i).
343                // Step 1.3.5. Let status be CreateDataProperty(result, p, key).
344                // Step 1.3.6. Assert: status is true.
345                set_dictionary_property(cx, result.handle(), &i.to_string(), key.handle())
346                    .map_err(|_| Error::JSFailed)?;
347
348                // Step 1.3.7. Increase i by 1.
349                // Done by for loop with enumerate()
350            }
351
352            // Step 1.4. Return result.
353            result.safe_to_jsval(cx, return_val, CanGc::note());
354        },
355        KeyPath::String(key_path) => {
356            // Step 2. If keyPath is the empty string, return value and skip the remaining steps.
357            if key_path.is_empty() {
358                return_val.set(*value);
359                return Ok(EvaluationResult::Success);
360            }
361
362            // NOTE: Use current_value, instead of value described in spec, in the following steps.
363            rooted!(in(*cx) let mut current_value = *value);
364
365            // Step 3. Let identifiers be the result of strictly splitting keyPath on U+002E
366            // FULL STOP characters (.).
367            // Step 4. For each identifier of identifiers, jump to the appropriate step below:
368            for identifier in key_path.str().split('.') {
369                // If Type(value) is String, and identifier is "length"
370                if identifier == "length" && current_value.is_string() {
371                    // Let value be a Number equal to the number of elements in value.
372                    rooted!(in(*cx) let string_value = current_value.to_string());
373                    unsafe {
374                        let string_length = JS_GetStringLength(*string_value) as u64;
375                        string_length.safe_to_jsval(cx, current_value.handle_mut(), CanGc::note());
376                    }
377                    continue;
378                }
379
380                // If value is an Array and identifier is "length"
381                if identifier == "length" {
382                    unsafe {
383                        let mut is_array = false;
384                        if !IsArrayObject(*cx, current_value.handle(), &mut is_array) {
385                            return Err(Error::JSFailed);
386                        }
387                        if is_array {
388                            // Let value be ! ToLength(! Get(value, "length")).
389                            rooted!(in(*cx) let object = current_value.to_object());
390                            get_property_jsval(
391                                cx,
392                                object.handle(),
393                                "length",
394                                current_value.handle_mut(),
395                            )?;
396
397                            continue;
398                        }
399                    }
400                }
401
402                // If value is a Blob and identifier is "size"
403                if identifier == "size" {
404                    if let Ok(blob) = root_from_handlevalue::<Blob>(current_value.handle(), cx) {
405                        // Let value be a Number equal to value’s size.
406                        blob.Size()
407                            .safe_to_jsval(cx, current_value.handle_mut(), CanGc::note());
408
409                        continue;
410                    }
411                }
412
413                // If value is a Blob and identifier is "type"
414                if identifier == "type" {
415                    if let Ok(blob) = root_from_handlevalue::<Blob>(current_value.handle(), cx) {
416                        // Let value be a String equal to value’s type.
417                        blob.Type()
418                            .safe_to_jsval(cx, current_value.handle_mut(), CanGc::note());
419
420                        continue;
421                    }
422                }
423
424                // If value is a File and identifier is "name"
425                if identifier == "name" {
426                    if let Ok(file) = root_from_handlevalue::<File>(current_value.handle(), cx) {
427                        // Let value be a String equal to value’s name.
428                        file.name()
429                            .safe_to_jsval(cx, current_value.handle_mut(), CanGc::note());
430
431                        continue;
432                    }
433                }
434
435                // If value is a File and identifier is "lastModified"
436                if identifier == "lastModified" {
437                    if let Ok(file) = root_from_handlevalue::<File>(current_value.handle(), cx) {
438                        // Let value be a Number equal to value’s lastModified.
439                        file.LastModified().safe_to_jsval(
440                            cx,
441                            current_value.handle_mut(),
442                            CanGc::note(),
443                        );
444
445                        continue;
446                    }
447                }
448
449                // If value is a File and identifier is "lastModifiedDate"
450                if identifier == "lastModifiedDate" {
451                    if let Ok(file) = root_from_handlevalue::<File>(current_value.handle(), cx) {
452                        // Let value be a new Date object with [[DateValue]] internal slot equal to value’s lastModified.
453                        let time = ClippedTime {
454                            t: file.LastModified() as f64,
455                        };
456                        unsafe {
457                            NewDateObject(*cx, time).safe_to_jsval(
458                                cx,
459                                current_value.handle_mut(),
460                                CanGc::note(),
461                            );
462                        }
463
464                        continue;
465                    }
466                }
467
468                // Otherwise
469                unsafe {
470                    // If Type(value) is not Object, return failure.
471                    if !current_value.is_object() {
472                        return Ok(EvaluationResult::Failure);
473                    }
474
475                    rooted!(in(*cx) let object = current_value.to_object());
476                    let identifier_name =
477                        CString::new(identifier).expect("Failed to convert str to CString");
478
479                    // Let hop be ! HasOwnProperty(value, identifier).
480                    let mut hop = false;
481                    if !JS_HasOwnProperty(*cx, object.handle(), identifier_name.as_ptr(), &mut hop)
482                    {
483                        return Err(Error::JSFailed);
484                    }
485
486                    // If hop is false, return failure.
487                    if !hop {
488                        return Ok(EvaluationResult::Failure);
489                    }
490
491                    // Let value be ! Get(value, identifier).
492                    if !JS_GetProperty(
493                        *cx,
494                        object.handle(),
495                        identifier_name.as_ptr(),
496                        current_value.handle_mut(),
497                    ) {
498                        return Err(Error::JSFailed);
499                    }
500
501                    // If value is undefined, return failure.
502                    if current_value.get().is_undefined() {
503                        return Ok(EvaluationResult::Failure);
504                    }
505                }
506            }
507
508            // Step 5. Assert: value is not an abrupt completion.
509            // Done within Step 4.
510
511            // Step 6. Return value.
512            return_val.set(*current_value);
513        },
514    }
515    Ok(EvaluationResult::Success)
516}

526/// <https://www.w3.org/TR/IndexedDB-2/#extract-a-key-from-a-value-using-a-key-path>
527pub(crate) fn extract_key(
528    cx: SafeJSContext,
529    value: HandleValue,
530    key_path: &KeyPath,
531    multi_entry: Option<bool>,
532) -> Result<ExtractionResult, Error> {
533    // Step 1. Let r be the result of running the steps to evaluate a key path on a value with
534    // value and keyPath. Rethrow any exceptions.
535    // Step 2. If r is failure, return failure.
536    rooted!(in(*cx) let mut r = UndefinedValue());
537    if let EvaluationResult::Failure =
538        evaluate_key_path_on_value(cx, value, key_path, r.handle_mut())?
539    {
540        return Ok(ExtractionResult::Failure);
541    }
542
543    // Step 3. Let key be the result of running the steps to convert a value to a key with r if the
544    // multiEntry flag is unset, and the result of running the steps to convert a value to a
545    // multiEntry key with r otherwise. Rethrow any exceptions.
546    let key = match multi_entry {
547        Some(true) => {
548            // TODO: implement convert_value_to_multientry_key
549            unimplemented!("multiEntry keys are not yet supported");
550        },
551        _ => match convert_value_to_key(cx, r.handle(), None)? {
552            ConversionResult::Valid(key) => key,
553            // Step 4. If key is invalid, return invalid.
554            ConversionResult::Invalid => return Ok(ExtractionResult::Invalid),
555        },
556    };
557
558    // Step 5. Return key.
559    Ok(ExtractionResult::Key(key))
