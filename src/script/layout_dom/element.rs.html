147pub enum DOMDescendantIterator<E>
148where
149    E: TElement,
150{
151    /// Iterating over the children of a node, including children of a potential
152    /// [ShadowRoot](crate::dom::shadow_root::ShadowRoot)
153    Children(DomChildren<E::ConcreteNode>),
154    /// Iterating over the content's of a [`<slot>`](HTMLSlotElement) element.
155    Slottables { slot: E, index: usize },
156}
157
158impl<E> Iterator for DOMDescendantIterator<E>
159where
160    E: TElement,
161{
162    type Item = E::ConcreteNode;
163
164    fn next(&mut self) -> Option<Self::Item> {
165        match self {
166            Self::Children(children) => children.next(),
167            Self::Slottables { slot, index } => {
168                let slottables = slot.slotted_nodes();
169                let slot = slottables.get(*index)?;
170                *index += 1;
171                Some(*slot)
172            },
173        }
174    }
175}

475    fn match_element_lang(
476        &self,
477        override_lang: Option<Option<SelectorAttrValue>>,
478        value: &Lang,
479    ) -> bool {
480        // Servo supports :lang() from CSS Selectors 4, which can take a comma-
481        // separated list of language tags in the pseudo-class, and which
482        // performs RFC 4647 extended filtering matching on them.
483        //
484        // FIXME(heycam): This is wrong, since extended_filtering accepts
485        // a string containing commas (separating each language tag in
486        // a list) but the pseudo-class instead should be parsing and
487        // storing separate <ident> or <string>s for each language tag.
488        //
489        // FIXME(heycam): Look at `element`'s document's Content-Language
490        // HTTP header for language tags to match `value` against.  To
491        // do this, we should make `get_lang_for_layout` return an Option,
492        // so we can decide when to fall back to the Content-Language check.
493        let element_lang = match override_lang {
494            Some(Some(lang)) => lang,
495            Some(None) => AtomString::default(),
496            None => AtomString::from(&*self.element.get_lang_for_layout()),
497        };
498        extended_filtering(&element_lang, value)
499    }

581    fn slotted_nodes(&self) -> &[Self::ConcreteNode] {
582        let Some(slot_element) = self.element.unsafe_get().downcast::<HTMLSlotElement>() else {
583            return &[];
584        };
585        let assigned_nodes = slot_element.assigned_nodes();
586
587        // SAFETY:
588        // Self::ConcreteNode (aka ServoLayoutNode) and Slottable are guaranteed to have the same
589        // layout and alignment as ptr::NonNull<T>. Lifetimes are not an issue because the
590        // slottables are being kept alive by the slot element.
591        unsafe {
592            slice::from_raw_parts(
593                assigned_nodes.as_ptr() as *const Self::ConcreteNode,
594                assigned_nodes.len(),
595            )
596        }
597    }

599    fn compute_layout_damage(old: &ComputedValues, new: &ComputedValues) -> RestyleDamage {
600        let box_tree_needs_rebuild = || {
601            let old_box = old.get_box();
602            let new_box = new.get_box();
603
604            if old_box.display != new_box.display ||
605                old_box.float != new_box.float ||
606                old_box.position != new_box.position
607            {
608                return true;
609            }
610
611            if new_box.position.is_absolutely_positioned() &&
612                old_box.original_display != new_box.original_display
613            {
614                // The original display only affects the static position, which is only used
615                // when both insets in some axis are auto.
616                // <https://drafts.csswg.org/css-position/#resolving-insets>
617                let position = new.get_position();
618                if (position.top.is_auto() && position.bottom.is_auto()) ||
619                    (position.left.is_auto() && position.right.is_auto())
620                {
621                    return true;
622                }
623            }
624
625            if old.get_font() != new.get_font() {
626                return true;
627            }
628
629            if old.get_position().order != new.get_position().order {
630                return true;
631            }
632
633            // Only consider changes to the `quotes` attribute if they actually apply to this
634            // style (if it is a pseudo-element that supports it).
635            if matches!(
636                new.pseudo(),
637                Some(PseudoElement::Before | PseudoElement::After | PseudoElement::Marker),
638            ) && old.get_list().quotes != new.get_list().quotes
639            {
640                return true;
641            }
642
643            // NOTE: This should be kept in sync with the checks in `impl
644            // StyleExt::establishes_block_formatting_context` for `ComputedValues` in
645            // `components/layout/style_ext.rs`.
646            if new_box.display.outside() == DisplayOutside::Block &&
647                new_box.display.inside() == DisplayInside::Flow
648            {
649                let alignment_establishes_new_block_formatting_context =
650                    |style: &ComputedValues| {
651                        style.get_position().align_content.primary() != AlignFlags::NORMAL
652                    };
653
654                let old_column = old.get_column();
655                let new_column = new.get_column();
656                if old_box.overflow_x.is_scrollable() != new_box.overflow_x.is_scrollable() ||
657                    old_column.is_multicol() != new_column.is_multicol() ||
658                    old_column.column_span != new_column.column_span ||
659                    alignment_establishes_new_block_formatting_context(old) !=
660                        alignment_establishes_new_block_formatting_context(new)
661                {
662                    return true;
663                }
664            }
665
666            if old_box.display.is_list_item() {
667                let old_list = old.get_list();
668                let new_list = new.get_list();
669                if old_list.list_style_position != new_list.list_style_position ||
670                    old_list.list_style_image != new_list.list_style_image ||
671                    (new_list.list_style_image == Image::None &&
672                        old_list.list_style_type != new_list.list_style_type)
673                {
674                    return true;
675                }
676            }
677
678            if new.is_pseudo_style() && old.get_counters().content != new.get_counters().content {
679                return true;
680            }
681
682            false
683        };
684
685        let text_shaping_needs_recollect = || {
686            if old.clone_direction() != new.clone_direction() ||
687                old.clone_unicode_bidi() != new.clone_unicode_bidi()
688            {
689                return true;
690            }
691
692            let old_text = old.get_inherited_text().clone();
693            let new_text = new.get_inherited_text().clone();
694            if old_text.white_space_collapse != new_text.white_space_collapse ||
695                old_text.text_transform != new_text.text_transform ||
696                old_text.word_break != new_text.word_break ||
697                old_text.overflow_wrap != new_text.overflow_wrap ||
698                old_text.letter_spacing != new_text.letter_spacing ||
699                old_text.word_spacing != new_text.word_spacing ||
700                old_text.text_rendering != new_text.text_rendering
701            {
702                return true;
703            }
704
705            false
706        };
707
708        if box_tree_needs_rebuild() {
709            RestyleDamage::from_bits_retain(LayoutDamage::REBUILD_BOX.bits())
710        } else if text_shaping_needs_recollect() {
711            RestyleDamage::from_bits_retain(LayoutDamage::RECOLLECT_BOX_TREE_CHILDREN.bits())
712        } else {
713            // This element needs to be laid out again, but does not have any damage to
714            // its box. In the future, we will distinguish between types of damage to the
715            // fragment as well.
716            RestyleDamage::RELAYOUT
717        }
718    }
719}
