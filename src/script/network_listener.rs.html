25pub(crate) trait ResourceTimingListener {
26    fn resource_timing_information(&self) -> (InitiatorType, ServoUrl);
27    fn resource_timing_global(&self) -> DomRoot<GlobalScope>;
28}

30pub(crate) fn submit_timing<T: ResourceTimingListener>(
31    listener: &T,
32    resource_timing: &ResourceFetchTiming,
33    can_gc: CanGc,
34) {
35    // Resource timings should only be submitted for the initial preload request,
36    // not for the request that consumes the preload: https://github.com/whatwg/html/issues/12047
37    if resource_timing.preloaded {
38        return;
39    }
40    // TODO timing check https://w3c.github.io/resource-timing/#dfn-timing-allow-check
41    //
42    // TODO Resources for which the fetch was initiated, but was later aborted
43    // (e.g. due to a network error) MAY be included as PerformanceResourceTiming
44    // objects in the Performance Timeline and MUST contain initialized attribute
45    // values for processed substeps of the processing model.
46    if resource_timing.timing_type != ResourceTimingType::Resource {
47        warn!(
48            "Submitting non-resource ({:?}) timing as resource",
49            resource_timing.timing_type
50        );
51        return;
52    }
53
54    let (initiator_type, url) = listener.resource_timing_information();
55    if initiator_type == InitiatorType::Other {
56        warn!("Ignoring InitiatorType::Other resource {:?}", url);
57        return;
58    }
59
60    submit_timing_data(
61        &listener.resource_timing_global(),
62        url,
63        initiator_type,
64        resource_timing,
65        can_gc,
66    );
67}
68
69pub(crate) fn submit_timing_data(
70    global: &GlobalScope,
71    url: ServoUrl,
72    initiator_type: InitiatorType,
73    resource_timing: &ResourceFetchTiming,
74    can_gc: CanGc,
75) {
76    let performance_entry =
77        PerformanceResourceTiming::new(global, url, initiator_type, None, resource_timing, can_gc);
78    global
79        .performance()
80        .queue_entry(performance_entry.upcast::<PerformanceEntry>());
81}

83pub(crate) trait FetchResponseListener: Send + 'static {
84    /// A gating mechanism that runs before invoking the listener methods on the target
85    /// thread. If the `should_invoke` method returns false, the listener does not receive
86    /// the notification.
87    fn should_invoke(&self) -> bool {
88        true
89    }
90
91    fn process_request_body(&mut self, request_id: RequestId);
92    fn process_request_eof(&mut self, request_id: RequestId);
93    fn process_response(
94        &mut self,
95        request_id: RequestId,
96        metadata: Result<FetchMetadata, NetworkError>,
97    );
98    fn process_response_chunk(&mut self, request_id: RequestId, chunk: Vec<u8>);
99    fn process_response_eof(
100        self,
101        request_id: RequestId,
102        response: Result<ResourceFetchTiming, NetworkError>,
103    );
104    fn process_csp_violations(&mut self, request_id: RequestId, violations: Vec<Violation>);
105}

107/// An off-thread sink for async network event tasks. All such events are forwarded to
108/// a target thread, where they are invoked on the provided context object.
109pub(crate) struct NetworkListener<Listener: FetchResponseListener> {
110    pub(crate) context: Arc<Mutex<Option<Listener>>>,
111    pub(crate) task_source: SendableTaskSource,
112}
113
114impl<Listener: FetchResponseListener> NetworkListener<Listener> {
115    pub(crate) fn new(context: Listener, task_source: SendableTaskSource) -> Self {
116        Self {
117            context: Arc::new(Mutex::new(Some(context))),
118            task_source,
119        }
120    }
121
122    pub(crate) fn notify(&mut self, message: FetchResponseMsg) {
123        let context = self.context.clone();
124        self.task_source
125            .queue(task!(network_listener_response: move || {
126                let mut context = context.lock().unwrap();
127                let Some(fetch_listener) = &mut *context else {
128                    return;
129                };
130
131                if !fetch_listener.should_invoke() {
132                    return;
133                }
134
135                match message {
136                    FetchResponseMsg::ProcessRequestBody(request_id) => {
137                        fetch_listener.process_request_body(request_id)
138                    },
139                    FetchResponseMsg::ProcessRequestEOF(request_id) => {
140                        fetch_listener.process_request_eof(request_id)
141                    },
142                    FetchResponseMsg::ProcessResponse(request_id, meta) => {
143                        fetch_listener.process_response(request_id, meta)
144                    },
145                    FetchResponseMsg::ProcessResponseChunk(request_id, data) => {
146                        fetch_listener.process_response_chunk(request_id, data.0)
147                    },
148                    FetchResponseMsg::ProcessResponseEOF(request_id, resource_timing_result) => {
149                        if let Some(fetch_listener) = context.take() {
150                            fetch_listener.process_response_eof(request_id, resource_timing_result);
151                        };
152                    },
153                    FetchResponseMsg::ProcessCspViolations(request_id, violations) => {
154                        fetch_listener.process_csp_violations(request_id, violations)
155                    },
156                }
157            }));
158    }
159
160    pub(crate) fn into_callback(mut self) -> BoxedFetchCallback {
161        Box::new(move |response_msg| self.notify(response_msg))
162    }
