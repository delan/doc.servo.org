145/// Identity for a module which will be
146/// used to retrieve the module when we'd
147/// like to get it from module map.
148///
149/// For example, we will save module parents with
150/// module identity so that we can get module tree
151/// from a descendant no matter the parent is an
152/// inline script or a external script
153#[derive(Clone, Debug, Eq, Hash, JSTraceable, PartialEq, MallocSizeOf)]
154pub(crate) enum ModuleIdentity {
155    ScriptId(ScriptId),
156    ModuleUrl(#[no_trace] ServoUrl),
157}

159impl ModuleIdentity {
160    pub(crate) fn get_module_tree(&self, global: &GlobalScope) -> Rc<ModuleTree> {
161        match self {
162            ModuleIdentity::ModuleUrl(url) => {
163                let module_map = global.get_module_map().borrow();
164                module_map.get(&url.clone()).unwrap().clone()
165            },
166            ModuleIdentity::ScriptId(script_id) => {
167                let inline_module_map = global.get_inline_module_map().borrow();
168                inline_module_map.get(script_id).unwrap().clone()
169            },
170        }
171    }

305    /// recursively checks if all of the transitive descendants are
306    /// in the FetchingDescendants or later status
307    fn recursive_check_descendants(
308        module_tree: &ModuleTree,
309        module_map: &HashMap<ServoUrl, Rc<ModuleTree>>,
310        discovered_urls: &mut HashSet<ServoUrl>,
311    ) -> bool {
312        discovered_urls.insert(module_tree.url.clone());
313
314        let descendant_urls = module_tree.descendant_urls.borrow();
315
316        for descendant_url in descendant_urls.iter() {
317            match module_map.get(&descendant_url.clone()) {
318                None => return false,
319                Some(descendant_module) => {
320                    if discovered_urls.contains(&descendant_module.url) {
321                        continue;
322                    }
323
324                    let descendant_status = descendant_module.get_status();
325                    if descendant_status < ModuleStatus::FetchingDescendants {
326                        return false;
327                    }
328
329                    let all_ready_descendants = ModuleTree::recursive_check_descendants(
330                        descendant_module,
331                        module_map,
332                        discovered_urls,
333                    );
334
335                    if !all_ready_descendants {
336                        return false;
337                    }
338                },
339            }
340        }
341
342        true
343    }

352    // We just leverage the power of Promise to run the task for `finish` the owner.
353    // Thus, we will always `resolve` it and no need to register a callback for `reject`
354    fn append_handler(&self, owner: ModuleOwner, module_identity: ModuleIdentity, can_gc: CanGc) {
355        let this = owner.clone();
356        let identity = module_identity.clone();
357
358        let handler = PromiseNativeHandler::new(
359            &owner.global(),
360            Some(ModuleHandler::new_boxed(Box::new(
361                task!(fetched_resolve: move || {
362                    this.notify_owner_to_finish(identity, CanGc::note());
363                }),
364            ))),
365            None,
366            can_gc,
367        );
368
369        let realm = enter_realm(&*owner.global());
370        let comp = InRealm::Entered(&realm);
371        let _ais = AutoIncumbentScript::new(&owner.global());
372
373        if let Some(promise) = self.promise.borrow().as_ref() {
374            promise.append_native_handler(&handler, comp, can_gc);
375            return;
376        }
377
378        let new_promise = Promise::new_in_current_realm(comp, can_gc);
379        new_promise.append_native_handler(&handler, comp, can_gc);
380        *self.promise.borrow_mut() = Some(new_promise);
381    }

459    #[expect(unsafe_code)]
460    #[allow(clippy::too_many_arguments)]
461    /// <https://html.spec.whatwg.org/multipage/#creating-a-module-script>
462    /// Step 7-11.
463    /// Although the CanGc argument appears unused, it represents the GC operations that
464    /// can occur as part of compiling a script.
465    fn compile_module_script(
466        &self,
467        global: &GlobalScope,
468        owner: ModuleOwner,
469        module_script_text: Rc<DOMString>,
470        url: &ServoUrl,
471        options: ScriptFetchOptions,
472        mut module_script: RustMutableHandleObject,
473        inline: bool,
474        line_number: u64,
475        introduction_type: Option<&'static CStr>,
476        can_gc: CanGc,
477    ) -> Result<(), RethrowError> {
478        let cx = GlobalScope::get_cx();
479        let _ac = JSAutoRealm::new(*cx, *global.reflector().get_jsobject());
480
481        let mut compile_options =
482            unsafe { CompileOptionsWrapper::new_raw(*cx, url.as_str(), line_number as u32) };
483        if let Some(introduction_type) = introduction_type {
484            compile_options.set_introduction_type(introduction_type);
485        }
486        let mut module_source = ModuleSource {
487            source: module_script_text,
488            unminified_dir: global.unminified_js_dir(),
489            external: !inline,
490            url: url.clone(),
491        };
492        crate::unminify::unminify_js(&mut module_source);
493
494        unsafe {
495            module_script.set(CompileModule1(
496                *cx,
497                compile_options.ptr,
498                &mut transform_str_to_source_text(&module_source.source.str()),
499            ));
500
501            if module_script.is_null() {
502                warn!("fail to compile module script of {}", url);
503
504                rooted!(in(*cx) let mut exception = UndefinedValue());
505                assert!(JS_GetPendingException(*cx, exception.handle_mut()));
506                JS_ClearPendingException(*cx);
507
508                return Err(RethrowError(RootedTraceableBox::from_box(Heap::boxed(
509                    exception.get(),
510                ))));
511            }
512
513            let module_script_data = Rc::new(ModuleScript::new(url.clone(), options, Some(owner)));
514
515            SetModulePrivate(
516                module_script.get(),
517                &PrivateValue(Rc::into_raw(module_script_data) as *const _),
518            );
519
520            debug!("module script of {} compile done", url);
521
522            self.resolve_requested_module_specifiers(
523                global,
524                module_script.handle().into_handle(),
525                can_gc,
526            )
527            .map(|_| ())
528        }
529    }

561    /// Execute the provided module, storing the evaluation return value in the provided
562    /// mutable handle. Although the CanGc appears unused, it represents the GC operations
563    /// possible when evluating arbitrary JS.
564    #[expect(unsafe_code)]
565    pub(crate) fn execute_module(
566        &self,
567        global: &GlobalScope,
568        module_record: HandleObject,
569        eval_result: MutableHandleValue,
570        _can_gc: CanGc,
571    ) -> Result<(), RethrowError> {
572        let cx = GlobalScope::get_cx();
573        let _ac = JSAutoRealm::new(*cx, *global.reflector().get_jsobject());
574
575        unsafe {
576            let ok = ModuleEvaluate(*cx, module_record, eval_result);
577            assert!(ok, "module evaluation failed");
578
579            rooted!(in(*cx) let mut evaluation_promise = ptr::null_mut::<JSObject>());
580            if eval_result.is_object() {
581                evaluation_promise.set(eval_result.to_object());
582            }
583
584            let throw_result = ThrowOnModuleEvaluationFailure(
585                *cx,
586                evaluation_promise.handle().into(),
587                ModuleErrorBehaviour::ThrowModuleErrorsSync,
588            );
589            if !throw_result {
590                warn!("fail to evaluate module");
591
592                rooted!(in(*cx) let mut exception = UndefinedValue());
593                assert!(JS_GetPendingException(*cx, exception.handle_mut()));
594                JS_ClearPendingException(*cx);
595
596                Err(RethrowError(RootedTraceableBox::from_box(Heap::boxed(
597                    exception.get(),
598                ))))
599            } else {
600                debug!("module evaluated successfully");
601                Ok(())
602            }
603        }

623    #[expect(unsafe_code)]
624    fn resolve_requested_module_specifiers(
625        &self,
626        global: &GlobalScope,
627        module_object: HandleObject,
628        can_gc: CanGc,
629    ) -> Result<IndexSet<ServoUrl>, RethrowError> {
630        let cx = GlobalScope::get_cx();
631        let _ac = JSAutoRealm::new(*cx, *global.reflector().get_jsobject());
632
633        let mut specifier_urls = IndexSet::new();
634
635        unsafe {
636            let length = GetRequestedModulesCount(*cx, module_object);
637
638            for index in 0..length {
639                let jsstr =
640                    std::ptr::NonNull::new(GetRequestedModuleSpecifier(*cx, module_object, index))
641                        .unwrap();
642                let specifier = DOMString::from_string(jsstr_to_string(*cx, jsstr));
643
644                rooted!(in(*cx) let mut private = UndefinedValue());
645                JS_GetModulePrivate(module_object.get(), private.handle_mut());
646                let private = private.handle().into_handle();
647                let script = module_script_from_reference_private(&private);
648                let url = ModuleTree::resolve_module_specifier(global, script, specifier, can_gc);
649
650                if url.is_err() {
651                    let specifier_error =
652                        gen_type_error(global, "Wrong module specifier".to_owned(), can_gc);
653
654                    return Err(specifier_error);
655                }
656
657                specifier_urls.insert(url.unwrap());
658            }
659        }
660
661        Ok(specifier_urls)
662    }

664    /// <https://html.spec.whatwg.org/multipage/#resolve-a-module-specifier>
665    fn resolve_module_specifier(
666        global: &GlobalScope,
667        script: Option<&ModuleScript>,
668        specifier: DOMString,
669        can_gc: CanGc,
670    ) -> Fallible<ServoUrl> {
671        // Step 1~3 to get settingsObject and baseURL
672        let script_global = script.and_then(|s| s.owner.as_ref().map(|o| o.global()));
673        // Step 1. Let settingsObject and baseURL be null.
674        let (global, base_url): (&GlobalScope, &ServoUrl) = match script {
675            // Step 2. If referringScript is not null, then:
676            // Set settingsObject to referringScript's settings object.
677            // Set baseURL to referringScript's base URL.
678            Some(s) => (script_global.as_ref().map_or(global, |g| g), &s.base_url),
679            // Step 3. Otherwise:
680            // Set settingsObject to the current settings object.
681            // Set baseURL to settingsObject's API base URL.
682            // FIXME(#37553): Is this the correct current settings object?
683            None => (global, &global.api_base_url()),
684        };
685
686        // Step 4. Let importMap be an empty import map.
687        // Step 5. If settingsObject's global object implements Window, then set importMap to settingsObject's
688        // global object's import map.
689        let import_map = if global.is::<Window>() {
690            Some(global.import_map())
691        } else {
692            None
693        };
694
695        // Step 6. Let serializedBaseURL be baseURL, serialized.
696        let serialized_base_url = base_url.as_str();
697        // Step 7. Let asURL be the result of resolving a URL-like module specifier given specifier and baseURL.
698        let as_url = Self::resolve_url_like_module_specifier(&specifier, base_url);
699        // Step 8. Let normalizedSpecifier be the serialization of asURL, if asURL is non-null;
700        // otherwise, specifier.
701        let specifier = specifier.str();
702        let normalized_specifier = match &as_url {
703            Some(url) => url.as_str(),
704            None => &specifier,
705        };
706
707        // Step 9. Let result be a URL-or-null, initially null.
708        let mut result = None;
709        if let Some(map) = import_map {
710            // Step 10. For each scopePrefix  scopeImports of importMap's scopes:
711            for (prefix, imports) in &map.scopes {
712                // Step 10.1 If scopePrefix is serializedBaseURL, or if scopePrefix ends with U+002F (/)
713                // and scopePrefix is a code unit prefix of serializedBaseURL, then:
714                let prefix = prefix.as_str();
715                if prefix == serialized_base_url ||
716                    (serialized_base_url.starts_with(prefix) && prefix.ends_with('\u{002f}'))
717                {
718                    // Step 10.1.1 Let scopeImportsMatch be the result of resolving an imports match
719                    // given normalizedSpecifier, asURL, and scopeImports.
720                    // Step 10.1.2 If scopeImportsMatch is not null, then set result to scopeImportsMatch,
721                    // and break.
722                    result = resolve_imports_match(
723                        normalized_specifier,
724                        as_url.as_ref(),
725                        imports,
726                        can_gc,
727                    )?;
728                    break;
729                }
730            }
731
732            // Step 11. If result is null, set result to the result of resolving an imports match given
733            // normalizedSpecifier, asURL, and importMap's imports.
734            if result.is_none() {
735                result = resolve_imports_match(
736                    normalized_specifier,
737                    as_url.as_ref(),
738                    &map.imports,
739                    can_gc,
740                )?;
741            }
742        }
743
744        // Step 12. If result is null, set it to asURL.
745        if result.is_none() {
746            result = as_url.clone();
747        }
748
749        // Step 13. If result is not null, then:
750        match result {
751            Some(result) => {
752                // Step 13.1 Add module to resolved module set given settingsObject, serializedBaseURL,
753                // normalizedSpecifier, and asURL.
754                global.add_module_to_resolved_module_set(
755                    serialized_base_url,
756                    normalized_specifier,
757                    as_url.clone(),
758                );
759                // Step 13.2 Return result.
760                Ok(result)
761            },
762            // Step 14. Throw a TypeError indicating that specifier was a bare specifier,
763            // but was not remapped to anything by importMap.
764            None => Err(Error::Type(
765                "Specifier was a bare specifier, but was not remapped to anything by importMap."
766                    .to_owned(),
767            )),
768        }
769    }

771    /// <https://html.spec.whatwg.org/multipage/#resolving-a-url-like-module-specifier>
772    fn resolve_url_like_module_specifier(
773        specifier: &DOMString,
774        base_url: &ServoUrl,
775    ) -> Option<ServoUrl> {
776        // Step 1. If specifier starts with "/", "./", or "../", then:
777        if specifier.starts_with('/') ||
778            specifier.starts_with_str("./") ||
779            specifier.starts_with_str("../")
780        {
781            // Step 1.1. Let url be the result of URL parsing specifier with baseURL.
782            return ServoUrl::parse_with_base(Some(base_url), &specifier.str()).ok();
783        }
784        // Step 2. Let url be the result of URL parsing specifier (with no base URL).
785        ServoUrl::parse(&specifier.str()).ok()
786    }

788    /// <https://html.spec.whatwg.org/multipage/#finding-the-first-parse-error>
789    fn find_first_parse_error(
790        &self,
791        global: &GlobalScope,
792        discovered_urls: &mut HashSet<ServoUrl>,
793    ) -> (Option<NetworkError>, Option<RethrowError>) {
794        // 3.
795        discovered_urls.insert(self.url.clone());
796
797        // 4.
798        let record = self.get_record().borrow();
799        if record.is_none() {
800            return (
801                self.network_error.borrow().clone(),
802                self.rethrow_error.borrow().clone(),
803            );
804        }
805
806        let module_map = global.get_module_map().borrow();
807        let mut parse_error: Option<RethrowError> = None;
808
809        // 5-6.
810        let descendant_urls = self.descendant_urls.borrow();
811        for descendant_module in descendant_urls
812            .iter()
813            // 7.
814            .filter_map(|url| module_map.get(&url.clone()))
815        {
816            // 8-2.
817            if discovered_urls.contains(&descendant_module.url) {
818                continue;
819            }
820
821            // 8-3.
822            let (child_network_error, child_parse_error) =
823                descendant_module.find_first_parse_error(global, discovered_urls);
824
825            // Due to network error's priority higher than parse error,
826            // we will return directly when we meet a network error.
827            if child_network_error.is_some() {
828                return (child_network_error, None);
829            }
830
831            // 8-4.
832            //
833            // In case of having any network error in other descendants,
834            // we will store the "first" parse error and keep running this
835            // loop to ensure we don't have any network error.
836            if child_parse_error.is_some() && parse_error.is_none() {
837                parse_error = child_parse_error;
838            }
839        }
840
841        // Step 9.
842        (None, parse_error)
843    }

845    // FIXME: spec links in this function are all broken, so its unclear what this algorithm does
846    /// <https://html.spec.whatwg.org/multipage/#fetch-the-descendants-of-a-module-script>
847    fn fetch_module_descendants(
848        &self,
849        owner: &ModuleOwner,
850        destination: Destination,
851        options: &ScriptFetchOptions,
852        parent_identity: ModuleIdentity,
853        can_gc: CanGc,
854    ) {
855        debug!("Start to load dependencies of {}", self.url);
856
857        let global = owner.global();
858
859        self.set_status(ModuleStatus::FetchingDescendants);
860
861        let specifier_urls = {
862            let raw_record = self.record.borrow();
863            match raw_record.as_ref() {
864                // Step 1.
865                None => {
866                    self.set_status(ModuleStatus::Finished);
867                    debug!(
868                        "Module {} doesn't have module record but tried to load descendants.",
869                        self.url
870                    );
871                    return;
872                },
873                // Step 5.
874                Some(raw_record) => {
875                    self.resolve_requested_module_specifiers(&global, raw_record.handle(), can_gc)
876                },
877            }
878        };
879
880        match specifier_urls {
881            // Step 3.
882            Ok(valid_specifier_urls) if valid_specifier_urls.is_empty() => {
883                debug!("Module {} doesn't have any dependencies.", self.url);
884                self.advance_finished_and_link(&global, can_gc);
885            },
886            Ok(valid_specifier_urls) => {
887                self.descendant_urls
888                    .borrow_mut()
889                    .extend(valid_specifier_urls.clone());
890
891                let urls_to_fetch = {
892                    let mut urls = IndexSet::new();
893                    let mut visited_urls = self.visited_urls.borrow_mut();
894
895                    for parsed_url in &valid_specifier_urls {
896                        // Step 5-3.
897                        if !visited_urls.contains(parsed_url) {
898                            // Step 5-3-1.
899                            urls.insert(parsed_url.clone());
900                            // Step 5-3-2.
901                            visited_urls.insert(parsed_url.clone());
902
903                            self.insert_incomplete_fetch_url(parsed_url);
904                        }
905                    }
906                    urls
907                };
908
909                // Step 3.
910                if urls_to_fetch.is_empty() {
911                    debug!(
912                        "After checking with visited urls, module {} doesn't have dependencies to load.",
913                        &self.url
914                    );
915                    self.advance_finished_and_link(&global, can_gc);
916                    return;
917                }
918
919                // Step 8.
920
921                let visited_urls = self.visited_urls.borrow().clone();
922                let options = options.descendant_fetch_options();
923
924                for url in urls_to_fetch {
925                    // https://html.spec.whatwg.org/multipage/#internal-module-script-graph-fetching-procedure
926                    // Step 1.
927                    assert!(self.visited_urls.borrow().contains(&url));
928
929                    // Step 2.
930                    fetch_single_module_script(
931                        owner.clone(),
932                        url,
933                        visited_urls.clone(),
934                        destination,
935                        options.clone(),
936                        Some(parent_identity.clone()),
937                        false,
938                        None,
939                        // TODO: is this correct?
940                        Some(IntroductionType::IMPORTED_MODULE),
941                        can_gc,
942                    );
943                }
944            },
945            Err(error) => {
946                self.set_rethrow_error(error);
947                self.advance_finished_and_link(&global, can_gc);
948            },
949        }
950    }

952    /// <https://html.spec.whatwg.org/multipage/#fetch-the-descendants-of-and-link-a-module-script>
953    /// step 4-7.
954    fn advance_finished_and_link(&self, global: &GlobalScope, can_gc: CanGc) {
955        {
956            if !self.has_all_ready_descendants(global) {
957                return;
958            }
959        }
960
961        self.set_status(ModuleStatus::Finished);
962
963        debug!("Going to advance and finish for: {}", self.url);
964
965        {
966            // Notify parents of this module to finish
967            //
968            // Before notifying, if the parent module has already had zero incomplete
969            // fetches, then it means we don't need to notify it.
970            let parent_identities = self.parent_identities.borrow();
971            for parent_identity in parent_identities.iter() {
972                let parent_tree = parent_identity.get_module_tree(global);
973
974                let incomplete_count_before_remove = {
975                    let incomplete_urls = parent_tree.get_incomplete_fetch_urls().borrow();
976                    incomplete_urls.len()
977                };
978
979                if incomplete_count_before_remove > 0 {
980                    parent_tree.remove_incomplete_fetch_url(&self.url);
981                    parent_tree.advance_finished_and_link(global, can_gc);
982                }
983            }
984        }
985
986        let mut discovered_urls: HashSet<ServoUrl> = HashSet::new();
987        let (network_error, rethrow_error) =
988            self.find_first_parse_error(global, &mut discovered_urls);
989
990        match (network_error, rethrow_error) {
991            (Some(network_error), _) => {
992                self.set_network_error(network_error);
993            },
994            (None, None) => {
995                let module_record = self.get_record().borrow();
996                if let Some(record) = &*module_record {
997                    let instantiated = self.instantiate_module_tree(global, record.handle());
998
999                    if let Err(exception) = instantiated {
1000                        self.set_rethrow_error(exception);
1001                    }
1002                }
1003            },
1004            (None, Some(error)) => {
1005                self.set_rethrow_error(error);
1006            },
1007        }
1008
1009        let promise = self.promise.borrow();
1010        if let Some(promise) = promise.as_ref() {
1011            promise.resolve_native(&(), can_gc);
1012        }
1013    }
1014}

1098    ) {

1100
1101        let module = global.dynamic_module_list().remove(dynamic_module_id);
1102
1103        let cx = GlobalScope::get_cx();
1104        let module_tree = module_identity.get_module_tree(&global);
1105

1107        // we've run `find_first_parse_error` which means we've had the highest

1111        let existing_rethrow_error = module_tree.get_rethrow_error().borrow().as_ref().cloned();

1123                    .execute_module(&global, record, rval.handle_mut().into(), can_gc)
1124                    .err();
1125
1126                if let Some(exception) = evaluated.clone() {
1127                    module_tree.set_rethrow_error(exception);

1129            }

1144            },

1147        };

1150
1151        rooted!(in(*cx) let mut evaluation_promise = ptr::null_mut::<JSObject>());

1157            let ok = FinishDynamicModuleImport(
1158                *cx,
1159                evaluation_promise.handle().into(),
1160                module.referencing_private.handle(),
1161                module.specifier.handle(),
1162                module.promise.reflector().get_jsobject().into_handle(),
1163            );
1164            if ok {
1165                assert!(!JS_IsExceptionPending(*cx));
1166            } else {
1167                warn!("failed to finish dynamic module import");
1168            }
1169        }
1170    }

1293                Rc::new(DOMString::from(source_text)),
1294                meta.final_url,
1295                self.options.clone(),
1296                ScriptType::Module,
1297                global.unminified_js_dir(),
1298            ))
1299        });
1300
1301        let module_tree = {
1302            let module_map = global.get_module_map().borrow();
1303            module_map.get(&self.url).unwrap().clone()
1304        };
1305
1306        module_tree.remove_incomplete_fetch_url(&self.url);
1307
1308        // Step 12.
1309        match load {
1310            Err(err) => {
1311                error!("Failed to fetch {} with error {:?}", &self.url, err);
1312                module_tree.set_network_error(err);
1313                module_tree.advance_finished_and_link(&global, CanGc::note());
1314            },
1315            Ok(ref resp_mod_script) => {
1316                module_tree.set_text(resp_mod_script.text());
1317
1318                let cx = GlobalScope::get_cx();
1319                rooted!(in(*cx) let mut compiled_module: *mut JSObject = ptr::null_mut());
1320                let compiled_module_result = module_tree.compile_module_script(
1321                    &global,
1322                    self.owner.clone(),
1323                    resp_mod_script.text(),
1324                    &self.url,
1325                    self.options.clone(),
1326                    compiled_module.handle_mut(),
1327                    false,
1328                    1, // external scripts start at the first line of the file
1329                    self.introduction_type,
1330                    CanGc::note(),
1331                );
1332
1333                match compiled_module_result {
1334                    Err(exception) => {
1335                        module_tree.set_rethrow_error(exception);
1336                        module_tree.advance_finished_and_link(&global, CanGc::note());
1337                    },
1338                    Ok(_) => {
1339                        module_tree.set_record(ModuleObject::new(compiled_module.handle()));
1340
1341                        module_tree.fetch_module_descendants(
1342                            &self.owner,
1343                            self.destination,
1344                            &self.options,
1345                            ModuleIdentity::ModuleUrl(self.url.clone()),
1346                            CanGc::note(),
1347                        );
1348                    },
1349                }
1350            },
1351        }
1352
1353        if let Ok(response) = response {
1354            network_listener::submit_timing(&self, &response, CanGc::note());
1355        }
1356    }
1357
1358    fn process_csp_violations(&mut self, _request_id: RequestId, violations: Vec<Violation>) {
1359        let global = &self.resource_timing_global();
1360        global.report_csp_violations(violations, None, None);
1361    }

1489/// <https://html.spec.whatwg.org/multipage/#fetch-an-import()-module-script-graph>
1490#[expect(unsafe_code)]
1491fn fetch_an_import_module_script_graph(
1492    global: &GlobalScope,
1493    module_request: RawHandle<*mut JSObject>,
1494    reference_private: RawHandleValue,
1495    promise: Rc<Promise>,
1496    can_gc: CanGc,
1497) -> Result<(), RethrowError> {
1498    // Step 1.
1499    let cx = GlobalScope::get_cx();
1500    let specifier = unsafe {
1501        let jsstr = std::ptr::NonNull::new(GetModuleRequestSpecifier(*cx, module_request)).unwrap();
1502        DOMString::from_string(jsstr_to_string(*cx, jsstr))
1503    };
1504    let mut options = ScriptFetchOptions::default_classic_script(global);
1505    let module_data = unsafe { module_script_from_reference_private(&reference_private) };
1506    if let Some(data) = module_data {
1507        options = data.options.descendant_fetch_options();
1508    }
1509    let url = ModuleTree::resolve_module_specifier(global, module_data, specifier, can_gc);
1510
1511    // Step 2.
1512    if url.is_err() {
1513        let specifier_error = gen_type_error(global, "Wrong module specifier".to_owned(), can_gc);
1514        return Err(specifier_error);
1515    }
1516
1517    let dynamic_module_id = DynamicModuleId(Uuid::new_v4());
1518
1519    // Step 3.
1520    let owner = match unsafe { module_script_from_reference_private(&reference_private) } {
1521        Some(module_data) if module_data.owner.is_some() => module_data.owner.clone().unwrap(),
1522        _ => ModuleOwner::DynamicModule(Trusted::new(&DynamicModuleOwner::new(
1523            global,
1524            promise.clone(),
1525            dynamic_module_id,
1526            can_gc,
1527        ))),
1528    };
1529
1530    let dynamic_module = RootedTraceableBox::new(DynamicModule {
1531        promise,
1532        specifier: Heap::default(),
1533        referencing_private: Heap::default(),
1534        id: dynamic_module_id,
1535    });
1536    dynamic_module.specifier.set(module_request.get());
1537    dynamic_module
1538        .referencing_private
1539        .set(reference_private.get());
1540
1541    let url = url.unwrap();
1542
1543    let mut visited_urls = HashSet::new();
1544    visited_urls.insert(url.clone());
1545
1546    fetch_single_module_script(
1547        owner,
1548        url,
1549        visited_urls,
1550        Destination::Script,
1551        options,
1552        None,
1553        true,
1554        Some(dynamic_module),
1555        Some(IntroductionType::IMPORTED_MODULE),
1556        can_gc,
1557    );
1558    Ok(())
1559}

1563/// <https://tc39.es/ecma262/#sec-HostLoadImportedModule>
1564/// <https://html.spec.whatwg.org/multipage/#hostloadimportedmodule>
1565unsafe extern "C" fn HostResolveImportedModule(
1566    cx: *mut JSContext,
1567    reference_private: RawHandleValue,
1568    specifier: RawHandle<*mut JSObject>,
1569) -> *mut JSObject {
1570    let in_realm_proof = AlreadyInRealm::assert_for_cx(unsafe { SafeJSContext::from_ptr(cx) });
1571    let global_scope = unsafe { GlobalScope::from_context(cx, InRealm::Already(&in_realm_proof)) };
1572
1573    // Step 5.
1574    let module_data = unsafe { module_script_from_reference_private(&reference_private) };
1575    let jsstr =
1576        std::ptr::NonNull::new(unsafe { GetModuleRequestSpecifier(cx, specifier) }).unwrap();
1577    let specifier = DOMString::from_string(unsafe { jsstr_to_string(cx, jsstr) });
1578    let url =
1579        ModuleTree::resolve_module_specifier(&global_scope, module_data, specifier, CanGc::note());
1580
1581    // Step 6.
1582    assert!(url.is_ok());
1583
1584    let parsed_url = url.unwrap();
1585
1586    // Step 4 & 7.
1587    let module_map = global_scope.get_module_map().borrow();
1588
1589    let module_tree = module_map.get(&parsed_url);
1590
1591    // Step 9.
1592    assert!(module_tree.is_some());
1593
1594    let fetched_module_object = module_tree.unwrap().get_record().borrow();
1595
1596    // Step 8.
1597    assert!(fetched_module_object.is_some());
1598
1599    // Step 10.
1600    if let Some(record) = &*fetched_module_object {
1601        return record.handle().get();
1602    }
1603
1604    unreachable!()
1605}

1720/// <https://html.spec.whatwg.org/multipage/#fetch-a-single-module-script>
1721#[allow(clippy::too_many_arguments)]
1722fn fetch_single_module_script(
1723    owner: ModuleOwner,
1724    url: ServoUrl,
1725    visited_urls: HashSet<ServoUrl>,
1726    destination: Destination,
1727    options: ScriptFetchOptions,
1728    parent_identity: Option<ModuleIdentity>,
1729    top_level_module_fetch: bool,
1730    dynamic_module: Option<RootedTraceableBox<DynamicModule>>,
1731    introduction_type: Option<&'static CStr>,
1732    can_gc: CanGc,
1733) {
1734    {
1735        // Step 1.
1736        let global = owner.global();
1737        let module_map = global.get_module_map().borrow();
1738
1739        debug!("Start to fetch {}", url);
1740
1741        if let Some(module_tree) = module_map.get(&url.clone()) {
1742            let status = module_tree.get_status();
1743
1744            debug!("Meet a fetched url {} and its status is {:?}", url, status);
1745
1746            match dynamic_module {
1747                Some(module) => module_tree.append_dynamic_module_handler(
1748                    owner.clone(),
1749                    ModuleIdentity::ModuleUrl(url.clone()),
1750                    module,
1751                    can_gc,
1752                ),
1753                None if top_level_module_fetch => module_tree.append_handler(
1754                    owner.clone(),
1755                    ModuleIdentity::ModuleUrl(url.clone()),
1756                    can_gc,
1757                ),
1758                // do nothing if it's neither a dynamic module nor a top level module
1759                None => {},
1760            }
1761
1762            if let Some(parent_identity) = parent_identity {
1763                module_tree.insert_parent_identity(parent_identity);
1764            }
1765
1766            match status {
1767                ModuleStatus::Initial => unreachable!(
1768                    "We have the module in module map so its status should not be `initial`"
1769                ),
1770                // Step 2.
1771                ModuleStatus::Fetching => {},
1772                // Step 3.
1773                ModuleStatus::FetchingDescendants | ModuleStatus::Finished => {
1774                    module_tree.advance_finished_and_link(&global, can_gc);
1775                },
1776            }
1777
1778            return;
1779        }
1780    }
1781
1782    let global = owner.global();
1783    let is_external = true;
1784    let module_tree = ModuleTree::new(url.clone(), is_external, visited_urls);
1785    module_tree.set_status(ModuleStatus::Fetching);
1786
1787    match dynamic_module {
1788        Some(module) => module_tree.append_dynamic_module_handler(
1789            owner.clone(),
1790            ModuleIdentity::ModuleUrl(url.clone()),
1791            module,
1792            can_gc,
1793        ),
1794        None if top_level_module_fetch => module_tree.append_handler(
1795            owner.clone(),
1796            ModuleIdentity::ModuleUrl(url.clone()),
1797            can_gc,
1798        ),
1799        // do nothing if it's neither a dynamic module nor a top level module
1800        None => {},
1801    }
1802
1803    if let Some(parent_identity) = parent_identity {
1804        module_tree.insert_parent_identity(parent_identity);
1805    }
1806
1807    module_tree.insert_incomplete_fetch_url(&url);
1808
1809    // Step 4.
1810    global.set_module_map(url.clone(), module_tree);
1811
1812    // Step 5-6.
1813    let mode = match destination {
1814        Destination::Worker | Destination::SharedWorker if top_level_module_fetch => {
1815            RequestMode::SameOrigin
1816        },
1817        _ => RequestMode::CorsMode,
1818    };
1819
1820    let document: Option<DomRoot<Document>> = match &owner {
1821        ModuleOwner::Worker(_) | ModuleOwner::DynamicModule(_) => None,
1822        ModuleOwner::Window(script) => Some(script.root().owner_document()),
1823    };
1824    let webview_id = document.as_ref().map(|document| document.webview_id());
1825
1826    // Step 7-8.
1827    let request = RequestBuilder::new(webview_id, url.clone(), global.get_referrer())
1828        .destination(destination)
1829        .parser_metadata(options.parser_metadata)
1830        .integrity_metadata(options.integrity_metadata.clone())
1831        .credentials_mode(options.credentials_mode)
1832        .referrer_policy(options.referrer_policy)
1833        .mode(mode)
1834        .with_global_scope(&global)
1835        .cryptographic_nonce_metadata(options.cryptographic_nonce.clone());
1836
1837    let context = ModuleContext {
1838        owner,
1839        data: vec![],
1840        metadata: None,
1841        url: url.clone(),
1842        destination,
1843        options,
1844        status: Ok(()),
1845        introduction_type,
1846    };
1847
1848    let network_listener = NetworkListener::new(
1849        context,
1850        global.task_manager().networking_task_source().to_sendable(),
1851    );
1852    match document {
1853        Some(document) => {
1854            document.loader_mut().fetch_async_with_callback(
1855                LoadType::Script(url),
1856                request,
1857                network_listener.into_callback(),
1858            );
1859        },
1860        None => global.fetch_with_network_listener(request, network_listener),
1861    }
1862}

1864/// <https://html.spec.whatwg.org/multipage/#fetch-an-inline-module-script-graph>
1865pub(crate) fn fetch_inline_module_script(
1866    owner: ModuleOwner,
1867    module_script_text: Rc<DOMString>,
1868    url: ServoUrl,
1869    script_id: ScriptId,
1870    options: ScriptFetchOptions,
1871    line_number: u64,
1872    can_gc: CanGc,
1873) {
1874    let global = owner.global();
1875    let is_external = false;
1876    let module_tree = ModuleTree::new(url.clone(), is_external, HashSet::new());
1877
1878    let cx = GlobalScope::get_cx();
1879    rooted!(in(*cx) let mut compiled_module: *mut JSObject = ptr::null_mut());
1880    let compiled_module_result = module_tree.compile_module_script(
1881        &global,
1882        owner.clone(),
1883        module_script_text,
1884        &url,
1885        options.clone(),
1886        compiled_module.handle_mut(),
1887        true,
1888        line_number,
1889        Some(IntroductionType::INLINE_SCRIPT),
1890        can_gc,
1891    );
1892
1893    match compiled_module_result {
1894        Ok(_) => {
1895            module_tree.append_handler(owner.clone(), ModuleIdentity::ScriptId(script_id), can_gc);
1896            module_tree.set_record(ModuleObject::new(compiled_module.handle()));
1897
1898            // We need to set `module_tree` into inline module map in case
1899            // of that the module descendants finished right after the
1900            // fetch module descendants step.
1901            global.set_inline_module_map(script_id, module_tree);
1902
1903            // Due to needed to set `module_tree` to inline module_map first,
1904            // we will need to retrieve it again so that we can do the fetch
1905            // module descendants step.
1906            let inline_module_map = global.get_inline_module_map().borrow();
1907            let module_tree = inline_module_map.get(&script_id).unwrap().clone();
1908
1909            module_tree.fetch_module_descendants(
1910                &owner,
1911                Destination::Script,
1912                &options,
1913                ModuleIdentity::ScriptId(script_id),
1914                can_gc,
1915            );
1916        },
1917        Err(exception) => {
1918            module_tree.set_rethrow_error(exception);
1919            module_tree.set_status(ModuleStatus::Finished);
1920            global.set_inline_module_map(script_id, module_tree);

1985/// <https://html.spec.whatwg.org/multipage/#merge-existing-and-new-import-maps>
1986fn merge_existing_and_new_import_maps(
1987    global: &GlobalScope,
1988    new_import_map: ImportMap,
1989    can_gc: CanGc,
1990) {
1991    // Step 1. Let newImportMapScopes be a deep copy of newImportMap's scopes.
1992    let new_import_map_scopes = new_import_map.scopes;
1993
1994    // Step 2. Let oldImportMap be global's import map.
1995    let mut old_import_map = global.import_map_mut();
1996
1997    // Step 3. Let newImportMapImports be a deep copy of newImportMap's imports.
1998    let mut new_import_map_imports = new_import_map.imports;
1999
2000    let resolved_module_set = global.resolved_module_set();
2001    // Step 4. For each scopePrefix  scopeImports of newImportMapScopes:
2002    for (scope_prefix, mut scope_imports) in new_import_map_scopes {
2003        // Step 4.1. For each record of global's resolved module set:
2004        for record in resolved_module_set.iter() {
2005            // If scopePrefix is record's serialized base URL, or if scopePrefix ends with
2006            // U+002F (/) and scopePrefix is a code unit prefix of record's serialized base URL, then:
2007            let prefix = scope_prefix.as_str();
2008            if prefix == record.base_url ||
2009                (record.base_url.starts_with(prefix) && prefix.ends_with('\u{002f}'))
2010            {
2011                // For each specifierKey  resolutionResult of scopeImports:
2012                scope_imports.retain(|key, val| {
2013                    // If specifierKey is record's specifier, or if all of the following conditions are true:
2014                    // specifierKey ends with U+002F (/);
2015                    // specifierKey is a code unit prefix of record's specifier;
2016                    // either record's specifier as a URL is null or is special,
2017                    if *key == record.specifier ||
2018                        (key.ends_with('\u{002f}') &&
2019                            record.specifier.starts_with(key) &&
2020                            (record.specifier_url.is_none() ||
2021                                record
2022                                    .specifier_url
2023                                    .as_ref()
2024                                    .map(|u| u.is_special_scheme())
2025                                    .unwrap_or_default()))
2026                    {
2027                        // The user agent may report a warning to the console indicating the ignored rule.
2028                        // They may choose to avoid reporting if the rule is identical to an existing one.
2029                        Console::internal_warn(
2030                            global,
2031                            DOMString::from(format!("Ignored rule: {key} -> {val:?}.")),
2032                        );
2033                        // Remove scopeImports[specifierKey].
2034                        false
2035                    } else {
2036                        true
2037                    }
2038                })
2039            }
2040        }
2041
2042        // Step 4.2 If scopePrefix exists in oldImportMap's scopes
2043        if old_import_map.scopes.contains_key(&scope_prefix) {
2044            // set oldImportMap's scopes[scopePrefix] to the result of
2045            // merging module specifier maps, given scopeImports and oldImportMap's scopes[scopePrefix].
2046            let merged_module_specifier_map = merge_module_specifier_maps(
2047                global,
2048                scope_imports,
2049                &old_import_map.scopes[&scope_prefix],
2050                can_gc,
2051            );
2052            old_import_map
2053                .scopes
2054                .insert(scope_prefix, merged_module_specifier_map);
2055        } else {
2056            // Step 4.3 Otherwise, set oldImportMap's scopes[scopePrefix] to scopeImports.
2057            old_import_map.scopes.insert(scope_prefix, scope_imports);
2058        }
2059    }
2060
2061    // Step 5. For each url  integrity of newImportMap's integrity:
2062    for (url, integrity) in &new_import_map.integrity {
2063        // Step 5.1 If url exists in oldImportMap's integrity, then:
2064        if old_import_map.integrity.contains_key(url) {
2065            // Step 5.1.1 The user agent may report a warning to the console indicating the ignored rule.
2066            // They may choose to avoid reporting if the rule is identical to an existing one.
2067            Console::internal_warn(
2068                global,
2069                DOMString::from(format!("Ignored rule: {url} -> {integrity}.")),
2070            );
2071            // Step 5.1.2 Continue.
2072            continue;
2073        }
2074
2075        // Step 5.2 Set oldImportMap's integrity[url] to integrity.
2076        old_import_map
2077            .integrity
2078            .insert(url.clone(), integrity.clone());
2079    }
2080
2081    // Step 6. For each record of global's resolved module set:
2082    for record in resolved_module_set.iter() {
2083        // For each specifier  url of newImportMapImports:
2084        new_import_map_imports.retain(|specifier, val| {
2085            // If specifier starts with record's specifier, then:
2086            if specifier.starts_with(&record.specifier) {
2087                // The user agent may report a warning to the console indicating the ignored rule.
2088                // They may choose to avoid reporting if the rule is identical to an existing one.
2089                Console::internal_warn(
2090                    global,
2091                    DOMString::from(format!("Ignored rule: {specifier} -> {val:?}.")),
2092                );
2093                // Remove newImportMapImports[specifier].
2094                false
2095            } else {
2096                true
2097            }
2098        });
2099    }
2100
2101    // Step 7. Set oldImportMap's imports to the result of merge module specifier maps,
2102    // given newImportMapImports and oldImportMap's imports.
2103    let merged_module_specifier_map = merge_module_specifier_maps(
2104        global,
2105        new_import_map_imports,
2106        &old_import_map.imports,
2107        can_gc,
2108    );
2109    old_import_map.imports = merged_module_specifier_map;
2110}

2239/// <https://html.spec.whatwg.org/multipage/#sorting-and-normalizing-a-module-specifier-map>
2240fn sort_and_normalize_module_specifier_map(
2241    global: &GlobalScope,
2242    original_map: &JsonMap<String, JsonValue>,
2243    base_url: &ServoUrl,
2244    can_gc: CanGc,
2245) -> ModuleSpecifierMap {
2246    // Step 1. Let normalized be an empty ordered map.
2247    let mut normalized = ModuleSpecifierMap::new();
2248
2249    // Step 2. For each specifier_key -> value in originalMap
2250    for (specifier_key, value) in original_map {
2251        // Step 2.1 Let normalized_specifier_key be the result of
2252        // normalizing a specifier key given specifier_key and base_url.
2253        let Some(normalized_specifier_key) =
2254            normalize_specifier_key(global, specifier_key, base_url, can_gc)
2255        else {
2256            // Step 2.2 If normalized_specifier_key is null, then continue.
2257            continue;
2258        };
2259
2260        // Step 2.3 If value is not a string, then:
2261        let JsonValue::String(value) = value else {
2262            // Step 2.3.1 The user agent may report a warning to the console
2263            // indicating that addresses need to be strings.
2264            Console::internal_warn(global, DOMString::from("Addresses need to be strings."));
2265
2266            // Step 2.3.2 Set normalized[normalized_specifier_key] to null.
2267            normalized.insert(normalized_specifier_key, None);
2268            // Step 2.3.3 Continue.
2269            continue;
2270        };
2271
2272        // Step 2.4. Let address_url be the result of resolving a URL-like module specifier given value and baseURL.
2273        let value = DOMString::from(value.as_str());
2274        let Some(address_url) = ModuleTree::resolve_url_like_module_specifier(&value, base_url)
2275        else {
2276            // Step 2.5 If address_url is null, then:
2277            // Step 2.5.1. The user agent may report a warning to the console
2278            // indicating that the address was invalid.
2279            Console::internal_warn(
2280                global,
2281                DOMString::from(format!(
2282                    "Value failed to resolve to module specifier: {value}"
2283                )),
2284            );
2285
2286            // Step 2.5.2 Set normalized[normalized_specifier_key] to null.
2287            normalized.insert(normalized_specifier_key, None);
2288            // Step 2.5.3 Continue.
2289            continue;
2290        };
2291
2292        // Step 2.6 If specifier_key ends with U+002F (/), and the serialization of
2293        // address_url does not end with U+002F (/), then:
2294        if specifier_key.ends_with('\u{002f}') && !address_url.as_str().ends_with('\u{002f}') {
2295            // step 2.6.1. The user agent may report a warning to the console
2296            // indicating that an invalid address was given for the specifier key specifierKey;
2297            // since specifierKey ends with a slash, the address needs to as well.
2298            Console::internal_warn(
2299                global,
2300                DOMString::from(format!(
2301                    "Invalid address for specifier key '{specifier_key}': {address_url}.
2302                    Since specifierKey ends with a slash, the address needs to as well."
2303                )),
2304            );
2305
2306            // Step 2.6.2 Set normalized[normalized_specifier_key] to null.
2307            normalized.insert(normalized_specifier_key, None);
2308            // Step 2.6.3 Continue.
2309            continue;
2310        }
2311
2312        // Step 2.7 Set normalized[normalized_specifier_key] to address_url.
2313        normalized.insert(normalized_specifier_key, Some(address_url));
2314    }
2315
2316    // Step 3. Return the result of sorting in descending order normalized
2317    // with an entry a being less than an entry b if a's key is code unit less than b's key.
2318    normalized.sort_by(|a_key, _, b_key, _| b_key.cmp(a_key));

2320}

2472                // Step 1.1.2 Assert: resolutionResult is a URL.
2473                // This is checked by Url type already.
2474                // Step 1.1.3 Return resolutionResult.
2475                return Ok(Some(resolution_result.clone()));
2476            } else {
2477                // Step 1.1.1 If resolutionResult is null, then throw a TypeError.
2478                return Err(Error::Type(
2479                    "Resolution of specifierKey was blocked by a null entry.".to_owned(),
2480                ));
2481            }
2482        }
2483
2484        // Step 1.2 If all of the following are true:
2485        // - specifierKey ends with U+002F (/)
2486        // - specifierKey is a code unit prefix of normalizedSpecifier
2487        // - either asURL is null, or asURL is special, then:
2488        if specifier_key.ends_with('\u{002f}') &&
2489            normalized_specifier.starts_with(specifier_key) &&
2490            (as_url.is_none() || as_url.map(|u| u.is_special_scheme()).unwrap_or_default())
2491        {
2492            // Step 1.2.1 If resolutionResult is null, then throw a TypeError.
2493            // Step 1.2.2 Assert: resolutionResult is a URL.
2494            let Some(resolution_result) = resolution_result else {
2495                return Err(Error::Type(
2496                    "Resolution of specifierKey was blocked by a null entry.".to_owned(),
2497                ));
2498            };
2499
2500            // Step 1.2.3 Let afterPrefix be the portion of normalizedSpecifier after the initial specifierKey prefix.
2501            let after_prefix = normalized_specifier
2502                .strip_prefix(specifier_key)
2503                .expect("specifier_key should be the prefix of normalized_specifier");
2504
2505            // Step 1.2.4 Assert: resolutionResult, serialized, ends with U+002F (/), as enforced during parsing.
2506            debug_assert!(resolution_result.as_str().ends_with('\u{002f}'));
2507
2508            // Step 1.2.5 Let url be the result of URL parsing afterPrefix with resolutionResult.
2509            let url = ServoUrl::parse_with_base(Some(resolution_result), after_prefix);
2510
2511            // Step 1.2.6 If url is failure, then throw a TypeError
2512            // Step 1.2.7 Assert: url is a URL.
2513            let Ok(url) = url else {
2514                return Err(Error::Type(
2515                    "Resolution of normalizedSpecifier was blocked since
2516                    the afterPrefix portion could not be URL-parsed relative to
2517                    the resolutionResult mapped to by the specifierKey prefix."
2518                        .to_owned(),
2519                ));
2520            };
2521
2522            // Step 1.2.8 If the serialization of resolutionResult is not
2523            // a code unit prefix of the serialization of url, then throw a TypeError
2524            if !url.as_str().starts_with(resolution_result.as_str()) {
2525                return Err(Error::Type(
2526                    "Resolution of normalizedSpecifier was blocked due to
2527                    it backtracking above its prefix specifierKey."
2528                        .to_owned(),
2529                ));
2530            }
2531
2532            // Step 1.2.9 Return url.
2533            return Ok(Some(url));
2534        }
2535    }
2536
2537    // Step 2. Return null.
2538    Ok(None)
