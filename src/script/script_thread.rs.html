5//! The script thread is the thread that owns the DOM in memory, runs JavaScript, and triggers
6//! layout. It's in charge of processing events for all same-origin pages in a frame
7//! tree, and manages the entire lifetime of pages in the frame tree from initial request to
8//! teardown.
9//!
10//! Page loads follow a two-step process. When a request for a new page load is received, the
11//! network request is initiated and the relevant data pertaining to the new page is stashed.
12//! While the non-blocking request is ongoing, the script thread is free to process further events,
13//! noting when they pertain to ongoing loads (such as resizes/viewport adjustments). When the
14//! initial response is received for an ongoing load, the second phase starts - the frame tree
15//! entry is created, along with the Window and Document objects, and the appropriate parser
16//! takes over the response body. Once parsing is complete, the document lifecycle for loading
17//! a page runs its course and the script thread returns to processing events in the main event
18//! loop.

167fn with_optional_script_thread<R>(f: impl FnOnce(Option<&ScriptThread>) -> R) -> R {
168    SCRIPT_THREAD_ROOT.with(|root| {

403/// In the event of thread panic, all data on the stack runs its destructor. However, there
404/// are no reachable, owning pointers to the DOM memory, so it never gets freed by default
405/// when the script thread fails. The ScriptMemoryFailsafe uses the destructor bomb pattern
406/// to forcibly tear down the JS realms for pages associated with the failing ScriptThread.
407struct ScriptMemoryFailsafe<'a> {
408    owner: Option<&'a ScriptThread>,
409}
410
411impl<'a> ScriptMemoryFailsafe<'a> {
412    fn neuter(&mut self) {
413        self.owner = None;
414    }
415
416    fn new(owner: &'a ScriptThread) -> ScriptMemoryFailsafe<'a> {
417        ScriptMemoryFailsafe { owner: Some(owner) }
418    }
419}
420
421impl Drop for ScriptMemoryFailsafe<'_> {
422    fn drop(&mut self) {
423        if let Some(owner) = self.owner {
424            for (_, document) in owner.documents.borrow().iter() {
425                document.window().clear_js_runtime_for_script_deallocation();
426            }
427        }
428    }
429}

431impl ScriptThreadFactory for ScriptThread {
432    fn create(
433        state: InitialScriptState,
434        layout_factory: Arc<dyn LayoutFactory>,
435        image_cache_factory: Arc<dyn ImageCacheFactory>,
436        background_hang_monitor_register: Box<dyn BackgroundHangMonitorRegister>,
437    ) -> JoinHandle<()> {
438        // Setup pipeline-namespace-installing for all threads in this process.
439        // Idempotent in single-process mode.
440        PipelineNamespace::set_installer_sender(state.namespace_request_sender.clone());
441
442        let script_thread_id = state.id;
443        thread::Builder::new()
444            .name(format!("Script#{script_thread_id}"))
445            .spawn(move || {
446                thread_state::initialize(ThreadState::SCRIPT | ThreadState::LAYOUT);
447                PipelineNamespace::install(state.pipeline_namespace_id);
448                ScriptEventLoopId::install(state.id);
449                let memory_profiler_sender = state.memory_profiler_sender.clone();
450                let reporter_name = format!("script-reporter-{script_thread_id:?}");
451                let (script_thread, mut cx) = ScriptThread::new(
452                    state,
453                    layout_factory,
454                    image_cache_factory,
455                    background_hang_monitor_register,
456                );
457                SCRIPT_THREAD_ROOT.with(|root| {
458                    root.set(Some(Rc::as_ptr(&script_thread)));
459                });
460                let mut failsafe = ScriptMemoryFailsafe::new(&script_thread);
461
462                memory_profiler_sender.run_with_memory_reporting(
463                    || script_thread.start(&mut cx),
464                    reporter_name,
465                    ScriptEventLoopSender::MainThread(script_thread.senders.self_sender.clone()),
466                    CommonScriptMsg::CollectReports,
467                );
468
469                // This must always be the very last operation performed before the thread completes
470                failsafe.neuter();
471            })
472            .expect("Thread spawning failed")
473    }
474}

581    /// Step 13 of <https://html.spec.whatwg.org/multipage/#navigate>
582    pub(crate) fn navigate(
583        webview_id: WebViewId,
584        pipeline_id: PipelineId,
585        mut load_data: LoadData,
586        history_handling: NavigationHistoryBehavior,
587    ) {
588        with_script_thread(|script_thread| {
589            let is_javascript = load_data.url.scheme() == "javascript";
590            // If resource is a request whose url's scheme is "javascript"
591            // https://html.spec.whatwg.org/multipage/#navigate-to-a-javascript:-url
592            if is_javascript {
593                let Some(window) = script_thread.documents.borrow().find_window(pipeline_id) else {
594                    return;
595                };
596                let global = window.as_global_scope();
597                let trusted_global = Trusted::new(global);
598                let sender = script_thread
599                    .senders
600                    .pipeline_to_constellation_sender
601                    .clone();
602                let task = task!(navigate_javascript: move || {
603                    // Important re security. See https://github.com/servo/servo/issues/23373
604                    if trusted_global.root().is::<Window>() {
605                        let global = &trusted_global.root();
606                        if Self::navigate_to_javascript_url(global, global, &mut load_data, None, CanGc::note()) {
607                            sender
608                                .send((webview_id, pipeline_id, ScriptToConstellationMessage::LoadUrl(load_data, history_handling)))
609                                .unwrap();
610                        }
611                    }
612                });
613                // Step 19 of <https://html.spec.whatwg.org/multipage/#navigate>
614                global
615                    .task_manager()
616                    .dom_manipulation_task_source()
617                    .queue(task);
618            } else {
619                script_thread
620                    .senders
621                    .pipeline_to_constellation_sender
622                    .send((
623                        webview_id,
624                        pipeline_id,
625                        ScriptToConstellationMessage::LoadUrl(load_data, history_handling),
626                    ))
627                    .expect("Sending a LoadUrl message to the constellation failed");
628            }
629        });
630    }

633    pub(crate) fn can_navigate_to_javascript_url(
634        global: &GlobalScope,
635        load_data: &mut LoadData,
636        container: Option<&Element>,
637        can_gc: CanGc,
638    ) -> bool {
639        // Step 3. If initiatorOrigin is not same origin-domain with targetNavigable's active document's origin, then return.
640        //
641        // Important re security. See https://github.com/servo/servo/issues/23373
642        if !Self::check_load_origin(&load_data.load_origin, &global.get_url().origin()) {
643            return false;
644        }
645
646        // Step 5: If the result of should navigation request of type be blocked by
647        // Content Security Policy? given request and cspNavigationType is "Blocked", then return. [CSP]
648        if global
649            .get_csp_list()
650            .should_navigation_request_be_blocked(global, load_data, container, can_gc)
651        {
652            return false;
653        }
654
655        true
656    }

1079    /// <https://html.spec.whatwg.org/multipage/#update-the-rendering>
1080    ///
1081    /// Attempt to update the rendering and then do a microtask checkpoint if rendering was
1082    /// actually updated.
1083    ///
1084    /// Returns true if any reflows produced a new display list.
1085    pub(crate) fn update_the_rendering(&self, can_gc: CanGc) -> bool {
1086        self.last_render_opportunity_time.set(Some(Instant::now()));
1087        self.cancel_scheduled_update_the_rendering();
1088        self.needs_rendering_update.store(false, Ordering::Relaxed);
1089
1090        if !self.can_continue_running_inner() {
1091            return false;
1092        }
1093
1094        // TODO: The specification says to filter out non-renderable documents,
1095        // as well as those for which a rendering update would be unnecessary,
1096        // but this isn't happening here.
1097
1098        // TODO(#31242): the filtering of docs is extended to not exclude the ones that
1099        // has pending initial observation targets
1100        // https://w3c.github.io/IntersectionObserver/#pending-initial-observation
1101
1102        // > 2. Let docs be all fully active Document objects whose relevant agent's event loop
1103        // > is eventLoop, sorted arbitrarily except that the following conditions must be
1104        // > met:
1105        //
1106        // > Any Document B whose container document is A must be listed after A in the
1107        // > list.
1108        //
1109        // > If there are two documents A and B that both have the same non-null container
1110        // > document C, then the order of A and B in the list must match the
1111        // > shadow-including tree order of their respective navigable containers in C's
1112        // > node tree.
1113        //
1114        // > In the steps below that iterate over docs, each Document must be processed in
1115        // > the order it is found in the list.
1116        let documents_in_order = self.documents.borrow().documents_in_order();
1117
1118        // TODO: The specification reads: "for doc in docs" at each step whereas this runs all
1119        // steps per doc in docs. Currently `<iframe>` resizing depends on a parent being able to
1120        // queue resize events on a child and have those run in the same call to this method, so
1121        // that needs to be sorted out to fix this.
1122        let mut painters_generating_frames = HashSet::new();
1123        for pipeline_id in documents_in_order.iter() {
1124            let document = self
1125                .documents
1126                .borrow()
1127                .find_document(*pipeline_id)
1128                .expect("Got pipeline for Document not managed by this ScriptThread.");
1129
1130            if !document.is_fully_active() {
1131                continue;
1132            }
1133
1134            if document.waiting_on_canvas_image_updates() {
1135                continue;
1136            }
1137
1138            // Clear this as early as possible so that any callbacks that
1139            // trigger new reasons for updating the rendering don't get lost.
1140            document.clear_rendering_update_reasons();
1141
1142            // TODO(#31581): The steps in the "Revealing the document" section need to be implemented
1143            // `process_pending_input_events` handles the focusing steps as well as other events
1144            // from `Paint`.
1145
1146            // TODO: Should this be broken and to match the specification more closely? For instance see
1147            // https://html.spec.whatwg.org/multipage/#flush-autofocus-candidates.
1148            self.process_pending_input_events(*pipeline_id, can_gc);
1149
1150            // > 8. For each doc of docs, run the resize steps for doc. [CSSOMVIEW]
1151            let resized = document.window().run_the_resize_steps(can_gc);
1152
1153            // > 9. For each doc of docs, run the scroll steps for doc.
1154            document.run_the_scroll_steps(can_gc);
1155
1156            // Media queries is only relevant when there are resizing.
1157            if resized {
1158                // 10. For each doc of docs, evaluate media queries and report changes for doc.
1159                document
1160                    .window()
1161                    .evaluate_media_queries_and_report_changes(can_gc);
1162
1163                // https://html.spec.whatwg.org/multipage/#img-environment-changes
1164                // As per the spec, this can be run at any time.
1165                document.react_to_environment_changes()
1166            }
1167
1168            // > 11. For each doc of docs, update animations and send events for doc, passing
1169            // > in relative high resolution time given frameTimestamp and doc's relevant
1170            // > global object as the timestamp [WEBANIMATIONS]
1171            document.update_animations_and_send_events(can_gc);
1172
1173            // TODO(#31866): Implement "run the fullscreen steps" from
1174            // https://fullscreen.spec.whatwg.org/multipage/#run-the-fullscreen-steps.
1175
1176            // TODO(#31868): Implement the "context lost steps" from
1177            // https://html.spec.whatwg.org/multipage/#context-lost-steps.
1178
1179            // > 14. For each doc of docs, run the animation frame callbacks for doc, passing
1180            // > in the relative high resolution time given frameTimestamp and doc's
1181            // > relevant global object as the timestamp.
1182            document.run_the_animation_frame_callbacks(can_gc);
1183
1184            // Run the resize observer steps.
1185            let _realm = enter_realm(&*document);
1186            let mut depth = Default::default();
1187            while document.gather_active_resize_observations_at_depth(&depth) {
1188                // Note: this will reflow the doc.
1189                depth = document.broadcast_active_resize_observations(can_gc);
1190            }
1191
1192            if document.has_skipped_resize_observations() {
1193                document.deliver_resize_loop_error_notification(can_gc);
1194                // Ensure that another turn of the event loop occurs to process

1237    /// Schedule a rendering update ("update the rendering"), if necessary. This
1238    /// can be necessary for a couple reasons. For instance, when the DOM
1239    /// changes a scheduled rendering update becomes necessary if one isn't
1240    /// scheduled already. Another example is if rAFs are running but no display
1241    /// lists are being produced. In that case the [`ScriptThread`] is
1242    /// responsible for scheduling animation ticks.
1243    fn maybe_schedule_rendering_opportunity_after_ipc_message(
1244        &self,
1245        built_any_display_lists: bool,
1246    ) {
1247        let needs_rendering_update = self
1248            .documents
1249            .borrow()
1250            .iter()
1251            .any(|(_, document)| document.needs_rendering_update());
1252        let running_animations = self.documents.borrow().iter().any(|(_, document)| {
1253            document.is_fully_active() &&
1254                !document.window().throttled() &&
1255                (document.animations().running_animation_count() != 0 ||
1256                    document.has_active_request_animation_frame_callbacks())
1257        });
1258
1259        // If we are not running animations and no rendering update is
1260        // necessary, just exit early and schedule the next rendering update
1261        // when it becomes necessary.
1262        if !needs_rendering_update && !running_animations {
1263            return;
1264        }
1265
1266        // If animations are running and a reflow in this event loop iteration
1267        // produced a display list, rely on the renderer to inform us of the
1268        // next animation tick / rendering opportunity.
1269        if running_animations && built_any_display_lists {
1270            return;
1271        }
1272
1273        // There are two possibilities: rendering needs to be updated or we are
1274        // scheduling a new animation tick because animations are running, but
1275        // not changing the DOM. In the later case we can wait a bit longer
1276        // until the next "update the rendering" call as it's more efficient to
1277        // slow down rAFs that don't change the DOM.
1278        //
1279        // TODO: Should either of these delays be reduced to also reduce update latency?
1280        let animation_delay = if running_animations && !needs_rendering_update {
1281            // 30 milliseconds (33 FPS) is used here as the rendering isn't changing
1282            // so it isn't a problem to slow down rAF callback calls. In addition, this allows
1283            // renderer-based ticks to arrive first.
1284            Duration::from_millis(30)
1285        } else {
1286            // 20 milliseconds (50 FPS) is used here in order to allow any renderer-based
1287            // animation ticks to arrive first.
1288            Duration::from_millis(20)
1289        };
1290
1291        let time_since_last_rendering_opportunity = self
1292            .last_render_opportunity_time
1293            .get()
1294            .map(|last_render_opportunity_time| Instant::now() - last_render_opportunity_time)
1295            .unwrap_or(Duration::MAX)
1296            .min(animation_delay);
1297        self.schedule_update_the_rendering_timer_if_necessary(
1298            animation_delay - time_since_last_rendering_opportunity,
1299        );
1300    }

1326    /// Handle incoming messages from other tasks and the task queue.
1327    fn handle_msgs(&self, cx: &mut js::context::JSContext) -> bool {
1328        // Proritize rendering tasks and others, and gather all other events as `sequential`.
1329        let mut sequential = vec![];
1330
1331        // Notify the background-hang-monitor we are waiting for an event.
1332        self.background_hang_monitor.notify_wait();
1333
1334        // Receive at least one message so we don't spinloop.
1335        debug!("Waiting for event.");
1336        let fully_active = self.get_fully_active_document_ids();
1337        let mut event = self.receivers.recv(
1338            &self.task_queue,
1339            &self.timer_scheduler.borrow(),
1340            &fully_active,
1341        );
1342
1343        loop {
1344            debug!("Handling event: {event:?}");
1345
1346            // Dispatch any completed timers, so that their tasks can be run below.
1347            self.timer_scheduler
1348                .borrow_mut()
1349                .dispatch_completed_timers();
1350
1351            // https://html.spec.whatwg.org/multipage/#event-loop-processing-model step 7
1352            match event {
1353                // This has to be handled before the ResizeMsg below,
1354                // otherwise the page may not have been added to the
1355                // child list yet, causing the find() to fail.
1356                MixedMessage::FromConstellation(ScriptThreadMessage::SpawnPipeline(
1357                    new_pipeline_info,
1358                )) => {
1359                    self.spawn_pipeline(new_pipeline_info);
1360                },
1361                MixedMessage::FromScript(MainThreadScriptMsg::Inactive) => {
1362                    // An event came-in from a document that is not fully-active, it has been stored by the task-queue.
1363                    // Continue without adding it to "sequential".
1364                },
1365                MixedMessage::FromConstellation(ScriptThreadMessage::ExitFullScreen(id)) => self
1366                    .profile_event(ScriptThreadEventCategory::ExitFullscreen, Some(id), || {
1367                        self.handle_exit_fullscreen(id, cx);
1368                    }),
1369                _ => {
1370                    sequential.push(event);
1371                },
1372            }
1373
1374            // If any of our input sources has an event pending, we'll perform another
1375            // iteration and check for events. If there are no events pending, we'll move
1376            // on and execute the sequential events.
1377            match self.receivers.try_recv(&self.task_queue, &fully_active) {
1378                Some(new_event) => event = new_event,
1379                None => break,
1380            }
1381        }
1382
1383        // Process the gathered events.
1384        debug!("Processing events.");
1385        for msg in sequential {
1386            debug!("Processing event {:?}.", msg);
1387            let category = self.categorize_msg(&msg);
1388            let pipeline_id = msg.pipeline_id();
1389            let _realm = pipeline_id.and_then(|id| {
1390                let global = self.documents.borrow().find_global(id);
1391                global.map(|global| enter_realm(&*global))
1392            });
1393
1394            if self.closing.load(Ordering::SeqCst) {
1395                // If we've received the closed signal from the BHM, only handle exit messages.
1396                match msg {
1397                    MixedMessage::FromConstellation(ScriptThreadMessage::ExitScriptThread) => {
1398                        self.handle_exit_script_thread_msg(CanGc::from_cx(cx));
1399                        return false;
1400                    },
1401                    MixedMessage::FromConstellation(ScriptThreadMessage::ExitPipeline(
1402                        webview_id,
1403                        pipeline_id,
1404                        discard_browsing_context,
1405                    )) => {
1406                        self.handle_exit_pipeline_msg(
1407                            webview_id,
1408                            pipeline_id,
1409                            discard_browsing_context,
1410                            CanGc::from_cx(cx),
1411                        );
1412                    },
1413                    _ => {},
1414                }
1415                continue;
1416            }
1417
1418            let exiting = self.profile_event(category, pipeline_id, || {
1419                match msg {
1420                    MixedMessage::FromConstellation(ScriptThreadMessage::ExitScriptThread) => {
1421                        self.handle_exit_script_thread_msg(CanGc::from_cx(cx));
1422                        return true;
1423                    },
1424                    MixedMessage::FromConstellation(inner_msg) => {
1425                        self.handle_msg_from_constellation(inner_msg, cx)
1426                    },
1427                    MixedMessage::FromScript(inner_msg) => {
1428                        self.handle_msg_from_script(inner_msg, cx)
1429                    },
1430                    MixedMessage::FromDevtools(inner_msg) => {
1431                        self.handle_msg_from_devtools(inner_msg, cx)
1432                    },
1433                    MixedMessage::FromImageCache(inner_msg) => {
1434                        self.handle_msg_from_image_cache(inner_msg)
1435                    },
1436                    #[cfg(feature = "webgpu")]
1437                    MixedMessage::FromWebGPUServer(inner_msg) => {
1438                        self.handle_msg_from_webgpu_server(inner_msg, cx)
1439                    },
1440                    MixedMessage::TimerFired => {},
1441                }
1442
1443                false
1444            });
1445
1446            // If an `ExitScriptThread` message was handled above, bail out now.
1447            if exiting {
1448                return false;
1449            }
1450
1451            // https://html.spec.whatwg.org/multipage/#event-loop-processing-model step 6
1452            // TODO(#32003): A microtask checkpoint is only supposed to be performed after running a task.
1453            self.perform_a_microtask_checkpoint(CanGc::from_cx(cx));
1454        }
1455
1456        for (_, doc) in self.documents.borrow().iter() {
1457            let window = doc.window();
1458            window
1459                .upcast::<GlobalScope>()
1460                .perform_a_dom_garbage_collection_checkpoint();
1461        }
1462
1463        {
1464            // https://html.spec.whatwg.org/multipage/#the-end step 6
1465            let mut docs = self.docs_with_no_blocking_loads.borrow_mut();
1466            for document in docs.iter() {
1467                let _realm = enter_auto_realm(cx, &**document);
1468                document.maybe_queue_document_completion();
1469            }
1470            docs.clear();
1471        }
1472
1473        let built_any_display_lists = self.needs_rendering_update.load(Ordering::Relaxed) &&
1474            self.update_the_rendering(CanGc::from_cx(cx));
1475
1476        self.maybe_fulfill_font_ready_promises(CanGc::from_cx(cx));
1477        self.maybe_resolve_pending_screenshot_readiness_requests(CanGc::from_cx(cx));
1478
1479        // This must happen last to detect if any change above makes a rendering update necessary.
1480        self.maybe_schedule_rendering_opportunity_after_ipc_message(built_any_display_lists);
1481
1482        true
1483    }

1485    fn categorize_msg(&self, msg: &MixedMessage) -> ScriptThreadEventCategory {
1486        match *msg {
1487            MixedMessage::FromConstellation(ref inner_msg) => match *inner_msg {
1488                ScriptThreadMessage::SendInputEvent(..) => ScriptThreadEventCategory::InputEvent,
1489                _ => ScriptThreadEventCategory::ConstellationMsg,
1490            },
1491            MixedMessage::FromDevtools(_) => ScriptThreadEventCategory::DevtoolsMsg,
1492            MixedMessage::FromImageCache(_) => ScriptThreadEventCategory::ImageCacheMsg,
1493            MixedMessage::FromScript(ref inner_msg) => match *inner_msg {
1494                MainThreadScriptMsg::Common(CommonScriptMsg::Task(category, ..)) => category,
1495                MainThreadScriptMsg::RegisterPaintWorklet { .. } => {
1496                    ScriptThreadEventCategory::WorkletEvent
1497                },
1498                _ => ScriptThreadEventCategory::ScriptEvent,
1499            },
1500            #[cfg(feature = "webgpu")]
1501            MixedMessage::FromWebGPUServer(_) => ScriptThreadEventCategory::WebGPUMsg,
1502            MixedMessage::TimerFired => ScriptThreadEventCategory::TimerEvent,
1503        }
1504    }

2557    pub(crate) fn spawn_pipeline(&self, new_pipeline_info: NewPipelineInfo) {
2558        self.profile_event(
2559            ScriptThreadEventCategory::SpawnPipeline,
2560            Some(new_pipeline_info.new_pipeline_id),
2561            || {
2562                // If this is an about:blank or about:srcdoc load, it must share the
2563                // creator's origin. This must match the logic in the constellation
2564                // when creating a new pipeline
2565                let not_an_about_blank_and_about_srcdoc_load =
2566                    new_pipeline_info.load_data.url.as_str() != "about:blank" &&
2567                        new_pipeline_info.load_data.url.as_str() != "about:srcdoc";
2568                let origin = if not_an_about_blank_and_about_srcdoc_load {
2569                    MutableOrigin::new(new_pipeline_info.load_data.url.origin())
2570                } else if let Some(parent) = new_pipeline_info
2571                    .parent_info
2572                    .and_then(|pipeline_id| self.documents.borrow().find_document(pipeline_id))
2573                {
2574                    parent.origin().clone()
2575                } else if let Some(creator) = new_pipeline_info
2576                    .load_data
2577                    .creator_pipeline_id
2578                    .and_then(|pipeline_id| self.documents.borrow().find_document(pipeline_id))
2579                {
2580                    creator.origin().clone()
2581                } else {
2582                    MutableOrigin::new(ImmutableOrigin::new_opaque())
2583                };
2584
2585                // Kick off the fetch for the new resource.
2586                self.pre_page_load(InProgressLoad::new(new_pipeline_info, origin));

2776    /// <https://html.spec.whatwg.org/multipage/#window-post-message-steps>
2777    fn handle_post_message_msg(
2778        &self,
2779        pipeline_id: PipelineId,
2780        source_webview: WebViewId,
2781        source_with_ancestry: Vec<BrowsingContextId>,
2782        origin: Option<ImmutableOrigin>,
2783        source_origin: ImmutableOrigin,
2784        data: StructuredSerializedData,
2785    ) {
2786        let window = self.documents.borrow().find_window(pipeline_id);
2787        match window {
2788            None => warn!("postMessage after target pipeline {} closed.", pipeline_id),
2789            Some(window) => {
2790                let mut last = None;
2791                for browsing_context_id in source_with_ancestry.into_iter().rev() {
2792                    if let Some(window_proxy) = self.window_proxies.get(browsing_context_id) {
2793                        last = Some(window_proxy);
2794                        continue;
2795                    }
2796                    let window_proxy = WindowProxy::new_dissimilar_origin(
2797                        window.upcast::<GlobalScope>(),
2798                        browsing_context_id,
2799                        source_webview,
2800                        last.as_deref(),
2801                        None,
2802                        CreatorBrowsingContextInfo::from(last.as_deref(), None),
2803                    );
2804                    self.window_proxies
2805                        .insert(browsing_context_id, window_proxy.clone());
2806                    last = Some(window_proxy);
2807                }
2808
2809                // Step 8.3: Let source be the WindowProxy object corresponding to
2810                // incumbentSettings's global object (a Window object).
2811                let source = last.expect("Source with ancestry should contain at least one bc.");
2812
2813                // FIXME(#22512): enqueues a task; unnecessary delay.
2814                window.post_message(origin, source_origin, &source, data)
2815            },
2816        }

2907    /// We have received notification that the response associated with a load has completed.
2908    /// Kick off the document and frame tree creation process using the result.
2909    fn handle_page_headers_available(
2910        &self,
2911        webview_id: WebViewId,
2912        pipeline_id: PipelineId,
2913        metadata: Option<Metadata>,
2914        can_gc: CanGc,
2915    ) -> Option<DomRoot<ServoParser>> {
2916        if self.closed_pipelines.borrow().contains(&pipeline_id) {
2917            // If the pipeline closed, do not process the headers.
2918            return None;
2919        }
2920
2921        let Some(idx) = self
2922            .incomplete_loads
2923            .borrow()
2924            .iter()
2925            .position(|load| load.pipeline_id == pipeline_id)
2926        else {
2927            unreachable!("Pipeline shouldn't have finished loading.");
2928        };
2929
2930        // https://html.spec.whatwg.org/multipage/#process-a-navigate-response
2931        // 2. If response's status is 204 or 205, then abort these steps.
2932        let is_204_205 = match metadata {
2933            Some(ref metadata) => metadata.status.in_range(204..=205),
2934            _ => false,
2935        };
2936
2937        if is_204_205 {
2938            // If we have an existing window that is being navigated:
2939            if let Some(window) = self.documents.borrow().find_window(pipeline_id) {
2940                let window_proxy = window.window_proxy();
2941                // https://html.spec.whatwg.org/multipage/
2942                // #navigating-across-documents:delaying-load-events-mode-2
2943                if window_proxy.parent().is_some() {
2944                    // The user agent must take this nested browsing context
2945                    // out of the delaying load events mode
2946                    // when this navigation algorithm later matures,
2947                    // or when it terminates (whether due to having run all the steps,
2948                    // or being canceled, or being aborted), whichever happens first.
2949                    window_proxy.stop_delaying_load_events_mode();
2950                }
2951            }
2952            self.senders
2953                .pipeline_to_constellation_sender
2954                .send((
2955                    webview_id,
2956                    pipeline_id,
2957                    ScriptToConstellationMessage::AbortLoadUrl,
2958                ))
2959                .unwrap();
2960            return None;
2961        };
2962
2963        let load = self.incomplete_loads.borrow_mut().remove(idx);
2964        metadata.map(|meta| self.load(meta, load, can_gc))
2965    }

2976    fn handle_exit_pipeline_msg(
2977        &self,
2978        webview_id: WebViewId,
2979        pipeline_id: PipelineId,
2980        discard_bc: DiscardBrowsingContext,
2981        can_gc: CanGc,
2982    ) {

2984
2985        // Abort the parser, if any,
2986        // to prevent any further incoming networking messages from being handled.
2987        let document = self.documents.borrow_mut().remove(pipeline_id);
2988        if let Some(document) = document {

2990            debug_assert!(

2992                    .incomplete_loads
2993                    .borrow()
2994                    .iter()
2995                    .any(|load| load.pipeline_id == pipeline_id)
2996            );
2997
2998            if let Some(parser) = document.get_current_parser() {
2999                parser.abort(can_gc);
3000            }
3001
3002            debug!("{pipeline_id}: Shutting down layout");
3003            document.window().layout_mut().exit_now();
3004
3005            // Clear any active animations and unroot all of the associated DOM objects.
3006            debug!("{pipeline_id}: Clearing animations");
3007            document.animations().clear();
3008
3009            // We discard the browsing context after requesting layout shut down,
3010            // to avoid running layout on detached iframes.
3011            let window = document.window();
3012            if discard_bc == DiscardBrowsingContext::Yes {
3013                window.discard_browsing_context();
3014            }
3015
3016            debug!("{pipeline_id}: Clearing JavaScript runtime");
3017            window.clear_js_runtime();
3018        }
3019
3020        // Prevent any further work for this Pipeline.
3021        self.closed_pipelines.borrow_mut().insert(pipeline_id);
3022
3023        debug!("{pipeline_id}: Sending PipelineExited message to constellation");
3024        self.senders
3025            .pipeline_to_constellation_sender

3027                webview_id,

3702    fn handle_navigation_redirect(&self, id: PipelineId, metadata: &Metadata) {
3703        // TODO(mrobinson): This tries to accomplish some steps from
3704        // <https://html.spec.whatwg.org/multipage/#process-a-navigate-fetch>, but it's
3705        // very out of sync with the specification.
3706        assert!(metadata.location_url.is_some());
3707
3708        let mut incomplete_loads = self.incomplete_loads.borrow_mut();
3709        let Some(incomplete_load) = incomplete_loads
3710            .iter_mut()
3711            .find(|incomplete_load| incomplete_load.pipeline_id == id)
3712        else {
3713            return;
3714        };
3715
3716        // Update the `url_list` of the incomplete load to track all redirects. This will be reflected
3717        // in the new `RequestBuilder` as well.
3718        incomplete_load.url_list.push(metadata.final_url.clone());
3719
3720        let mut request_builder = incomplete_load.request_builder();
3721        request_builder.referrer = metadata
3722            .referrer
3723            .clone()
3724            .map(Referrer::ReferrerUrl)
3725            .unwrap_or(Referrer::NoReferrer);
3726        request_builder.referrer_policy = metadata.referrer_policy;
3727
3728        let headers = metadata
3729            .headers
3730            .as_ref()
3731            .map(|headers| headers.clone().into_inner())
3732            .unwrap_or_default();
3733
3734        let response_init = Some(ResponseInit {
3735            url: metadata.final_url.clone(),
3736            location_url: metadata.location_url.clone(),
3737            headers,
3738            referrer: metadata.referrer.clone(),
3739            status_code: metadata
3740                .status
3741                .try_code()
3742                .map(|code| code.as_u16())
3743                .unwrap_or(200),
3744        });
3745
3746        incomplete_load.canceller = FetchCanceller::new(
3747            request_builder.id,
3748            false,
3749            self.resource_threads.core_thread.clone(),
3750        );
3751        NavigationListener::new(request_builder, self.senders.self_sender.clone())
3752            .initiate_fetch(&self.resource_threads.core_thread, response_init);
3753    }
