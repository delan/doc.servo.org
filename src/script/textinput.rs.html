205/// The length in bytes of the first n code units in a string when encoded in UTF-16.
206///
207/// If the string is fewer than n code units, returns the length of the whole string.
208fn len_of_first_n_code_units(text: &DOMString, n: Utf16CodeUnitLength) -> Utf8CodeUnitLength {
209    let mut utf8_len = Utf8CodeUnitLength::zero();
210    let mut utf16_len = Utf16CodeUnitLength::zero();
211    for c in text.str().chars() {
212        utf16_len += Utf16CodeUnitLength(c.len_utf16());
213        if utf16_len > n {
214            break;
215        }
216        utf8_len += Utf8CodeUnitLength(c.len_utf8());
217    }
218    utf8_len
219}

274    /// Remove a character at the current editing point
275    ///
276    /// Returns true if any character was deleted
277    pub fn delete_char(&mut self, direction: Direction) -> bool {
278        if self.selection_origin.is_none() || self.selection_origin == Some(self.edit_point) {
279            let amount = match direction {
280                Direction::Forward => 1,
281                Direction::Backward => -1,
282            };
283            self.modify_selection(amount, RopeMovement::Grapheme);
284        }
285
286        if self.selection_start() == self.selection_end() {
287            return false;
288        }
289
290        self.replace_selection(&DOMString::new());
291        true
292    }

368    // Check that the selection is valid.
369    fn assert_ok_selection(&self) {
370        debug!(
371            "edit_point: {:?}, selection_origin: {:?}, direction: {:?}",
372            self.edit_point, self.selection_origin, self.selection_direction
373        );
374
375        debug_assert_eq!(self.edit_point, self.rope.clamp_index(self.edit_point));
376        if let Some(selection_origin) = self.selection_origin {
377            debug_assert_eq!(selection_origin, self.rope.clamp_index(selection_origin));
378            match self.selection_direction {
379                SelectionDirection::None | SelectionDirection::Forward => {
380                    debug_assert!(selection_origin <= self.edit_point)
381                },
382                SelectionDirection::Backward => debug_assert!(self.edit_point <= selection_origin),
383            }
384        }
385    }

410    pub fn replace_selection(&mut self, insert: &DOMString) {
411        if !self.has_selection() {
412            return;
413        }
414
415        let string_to_insert = if let Some(max_length) = self.max_length {
416            let utf16_length_without_selection =
417                self.len_utf16().saturating_sub(self.selection_utf16_len());
418            let utf16_length_that_can_be_inserted =
419                max_length.saturating_sub(utf16_length_without_selection);
420            let Utf8CodeUnitLength(last_char_index) =
421                len_of_first_n_code_units(insert, utf16_length_that_can_be_inserted);
422            &insert.str()[..last_char_index]
423        } else {
424            &insert.str()
425        };
426
427        let start = self.selection_start();
428        let end = self.selection_end();
429        let end_index_of_insertion = self.rope.replace_range(start..end, string_to_insert);
430
431        self.was_last_change_by_set_content = false;
432        self.clear_selection();
433        self.edit_point = end_index_of_insertion;
434    }

436    pub fn modify_edit_point(&mut self, amount: isize, movement: RopeMovement) {
437        if amount == 0 {
438            return;
439        }
440
441        // When moving by lines or if we do not have a selection, we do actually move
442        // the edit point from its position.
443        if matches!(movement, RopeMovement::Line) || !self.has_selection() {
444            self.clear_selection();
445            self.edit_point = self.rope.move_by(self.edit_point, movement, amount);
446            return;
447        }
448
449        // If there's a selection and we are moving by words or characters, we just collapse
450        // the selection in the direction of the motion.
451        let new_edit_point = if amount > 0 {
452            self.selection_end()
453        } else {
454            self.selection_start()
455        };
456        self.clear_selection();
457        self.edit_point = new_edit_point;
458    }
459
460    pub fn modify_selection(&mut self, amount: isize, movement: RopeMovement) {
461        let old_edit_point = self.edit_point;
462        self.edit_point = self.rope.move_by(old_edit_point, movement, amount);
463
464        if self.selection_origin.is_none() {
465            self.selection_origin = Some(old_edit_point);
466        }
467        self.update_selection_direction();
468    }
469
470    pub fn modify_selection_or_edit_point(
471        &mut self,
472        amount: isize,
473        movement: RopeMovement,
474        select: Selection,
475    ) {
476        match select {
477            Selection::Selected => self.modify_selection(amount, movement),
478            Selection::NotSelected => self.modify_edit_point(amount, movement),
479        }
480        self.assert_ok_selection();
481    }

483    /// Update the field selection_direction.
484    ///
485    /// When the edit_point (or focus) is before the selection_origin (or anchor)
486    /// you have a backward selection. Otherwise you have a forward selection.
487    fn update_selection_direction(&mut self) {
488        debug!(
489            "edit_point: {:?}, selection_origin: {:?}",
490            self.edit_point, self.selection_origin
491        );
492        self.selection_direction = if Some(self.edit_point) < self.selection_origin {
493            SelectionDirection::Backward
494        } else {
495            SelectionDirection::Forward
496        }
497    }

539    /// Process a given `KeyboardEvent` and return an action for the caller to execute.
540    pub(crate) fn handle_keydown(&mut self, event: &KeyboardEvent) -> KeyReaction {
541        let key = event.key();
542        let mods = event.modifiers();
543        self.handle_keydown_aux(key, mods, cfg!(target_os = "macos"))
544    }

546    // This function exists for easy unit testing.
547    // To test Mac OS shortcuts on other systems a flag is passed.
548    pub fn handle_keydown_aux(
549        &mut self,
550        key: Key,
551        mut mods: Modifiers,
552        macos: bool,
553    ) -> KeyReaction {
554        let maybe_select = if mods.contains(Modifiers::SHIFT) {
555            Selection::Selected
556        } else {
557            Selection::NotSelected
558        };
559        mods.remove(Modifiers::SHIFT);
560        ShortcutMatcher::new(KeyState::Down, key.clone(), mods)
561            .shortcut(Modifiers::CONTROL | Modifiers::ALT, 'B', || {
562                self.modify_selection_or_edit_point(-1, RopeMovement::Word, maybe_select);
563                KeyReaction::RedrawSelection
564            })
565            .shortcut(Modifiers::CONTROL | Modifiers::ALT, 'F', || {
566                self.modify_selection_or_edit_point(1, RopeMovement::Word, maybe_select);
567                KeyReaction::RedrawSelection
568            })
569            .shortcut(Modifiers::CONTROL | Modifiers::ALT, 'A', || {
570                self.modify_selection_or_edit_point(-1, RopeMovement::LineStartOrEnd, maybe_select);
571                KeyReaction::RedrawSelection
572            })
573            .shortcut(Modifiers::CONTROL | Modifiers::ALT, 'E', || {
574                self.modify_selection_or_edit_point(1, RopeMovement::LineStartOrEnd, maybe_select);
575                KeyReaction::RedrawSelection
576            })
577            .optional_shortcut(macos, Modifiers::CONTROL, 'A', || {
578                self.modify_selection_or_edit_point(-1, RopeMovement::LineStartOrEnd, maybe_select);
579                KeyReaction::RedrawSelection
580            })
581            .optional_shortcut(macos, Modifiers::CONTROL, 'E', || {
582                self.modify_selection_or_edit_point(1, RopeMovement::LineStartOrEnd, maybe_select);
583                KeyReaction::RedrawSelection
584            })
585            .shortcut(CMD_OR_CONTROL, 'A', || {
586                self.select_all();
587                KeyReaction::RedrawSelection
588            })
589            .shortcut(CMD_OR_CONTROL, 'X', || {
590                if let Some(text) = self.get_selection_text() {
591                    self.clipboard_provider.set_text(text);
592                    self.delete_char(Direction::Backward);
593                }
594                KeyReaction::DispatchInput(None, IsComposing::NotComposing, InputType::DeleteByCut)
595            })
596            .shortcut(CMD_OR_CONTROL, 'C', || {
597                // TODO(stevennovaryo): we should not provide text to clipboard for type=password
598                if let Some(text) = self.get_selection_text() {
599                    self.clipboard_provider.set_text(text);
600                }
601                KeyReaction::DispatchInput(None, IsComposing::NotComposing, InputType::Nothing)
602            })
603            .shortcut(CMD_OR_CONTROL, 'V', || {
604                if let Ok(text_content) = self.clipboard_provider.get_text() {
605                    self.insert_string(&text_content);
606                    KeyReaction::DispatchInput(
607                        Some(text_content),
608                        IsComposing::NotComposing,
609                        InputType::InsertFromPaste,
610                    )
611                } else {
612                    KeyReaction::DispatchInput(
613                        Some("".to_string()),
614                        IsComposing::NotComposing,
615                        InputType::InsertFromPaste,
616                    )
617                }
618            })
619            .shortcut(Modifiers::empty(), Key::Named(NamedKey::Delete), || {
620                if self.delete_char(Direction::Forward) {
621                    KeyReaction::DispatchInput(
622                        None,
623                        IsComposing::NotComposing,
624                        InputType::DeleteContentForward,
625                    )
626                } else {
627                    KeyReaction::Nothing
628                }
629            })
630            .shortcut(Modifiers::empty(), Key::Named(NamedKey::Backspace), || {
631                if self.delete_char(Direction::Backward) {
632                    KeyReaction::DispatchInput(
633                        None,
634                        IsComposing::NotComposing,
635                        InputType::DeleteContentBackward,
636                    )
637                } else {
638                    KeyReaction::Nothing
639                }
640            })
641            .optional_shortcut(
642                macos,
643                Modifiers::META,
644                Key::Named(NamedKey::ArrowLeft),
645                || {
646                    self.modify_selection_or_edit_point(
647                        -1,
648                        RopeMovement::LineStartOrEnd,
649                        maybe_select,
650                    );
651                    KeyReaction::RedrawSelection
652                },
653            )
654            .optional_shortcut(
655                macos,
656                Modifiers::META,
657                Key::Named(NamedKey::ArrowRight),
658                || {
659                    self.modify_selection_or_edit_point(
660                        1,
661                        RopeMovement::LineStartOrEnd,
662                        maybe_select,
663                    );
664                    KeyReaction::RedrawSelection
665                },
666            )
667            .optional_shortcut(
668                macos,
669                Modifiers::META,
670                Key::Named(NamedKey::ArrowUp),
671                || {
672                    self.modify_selection_or_edit_point(
673                        -1,
674                        RopeMovement::RopeStartOrEnd,
675                        maybe_select,
676                    );
677                    KeyReaction::RedrawSelection
678                },
679            )
680            .optional_shortcut(
681                macos,
682                Modifiers::META,
683                Key::Named(NamedKey::ArrowDown),
684                || {
685                    self.modify_selection_or_edit_point(
686                        1,
687                        RopeMovement::RopeStartOrEnd,
688                        maybe_select,
689                    );
690                    KeyReaction::RedrawSelection
691                },
692            )
693            .shortcut(Modifiers::ALT, Key::Named(NamedKey::ArrowLeft), || {
694                self.modify_selection_or_edit_point(-1, RopeMovement::Word, maybe_select);
695                KeyReaction::RedrawSelection
696            })
697            .shortcut(Modifiers::ALT, Key::Named(NamedKey::ArrowRight), || {
698                self.modify_selection_or_edit_point(1, RopeMovement::Word, maybe_select);
699                KeyReaction::RedrawSelection
700            })
701            .shortcut(Modifiers::empty(), Key::Named(NamedKey::ArrowLeft), || {
702                self.modify_selection_or_edit_point(-1, RopeMovement::Grapheme, maybe_select);
703                KeyReaction::RedrawSelection
704            })
705            .shortcut(Modifiers::empty(), Key::Named(NamedKey::ArrowRight), || {
706                self.modify_selection_or_edit_point(1, RopeMovement::Grapheme, maybe_select);
707                KeyReaction::RedrawSelection
708            })
709            .shortcut(Modifiers::empty(), Key::Named(NamedKey::ArrowUp), || {
710                self.modify_selection_or_edit_point(-1, RopeMovement::Line, maybe_select);
711                KeyReaction::RedrawSelection
712            })
713            .shortcut(Modifiers::empty(), Key::Named(NamedKey::ArrowDown), || {
714                self.modify_selection_or_edit_point(1, RopeMovement::Line, maybe_select);
715                KeyReaction::RedrawSelection
716            })
717            .shortcut(Modifiers::empty(), Key::Named(NamedKey::Enter), || {
718                self.handle_return()
719            })
720            .optional_shortcut(
721                macos,
722                Modifiers::empty(),
723                Key::Named(NamedKey::Home),
724                || {
725                    self.modify_selection_or_edit_point(
726                        -1,
727                        RopeMovement::RopeStartOrEnd,
728                        maybe_select,
729                    );
730                    KeyReaction::RedrawSelection
731                },
732            )
733            .optional_shortcut(macos, Modifiers::empty(), Key::Named(NamedKey::End), || {
734                self.modify_selection_or_edit_point(1, RopeMovement::RopeStartOrEnd, maybe_select);
735                KeyReaction::RedrawSelection
736            })
737            .shortcut(Modifiers::empty(), Key::Named(NamedKey::PageUp), || {
738                self.modify_selection_or_edit_point(-28, RopeMovement::Line, maybe_select);
739                KeyReaction::RedrawSelection
740            })
741            .shortcut(Modifiers::empty(), Key::Named(NamedKey::PageDown), || {
742                self.modify_selection_or_edit_point(28, RopeMovement::Line, maybe_select);
743                KeyReaction::RedrawSelection
744            })
745            .otherwise(|| {
746                if let Key::Character(ref c) = key {
747                    self.insert_string(c.as_str());
748                    return KeyReaction::DispatchInput(
749                        Some(c.to_string()),
750                        IsComposing::NotComposing,
751                        InputType::InsertText,
752                    );
753                }
754                if matches!(key, Key::Named(NamedKey::Process)) {
755                    return KeyReaction::DispatchInput(
756                        None,
757                        IsComposing::Composing,
758                        InputType::Nothing,
759                    );
760                }
761                KeyReaction::Nothing
762            })
763            .unwrap()
764    }

832    pub fn set_selection_range_utf8(
833        &mut self,
834        mut start: Utf8CodeUnitLength,
835        mut end: Utf8CodeUnitLength,
836        direction: SelectionDirection,
837    ) {
838        let text_end = self.get_content().len_utf8();
839        if end > text_end {
840            end = text_end;
841        }
842        if start > end {
843            start = end;
844        }
845
846        self.selection_direction = direction;
847
848        match direction {
849            SelectionDirection::None | SelectionDirection::Forward => {
850                self.selection_origin = Some(self.rope.utf8_offset_to_rope_index(start));
851                self.edit_point = self.rope.utf8_offset_to_rope_index(end);
852            },
853            SelectionDirection::Backward => {
854                self.selection_origin = Some(self.rope.utf8_offset_to_rope_index(end));
855                self.edit_point = self.rope.utf8_offset_to_rope_index(start);
856            },
857        }
858        self.assert_ok_selection();
859    }

861    /// This implements step 3 onward from:
862    ///
863    ///  - <https://www.w3.org/TR/clipboard-apis/#copy-action>
864    ///  - <https://www.w3.org/TR/clipboard-apis/#cut-action>
865    ///  - <https://www.w3.org/TR/clipboard-apis/#paste-action>
866    ///
867    /// Earlier steps should have already been run by the callers.
868    pub(crate) fn handle_clipboard_event(
869        &mut self,
870        clipboard_event: &ClipboardEvent,
871    ) -> ClipboardEventReaction {
872        let event = clipboard_event.upcast::<Event>();
873        if !event.IsTrusted() {
874            return ClipboardEventReaction::empty();
875        }
876
877        // This step is common to all event types in the specification.
878        // Step 3: If the event was not canceled, then
879        if event.DefaultPrevented() {
880            // Step 4: Else, if the event was canceled
881            // Step 4.1: Return false.
882            return ClipboardEventReaction::empty();
883        }
884
885        let event_type = event.Type();
886        match_domstring_ascii!(event_type,
887            "copy" => {
888                // These steps are from <https://www.w3.org/TR/clipboard-apis/#copy-action>:
889                let selection = self.get_selection_text();
890
891                // Step 3.1 Copy the selected contents, if any, to the clipboard
892                if let Some(text) = selection {
893                    self.clipboard_provider.set_text(text);
894                }
895
896                // Step 3.2 Fire a clipboard event named clipboardchange
897                ClipboardEventReaction::new(ClipboardEventFlags::FireClipboardChangedEvent)
898            },
899            "cut" => {
900                // These steps are from <https://www.w3.org/TR/clipboard-apis/#cut-action>:
901                let selection = self.get_selection_text();
902
903                // Step 3.1 If there is a selection in an editable context where cutting is enabled, then
904                let Some(text) = selection else {
905                    // Step 3.2 Else, if there is no selection or the context is not editable, then
906                    return ClipboardEventReaction::empty();
907                };
908
909                // Step 3.1.1 Copy the selected contents, if any, to the clipboard
910                self.clipboard_provider.set_text(text);
911
912                // Step 3.1.2 Remove the contents of the selection from the document and collapse the selection.
913                self.delete_char(Direction::Backward);
914
915                // Step 3.1.3 Fire a clipboard event named clipboardchange
916                // Step 3.1.4 Queue tasks to fire any events that should fire due to the modification.
917                ClipboardEventReaction::new(
918                    ClipboardEventFlags::FireClipboardChangedEvent |
919                        ClipboardEventFlags::QueueInputEvent,
920                )
921                .with_input_type(InputType::DeleteByCut)
922            },
923            "paste" => {
924                // These steps are from <https://www.w3.org/TR/clipboard-apis/#paste-action>:
925                let Some(data_transfer) = clipboard_event.get_clipboard_data() else {
926                    return ClipboardEventReaction::empty();
927                };
928                let Some(drag_data_store) = data_transfer.data_store() else {
929                    return ClipboardEventReaction::empty();
930                };
931
932                // Step 3.1: If there is a selection or cursor in an editable context where pasting is
933                // enabled, then:
934                // TODO: Our TextInput always has a selection or an input point. It's likely that this
935                // shouldn't be the case when the entry loses the cursor.
936
937                // Step 3.1.1: Insert the most suitable content found on the clipboard, if any, into the
938                // context.
939                // TODO: Only text content is currently supported, but other data types should be supported
940                // in the future.
941                let Some(text_content) =
942                    drag_data_store
943                        .iter_item_list()
944                        .find_map(|item| match item {
945                            Kind::Text { data, .. } => Some(data.to_string()),
946                            _ => None,
947                        })
948                else {
949                    return ClipboardEventReaction::empty();
950                };
951                if text_content.is_empty() {
952                    return ClipboardEventReaction::empty();
953                }
954
955                self.insert_string(&text_content);
956
957                // Step 3.1.2: Queue tasks to fire any events that should fire due to the
958                // modification, see  5.3 Integration with other scripts and events for details.
959                ClipboardEventReaction::new(ClipboardEventFlags::QueueInputEvent)
960                    .with_text(text_content)
961                    .with_input_type(InputType::InsertFromPaste)
962            },
963        _ => ClipboardEventReaction::empty(),)
964    }

966    /// <https://w3c.github.io/uievents/#event-type-input>
967    pub(crate) fn queue_input_event(
968        &self,
969        target: &EventTarget,
970        data: Option<String>,
971        is_composing: IsComposing,
972        input_type: InputType,
973    ) {
974        let global = target.global();
975        let target = Trusted::new(target);
976        global.task_manager().user_interaction_task_source().queue(
977            task!(fire_input_event: move || {
978                let target = target.root();
979                let global = target.global();
980                let window = global.as_window();
981                let event = InputEvent::new(
982                    window,
983                    None,
984                    DOMString::from("input"),
985                    true,
986                    false,
987                    Some(window),
988                    0,
989                    data.map(DOMString::from),
990                    is_composing.into(),
991                    input_type.as_str().into(),
992                    CanGc::note(),
993                );
994                let event = event.upcast::<Event>();
995                event.set_composed(true);
996                event.fire(&target, CanGc::note());
997            }),
998        );
999    }
