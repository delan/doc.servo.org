16/// A struct to store a reference to the reflector of a DOM object.
17#[derive(MallocSizeOf)]
18#[cfg_attr(crown, crown::unrooted_must_root_lint::must_root)]
19// If you're renaming or moving this field, update the path in plugins::reflector as well
20pub struct Reflector {
21    #[ignore_malloc_size_of = "defined and measured in rust-mozjs"]
22    object: Heap<*mut JSObject>,
23}
24
25unsafe impl js::gc::Traceable for Reflector {
26    unsafe fn trace(&self, _: *mut js::jsapi::JSTracer) {}
27}
28
29impl PartialEq for Reflector {
30    fn eq(&self, other: &Reflector) -> bool {
31        self.object.get() == other.object.get()
32    }
33}
34
35impl Reflector {
36    /// Get the reflector.
37    #[inline]
38    pub fn get_jsobject(&self) -> HandleObject<'_> {
39        // We're rooted, so it's safe to hand out a handle to object in Heap
40        unsafe { HandleObject::from_raw(self.object.handle()) }
41    }
42
43    /// Initialize the reflector. (May be called only once.)
44    ///
45    /// # Safety
46    ///
47    /// The provided [`JSObject`] pointer must point to a valid [`JSObject`].
48    pub unsafe fn set_jsobject(&self, object: *mut JSObject) {
49        assert!(self.object.get().is_null());
50        assert!(!object.is_null());
51        self.object.set(object);
52    }
53
54    /// Return a pointer to the memory location at which the JS reflector
55    /// object is stored. Used to root the reflector, as
56    /// required by the JSAPI rooting APIs.
57    pub fn rootable(&self) -> &Heap<*mut JSObject> {
58        &self.object
59    }
60
61    /// Create an uninitialized `Reflector`.
62    // These are used by the bindings and do not need `default()` functions.
63    #[expect(clippy::new_without_default)]
64    pub fn new() -> Reflector {
65        Reflector {
66            object: Heap::default(),
67        }
68    }
69}

71/// A trait to provide access to the `Reflector` for a DOM object.
72pub trait DomObject: js::gc::Traceable + 'static {
73    /// Returns the receiver's reflector.
74    fn reflector(&self) -> &Reflector;
75}
76
77impl DomObject for Reflector {
78    fn reflector(&self) -> &Self {
79        self
80    }
81}

83/// A trait to initialize the `Reflector` for a DOM object.
84pub trait MutDomObject: DomObject {
85    /// Initializes the Reflector
86    ///
87    /// # Safety
88    ///
89    /// The provided [`JSObject`] pointer must point to a valid [`JSObject`].
90    unsafe fn init_reflector(&self, obj: *mut JSObject);
91}
92
93impl MutDomObject for Reflector {
94    unsafe fn init_reflector(&self, obj: *mut JSObject) {
95        self.set_jsobject(obj)
96    }
97}

99pub trait DomGlobalGeneric<D: DomTypes>: DomObject {
100    /// Returns the [`GlobalScope`] of the realm that the [`DomObject`] was created in.  If this
101    /// object is a `Node`, this will be different from it's owning `Document` if adopted by. For
102    /// `Node`s it's almost always better to use `NodeTraits::owning_global`.
103    fn global_(&self, realm: InRealm) -> DomRoot<D::GlobalScope>
104    where
105        Self: Sized,
106    {
107        D::GlobalScope::from_reflector(self, realm)
108    }
109}
110
111impl<D: DomTypes, T: DomObject> DomGlobalGeneric<D> for T {}
