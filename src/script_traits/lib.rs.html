93/// Is a document fully active, active or inactive?
94/// A document is active if it is the current active document in its session history,
95/// it is fuly active if it is active and all of its ancestors are active,
96/// and it is inactive otherwise.
97///
98/// * <https://html.spec.whatwg.org/multipage/#active-document>
99/// * <https://html.spec.whatwg.org/multipage/#fully-active>
100#[derive(Clone, Copy, Debug, Deserialize, Eq, Hash, MallocSizeOf, PartialEq, Serialize)]
101pub enum DocumentActivity {
102    /// An inactive document
103    Inactive,
104    /// An active but not fully active document
105    Active,
106    /// A fully active document
107    FullyActive,
108}

110/// Type of recorded progressive web metric
111#[derive(Clone, Copy, Debug, Deserialize, Serialize)]
112pub enum ProgressiveWebMetricType {
113    /// Time to first Paint
114    FirstPaint,
115    /// Time to first contentful paint
116    FirstContentfulPaint,
117    /// Time for the largest contentful paint
118    LargestContentfulPaint {
119        /// The pixel area of the largest contentful element.
120        area: usize,
121        /// The type of the largest contentful paint element.
122        lcp_type: LargestContentfulPaintType,
123    },
124    /// Time to interactive
125    TimeToInteractive,
126}
127
128impl ProgressiveWebMetricType {
129    /// Returns the area if the metric type is LargestContentfulPaint
130    pub fn area(&self) -> usize {
131        match self {
132            ProgressiveWebMetricType::LargestContentfulPaint { area, .. } => *area,
133            _ => 0,
134        }
135    }
136}

147/// Messages sent to the `ScriptThread` event loop from the `Constellation`, `Paint`, and (for
148/// now) `Layout`.
149#[derive(Deserialize, IntoStaticStr, Serialize)]
150pub enum ScriptThreadMessage {
151    /// Span a new `Pipeline` in this `ScriptThread` and start fetching the contents
152    /// according to the provided `LoadData`. This will ultimately create a `Window`
153    /// and all associated data structures such as `Layout` in the `ScriptThread`.
154    SpawnPipeline(NewPipelineInfo),
155    /// Takes the associated window proxy out of "delaying-load-events-mode",
156    /// used if a scheduled navigated was refused by the embedder.
157    /// <https://html.spec.whatwg.org/multipage/#delaying-load-events-mode>
158    StopDelayingLoadEventsMode(PipelineId),
159    /// Window resized.  Sends a DOM event eventually, but first we combine events.
160    Resize(PipelineId, ViewportDetails, WindowSizeType),
161    /// Theme changed.
162    ThemeChange(PipelineId, Theme),
163    /// Notifies script that window has been resized but to not take immediate action.
164    ResizeInactive(PipelineId, ViewportDetails),
165    /// Window switched from fullscreen mode.
166    ExitFullScreen(PipelineId),
167    /// Notifies the script that the document associated with this pipeline should 'unload'.
168    UnloadDocument(PipelineId),
169    /// Notifies the script that a pipeline should be closed.
170    ExitPipeline(WebViewId, PipelineId, DiscardBrowsingContext),
171    /// Notifies the script that the whole thread should be closed.
172    ExitScriptThread,
173    /// Sends a DOM event.
174    SendInputEvent(WebViewId, PipelineId, ConstellationInputEvent),
175    /// Request that the given pipeline refresh the cursor by doing a hit test at the most
176    /// recently hovered cursor position and resetting the cursor. This happens after a
177    /// display list update is rendered.
178    RefreshCursor(PipelineId),
179    /// Requests that the script thread immediately send the constellation the title of a pipeline.
180    GetTitle(PipelineId),
181    /// Notifies script thread of a change to one of its document's activity
182    SetDocumentActivity(PipelineId, DocumentActivity),
183    /// Set whether to use less resources by running timers at a heavily limited rate.
184    SetThrottled(WebViewId, PipelineId, bool),
185    /// Notify the containing iframe (in PipelineId) that the nested browsing context (BrowsingContextId) is throttled.
186    SetThrottledInContainingIframe(WebViewId, PipelineId, BrowsingContextId, bool),
187    /// Notifies script thread that a url should be loaded in this iframe.
188    /// PipelineId is for the parent, BrowsingContextId is for the nested browsing context
189    NavigateIframe(
190        PipelineId,
191        BrowsingContextId,
192        LoadData,
193        NavigationHistoryBehavior,
194    ),
195    /// Post a message to a given window.
196    PostMessage {
197        /// The target of the message.
198        target: PipelineId,
199        /// The webview associated with the source pipeline.
200        source_webview: WebViewId,
201        /// The ancestry of browsing context associated with the source,
202        /// starting with the source itself.
203        source_with_ancestry: Vec<BrowsingContextId>,
204        /// The expected origin of the target.
205        target_origin: Option<ImmutableOrigin>,
206        /// The source origin of the message.
207        /// <https://html.spec.whatwg.org/multipage/#dom-messageevent-origin>
208        source_origin: ImmutableOrigin,
209        /// The data to be posted.
210        data: Box<StructuredSerializedData>,
211    },
212    /// Updates the current pipeline ID of a given iframe.
213    /// First PipelineId is for the parent, second is the new PipelineId for the frame.
214    UpdatePipelineId(
215        PipelineId,
216        BrowsingContextId,
217        WebViewId,
218        PipelineId,
219        UpdatePipelineIdReason,
220    ),
221    /// Updates the history state and url of a given pipeline.
222    UpdateHistoryState(PipelineId, Option<HistoryStateId>, ServoUrl),
223    /// Removes inaccesible history states.
224    RemoveHistoryStates(PipelineId, Vec<HistoryStateId>),
225    /// Set an iframe to be focused. Used when an element in an iframe gains focus.
226    /// PipelineId is for the parent, BrowsingContextId is for the nested browsing context
227    FocusIFrame(PipelineId, BrowsingContextId, FocusSequenceNumber),
228    /// Focus the document. Used when the container gains focus.
229    FocusDocument(PipelineId, FocusSequenceNumber),
230    /// Notifies that the document's container (e.g., an iframe) is not included
231    /// in the top-level browsing context's focus chain (not considering system
232    /// focus) anymore.
233    ///
234    /// Obviously, this message is invalid for a top-level document.
235    Unfocus(PipelineId, FocusSequenceNumber),
236    /// Passes a webdriver command to the script thread for execution
237    WebDriverScriptCommand(PipelineId, WebDriverScriptCommand),
238    /// Notifies script thread that all animations are done
239    TickAllAnimations(Vec<WebViewId>),
240    /// Notifies the script thread that a new Web font has been loaded, and thus the page should be
241    /// reflowed.
242    WebFontLoaded(PipelineId, bool /* success */),
243    /// Cause a `load` event to be dispatched at the appropriate iframe element.
244    DispatchIFrameLoadEvent {
245        /// The frame that has been marked as loaded.
246        target: BrowsingContextId,
247        /// The pipeline that contains a frame loading the target pipeline.
248        parent: PipelineId,
249        /// The pipeline that has completed loading.
250        child: PipelineId,
251    },
252    /// Cause a `storage` event to be dispatched at the appropriate window.
253    /// The strings are key, old value and new value.
254    DispatchStorageEvent(
255        PipelineId,
256        WebStorageType,
257        ServoUrl,
258        Option<String>,
259        Option<String>,
260        Option<String>,
261    ),
262    /// Report an error from a CSS parser for the given pipeline
263    ReportCSSError(PipelineId, String, u32, u32, String),
264    /// Reload the given page.
265    Reload(PipelineId),
266    /// Notifies the script thread about a new recorded paint metric.
267    PaintMetric(
268        PipelineId,
269        ProgressiveWebMetricType,
270        CrossProcessInstant,
271        bool, /* first_reflow */
272    ),
273    /// Notifies the media session about a user requested media session action.
274    MediaSessionAction(PipelineId, MediaSessionActionType),
275    /// Notifies script thread that WebGPU server has started
276    #[cfg(feature = "webgpu")]
277    SetWebGPUPort(IpcReceiver<WebGPUMsg>),
278    /// `Paint` scrolled and is updating the scroll states of the nodes in the given
279    /// pipeline via the Constellation.
280    SetScrollStates(PipelineId, FxHashMap<ExternalScrollId, LayoutVector2D>),
281    /// Evaluate the given JavaScript and return a result via a corresponding message
282    /// to the Constellation.
283    EvaluateJavaScript(WebViewId, PipelineId, JavaScriptEvaluationId, String),
284    /// A new batch of keys for the image cache for the specific pipeline.
285    SendImageKeysBatch(PipelineId, Vec<ImageKey>),
286    /// Preferences were updated in the parent process.
287    PreferencesUpdated(Vec<(String, PrefValue)>),
288    /// Notify the `ScriptThread` that the Servo renderer is no longer waiting on
289    /// asynchronous image uploads for the given `Pipeline`. These are mainly used
290    /// by canvas to perform uploads while the display list is being built.
291    NoLongerWaitingOnAsychronousImageUpdates(PipelineId),
292    /// Forward a keyboard scroll operation from an `<iframe>` to a parent pipeline.
293    ForwardKeyboardScroll(PipelineId, KeyboardScroll),
294    /// Request readiness for a screenshot from the given pipeline. The pipeline will
295    /// respond when it is ready to take the screenshot or will not be able to take it
296    /// in the future.
297    RequestScreenshotReadiness(WebViewId, PipelineId),
298    /// A response to a request to show an embedder user interface control.
299    EmbedderControlResponse(EmbedderControlId, EmbedderControlResponse),
300    /// Set the `UserContents` for the given `UserContentManagerId`. A `ScriptThread` can host many
301    /// `WebView`s which share the same `UserContentManager`. Only documents loaded after
302    /// the processing of this message will observe the new `UserContents` of the specified
303    /// `UserContentManagerId`.
304    SetUserContents(UserContentManagerId, UserContents),
305    /// Release all data for the given `UserContentManagerId` from the `ScriptThread`'s
306    /// `user_contents_for_manager_id` map.
307    DestroyUserContentManager(UserContentManagerId),
308}
