12/// Sender for errors raised by delegate request objects.
13///
14/// This allows errors to be raised asynchronously.
15pub(crate) struct ServoErrorSender {
16    sender: Sender<ServoError>,
17}
18
19impl ServoErrorSender {
20    pub(crate) fn raise_response_send_error(&self, error: SendError) {
21        if let Err(error) = self.sender.send(ServoError::ResponseFailedToSend(error)) {
22            warn!("Failed to send Servo error: {error:?}");
23        }
24    }
25}

27/// Channel for errors raised by [`WebViewDelegate`] request objects.
28///
29/// This allows errors to be raised asynchronously.
30pub(crate) struct ServoErrorChannel {
31    sender: Sender<ServoError>,
32    receiver: Receiver<ServoError>,
33}
34
35impl Default for ServoErrorChannel {
36    fn default() -> Self {
37        let (sender, receiver) = unbounded();
38        Self { sender, receiver }
39    }
40}
41
42impl ServoErrorChannel {
43    pub(crate) fn sender(&self) -> ServoErrorSender {
44        ServoErrorSender {
45            sender: self.sender.clone(),
46        }
47    }
48
49    pub(crate) fn try_recv(&self) -> Option<ServoError> {
50        match self.receiver.try_recv() {
51            Ok(result) => Some(result),
52            Err(error) => {
53                debug_assert_eq!(error, TryRecvError::Empty);
54                None
55            },
56        }
57    }
58}

60/// Sends a response over an IPC channel, or a default response on [`Drop`] if no response was sent.
61pub(crate) struct IpcResponder<T: Serialize> {
62    response_sender: GenericSender<T>,
63    response_sent: bool,
64    /// Always present, except when taken by [`Drop`].
65    default_response: Option<T>,
66}
67
68impl<T: Serialize> IpcResponder<T> {
69    pub(crate) fn new(response_sender: GenericSender<T>, default_response: T) -> Self {
70        Self {
71            response_sender,
72            response_sent: false,
73            default_response: Some(default_response),
74        }
75    }
76
77    pub(crate) fn send(&mut self, response: T) -> SendResult {
78        let result = self.response_sender.send(response);
79        self.response_sent = true;
80        result
81    }
82
83    pub(crate) fn into_inner(self) -> GenericSender<T> {
84        self.response_sender.clone()
85    }
86}
87
88impl<T: Serialize> Drop for IpcResponder<T> {
89    fn drop(&mut self) {
90        if !self.response_sent {
91            let response = self
92                .default_response
93                .take()
94                .expect("Guaranteed by inherent impl");
95            // Don’t notify embedder about send errors for the default response,
96            // since they didn’t send anything and probably don’t care.
97            let _ = self.response_sender.send(response);
98        }
99    }
100}
