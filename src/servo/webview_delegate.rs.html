30/// A request to navigate a [`WebView`] or one of its inner frames. This can be handled
31/// asynchronously. If not handled, the request will automatically be allowed.
32pub struct NavigationRequest {
33    pub url: Url,
34    pub(crate) pipeline_id: PipelineId,
35    pub(crate) constellation_proxy: ConstellationProxy,
36    pub(crate) response_sent: bool,
37}
38
39impl NavigationRequest {
40    pub fn allow(mut self) {
41        self.constellation_proxy
42            .send(EmbedderToConstellationMessage::AllowNavigationResponse(
43                self.pipeline_id,
44                true,
45            ));
46        self.response_sent = true;
47    }
48
49    pub fn deny(mut self) {
50        self.constellation_proxy
51            .send(EmbedderToConstellationMessage::AllowNavigationResponse(
52                self.pipeline_id,
53                false,
54            ));
55        self.response_sent = true;
56    }
57}
58
59impl Drop for NavigationRequest {
60    fn drop(&mut self) {
61        if !self.response_sent {
62            self.constellation_proxy
63                .send(EmbedderToConstellationMessage::AllowNavigationResponse(
64                    self.pipeline_id,
65                    true,
66                ));
67        }
68    }
69}

127/// A request to authenticate a [`WebView`] navigation. Embedders may choose to prompt
128/// the user to enter credentials or simply ignore this request (in which case credentials
129/// will not be used).
130pub struct AuthenticationRequest {
131    pub(crate) url: Url,
132    pub(crate) for_proxy: bool,
133    pub(crate) responder: IpcResponder<Option<AuthenticationResponse>>,
134    pub(crate) error_sender: ServoErrorSender,
135}
136
137impl AuthenticationRequest {
138    pub(crate) fn new(
139        url: Url,
140        for_proxy: bool,
141        response_sender: GenericSender<Option<AuthenticationResponse>>,
142        error_sender: ServoErrorSender,
143    ) -> Self {
144        Self {
145            url,
146            for_proxy,
147            responder: IpcResponder::new(response_sender, None),
148            error_sender,
149        }
150    }
151
152    /// The URL of the request that triggered this authentication.
153    pub fn url(&self) -> &Url {
154        &self.url
155    }
156    /// Whether or not this authentication request is associated with a proxy server authentication.
157    pub fn for_proxy(&self) -> bool {
158        self.for_proxy
159    }
160    /// Respond to the [`AuthenticationRequest`] with the given username and password.
161    pub fn authenticate(mut self, username: String, password: String) {
162        if let Err(error) = self
163            .responder
164            .send(Some(AuthenticationResponse { username, password }))
165        {
166            self.error_sender.raise_response_send_error(error);
167        }
168    }
169}

171/// Information related to the loading of a web resource. These are created for all HTTP requests.
172/// The client may choose to intercept the load of web resources and send an alternate response
173/// by calling [`WebResourceLoad::intercept`].
174pub struct WebResourceLoad {
175    pub request: WebResourceRequest,
176    pub(crate) responder: IpcResponder<WebResourceResponseMsg>,
177    pub(crate) error_sender: ServoErrorSender,
178}
179
180impl WebResourceLoad {
181    pub(crate) fn new(
182        web_resource_request: WebResourceRequest,
183        response_sender: GenericSender<WebResourceResponseMsg>,
184        error_sender: ServoErrorSender,
185    ) -> Self {
186        Self {
187            request: web_resource_request,
188            responder: IpcResponder::new(response_sender, WebResourceResponseMsg::DoNotIntercept),
189            error_sender,
190        }
191    }
192
193    /// The [`WebResourceRequest`] associated with this [`WebResourceLoad`].
194    pub fn request(&self) -> &WebResourceRequest {
195        &self.request
196    }
197    /// Intercept this [`WebResourceLoad`] and control the response via the returned
198    /// [`InterceptedWebResourceLoad`].
199    pub fn intercept(mut self, response: WebResourceResponse) -> InterceptedWebResourceLoad {
200        if let Err(error) = self.responder.send(WebResourceResponseMsg::Start(response)) {
201            self.error_sender.raise_response_send_error(error);
202        }
203        InterceptedWebResourceLoad {
204            request: self.request.clone(),
205            response_sender: self.responder.into_inner(),
206            finished: false,
207            error_sender: self.error_sender,
208        }
209    }
210}

212/// An intercepted web resource load. This struct allows the client to send an alternative response
213/// after calling [`WebResourceLoad::intercept`]. In order to send chunks of body data, the client
214/// must call [`InterceptedWebResourceLoad::send_body_data`]. When the interception is complete, the client
215/// should call [`InterceptedWebResourceLoad::finish`]. If neither `finish()` or `cancel()` are called,
216/// this interception will automatically be finished when dropped.
217pub struct InterceptedWebResourceLoad {
218    pub request: WebResourceRequest,
219    pub(crate) response_sender: GenericSender<WebResourceResponseMsg>,
220    pub(crate) finished: bool,
221    pub(crate) error_sender: ServoErrorSender,
222}
223
224impl InterceptedWebResourceLoad {
225    /// Send a chunk of response body data. It's possible to make subsequent calls to
226    /// this method when streaming body data.
227    pub fn send_body_data(&self, data: Vec<u8>) {
228        if let Err(error) = self
229            .response_sender
230            .send(WebResourceResponseMsg::SendBodyData(data))
231        {
232            self.error_sender.raise_response_send_error(error);
233        }
234    }
235    /// Finish this [`InterceptedWebResourceLoad`] and complete the response.
236    pub fn finish(mut self) {
237        if let Err(error) = self
238            .response_sender
239            .send(WebResourceResponseMsg::FinishLoad)
240        {
241            self.error_sender.raise_response_send_error(error);
242        }
243        self.finished = true;
244    }
245    /// Cancel this [`InterceptedWebResourceLoad`], which will trigger a network error.
246    pub fn cancel(mut self) {
247        if let Err(error) = self
248            .response_sender
249            .send(WebResourceResponseMsg::CancelLoad)
250        {
251            self.error_sender.raise_response_send_error(error);
252        }
253        self.finished = true;
254    }
255}
256
257impl Drop for InterceptedWebResourceLoad {
258    fn drop(&mut self) {
259        if !self.finished {
260            if let Err(error) = self
261                .response_sender
262                .send(WebResourceResponseMsg::FinishLoad)
263            {
264                self.error_sender.raise_response_send_error(error);
265            }
266        }
267    }
268}

270/// The controls of an interactive form element.
271pub enum EmbedderControl {
272    /// The picker of a `<select>` element.
273    SelectElement(SelectElement),
274    /// The picker of a `<input type=color>` element.
275    ColorPicker(ColorPicker),
276    /// The picker of a `<input type=file>` element.
277    FilePicker(FilePicker),
278    /// Request to present an input method (IME) interface to the user when an
279    /// editable element is focused.
280    InputMethod(InputMethodControl),
281    /// A [simple dialog](https://html.spec.whatwg.org/multipage/#simple-dialogs) initiated by
282    /// script (`alert()`, `confirm()`, or `prompt()`). Since their messages are controlled by web
283    /// content, they should be presented to the user in a way that makes them impossible to
284    /// mistake for browser UI.
285    SimpleDialog(SimpleDialog),
286    /// A context menu. This can be triggered by things like right-clicking on web content.
287    /// The menu that is actually shown the user may be customized, but custom menu entries
288    /// must be handled by the embedder.
289    ContextMenu(ContextMenu),
290}
291
292impl EmbedderControl {
293    pub fn id(&self) -> EmbedderControlId {
294        match self {
295            EmbedderControl::SelectElement(select_element) => select_element.id,
296            EmbedderControl::ColorPicker(color_picker) => color_picker.id,
297            EmbedderControl::FilePicker(file_picker) => file_picker.id,
298            EmbedderControl::InputMethod(input_method) => input_method.id,
299            EmbedderControl::SimpleDialog(simple_dialog) => simple_dialog.id(),
300            EmbedderControl::ContextMenu(context_menu) => context_menu.id,
301        }
302    }
303}

587/// [Simple dialogs](https://html.spec.whatwg.org/multipage/#simple-dialogs) are synchronous dialogs
588/// that can be opened by web content. Since their messages are controlled by web content, they
589/// should be presented to the user in a way that makes them impossible to mistake for browser UI.
590pub enum SimpleDialog {
591    Alert(AlertDialog),
592    Confirm(ConfirmDialog),
593    Prompt(PromptDialog),
594}
595
596impl SimpleDialog {
597    pub fn message(&self) -> &str {
598        match self {
599            SimpleDialog::Alert(alert_dialog) => alert_dialog.message(),
600            SimpleDialog::Confirm(confirm_dialog) => confirm_dialog.message(),
601            SimpleDialog::Prompt(prompt_dialog) => prompt_dialog.message(),
602        }
603    }
604
605    pub fn confirm(self) {
606        match self {
607            SimpleDialog::Alert(alert_dialog) => alert_dialog.confirm(),
608            SimpleDialog::Confirm(confirm_dialog) => confirm_dialog.confirm(),
609            SimpleDialog::Prompt(prompt_dialog) => prompt_dialog.confirm(),
610        }
611    }
612
613    pub fn dismiss(self) {
614        match self {
615            SimpleDialog::Alert(alert_dialog) => alert_dialog.confirm(),
616            SimpleDialog::Confirm(confirm_dialog) => confirm_dialog.dismiss(),
617            SimpleDialog::Prompt(prompt_dialog) => prompt_dialog.dismiss(),
618        }
619    }
620}
621
622impl SimpleDialog {
623    fn id(&self) -> EmbedderControlId {
624        match self {
625            SimpleDialog::Alert(alert_dialog) => alert_dialog.id,
626            SimpleDialog::Confirm(confirm_dialog) => confirm_dialog.id,
627            SimpleDialog::Prompt(prompt_dialog) => prompt_dialog.id,
628        }
629    }
630}
631
632impl From<SimpleDialogRequest> for SimpleDialog {
633    fn from(simple_dialog_request: SimpleDialogRequest) -> Self {
634        match simple_dialog_request {
635            SimpleDialogRequest::Alert {
636                id,
637                message,
638                response_sender,
639            } => Self::Alert(AlertDialog {
640                id,
641                message,
642                response_sender,
643                response_sent: false,
644            }),
645            SimpleDialogRequest::Confirm {
646                id,
647                message,
648                response_sender,
649            } => Self::Confirm(ConfirmDialog {
650                id,
651                message,
652                response_sender,
653                response_sent: false,
654            }),
655            SimpleDialogRequest::Prompt {
656                id,
657                message,
658                default,
659                response_sender,
660            } => Self::Prompt(PromptDialog {
661                id,
662                message,
663                current_value: default,
664                response_sender,
665                response_sent: false,
666            }),
667        }
668    }
669}

753}

759        }
760    }

808        None
809    }

811    /// URL can accessed via [`WebView::url`].
812    fn notify_url_changed(&self, _webview: WebView, _url: Url) {}

814    /// title can accessed via [`WebView::page_title`].

817    /// status text can accessed via [`WebView::status_text`].

819    /// This [`WebView`] has either become focused or lost focus. Whether or not the
820    /// [`WebView`] is focused can be accessed via [`WebView::focused`].

822    /// This [`WebView`] has either started to animate or stopped animating. When a
823    /// [`WebView`] is animating, it is up to the embedding application ensure that

830    /// The [`Cursor`] of the currently loaded page in this [`WebView`] has changed. The new
831    /// cursor can accessed via [`WebView::cursor`].
832    fn notify_cursor_changed(&self, _webview: WebView, _: Cursor) {}
833    /// The favicon of the currently loaded page in this [`WebView`] has changed. The new
834    /// favicon [`Image`] can accessed via [`WebView::favicon`].
835    fn notify_favicon_changed(&self, _webview: WebView) {}

837    fn notify_new_frame_ready(&self, _webview: WebView) {}
838    /// The navigation history of this [`WebView`] has changed. The navigation history is represented

840    /// back navigation, and forward navigation modify this index.
841    fn notify_history_changed(&self, _webview: WebView, _entries: Vec<Url>, _current: usize) {}
842    /// A history traversal operation is complete.
843    fn notify_traversal_complete(&self, _webview: WebView, _: TraversalId) {}
844    /// Page content has closed this [`WebView`] via `window.close()`. It's the embedder's
845    /// responsibility to remove the [`WebView`] from the interface when this notification
846    /// occurs.
847    fn notify_closed(&self, _webview: WebView) {}
848
849    /// An input event passed to this [`WebView`] via [`WebView::notify_input_event`] has been handled
850    /// by Servo. This allows post-procesing of input events, such as chaining up unhandled events
851    /// to parent UI elements.

853    /// A pipeline in the webview panicked. First string is the reason, second one is the backtrace.

855    /// Notifies the embedder about media session events
856    /// (i.e. when there is metadata for the active media session, playback state changes...).
857    fn notify_media_session_event(&self, _webview: WebView, _event: MediaSessionEvent) {}
858    /// A notification that the [`WebView`] has entered or exited fullscreen mode. This is an
859    /// opportunity for the embedder to transition the containing window into or out of fullscreen
860    /// mode and to show or hide extra UI elements. Regardless of how the notification is handled,
861    /// the page will enter or leave fullscreen state internally according to the [Fullscreen
862    /// API](https://fullscreen.spec.whatwg.org/).
863    fn notify_fullscreen_state_changed(&self, _webview: WebView, _: bool) {}
864
865    /// Whether or not to allow a [`WebView`] to load a URL in its main frame or one of its
866    /// nested `<iframe>`s. [`NavigationRequest`]s are accepted by default.
867    fn request_navigation(&self, _webview: WebView, _navigation_request: NavigationRequest) {}
868    /// Whether or not to allow a [`WebView`]  to unload a `Document` in its main frame or one
869    /// of its nested `<iframe>`s. By default, unloads are allowed.
870    fn request_unload(&self, _webview: WebView, _unload_request: AllowOrDenyRequest) {}
871    /// Move the window to a point.
872    fn request_move_to(&self, _webview: WebView, _: DeviceIntPoint) {}
873    /// Whether or not to allow a [`WebView`] to (un)register a protocol handler (e.g. `mailto:`).
874    /// Typically an embedder application will show a permissions prompt when this happens
875    /// to confirm a protocol handler is allowed. By default, requests are denied.
876    /// For more information, see the specification:
877    /// <https://html.spec.whatwg.org/multipage/#custom-handlers>
878    fn request_protocol_handler(
879        &self,
880        _webview: WebView,
881        _protocol_handler_registration: ProtocolHandlerRegistration,
882        _allow_deny_request: AllowOrDenyRequest,
883    ) {
884    }
885    /// Try to resize the window that contains this [`WebView`] to the provided outer
886    /// size. These resize requests can come from page content. Servo will ensure that the
887    /// values are greater than zero, but it is up to the embedder to limit the maximum
888    /// size. For instance, a reasonable limitation might be that the final size is no
889    /// larger than the screen size.
890    fn request_resize_to(&self, _webview: WebView, _requested_outer_size: DeviceIntSize) {}
891    /// This method is called when web content makes a request to open a new
892    /// `WebView`, such as via the [`window.open`] DOM API. If this request is
893    /// ignored, no new `WebView` will be opened. Embedders can handle this method by
894    /// using the provided [`CreateNewWebViewRequest`] to build a new `WebView`.
895    ///
896    /// ```rust
897    /// fn request_create_new(&self, parent_webview: WebView, request: CreateNewWebViewRequest) {
898    ///     let webview = request
899    ///         .builder(self.rendering_context())
900    ///         .delegate(parent_webview.delegate())

904    /// ```
905    ///

909    /// [`window.open`]: https://developer.mozilla.org/en-US/docs/Web/API/Window/open
910    fn request_create_new(&self, _parent_webview: WebView, _: CreateNewWebViewRequest) {}
911    /// Content in a [`WebView`] is requesting permission to access a feature requiring
912    /// permission from the user. The embedder should allow or deny the request, either by
913    /// reading a cached value or querying the user for permission via the user interface.
914    fn request_permission(&self, _webview: WebView, _: PermissionRequest) {}
