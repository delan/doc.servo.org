26impl GStreamerAudioSink {
27    pub fn new() -> Result<Self, AudioSinkError> {
28        if let Some(category) = gst::DebugCategory::get("openslessink") {
29            category.set_threshold(gst::DebugLevel::Trace);
30        }
31        gst::init().map_err(|error| {
32            AudioSinkError::Backend(format!("GStreamer init failed: {error:?}"))
33        })?;
34
35        let appsrc = gst::ElementFactory::make("appsrc")
36            .build()
37            .map_err(|error| {
38                AudioSinkError::Backend(format!("appsrc creation failed: {error:?}"))
39            })?;
40        let appsrc = appsrc.downcast::<AppSrc>().unwrap();
41
42        Ok(Self {
43            pipeline: gst::Pipeline::new(),
44            appsrc: Arc::new(appsrc),
45            sample_rate: Cell::new(DEFAULT_SAMPLE_RATE),
46            audio_info: RefCell::new(None),
47            sample_offset: Cell::new(0),
48        })
49    }
50}

52impl GStreamerAudioSink {
53    fn set_audio_info(&self, sample_rate: f32, channels: u8) -> Result<(), AudioSinkError> {
54        let audio_info = gst_audio::AudioInfo::builder(
55            gst_audio::AUDIO_FORMAT_F32,
56            sample_rate as u32,
57            channels.into(),
58        )
59        .build()
60        .map_err(|error| AudioSinkError::Backend(format!("AudioInfo failed: {error:?}")))?;
61        self.appsrc.set_caps(audio_info.to_caps().ok().as_ref());
62        *self.audio_info.borrow_mut() = Some(audio_info);
63        Ok(())
64    }
65
66    fn set_channels_if_changed(&self, channels: u8) -> Result<(), AudioSinkError> {
67        let curr_channels = match self.audio_info.borrow().as_ref() {
68            Some(ch) => ch.channels(),
69            _ => {
70                return Ok(());
71            },
72        };
73        if channels != curr_channels as u8 {
74            self.set_audio_info(self.sample_rate.get(), channels)?;
75        }
76        Ok(())
77    }
78}

80impl AudioSink for GStreamerAudioSink {
81    fn init(
82        &self,
83        sample_rate: f32,
84        graph_thread_channel: Sender<AudioRenderThreadMsg>,
85    ) -> Result<(), AudioSinkError> {
86        self.sample_rate.set(sample_rate);
87        self.set_audio_info(sample_rate, 2)?;
88        self.appsrc.set_format(gst::Format::Time);
89
90        // Allow only a single chunk.
91        self.appsrc.set_max_bytes(1);
92
93        let appsrc = self.appsrc.clone();
94        Builder::new()
95            .name("GstAppSrcCallbacks".to_owned())
96            .spawn(move || {
97                let need_data = move |_: &AppSrc, _: u32| {
98                    if let Err(e) = graph_thread_channel.send(AudioRenderThreadMsg::SinkNeedData) {
99                        log::warn!("Error sending need data event: {:?}", e);
100                    }
101                };
102                appsrc.set_callbacks(AppSrcCallbacks::builder().need_data(need_data).build());
103            })
104            .unwrap();
105
106        let appsrc = self.appsrc.as_ref().clone().upcast();
107        let resample = gst::ElementFactory::make("audioresample")
108            .build()
109            .map_err(|error| {
110                AudioSinkError::Backend(format!("audioresample creation failed: {error:?}"))
111            })?;
112        let convert = gst::ElementFactory::make("audioconvert")
113            .build()
114            .map_err(|error| {
115                AudioSinkError::Backend(format!("audioconvert creation failed: {error:?}"))
116            })?;
117        let sink = gst::ElementFactory::make("autoaudiosink")
118            .build()
119            .map_err(|error| {
120                AudioSinkError::Backend(format!("autoaudiosink creation failed: {error:?}"))
121            })?;
122        self.pipeline
123            .add_many(&[&appsrc, &resample, &convert, &sink])
124            .map_err(|error| AudioSinkError::Backend(error.to_string()))?;
125        gst::Element::link_many(&[&appsrc, &resample, &convert, &sink])
126            .map_err(|error| AudioSinkError::Backend(error.to_string()))?;
127
128        Ok(())
129    }
130
131    fn init_stream(
132        &self,
133        channels: u8,
134        sample_rate: f32,
135        socket: Box<dyn MediaSocket>,
136    ) -> Result<(), AudioSinkError> {
137        self.sample_rate.set(sample_rate);
138        self.set_audio_info(sample_rate, channels)?;
139        self.appsrc.set_format(gst::Format::Time);
140
141        // Do not set max bytes or callback, we will push as needed
142
143        let appsrc = self.appsrc.as_ref().clone().upcast();
144        let convert = gst::ElementFactory::make("audioconvert")
145            .build()
146            .map_err(|error| {
147                AudioSinkError::Backend(format!("audioconvert creation failed: {error:?}"))
148            })?;
149        let sink = socket
150            .as_any()
151            .downcast_ref::<GstreamerMediaSocket>()
152            .unwrap()
153            .proxy_sink()
154            .clone();
155
156        self.pipeline
157            .add_many(&[&appsrc, &convert, &sink])
158            .map_err(|error| AudioSinkError::Backend(error.to_string()))?;
159        gst::Element::link_many(&[&appsrc, &convert, &sink])
160            .map_err(|error| AudioSinkError::Backend(error.to_string()))?;
161
162        Ok(())
163    }
164
165    fn play(&self) -> Result<(), AudioSinkError> {
166        self.pipeline
167            .set_state(gst::State::Playing)
168            .map(|_| ())
169            .map_err(|_| AudioSinkError::StateChangeFailed)
170    }

172    fn stop(&self) -> Result<(), AudioSinkError> {
173        self.pipeline
174            .set_state(gst::State::Paused)
175            .map(|_| ())
176            .map_err(|_| AudioSinkError::StateChangeFailed)
177    }

179    fn has_enough_data(&self) -> bool {
180        self.appsrc.current_level_bytes() >= self.appsrc.max_bytes()
181    }
182
183    fn push_data(&self, mut chunk: Chunk) -> Result<(), AudioSinkError> {
184        if let Some(block) = chunk.blocks.get(0) {
185            self.set_channels_if_changed(block.chan_count())?;
186        }
187
188        let sample_rate = self.sample_rate.get() as u64;
189        let audio_info = self.audio_info.borrow();
190        let audio_info = audio_info.as_ref().unwrap();
191        let channels = audio_info.channels();
192        let bpf = audio_info.bpf() as usize;
193        assert_eq!(bpf, 4 * channels as usize);
194        let n_samples = FRAMES_PER_BLOCK.0;
195        let buf_size = (n_samples as usize) * (bpf);
196        let mut buffer = gst::Buffer::with_size(buf_size).unwrap();
197        {
198            let buffer = buffer.get_mut().unwrap();
199            let mut sample_offset = self.sample_offset.get();
200            // Calculate the current timestamp (PTS) and the next one,
201            // and calculate the duration from the difference instead of
202            // simply the number of samples to prevent rounding errors
203            let pts = gst::ClockTime::from_nseconds(
204                sample_offset
205                    .mul_div_floor(gst::ClockTime::SECOND.nseconds(), sample_rate)
206                    .unwrap(),
207            );
208            let next_pts: gst::ClockTime = gst::ClockTime::from_nseconds(
209                (sample_offset + n_samples)
210                    .mul_div_floor(gst::ClockTime::SECOND.nseconds(), sample_rate)
211                    .unwrap(),
212            );
213            buffer.set_pts(Some(pts));
214            buffer.set_duration(next_pts - pts);
215
216            // sometimes nothing reaches the output
217            if chunk.len() == 0 {
218                chunk.blocks.push(Default::default());
219                chunk.blocks[0].repeat(channels as u8);
220            }
221            debug_assert!(chunk.len() == 1);
222            let mut data = chunk.blocks[0].interleave();
223            let data = data.as_mut_byte_slice();
224
225            // XXXManishearth if we have a safe way to convert
226            // from Box<[f32]> to Box<[u8]> (similarly for Vec)
227            // we can use Buffer::from_slice instead
228            buffer.copy_from_slice(0, data).expect("copying failed");
229
230            sample_offset += n_samples;
231            self.sample_offset.set(sample_offset);
232        }
233
234        self.appsrc
235            .push_buffer(buffer)
236            .map(|_| ())
237            .map_err(|_| AudioSinkError::BufferPushFailed)
238    }
239
240    fn set_eos_callback(&self, _: Box<dyn Fn(Box<dyn AsRef<[f32]>>) + Send + Sync + 'static>) {}
241}
