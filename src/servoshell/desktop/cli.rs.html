5use std::{env, panic};
6
7use crate::desktop::app::App;
8use crate::desktop::event_loop::ServoShellEventLoop;
9use crate::panic_hook;
10use crate::prefs::{ArgumentParsingResult, parse_command_line_arguments};
11
12pub fn main() {
13    crate::crash_handler::install();
14    crate::init_crypto();
15    crate::resources::init();
16
17    // TODO: once log-panics is released, can this be replaced by
18    // log_panics::init()?
19    panic::set_hook(Box::new(panic_hook::panic_hook));
20
21    // Skip the first argument, which is the binary name.
22    let args: Vec<String> = env::args().skip(1).collect();
23    let (opts, preferences, servoshell_preferences) = match parse_command_line_arguments(&*args) {
24        ArgumentParsingResult::ContentProcess(token) => return servo::run_content_process(token),
25        ArgumentParsingResult::ChromeProcess(opts, preferences, servoshell_preferences) => {
26            (opts, preferences, servoshell_preferences)
27        },
28        ArgumentParsingResult::Exit => {
29            std::process::exit(0);
30        },
31        ArgumentParsingResult::ErrorParsing => {
32            std::process::exit(1);
33        },
34    };
35
36    crate::init_tracing(servoshell_preferences.tracing_filter.as_deref());
37
38    let clean_shutdown = servoshell_preferences.clean_shutdown;
39    let event_loop = match servoshell_preferences.headless {
40        true => ServoShellEventLoop::headless(),
41        false => ServoShellEventLoop::headed(),
42    };
43
44    {
45        let mut app = App::new(opts, preferences, servoshell_preferences, &event_loop);
46        event_loop.run_app(&mut app);
47    }
48
49    crate::platform::deinit(clean_shutdown)
50}
