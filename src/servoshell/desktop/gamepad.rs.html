15pub struct HapticEffect {
16    pub effect: Effect,
17    pub callback: Option<Box<dyn FnOnce(bool)>>,
18}

20pub(crate) struct GamepadSupport {
21    handle: Gilrs,
22    haptic_effects: HashMap<usize, HapticEffect>,
23}

25impl GamepadSupport {
26    pub(crate) fn maybe_new() -> Option<Self> {
27        let handle = match Gilrs::new() {
28            Ok(handle) => handle,
29            Err(error) => {
30                warn!("Error creating gamepad input connection ({error})");
31                return None;
32            },
33        };
34        Some(Self {
35            handle,
36            haptic_effects: Default::default(),
37        })
38    }

40    /// Handle updates to connected gamepads from GilRs
41    pub(crate) fn handle_gamepad_events(&mut self, active_webview: WebView) {
42        while let Some(event) = self.handle.next_event() {
43            let gamepad = self.handle.gamepad(event.id);
44            let name = gamepad.name();
45            let index = GamepadIndex(event.id.into());
46            let mut gamepad_event: Option<GamepadEvent> = None;
47            match event.event {
48                EventType::ButtonPressed(button, _) => {
49                    let mapped_index = Self::map_gamepad_button(button);
50                    // We only want to send this for a valid digital button, aka on/off only
51                    if !matches!(mapped_index, 6 | 7 | 17) {
52                        let update_type = GamepadUpdateType::Button(mapped_index, 1.0);
53                        gamepad_event = Some(GamepadEvent::Updated(index, update_type));
54                    }
55                },
56                EventType::ButtonReleased(button, _) => {
57                    let mapped_index = Self::map_gamepad_button(button);
58                    // We only want to send this for a valid digital button, aka on/off only
59                    if !matches!(mapped_index, 6 | 7 | 17) {
60                        let update_type = GamepadUpdateType::Button(mapped_index, 0.0);
61                        gamepad_event = Some(GamepadEvent::Updated(index, update_type));
62                    }
63                },
64                EventType::ButtonChanged(button, value, _) => {
65                    let mapped_index = Self::map_gamepad_button(button);
66                    // We only want to send this for a valid non-digital button, aka the triggers
67                    if matches!(mapped_index, 6 | 7) {
68                        let update_type = GamepadUpdateType::Button(mapped_index, value as f64);
69                        gamepad_event = Some(GamepadEvent::Updated(index, update_type));
70                    }
71                },
72                EventType::AxisChanged(axis, value, _) => {
73                    // Map axis index and value to represent Standard Gamepad axis
74                    // <https://www.w3.org/TR/gamepad/#dfn-represents-a-standard-gamepad-axis>
75                    let mapped_axis: usize = match axis {
76                        gilrs::Axis::LeftStickX => 0,
77                        gilrs::Axis::LeftStickY => 1,
78                        gilrs::Axis::RightStickX => 2,
79                        gilrs::Axis::RightStickY => 3,
80                        _ => 4, // Other axes do not map to "standard" gamepad mapping and are ignored
81                    };
82                    if mapped_axis < 4 {
83                        // The Gamepad spec designates down as positive and up as negative.
84                        // GilRs does the inverse of this, so correct for it here.
85                        let axis_value = match mapped_axis {
86                            0 | 2 => value,
87                            1 | 3 => -value,
88                            _ => 0., // Should not reach here
89                        };
90                        let update_type = GamepadUpdateType::Axis(mapped_axis, axis_value as f64);
91                        gamepad_event = Some(GamepadEvent::Updated(index, update_type));
92                    }
93                },
94                EventType::Connected => {
95                    let name = String::from(name);
96                    let bounds = GamepadInputBounds {
97                        axis_bounds: (-1.0, 1.0),
98                        button_bounds: (0.0, 1.0),
99                    };
100                    // GilRs does not yet support trigger rumble
101                    let supported_haptic_effects = GamepadSupportedHapticEffects {
102                        supports_dual_rumble: true,
103                        supports_trigger_rumble: false,
104                    };
105                    gamepad_event = Some(GamepadEvent::Connected(
106                        index,
107                        name,
108                        bounds,
109                        supported_haptic_effects,
110                    ));
111                },
112                EventType::Disconnected => {
113                    gamepad_event = Some(GamepadEvent::Disconnected(index));
114                },
115                EventType::ForceFeedbackEffectCompleted => {
116                    let Some(effect) = self.haptic_effects.get_mut(&event.id.into()) else {
117                        warn!("Failed to find haptic effect for id {}", event.id);
118                        return;
119                    };
120                    if let Some(callback) = effect.callback.take() {
121                        callback(true);
122                    }
123                    self.haptic_effects.remove(&event.id.into());
124                },
125                _ => {},
126            }
127
128            if let Some(event) = gamepad_event {
129                active_webview.notify_input_event(InputEvent::Gamepad(event));
130            }
131        }
132    }

134    // Map button index and value to represent Standard Gamepad button
135    // <https://www.w3.org/TR/gamepad/#dfn-represents-a-standard-gamepad-button>
136    fn map_gamepad_button(button: gilrs::Button) -> usize {
137        match button {
138            gilrs::Button::South => 0,
139            gilrs::Button::East => 1,
140            gilrs::Button::West => 2,
141            gilrs::Button::North => 3,
142            gilrs::Button::LeftTrigger => 4,
143            gilrs::Button::RightTrigger => 5,
144            gilrs::Button::LeftTrigger2 => 6,
145            gilrs::Button::RightTrigger2 => 7,
146            gilrs::Button::Select => 8,
147            gilrs::Button::Start => 9,
148            gilrs::Button::LeftThumb => 10,
149            gilrs::Button::RightThumb => 11,
150            gilrs::Button::DPadUp => 12,
151            gilrs::Button::DPadDown => 13,
152            gilrs::Button::DPadLeft => 14,
153            gilrs::Button::DPadRight => 15,
154            gilrs::Button::Mode => 16,
155            _ => 17, // Other buttons do not map to "standard" gamepad mapping and are ignored
156        }
157    }

159    pub(crate) fn play_haptic_effect(
160        &mut self,
161        index: usize,
162        effect_type: GamepadHapticEffectType,
163        callback: Box<dyn FnOnce(bool)>,
164    ) {
165        let GamepadHapticEffectType::DualRumble(params) = effect_type;
166        if let Some(connected_gamepad) = self
167            .handle
168            .gamepads()
169            .find(|gamepad| usize::from(gamepad.0) == index)
170        {
171            let start_delay = Ticks::from_ms(params.start_delay as u32);
172            let duration = Ticks::from_ms(params.duration as u32);
173            let strong_magnitude = (params.strong_magnitude * u16::MAX as f64).round() as u16;
174            let weak_magnitude = (params.weak_magnitude * u16::MAX as f64).round() as u16;
175
176            let scheduling = Replay {
177                after: start_delay,
178                play_for: duration,
179                with_delay: Ticks::from_ms(0),
180            };
181            let effect = EffectBuilder::new()
182                .add_effect(BaseEffect {
183                    kind: BaseEffectType::Strong { magnitude: strong_magnitude },
184                    scheduling,
185                    envelope: Default::default(),
186                })
187                .add_effect(BaseEffect {
188                    kind: BaseEffectType::Weak { magnitude: weak_magnitude },
189                    scheduling,
190                    envelope: Default::default(),
191                })
192                .repeat(Repeat::For(start_delay + duration))
193                .add_gamepad(&connected_gamepad.1)
194                .finish(&mut self.handle)
195                .expect("Failed to create haptic effect, ensure connected gamepad supports force feedback.");
196            self.haptic_effects.insert(
197                index,
198                HapticEffect {
199                    effect,
200                    callback: Some(callback),
201                },
202            );
203            self.haptic_effects[&index]
204                .effect
205                .play()
206                .expect("Failed to play haptic effect.");
207        } else {
208            debug!("Couldn't find connected gamepad to play haptic effect on");
209        }
210    }

212    pub(crate) fn stop_haptic_effect(&mut self, index: usize) -> bool {
213        let Some(haptic_effect) = self.haptic_effects.get(&index) else {
214            return false;
215        };
216
217        let stopped_successfully = match haptic_effect.effect.stop() {
218            Ok(()) => true,
219            Err(e) => {
220                debug!("Failed to stop haptic effect: {:?}", e);
221                false
222            },
223        };
224        self.haptic_effects.remove(&index);
225
226        stopped_successfully
227    }
