5//! A headless window implementation.
6
7#![deny(clippy::panic)]
8#![deny(clippy::unwrap_used)]
9
10use std::cell::Cell;
11use std::rc::Rc;
12use std::sync::atomic::AtomicU64;
13
14use euclid::{Point2D, Scale, Size2D};
15use log::error;
16use servo::{
17    DeviceIndependentIntRect, DeviceIndependentPixel, DeviceIntPoint, DeviceIntRect, DeviceIntSize,
18    DevicePixel, RenderingContext, ScreenGeometry, SoftwareRenderingContext, WebView,
19    convert_rect_to_css_pixel,
20};
21use winit::dpi::PhysicalSize;
22
23use crate::prefs::ServoShellPreferences;
24use crate::window::{MIN_WINDOW_INNER_SIZE, PlatformWindow, ServoShellWindow, ServoShellWindowId};
25
26pub struct HeadlessWindow {
27    id: ServoShellWindowId,
28    fullscreen: Cell<bool>,
29    device_pixel_ratio_override: Option<Scale<f32, DeviceIndependentPixel, DevicePixel>>,
30    inner_size: Cell<DeviceIntSize>,
31    screen_size: Size2D<i32, DevicePixel>,
32    // virtual top-left position of the window in device pixels.
33    window_position: Cell<Point2D<i32, DevicePixel>>,
34    rendering_context: Rc<SoftwareRenderingContext>,
35}
36
37impl HeadlessWindow {
38    pub fn new(servoshell_preferences: &ServoShellPreferences) -> Rc<Self> {
39        let size = servoshell_preferences.initial_window_size;
40
41        let device_pixel_ratio_override = servoshell_preferences.device_pixel_ratio_override;
42        let device_pixel_ratio_override: Option<Scale<f32, DeviceIndependentPixel, DevicePixel>> =
43            device_pixel_ratio_override.map(Scale::new);
44        let hidpi_factor = device_pixel_ratio_override.unwrap_or_else(Scale::identity);
45
46        let inner_size = (size.to_f32() * hidpi_factor).to_i32();
47        let physical_size = PhysicalSize::new(inner_size.width as u32, inner_size.height as u32);
48        let rendering_context =
49            SoftwareRenderingContext::new(physical_size).expect("Failed to create WR surfman");
50
51        let screen_size = servoshell_preferences
52            .screen_size_override
53            .map_or(inner_size * 2, |screen_size_override| {
54                (screen_size_override.to_f32() * hidpi_factor).to_i32()
55            });
56
57        static CURRENT_WINDOW_ID: AtomicU64 = AtomicU64::new(0);
58        let window = HeadlessWindow {
59            id: CURRENT_WINDOW_ID
60                .fetch_add(1, std::sync::atomic::Ordering::Relaxed)
61                .into(),
62            fullscreen: Cell::new(false),
63            device_pixel_ratio_override,
64            inner_size: Cell::new(inner_size),
65            screen_size,
66            window_position: Cell::new(Point2D::zero()),
67            rendering_context: Rc::new(rendering_context),
68        };
69
70        Rc::new(window)
71    }
72}
73
74impl Drop for HeadlessWindow {
75    fn drop(&mut self) {
76        if let Err(error) = self.rendering_context.make_current() {
77            error!("Failed to make the rendering context current: {error:?}");
78        }
79    }
80}
81
82impl PlatformWindow for HeadlessWindow {
83    fn id(&self) -> ServoShellWindowId {
84        self.id
85    }
86
87    fn screen_geometry(&self) -> servo::ScreenGeometry {
88        ScreenGeometry {
89            size: self.screen_size,
90            available_size: self.screen_size,
91            window_rect: DeviceIntRect::from_origin_and_size(
92                self.window_position.get(),
93                self.inner_size.get(),
94            ),
95        }
96    }
97
98    fn set_position(&self, point: DeviceIntPoint) {
99        self.window_position.set(point);
100    }
101
102    fn request_repaint(&self, window: &ServoShellWindow) {
103        window.repaint_webviews();
104    }
105
106    fn request_resize(&self, webview: &WebView, new_size: DeviceIntSize) -> Option<DeviceIntSize> {
107        // Do not let the window size get smaller than `MIN_WINDOW_INNER_SIZE` or larger
108        // than twice the screen size.
109        let new_size = new_size.clamp(MIN_WINDOW_INNER_SIZE, self.screen_size * 2);
110        if self.inner_size.get() == new_size {
111            return Some(new_size);
112        }
113
114        self.inner_size.set(new_size);
115
116        // Because we are managing the rendering surface ourselves, there will be no other
117        // notification (such as from the display manager) that it has changed size, so we
118        // must notify `Paint` here.
119        webview.resize(PhysicalSize::new(
120            new_size.width as u32,
121            new_size.height as u32,
122        ));
123
124        Some(new_size)
125    }
126
127    fn device_hidpi_scale_factor(&self) -> Scale<f32, DeviceIndependentPixel, DevicePixel> {
128        Scale::new(1.0)
129    }
130
131    fn hidpi_scale_factor(&self) -> Scale<f32, DeviceIndependentPixel, DevicePixel> {
132        self.device_pixel_ratio_override
133            .unwrap_or_else(|| self.device_hidpi_scale_factor())
134    }
135
136    fn set_fullscreen(&self, state: bool) {
137        self.fullscreen.set(state);
138    }
139
140    fn get_fullscreen(&self) -> bool {
141        self.fullscreen.get()
142    }
143
144    #[cfg(feature = "webxr")]
145    fn new_glwindow(
146        &self,
147        _event_loop: &winit::event_loop::ActiveEventLoop,
148    ) -> Rc<dyn servo::webxr::GlWindow> {
149        unimplemented!()
150    }
151
152    fn window_rect(&self) -> DeviceIndependentIntRect {
153        convert_rect_to_css_pixel(
154            DeviceIntRect::from_origin_and_size(self.window_position.get(), self.inner_size.get()),
155            self.hidpi_scale_factor(),
156        )
157    }
158
159    fn rendering_context(&self) -> Rc<dyn RenderingContext> {
160        self.rendering_context.clone()
161    }
162
163    fn maximize(&self, webview: &WebView) {
164        self.window_position.set(Point2D::zero());
165        self.inner_size.set(self.screen_size);
166        // Because we are managing the rendering surface ourselves, there will be no other
167        // notification (such as from the display manager) that it has changed size, so we
168        // must notify the `Paint` here.
169        webview.resize(PhysicalSize::new(
170            self.screen_size.width as u32,
171            self.screen_size.height as u32,
172        ));
173    }
174
175    fn show_console_message(&self, level: servo::ConsoleLogLevel, message: &str) {
176        println!("{message}");
177        log::log!(level.into(), "{message}");
178    }
179}
