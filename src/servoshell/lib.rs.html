58pub fn init_tracing(filter_directives: Option<&str>) {
59    #[cfg(not(feature = "tracing"))]
60    {
61        if filter_directives.is_some() {
62            log::debug!("The tracing feature was not selected - ignoring trace filter directives");
63        }
64    }
65    #[cfg(feature = "tracing")]
66    {
67        use tracing_subscriber::layer::SubscriberExt;
68        let subscriber = tracing_subscriber::registry();
69
70        #[cfg(feature = "tracing-perfetto")]
71        let subscriber = {
72            // Set up a PerfettoLayer for performance tracing.
73            // The servo.pftrace file can be uploaded to https://ui.perfetto.dev for analysis.
74            let file = std::fs::File::create("servo.pftrace").unwrap();
75            let perfetto_layer = tracing_perfetto::PerfettoLayer::new(std::sync::Mutex::new(file))
76                .with_filter_by_marker(|field_name| field_name == "servo_profiling")
77                .with_debug_annotations(true);
78            subscriber.with(perfetto_layer)
79        };
80
81        #[cfg(feature = "tracing-hitrace")]
82        let subscriber = {
83            // Set up a HitraceLayer for performance tracing.
84            subscriber.with(HitraceLayer::default())
85        };
86
87        // Filter events and spans by the directives in SERVO_TRACING, using EnvFilter as a global filter.
88        // <https://docs.rs/tracing-subscriber/0.3.18/tracing_subscriber/layer/index.html#global-filtering>
89        let filter_builder = tracing_subscriber::EnvFilter::builder()
90            .with_default_directive(tracing::level_filters::LevelFilter::OFF.into());
91        let filter = if let Some(filters) = &filter_directives {
92            filter_builder.parse_lossy(filters)
93        } else {
94            filter_builder
95                .with_env_var("SERVO_TRACING")
96                .from_env_lossy()
97        };
98
99        let subscriber = subscriber.with(filter);
100
101        // Same as SubscriberInitExt::init, but avoids initialising the tracing-log compat layer,
102        // since it would break Servo’s FromScriptLogger and FromEmbederLogger.
103        // <https://docs.rs/tracing-subscriber/0.3.18/tracing_subscriber/util/trait.SubscriberInitExt.html#method.init>
104        // <https://docs.rs/tracing/0.1.40/tracing/#consuming-log-records>
105        tracing::subscriber::set_global_default(subscriber)
106            .expect("Failed to set tracing subscriber");
107    }
108}

112/// Plumbs tracing spans into HiTrace, with the following caveats:
113///
114/// - We ignore spans unless they have a `servo_profiling` field.
115/// - We map span entry ([`Layer::on_enter`]) to `OH_HiTrace_StartTrace(metadata.name())`.
116/// - We map span exit ([`Layer::on_exit`]) to `OH_HiTrace_FinishTrace()`.
117///
118/// As a result, within each thread, spans must exit in reverse order of their entry, otherwise the
119/// resultant profiling data will be incorrect (see the section below). This is not necessarily the
120/// case for tracing spans, since there can be multiple [trace trees], so we check that this
121/// invariant is upheld when debug assertions are enabled, logging errors if it is violated.
122///
123/// [trace trees]: https://docs.rs/tracing/0.1.40/tracing/span/index.html#span-relationships
124///
125/// # Uniquely identifying spans
126///
127/// We need to ensure that the start and end points of one span are not mixed up with other spans.
128/// For now, we use the HiTrace [synchronous API], which restricts how spans must behave.
129///
130/// In the HiTrace [synchronous API], spans must have stack-like behaviour, because spans are keyed
131/// entirely on their *name* string, and OH_HiTrace_FinishTrace always ends the most recent span.
132/// While synchronous API spans are thread-local, callers could still violate this invariant with
133/// reentrant or asynchronous code.
134///
135/// In the [asynchronous API], spans are keyed on a (*name*,*taskId*) pair, where *name* is again
136/// a string, and *taskId* is an arbitrary [`i32`]. This makes *taskId* a good place for a unique
137/// identifier, but asynchronous spans can cross thread boundaries, so the identifier needs to be
138/// temporally unique in the whole process.
139///
140/// Tracing spans have such an identifier ([`Id`]), but they’re [`u64`]-based, and their format
141/// is an internal implementation detail of the [`Subscriber`]. For [`Registry`], those values
142/// [come from] a [packed representation] of a generation number, thread number, page number, and
143/// variable-length index. This makes them hard to compress robustly into an [`i32`].
144///
145/// If we move to the asynchronous API, we will need to generate our own *taskId* values, perhaps
146/// by combining some sort of thread id with a thread-local atomic counter. [`ThreadId`] is opaque
147/// in stable Rust, and converts to a [`u64`] in unstable Rust, so we would also need to make our
148/// own thread ids, perhaps by having a global atomic counter cached in a thread-local.
149///
150/// [synchronous API]: https://docs.rs/hitrace-sys/0.1.2/hitrace_sys/fn.OH_HiTrace_StartTrace.html
151/// [asynchronous API]: https://docs.rs/hitrace-sys/0.1.2/hitrace_sys/fn.OH_HiTrace_StartAsyncTrace.html
152/// [`Registry`]: tracing_subscriber::Registry
153/// [come from]: https://docs.rs/tracing-subscriber/0.3.18/src/tracing_subscriber/registry/sharded.rs.html#237-269
154/// [packed representation]: https://docs.rs/sharded-slab/0.1.7/sharded_slab/trait.Config.html
155/// [`ThreadId`]: std::thread::ThreadId
156#[cfg(feature = "tracing-hitrace")]
157#[derive(Default)]
158struct HitraceLayer {}
159
160cfg_if! {
161    if #[cfg(feature = "tracing-hitrace")] {
162        use std::cell::RefCell;
163
164        use tracing::span::Id;
165        use tracing::Subscriber;
166        use tracing_subscriber::Layer;
167
168        #[cfg(debug_assertions)]
169        thread_local! {
170            /// Stack of span names, to ensure the HiTrace synchronous API is not misused.
171            static HITRACE_NAME_STACK: RefCell<Vec<String>> = RefCell::default();
172        }
173
174        impl<S: Subscriber + for<'lookup> tracing_subscriber::registry::LookupSpan<'lookup>>
175            Layer<S> for HitraceLayer
176        {
177            fn on_enter(&self, id: &Id, ctx: tracing_subscriber::layer::Context<'_, S>) {
178                if let Some(metadata) = ctx.metadata(id) {
179                    // TODO: is this expensive? Would extensions be faster?
180                    // <https://docs.rs/tracing-subscriber/0.3.18/tracing_subscriber/registry/struct.ExtensionsMut.html>
181                    if metadata.fields().field("servo_profiling").is_some() {
182                        #[cfg(debug_assertions)]
183                        HITRACE_NAME_STACK.with_borrow_mut(|stack|
184                            stack.push(metadata.name().to_owned()));
185
186                        hitrace::start_trace(
187                            &std::ffi::CString::new(metadata.name())
188                                .expect("Failed to convert str to CString"),
189                        );
190                    }
191                }
192            }
193
194            fn on_event(&self, event: &tracing::Event<'_>, _ctx: tracing_subscriber::layer::Context<'_, S>) {
195                hitrace::start_trace(
196                    &std::ffi::CString::new(event.metadata().name())
197                        .expect("Failed to convert str to CString"),
198                );
199                hitrace::finish_trace();
200            }
201
202
203            fn on_exit(&self, id: &Id, ctx: tracing_subscriber::layer::Context<'_, S>) {
204                if let Some(metadata) = ctx.metadata(id) {
205                    if metadata.fields().field("servo_profiling").is_some() {
206                        hitrace::finish_trace();
207
208                        #[cfg(debug_assertions)]
209                        HITRACE_NAME_STACK.with_borrow_mut(|stack| {
210                            if stack.last().map(|name| &**name) != Some(metadata.name()) {
211                                log::error!(
212                                    "Tracing span out of order: {} (stack: {:?})",
213                                    metadata.name(),
214                                    stack
215                                );
216                            }
217                            if !stack.is_empty() {
218                                stack.pop();
219                            }
220                        });
221                    }
222                }
223            }
224        }
225    }
