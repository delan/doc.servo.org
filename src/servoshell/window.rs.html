41pub(crate) struct ServoShellWindow {
42    /// The [`WebView`]s that have been added to this window.
43    pub(crate) webview_collection: RefCell<WebViewCollection>,
44    /// A handle to the [`PlatformWindow`] that servoshell is rendering in.
45    platform_window: Rc<dyn PlatformWindow>,
46    /// Whether or not this window should be closed at the end of the spin of the next event loop.
47    close_scheduled: Cell<bool>,
48    /// Whether or not the application interface needs to be updated.
49    needs_update: Cell<bool>,
50    /// Whether or not Servo needs to repaint its display. Currently this is global
51    /// because every `WebView` shares a `RenderingContext`.
52    needs_repaint: Cell<bool>,
53    /// List of webviews that have favicon textures which are not yet uploaded
54    /// to the GPU by egui.
55    pending_favicon_loads: RefCell<Vec<WebViewId>>,
56    /// Pending [`UserInterfaceCommand`] that have yet to be processed by the main loop.
57    pending_commands: RefCell<Vec<UserInterfaceCommand>>,
58}

87    pub(crate) fn create_toplevel_webview(&self, state: Rc<RunningAppState>, url: Url) -> WebView {
88        let webview = WebViewBuilder::new(state.servo(), self.platform_window.rendering_context())
89            .url(url)
90            .hidpi_scale_factor(self.platform_window.hidpi_scale_factor())
91            .user_content_manager(state.user_content_manager.clone())
92            .delegate(state.clone())
93            .build();
94
95        webview.notify_theme_change(self.platform_window.theme());
96        self.add_webview(webview.clone());
97        webview
98    }

100    /// Repaint the focused [`WebView`].
101    pub(crate) fn repaint_webviews(&self) {
102        let Some(webview) = self.active_webview() else {
103            return;
104        };
105
106        self.platform_window()
107            .rendering_context()
108            .make_current()
109            .expect("Could not make PlatformWindow RenderingContext current");
110        webview.paint();
111        self.platform_window().rendering_context().present();
112    }

190    pub(crate) fn update_and_request_repaint_if_necessary(&self, state: &RunningAppState) {
191        let updated_user_interface = self.needs_update.take() &&
192            self.platform_window
193                .update_user_interface_state(state, self);
194
195        // Delegate handlers may have asked us to present or update painted WebView contents.
196        // Currently, egui-file-dialog dialogs need to be constantly redrawn or animations aren't fluid.
197        let needs_repaint = self.needs_repaint.take();
198        if updated_user_interface || needs_repaint {
199            self.platform_window.request_repaint(self);
200        }
201    }

203    /// Close the given [`WebView`] via its [`WebViewId`].
204    ///
205    /// Note: This can happen because we can trigger a close with a UI action and then get
206    /// the close notification via the [`WebViewDelegate`] later.
207    pub(crate) fn close_webview(&self, webview_id: WebViewId) {
208        let mut webview_collection = self.webview_collection.borrow_mut();
209        if webview_collection.remove(webview_id).is_none() {
210            return;
211        }
212        self.platform_window
213            .dismiss_embedder_controls_for_webview(webview_id);
214
215        self.set_needs_update();
216        self.set_needs_repaint();
217    }

280    /// Takes any events generated during UI updates and performs their actions.
281    pub(crate) fn handle_interface_commands(
282        &self,
283        state: &Rc<RunningAppState>,
284        create_platform_window: Option<&dyn Fn(Url) -> Rc<dyn PlatformWindow>>,
285    ) {
286        let commands = std::mem::take(&mut *self.pending_commands.borrow_mut());
287        for event in commands {
288            match event {
289                UserInterfaceCommand::Go(location) => {
290                    self.set_needs_update();
291                    let Some(url) = location_bar_input_to_url(
292                        &location.clone(),
293                        &state.servoshell_preferences.searchpage,
294                    ) else {
295                        warn!("failed to parse location");
296                        break;
297                    };
298                    if let Some(active_webview) = self.active_webview() {
299                        active_webview.load(url.into_url());
300                    }
301                },
302                UserInterfaceCommand::Back => {
303                    if let Some(active_webview) = self.active_webview() {
304                        active_webview.go_back(1);
305                    }
306                },
307                UserInterfaceCommand::Forward => {
308                    if let Some(active_webview) = self.active_webview() {
309                        active_webview.go_forward(1);
310                    }
311                },
312                UserInterfaceCommand::Reload => {
313                    self.set_needs_update();
314                    if let Some(active_webview) = self.active_webview() {
315                        active_webview.reload();
316                    }
317                },
318                UserInterfaceCommand::ReloadAll => {
319                    for window in state.windows().values() {
320                        window.set_needs_update();
321                        for (_, webview) in window.webviews() {
322                            webview.reload();
323                        }
324                    }
325                },
326                UserInterfaceCommand::NewWebView => {
327                    self.set_needs_update();
328                    let url = Url::parse("servo:newtab").expect("Should always be able to parse");
329                    self.create_and_activate_toplevel_webview(state.clone(), url);
330                },
331                UserInterfaceCommand::CloseWebView(id) => {
332                    self.set_needs_update();
333                    self.close_webview(id);
334                },
335                UserInterfaceCommand::NewWindow => {
336                    if let Some(create_platform_window) = create_platform_window {
337                        let url = Url::parse("servo:newtab").unwrap();
338                        let platform_window = create_platform_window(url.clone());
339                        state.open_window(platform_window, url);
340                    }
341                },
342            }
343        }
344    }

347/// A `PlatformWindow` abstracts away the differents kinds of platform windows that might
348/// be used in a servoshell execution. This currently includes headed (winit) and headless
349/// windows.
350pub(crate) trait PlatformWindow {
351    fn id(&self) -> ServoShellWindowId;
352    fn screen_geometry(&self) -> ScreenGeometry;
353    #[cfg_attr(any(target_os = "android", target_env = "ohos"), expect(dead_code))]
354    fn device_hidpi_scale_factor(&self) -> Scale<f32, DeviceIndependentPixel, DevicePixel>;
355    fn hidpi_scale_factor(&self) -> Scale<f32, DeviceIndependentPixel, DevicePixel>;
356    #[cfg_attr(any(target_os = "android", target_env = "ohos"), expect(dead_code))]
357    fn get_fullscreen(&self) -> bool;
358    /// Request that the `Window` rebuild its user interface, if it has one. This should
359    /// not repaint, but should prepare the user interface for painting when it is
360    /// actually requested.
361    #[cfg_attr(any(target_os = "android", target_env = "ohos"), expect(dead_code))]
362    fn rebuild_user_interface(&self, _: &RunningAppState, _: &ServoShellWindow) {}
363    /// Inform the `Window` that the state of a `WebView` has changed and that it should
364    /// do an incremental update of user interface state. Returns `true` if the user
365    /// interface actually changed and a rebuild  and repaint is needed, `false` otherwise.
366    fn update_user_interface_state(&self, _: &RunningAppState, _: &ServoShellWindow) -> bool {
367        false
368    }
369    /// Request that the window redraw itself. It is up to the window to do this
370    /// once the windowing system is ready. If this is a headless window, the redraw
371    /// will happen immediately.
372    fn request_repaint(&self, _: &ServoShellWindow);
373    /// Request a new outer size for the window, including external decorations.
374    /// This should be the same as `window.outerWidth` and `window.outerHeight``
375    fn request_resize(&self, webview: &WebView, outer_size: DeviceIntSize)
376    -> Option<DeviceIntSize>;
377    fn set_position(&self, _point: DeviceIntPoint) {}
378    fn set_fullscreen(&self, _state: bool) {}
379    fn set_cursor(&self, _cursor: Cursor) {}
380    #[cfg(all(
381        feature = "webxr",
382        not(any(target_os = "android", target_env = "ohos"))
383    ))]
384    fn new_glwindow(
385        &self,
386        event_loop: &winit::event_loop::ActiveEventLoop,
387    ) -> Rc<dyn servo::webxr::GlWindow>;
388    /// This returns [`RenderingContext`] matching the viewport.
389    fn rendering_context(&self) -> Rc<dyn RenderingContext>;
390    fn theme(&self) -> servo::Theme {
391        servo::Theme::Light
392    }
393    fn window_rect(&self) -> DeviceIndependentIntRect;
394    fn maximize(&self, _: &WebView) {}
395    fn focus(&self) {}
396    fn has_platform_focus(&self) -> bool {
397        true
398    }
399
400    fn show_embedder_control(&self, _: WebViewId, _: EmbedderControl) {}
401    fn hide_embedder_control(&self, _: WebViewId, _: EmbedderControlId) {}
402    fn dismiss_embedder_controls_for_webview(&self, _: WebViewId) {}
403    fn show_bluetooth_device_dialog(
404        &self,
405        _: WebViewId,
406        _devices: Vec<String>,
407        _: GenericSender<Option<String>>,
408    ) {
409    }
410    fn show_permission_dialog(&self, _: WebViewId, _: PermissionRequest) {}
411    fn show_http_authentication_dialog(&self, _: WebViewId, _: AuthenticationRequest) {}
412
413    fn notify_input_event_handled(
414        &self,
415        _webview: &WebView,
416        _id: InputEventId,
417        _result: InputEventResult,
418    ) {
419    }
420
421    fn notify_media_session_event(&self, _: MediaSessionEvent) {}
422    fn notify_crashed(&self, _: WebView, _reason: String, _backtrace: Option<String>) {}
423    fn show_console_message(&self, _level: ConsoleLogLevel, _message: &str) {}
424
425    #[cfg(not(any(target_os = "android", target_env = "ohos")))]
426    /// If this window is a headed window, access the concrete type.
427    fn as_headed_window(&self) -> Option<&crate::desktop::headed_window::HeadedWindow> {
428        None
429    }

431    #[cfg(any(target_os = "android", target_env = "ohos"))]
432    /// If this window is a headed window, access the concrete type.
433    fn as_headed_window(&self) -> Option<&crate::egl::app::EmbeddedPlatformWindow> {
434        None
