116fn cookie_msg_to_cookie(cookie: cookie::Cookie) -> Cookie {
117    Cookie {
118        name: cookie.name().to_owned(),
119        value: cookie.value().to_owned(),
120        path: cookie.path().map(|s| s.to_owned()),
121        domain: cookie.domain().map(|s| s.to_owned()),
122        expiry: cookie.expires().and_then(|expiration| match expiration {
123            Expiration::DateTime(date_time) => Some(Date(date_time.unix_timestamp() as u64)),
124            Expiration::Session => None,
125        }),
126        secure: cookie.secure().unwrap_or(false),
127        http_only: cookie.http_only().unwrap_or(false),
128        same_site: cookie.same_site().map(|s| s.to_string()),
129    }

132pub fn start_server(
133    port: u16,
134    embedder_sender: Sender<WebDriverCommandMsg>,
135    event_loop_waker: Box<dyn EventLoopWaker>,
136) {
137    let handler = Handler::new(embedder_sender, event_loop_waker);
138
139    thread::Builder::new()
140        .name("WebDriverHttpServer".to_owned())
141        .spawn(move || {
142            let address = SocketAddrV4::new("0.0.0.0".parse().unwrap(), port);
143            match server::start(
144                SocketAddr::V4(address),
145                vec![],
146                vec![],
147                handler,
148                extension_routes(),
149            ) {
150                Ok(listening) => info!("WebDriver server listening on {}", listening.socket),
151                Err(e) => panic!("Unable to start WebDriver HTTP server {e:?}"),
152            }
153        })
154        .expect("Thread spawning failed");
155}

157struct Handler {
158    /// The threaded receiver on which we can block for a load-status.
159    /// It will receive messages sent on the load_status_sender,
160    /// and forwarded by the IPC router.
161    load_status_receiver: RoutedReceiver<WebDriverLoadStatus>,
162    /// The IPC sender which we can clone and pass along to the constellation,
163    /// for it to send us a load-status. Messages sent on it
164    /// will be forwarded to the load_status_receiver.
165    load_status_sender: GenericSender<WebDriverLoadStatus>,
166
167    session: Option<WebDriverSession>,
168
169    /// A [`Sender`] that sends messages to the embedder that this `WebDriver instance controls.
170    /// In addition to sending a message, we must always wake up the embedder's event loop so it
171    /// knows that more messages are available for processing.
172    embedder_sender: Sender<WebDriverCommandMsg>,
173
174    /// An [`EventLoopWaker`] which is used to wake up the embedder event loop.
175    event_loop_waker: Box<dyn EventLoopWaker>,
176
177    /// A list of [`Receiver`]s that are used to track when input events are handled in the DOM.
178    /// Once these receivers receive a response, we know that the event has been handled.
179    ///
180    /// TODO: Once we upgrade crossbeam-channel this can be replaced with a `WaitGroup`.
181    pending_input_event_receivers: RefCell<Vec<Receiver<()>>>,
182
183    /// Number of pending actions of which WebDriver is waiting for responses.
184    num_pending_actions: Cell<u32>,
185}

202impl WebDriverExtensionRoute for ServoExtensionRoute {
203    type Command = ServoExtensionCommand;
204
205    fn command(
206        &self,
207        _parameters: &Parameters,
208        body_data: &Value,
209    ) -> WebDriverResult<WebDriverCommand<ServoExtensionCommand>> {
210        let command = match *self {
211            ServoExtensionRoute::GetPrefs => {
212                let parameters: GetPrefsParameters = serde_json::from_value(body_data.clone())?;
213                ServoExtensionCommand::GetPrefs(parameters)
214            },
215            ServoExtensionRoute::SetPrefs => {
216                let parameters: SetPrefsParameters = serde_json::from_value(body_data.clone())?;
217                ServoExtensionCommand::SetPrefs(parameters)
218            },
219            ServoExtensionRoute::ResetPrefs => {
220                let parameters: GetPrefsParameters = serde_json::from_value(body_data.clone())?;
221                ServoExtensionCommand::ResetPrefs(parameters)
222            },
223            ServoExtensionRoute::CustomHandlersSetMode => {
224                let parameters: CustomHandlersSetModeParameters =
225                    serde_json::from_value(body_data.clone())?;
226                ServoExtensionCommand::CustomHandlersSetMode(parameters)
227            },
228            ServoExtensionRoute::Shutdown => ServoExtensionCommand::Shutdown,
229            ServoExtensionRoute::ResetAllCookies => ServoExtensionCommand::ResetAllCookies,
230        };
231        Ok(WebDriverCommand::Extension(command))
232    }
233}

260impl Serialize for SendableJSValue {
261    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
262    where
263        S: Serializer,
264    {
265        match self.0 {
266            JSValue::Undefined | JSValue::Null => serializer.serialize_unit(),
267            JSValue::Boolean(x) => serializer.serialize_bool(x),
268            JSValue::Number(x) => {
269                if x.fract() == 0.0 {
270                    serializer.serialize_i64(x as i64)
271                } else {
272                    serializer.serialize_f64(x)
273                }
274            },
275            JSValue::String(ref x) => serializer.serialize_str(x),
276            JSValue::Element(ref x) => WebElement(x.clone()).serialize(serializer),
277            JSValue::ShadowRoot(ref x) => ShadowRoot(x.clone()).serialize(serializer),
278            JSValue::Frame(ref x) => WebFrame(x.clone()).serialize(serializer),
279            JSValue::Window(ref x) => WebWindow(x.clone()).serialize(serializer),
280            JSValue::Array(ref x) => x
281                .iter()
282                .map(|element| SendableJSValue(element.clone()))
283                .collect::<Vec<SendableJSValue>>()
284                .serialize(serializer),
285            JSValue::Object(ref x) => x
286                .iter()
287                .map(|(k, v)| (k.clone(), SendableJSValue(v.clone())))
288                .collect::<HashMap<String, SendableJSValue>>()
289                .serialize(serializer),
290        }
291    }
292}

443impl Handler {
444    fn new(
445        embedder_sender: Sender<WebDriverCommandMsg>,
446        event_loop_waker: Box<dyn EventLoopWaker>,
447    ) -> Handler {
448        // Create a pair of both an IPC and a threaded channel,
449        // keep the IPC sender to clone and pass to the constellation for each load,
450        // and keep a threaded receiver to block on an incoming load-status.
451        // Pass the others to the IPC router so that IPC messages are forwarded to the threaded receiver.
452        // We need to use the router because IPC does not come with a timeout on receive/select.
453        let (load_status_sender, receiver) = generic_channel::channel().unwrap();
454        let load_status_receiver = receiver.route_preserving_errors();
455
456        Handler {
457            load_status_sender,
458            load_status_receiver,
459            session: None,
460            embedder_sender,
461            event_loop_waker,
462            pending_input_event_receivers: Default::default(),
463            num_pending_actions: Cell::new(0),
464        }
465    }
466
467    fn browsing_context_id(&self) -> WebDriverResult<BrowsingContextId> {
468        self.session()?
469            .current_browsing_context_id()
470            .ok_or_else(|| {
471                WebDriverError::new(ErrorStatus::UnknownError, "No browsing context available")
472            })
473    }
474
475    fn webview_id(&self) -> WebDriverResult<WebViewId> {
476        self.session()?
477            .current_webview_id()
478            .ok_or_else(|| WebDriverError::new(ErrorStatus::UnknownError, "No webview available"))
479    }
480
481    fn send_input_event_to_embedder(&self, input_event: InputEvent) {
482        let _ = self.send_message_to_embedder(WebDriverCommandMsg::InputEvent(
483            self.verified_webview_id(),
484            input_event,
485            None,
486        ));
487    }
488
489    fn send_blocking_input_event_to_embedder(&self, input_event: InputEvent) {
490        let (result_sender, result_receiver) = unbounded();
491        if self
492            .send_message_to_embedder(WebDriverCommandMsg::InputEvent(
493                self.verified_webview_id(),
494                input_event,
495                Some(result_sender),
496            ))
497            .is_ok()
498        {
499            self.pending_input_event_receivers
500                .borrow_mut()
501                .push(result_receiver);
502        }
503    }
504
505    fn send_message_to_embedder(&self, msg: WebDriverCommandMsg) -> WebDriverResult<()> {
506        self.embedder_sender.send(msg).map_err(|_| {
507            WebDriverError::new(
508                ErrorStatus::UnknownError,
509                "Failed to send message to embedder",
510            )
511        })?;
512        self.event_loop_waker.wake();
513        Ok(())
514    }
515
516    fn add_load_status_sender(&self) -> WebDriverResult<()> {
517        self.send_message_to_embedder(WebDriverCommandMsg::ScriptCommand(
518            self.browsing_context_id()?,
519            WebDriverScriptCommand::AddLoadStatusSender(
520                self.webview_id()?,
521                self.load_status_sender.clone(),
522            ),
523        ))
524    }
525
526    fn clear_load_status_sender(&self) -> WebDriverResult<()> {
527        self.send_message_to_embedder(WebDriverCommandMsg::ScriptCommand(
528            self.browsing_context_id()?,
529            WebDriverScriptCommand::RemoveLoadStatusSender(self.webview_id()?),
530        ))
531    }
532
533    // This function is called only if session and webview are verified.
534    fn verified_webview_id(&self) -> WebViewId {
535        self.session().unwrap().current_webview_id().unwrap()
536    }
537
538    fn focused_webview_id(&self) -> WebDriverResult<Option<WebViewId>> {
539        let (sender, receiver) = generic_channel::oneshot().unwrap();
540        self.send_message_to_embedder(WebDriverCommandMsg::GetFocusedWebView(sender))?;
541        // Wait until the document is ready before returning the top-level browsing context id.
542        wait_for_oneshot_response(receiver)
543    }

545    fn session(&self) -> WebDriverResult<&WebDriverSession> {
546        match self.session {
547            Some(ref x) => Ok(x),
548            // https://w3c.github.io/webdriver/#ref-for-dfn-invalid-session-id-1
549            None => Err(WebDriverError::new(
550                ErrorStatus::InvalidSessionId,
551                "Session not created",
552            )),
553        }
554    }

556    fn session_mut(&mut self) -> WebDriverResult<&mut WebDriverSession> {
557        match self.session {
558            Some(ref mut x) => Ok(x),
559            // https://w3c.github.io/webdriver/#ref-for-dfn-invalid-session-id-1
560            None => Err(WebDriverError::new(
561                ErrorStatus::InvalidSessionId,
562                "Session not created",
563            )),
564        }
565    }

567    /// <https://w3c.github.io/webdriver/#new-session>
568    fn handle_new_session(
569        &mut self,
570        parameters: &NewSessionParameters,
571    ) -> WebDriverResult<WebDriverResponse> {
572        if let Ok(value) = env::var("DELAY_AFTER_ACCEPT") {
573            let seconds = value.parse::<u64>().unwrap_or_default();
574            println!("Waiting for {} seconds...", seconds);
575            println!("lldb -p {}", process::id());
576            thread::sleep(Duration::from_secs(seconds));
577        }
578
579        // Step 1. If the list of active HTTP sessions is not empty
580        // return error with error code session not created.
581        if self.session.is_some() {
582            return Err(WebDriverError::new(
583                ErrorStatus::SessionNotCreated,
584                "Session already created",
585            ));
586        }
587
588        // Step 2. Skip because the step is only applied to an intermediary node.
589        // Step 3. Skip since all sessions are http for now.
590
591        // Step 4. Let capabilities be the result of trying to process capabilities
592        let mut servo_capabilities = ServoCapabilities::new();
593        let processed_capabilities = parameters.match_browser(&mut servo_capabilities)?;
594
595        // Step 5. If capabilities's is null, return error with error code session not created.
596        let mut capabilities = match processed_capabilities {
597            Some(capabilities) => capabilities,
598            None => {
599                return Err(WebDriverError::new(
600                    ErrorStatus::SessionNotCreated,
601                    "Session not created due to invalid capabilities",
602                ));
603            },
604        };
605
606        // Step 6. Create a session
607        let session_id = self.create_session(&mut capabilities, &servo_capabilities)?;
608
609        // Step 7. Let response be a JSON Object initialized with session's session ID and capabilities
610        let response = NewSessionResponse::new(session_id.to_string(), Value::Object(capabilities));
611
612        // Step 8. Set session' current top-level browsing context
613        match self.focused_webview_id()? {
614            Some(webview_id) => {
615                self.session_mut()?.set_webview_id(webview_id);
616                self.session_mut()?
617                    .set_browsing_context_id(BrowsingContextId::from(webview_id));
618            },
619            None => {
620                // This happens when there is no open webview.
621                // We need to create a new one. See https://github.com/servo/servo/issues/37408
622                let (sender, receiver) = generic_channel::oneshot().unwrap();
623
624                self.send_message_to_embedder(WebDriverCommandMsg::NewWindow(
625                    NewWindowTypeHint::Auto,
626                    sender,
627                    Some(self.load_status_sender.clone()),
628                ))?;
629                let webview_id = receiver
630                    .recv()
631                    .expect("IPC failure when creating new webview for new session");
632                self.focus_webview(webview_id)?;
633                self.session_mut()?.set_webview_id(webview_id);
634                self.session_mut()?
635                    .set_browsing_context_id(BrowsingContextId::from(webview_id));
636                let _ = self.wait_document_ready(Some(3000));
637            },
638        };
639
640        // Step 9. Set the request queue to a new queue.
641        // Skip here because the requests are handled in the external crate.
642
643        // Step 10. Return success with data body
644        Ok(WebDriverResponse::NewSession(response))
645    }
646
647    /// <https://w3c.github.io/webdriver/#dfn-delete-session>
648    fn handle_delete_session(&mut self) -> WebDriverResult<WebDriverResponse> {
649        // Step 1. If session is http, close the session
650        self.session = None;
651
652        // Step 2. Return success with data null
653        Ok(WebDriverResponse::DeleteSession)

735    fn wait_document_ready(&self, timeout: Option<u64>) -> WebDriverResult<WebDriverResponse> {
736        let timeout_channel = match timeout {
737            Some(timeout) => after(Duration::from_millis(timeout)),
738            None => crossbeam_channel::never(),
739        };
740
741        select! {
742            recv(self.load_status_receiver) -> res => {
743                match res {
744                    // If the navigation is navigation to IFrame, no document state event is fired.
745                    Ok(Ok(WebDriverLoadStatus::Blocked)) => {
746                        // TODO: evaluate the correctness later
747                        // Load status is block means an user prompt is shown.
748                        // Alot of tests expect this to return success
749                        // then the user prompt is handled in the next command.
750                        // If user prompt can't be handler, next command returns
751                        // an error anyway.
752                        Ok(WebDriverResponse::Void)
753                    },
754                    Ok(Ok(WebDriverLoadStatus::Complete)) |
755                    Ok(Ok(WebDriverLoadStatus::NavigationStop)) =>
756                        Ok(WebDriverResponse::Void)
757                    ,
758                    _ => Err(WebDriverError::new(
759                        ErrorStatus::UnknownError,
760                        "Unexpected load status received while waiting for document ready state",
761                    )),
762                }
763            },
764            recv(timeout_channel) -> _ => Err(
765                WebDriverError::new(ErrorStatus::Timeout, "Load timed out")
766            ),
767        }

771    fn wait_for_navigation_complete(&self) -> WebDriverResult<WebDriverResponse> {
772        debug!("waiting for load");
773
774        let session = self.session()?;
775
776        // Step 1. If session's page loading strategy is "none",
777        // return success with data null.
778        if session.page_loading_strategy() == PageLoadStrategy::None {

789        }

791        // Step 3. let timeout be the session's page load timeout.
792        let timeout = session.session_timeouts().page_load;

795
796        let result = self.wait_document_ready(timeout);
797        debug!("finished waiting for load with {:?}", result);

801    /// <https://w3c.github.io/webdriver/#dfn-wait-for-navigation-to-complete>
802    fn wait_for_navigation(&self) -> WebDriverResult<WebDriverResponse> {
803        let navigation_status = match self.load_status_receiver.try_recv() {
804            Ok(Ok(status)) => status,
805            // Empty channel means no navigation started. Nothing to wait for.
806            Err(crossbeam_channel::TryRecvError::Empty) => {
807                return Ok(WebDriverResponse::Void);
808            },
809            Err(crossbeam_channel::TryRecvError::Disconnected) => {
810                return Err(WebDriverError::new(
811                    ErrorStatus::UnknownError,
812                    "Load status channel disconnected",
813                ));
814            },
815            Ok(Err(ipc_error)) => {
816                return Err(WebDriverError::new(
817                    ErrorStatus::UnknownError,
818                    format!("Load status channel ipc error: {ipc_error}"),
819                ));
820            },
821        };
822
823        match navigation_status {
824            WebDriverLoadStatus::NavigationStart => self.wait_for_navigation_complete(),
825            // If the load status is timeout, return an error
826            WebDriverLoadStatus::Timeout => Err(WebDriverError::new(
827                ErrorStatus::Timeout,
828                "Navigation timed out",
829            )),
830            // If the load status is blocked, it means a user prompt is shown.
831            // We should handle the user prompt in the next command.
832            WebDriverLoadStatus::Blocked => Ok(WebDriverResponse::Void),
833            WebDriverLoadStatus::NavigationStop | WebDriverLoadStatus::Complete => {
834                unreachable!("Unexpected load status received")
835            },
836        }
837    }

1385    /// The boolean in callback result indicates whether implicit_wait can early return
1386    /// before timeout with current result.
1387    fn implicit_wait<T>(
1388        &self,
1389        callback: impl Fn() -> Result<(bool, T), (bool, WebDriverError)>,
1390    ) -> Result<T, WebDriverError> {
1391        let now = Instant::now();
1392        let (implicit_wait, sleep_interval) = {
1393            let timeouts = self.session()?.session_timeouts();
1394            (
1395                Duration::from_millis(timeouts.implicit_wait.unwrap_or(0)),
1396                Duration::from_millis(timeouts.sleep_interval),
1397            )
1398        };
1399
1400        loop {
1401            match callback() {
1402                Ok((can_early_return, value)) => {
1403                    if can_early_return || now.elapsed() >= implicit_wait {
1404                        return Ok(value);
1405                    }
1406                },
1407                Err((can_early_return, error)) => {
1408                    if can_early_return || now.elapsed() >= implicit_wait {
1409                        return Err(error);
1410                    }
1411                },
1412            }
1413            sleep(sleep_interval);
1414        }
1415    }

1419        &self,
1420        parameters: &LocatorParameters,
1421    ) -> WebDriverResult<WebDriverResponse> {

2361            id: id.clone(),
2362            actions: ActionsType::Pointer {
2363                parameters: PointerActionParameters {
2364                    pointer_type: PointerType::Mouse,
2365                },
2366                actions: vec![
2367                    PointerActionItem::Pointer(PointerAction::Move(pointer_move_action)),

2377            error!("handle_element_click: dispatch_actions failed: {:?}", e);
2378        }

2381        self.session_mut()?.input_state_table.remove(&id);
2382
2383        Ok(WebDriverResponse::Void)
2384    }
