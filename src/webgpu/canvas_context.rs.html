53/// Allocated buffer on GPU device
54#[derive(Clone, Copy, Debug)]
55struct Buffer {
56    device_id: DeviceId,
57    queue_id: QueueId,
58    size: u64,
59}
60
61impl Buffer {
62    /// Returns true if buffer is compatible with provided configuration
63    fn has_compatible_config(&self, config: &ContextConfiguration) -> bool {
64        config.device_id == self.device_id && self.size == config.buffer_size()
65    }
66}

68/// Mapped GPUBuffer
69#[derive(Debug)]
70struct MappedBuffer {
71    buffer: Buffer,
72    data: NonNull<u8>,
73    len: u64,
74    image_size: Size2D<u32>,
75    image_format: ImageFormat,
76    is_opaque: bool,
77}
78
79// Mapped buffer can be shared between safely (it's read-only)
80unsafe impl Send for MappedBuffer {}
81
82impl MappedBuffer {
83    const fn slice(&'_ self) -> &'_ [u8] {
84        // Safety: Pointer is from wgpu, and we only use it here
85        unsafe { std::slice::from_raw_parts(self.data.as_ptr(), self.len as usize) }
86    }
87
88    fn stride(&self) -> u32 {
89        (self.image_size.width * self.image_format.bytes_per_pixel() as u32)
90            .next_multiple_of(COPY_BYTES_PER_ROW_ALIGNMENT)
91    }
92}

94#[derive(Debug)]
95enum StagingBufferState {
96    /// The Initial state: the buffer has yet to be created with only an
97    /// id reserved for it.
98    Unassigned,
99    /// The buffer is allocated in the WGPU Device and is ready to be used.
100    Available(Buffer),
101    /// `mapAsync` is currently running on the buffer.
102    Mapping(Buffer),
103    /// The buffer is currently mapped.
104    Mapped(MappedBuffer),
105}

107/// A staging buffer used for texture to buffer to CPU copy operations.
108#[derive(Debug)]
109struct StagingBuffer {
110    global: Arc<Global>,
111    buffer_id: BufferId,
112    state: StagingBufferState,
113}
114
115// [`StagingBuffer`] only used for reading (never for writing)
116// so it is safe to share between threads.
117unsafe impl Sync for StagingBuffer {}
118
119impl StagingBuffer {
120    fn new(global: Arc<Global>, buffer_id: BufferId) -> Self {
121        Self {
122            global,
123            buffer_id,
124            state: StagingBufferState::Unassigned,
125        }
126    }
127
128    const fn is_mapped(&self) -> bool {
129        matches!(self.state, StagingBufferState::Mapped(..))
130    }
131
132    /// Return true if buffer can be used directly with provided config
133    /// without any additional work
134    fn is_available_and_has_compatible_config(&self, config: &ContextConfiguration) -> bool {
135        let StagingBufferState::Available(buffer) = &self.state else {
136            return false;
137        };
138        buffer.has_compatible_config(config)
139    }
140
141    /// Return true if buffer is not mapping or being mapped
142    const fn needs_assignment(&self) -> bool {
143        matches!(
144            self.state,
145            StagingBufferState::Unassigned | StagingBufferState::Available(_)
146        )
147    }
148
149    /// Make buffer available by unmapping / destroying it and then recreating it if needed.
150    fn ensure_available(&mut self, config: &ContextConfiguration) -> Result<(), CreateBufferError> {
151        let recreate = match &self.state {
152            StagingBufferState::Unassigned => true,
153            StagingBufferState::Available(buffer) |
154            StagingBufferState::Mapping(buffer) |
155            StagingBufferState::Mapped(MappedBuffer { buffer, .. }) => {
156                if buffer.has_compatible_config(config) {
157                    let _ = self.global.buffer_unmap(self.buffer_id);
158                    false
159                } else {
160                    self.global.buffer_drop(self.buffer_id);
161                    true
162                }
163            },
164        };
165        if recreate {
166            let buffer_size = config.buffer_size();
167            let (_, error) = self.global.device_create_buffer(
168                config.device_id,
169                &BufferDescriptor {
170                    label: None,
171                    size: buffer_size,
172                    usage: BufferUsages::MAP_READ | BufferUsages::COPY_DST,
173                    mapped_at_creation: false,
174                },
175                Some(self.buffer_id),
176            );
177            if let Some(error) = error {
178                return Err(error);
179            };
180            self.state = StagingBufferState::Available(Buffer {
181                device_id: config.device_id,
182                queue_id: config.queue_id,
183                size: buffer_size,
184            });
185        }
186        Ok(())
187    }
188
189    /// Makes buffer available and prepares command encoder
190    /// that will copy texture to this staging buffer.
191    ///
192    /// Caller must submit command buffer to queue.
193    fn prepare_load_texture_command_buffer(
194        &mut self,
195        texture_id: TextureId,
196        encoder_id: CommandEncoderId,
197        config: &ContextConfiguration,
198    ) -> Result<CommandBufferId, Box<dyn std::error::Error>> {
199        self.ensure_available(config)?;
200        let StagingBufferState::Available(buffer) = &self.state else {
201            unreachable!("Should be made available by `ensure_available`")
202        };
203        let device_id = buffer.device_id;
204        let command_descriptor = CommandEncoderDescriptor { label: None };
205        let (encoder_id, error) = self.global.device_create_command_encoder(
206            device_id,
207            &command_descriptor,
208            Some(encoder_id),
209        );
210        if let Some(error) = error {
211            return Err(error.into());
212        };
213        let buffer_info = TexelCopyBufferInfo {
214            buffer: self.buffer_id,
215            layout: TexelCopyBufferLayout {
216                offset: 0,
217                bytes_per_row: Some(config.stride()),
218                rows_per_image: None,
219            },
220        };
221        let texture_info = TexelCopyTextureInfo {
222            texture: texture_id,
223            mip_level: 0,
224            origin: Origin3d::ZERO,
225            aspect: TextureAspect::All,
226        };
227        let copy_size = Extent3d {
228            width: config.size.width,
229            height: config.size.height,
230            depth_or_array_layers: 1,
231        };
232        self.global.command_encoder_copy_texture_to_buffer(
233            encoder_id,
234            &texture_info,
235            &buffer_info,
236            &copy_size,
237        )?;
238        let (command_buffer_id, error) = self
239            .global
240            .command_encoder_finish(encoder_id, &CommandBufferDescriptor::default());
241        if let Some(error) = error {
242            return Err(error.into());
243        };
244        Ok(command_buffer_id)
245    }
246
247    /// Unmaps the buffer or cancels a mapping operation if one is in progress.
248    fn unmap(&mut self) {
249        match self.state {
250            StagingBufferState::Unassigned | StagingBufferState::Available(_) => {},
251            StagingBufferState::Mapping(buffer) |
252            StagingBufferState::Mapped(MappedBuffer { buffer, .. }) => {
253                let _ = self.global.buffer_unmap(self.buffer_id);
254                self.state = StagingBufferState::Available(buffer)
255            },
256        }
257    }
258
259    /// Obtain a snapshot from this buffer if is mapped or return `None` if it is not mapped.
260    fn snapshot(&self) -> Option<Snapshot> {
261        let StagingBufferState::Mapped(mapped) = &self.state else {
262            return None;
263        };
264        let format = match mapped.image_format {
265            ImageFormat::RGBA8 => SnapshotPixelFormat::RGBA,
266            ImageFormat::BGRA8 => SnapshotPixelFormat::BGRA,
267            _ => unreachable!("GPUCanvasContext does not support other formats per spec"),
268        };
269        let alpha_mode = if mapped.is_opaque {
270            SnapshotAlphaMode::AsOpaque {
271                premultiplied: false,
272            }
273        } else {
274            SnapshotAlphaMode::Transparent {
275                premultiplied: true,
276            }
277        };
278        let padded_byte_width = mapped.stride();
279        let data = mapped.slice();
280        let bytes_per_pixel = mapped.image_format.bytes_per_pixel() as usize;
281        let mut result_unpadded =
282            Vec::<u8>::with_capacity(mapped.image_size.area() as usize * bytes_per_pixel);
283        for row in 0..mapped.image_size.height {
284            let start = (row * padded_byte_width).try_into().ok()?;
285            result_unpadded
286                .extend(&data[start..start + mapped.image_size.width as usize * bytes_per_pixel]);
287        }
288        let mut snapshot =
289            Snapshot::from_vec(mapped.image_size, format, alpha_mode, result_unpadded);
290        if mapped.is_opaque {
291            snapshot.transform(SnapshotAlphaMode::Opaque, snapshot.format())
292        }
293        Some(snapshot)
294    }
295}
296
297impl Drop for StagingBuffer {
298    fn drop(&mut self) {
299        match self.state {
300            StagingBufferState::Unassigned => {},
301            StagingBufferState::Available(_) |
302            StagingBufferState::Mapping(_) |
303            StagingBufferState::Mapped(_) => {
304                self.global.buffer_drop(self.buffer_id);
305            },
306        }
307    }
308}

324impl WebRenderExternalImageApi for WebGpuExternalImages {
325    fn lock(&mut self, id: u64) -> (ExternalImageSource<'_>, Size2D<i32>) {
326        let id = WebGPUContextId(id);
327        let presentation = {
328            let mut webgpu_contexts = self.image_map.lock().unwrap();
329            webgpu_contexts
330                .get_mut(&id)
331                .and_then(|context_data| context_data.presentation.clone())
332        };
333        let Some(presentation) = presentation else {
334            return (ExternalImageSource::Invalid, Size2D::zero());
335        };
336        self.locked_ids.insert(id, presentation);
337        let presentation = self.locked_ids.get(&id).unwrap();
338        let StagingBufferState::Mapped(mapped_buffer) = &presentation.staging_buffer.state else {
339            unreachable!("Presentation staging buffer should be mapped")
340        };
341        let size = mapped_buffer.image_size;
342        (
343            ExternalImageSource::RawData(mapped_buffer.slice()),
344            size.cast().cast_unit(),
345        )
346    }
347
348    fn unlock(&mut self, id: u64) {
349        let id = WebGPUContextId(id);
350        let Some(presentation) = self.locked_ids.remove(&id) else {
351            return;
352        };
353        let mut webgpu_contexts = self.image_map.lock().unwrap();
354        if let Some(context_data) = webgpu_contexts.get_mut(&id) {
355            // We use this to return staging buffer if a newer one exists.
356            presentation.maybe_destroy(context_data);
357        } else {
358            // This will not free this buffer id in script,
359            // but that's okay because we still have many free ids.
360            drop(presentation);
361        }
362    }
363}

365/// Staging buffer currently used for presenting the epoch.
366///
367/// Users should [`ContextData::replace_presentation`] when done.
368#[derive(Clone)]
369pub struct PresentationStagingBuffer {
370    epoch: Epoch,
371    staging_buffer: Arc<StagingBuffer>,
372}
373
374impl PresentationStagingBuffer {
375    fn new(epoch: Epoch, staging_buffer: StagingBuffer) -> Self {
376        Self {
377            epoch,
378            staging_buffer: Arc::new(staging_buffer),
379        }
380    }
381
382    /// If the internal staging buffer is not shared,
383    /// unmap it and call [`ContextData::return_staging_buffer`] with it.
384    fn maybe_destroy(self, context_data: &mut ContextData) {
385        if let Some(mut staging_buffer) = Arc::into_inner(self.staging_buffer) {
386            staging_buffer.unmap();
387            context_data.return_staging_buffer(staging_buffer);
388        }
389    }
390}

392/// The embedder process-side representation of what is the `GPUCanvasContext` in script.
393pub struct ContextData {
394    /// The [`ImageKey`] of the WebRender image associated with this context.
395    image_key: Option<ImageKey>,
396    /// The current size of this context.
397    size: DeviceIntSize,
398    /// Staging buffers that are not actively used.
399    ///
400    /// Staging buffer here are either [`StagingBufferState::Unassigned`] or [`StagingBufferState::Available`].
401    /// They are removed from here when they are in process of being mapped or are already mapped.
402    inactive_staging_buffers: ArrayVec<StagingBuffer, PRESENTATION_BUFFER_COUNT>,
403    /// The [`PresentationStagingBuffer`] of the most recent presentation. This will
404    /// be `None` directly after initialization, as clearing is handled completely in
405    /// the `ScriptThread`.
406    presentation: Option<PresentationStagingBuffer>,
407    /// Next epoch to be used
408    next_epoch: Epoch,
409}
410
411impl ContextData {
412    fn new(
413        global: &Arc<Global>,
414        buffer_ids: ArrayVec<id::BufferId, PRESENTATION_BUFFER_COUNT>,
415        size: DeviceIntSize,
416    ) -> Self {
417        Self {
418            image_key: None,
419            size,
420            inactive_staging_buffers: buffer_ids
421                .iter()
422                .map(|buffer_id| StagingBuffer::new(global.clone(), *buffer_id))
423                .collect(),
424            presentation: None,
425            next_epoch: Epoch(1),
426        }
427    }
428
429    /// Returns `None` if no staging buffer is unused or failure when making it available
430    fn get_or_make_available_buffer(
431        &'_ mut self,
432        config: &ContextConfiguration,
433    ) -> Option<StagingBuffer> {
434        self.inactive_staging_buffers
435            .iter()
436            // Try to get first preallocated GPUBuffer.
437            .position(|staging_buffer| {
438                staging_buffer.is_available_and_has_compatible_config(config)
439            })
440            // Fall back to the first inactive staging buffer.
441            .or_else(|| {
442                self.inactive_staging_buffers
443                    .iter()
444                    .position(|staging_buffer| staging_buffer.needs_assignment())
445            })
446            // Or just the use first one.
447            .or_else(|| {
448                if self.inactive_staging_buffers.is_empty() {
449                    None
450                } else {
451                    Some(0)
452                }
453            })
454            .and_then(|index| {
455                let mut staging_buffer = self.inactive_staging_buffers.remove(index);
456                if staging_buffer.ensure_available(config).is_ok() {
457                    Some(staging_buffer)
458                } else {
459                    // If we fail to make it available, return it to the list of inactive staging buffers.
460                    self.inactive_staging_buffers.push(staging_buffer);
461                    None
462                }
463            })
464    }
465
466    /// Destroy the context that this [`ContextData`] represents,
467    /// freeing all of its buffers, and deleting the associated WebRender image.
468    fn destroy(mut self, script_sender: &IpcSender<WebGPUMsg>, paint_api: &CrossProcessPaintApi) {
469        // This frees the id in the `ScriptThread`.
470        for staging_buffer in self.inactive_staging_buffers {
471            if let Err(error) = script_sender.send(WebGPUMsg::FreeBuffer(staging_buffer.buffer_id))
472            {
473                warn!(
474                    "Unable to send FreeBuffer({:?}) ({error})",
475                    staging_buffer.buffer_id
476                );
477            };
478        }
479        if let Some(image_key) = self.image_key.take() {
480            paint_api.delete_image(image_key);
481        }
482    }
483
484    /// Advance the [`Epoch`] and return the new one.
485    fn next_epoch(&mut self) -> Epoch {
486        let epoch = self.next_epoch;
487        self.next_epoch.next();
488        epoch
489    }
490
491    /// If the given [`PresentationStagingBuffer`] is for a newer presentation, replace the existing
492    /// one. Deallocate the older one by calling [`Self::return_staging_buffer`] on it.
493    fn replace_presentation(&mut self, presentation: PresentationStagingBuffer) {
494        let stale_presentation = if presentation.epoch >=
495            self.presentation
496                .as_ref()
497                .map(|p| p.epoch)
498                .unwrap_or_default()
499        {
500            self.presentation.replace(presentation)
501        } else {
502            Some(presentation)
503        };
504        if let Some(stale_presentation) = stale_presentation {
505            stale_presentation.maybe_destroy(self);
506        }
507    }
508
509    fn clear_presentation(&mut self) {
510        if let Some(stale_presentation) = self.presentation.take() {
511            stale_presentation.maybe_destroy(self);
512        }
513    }
514
515    fn return_staging_buffer(&mut self, staging_buffer: StagingBuffer) {
516        self.inactive_staging_buffers.push(staging_buffer)
517    }
518}

520impl crate::WGPU {
521    pub(crate) fn create_context(
522        &self,
523        context_id: WebGPUContextId,
524        size: DeviceIntSize,
525        buffer_ids: ArrayVec<id::BufferId, PRESENTATION_BUFFER_COUNT>,
526    ) {
527        let context_data = ContextData::new(&self.global, buffer_ids, size);
528        assert!(
529            self.wgpu_image_map
530                .lock()
531                .unwrap()
532                .insert(context_id, context_data)
533                .is_none(),
534            "Context should be created only once!"
535        );
536    }
537
538    pub(crate) fn set_image_key(&self, context_id: WebGPUContextId, image_key: ImageKey) {
539        let mut webgpu_contexts = self.wgpu_image_map.lock().unwrap();
540        let context_data = webgpu_contexts.get_mut(&context_id).unwrap();
541
542        if let Some(old_image_key) = context_data.image_key.replace(image_key) {
543            self.paint_api.delete_image(old_image_key);
544        }
545
546        self.paint_api.add_image(
547            image_key,
548            ImageDescriptor {
549                format: ImageFormat::BGRA8,
550                size: context_data.size,
551                stride: None,
552                offset: 0,
553                flags: ImageDescriptorFlags::empty(),
554            },
555            SerializableImageData::External(image_data(context_id)),
556        );
557    }
558
559    pub(crate) fn get_image(
560        &self,
561        context_id: WebGPUContextId,
562        pending_texture: Option<PendingTexture>,
563        sender: IpcSender<SharedSnapshot>,
564    ) {
565        let mut webgpu_contexts = self.wgpu_image_map.lock().unwrap();
566        let context_data = webgpu_contexts.get_mut(&context_id).unwrap();
567        if let Some(PendingTexture {
568            texture_id,
569            encoder_id,
570            configuration,
571        }) = pending_texture
572        {
573            let Some(staging_buffer) = context_data.get_or_make_available_buffer(&configuration)
574            else {
575                warn!("Failure obtaining available staging buffer");
576                sender
577                    .send(SharedSnapshot::cleared(configuration.size))
578                    .unwrap();
579                return;
580            };
581
582            let epoch = context_data.next_epoch();
583            let wgpu_image_map = self.wgpu_image_map.clone();
584            let sender = sender.clone();
585            drop(webgpu_contexts);
586            self.texture_download(
587                texture_id,
588                encoder_id,
589                staging_buffer,
590                configuration,
591                move |staging_buffer| {
592                    let mut webgpu_contexts = wgpu_image_map.lock().unwrap();
593                    let context_data = webgpu_contexts.get_mut(&context_id).unwrap();
594                    sender
595                        .send(
596                            staging_buffer
597                                .snapshot()
598                                .as_ref()
599                                .map(Snapshot::to_shared)
600                                .unwrap_or_else(|| SharedSnapshot::cleared(configuration.size)),
601                        )
602                        .unwrap();
603                    if staging_buffer.is_mapped() {
604                        context_data.replace_presentation(PresentationStagingBuffer::new(
605                            epoch,
606                            staging_buffer,
607                        ));
608                    } else {
609                        // failure
610                        context_data.return_staging_buffer(staging_buffer);
611                    }
612                },
613            );
614        } else {
615            sender
616                .send(
617                    context_data
618                        .presentation
619                        .as_ref()
620                        .and_then(|presentation_staging_buffer| {
621                            presentation_staging_buffer.staging_buffer.snapshot()
622                        })
623                        .unwrap_or_else(Snapshot::empty)
624                        .to_shared(),
625                )
626                .unwrap();
627        }
628    }
629
630    /// Read the texture to the staging buffer, map it to CPU memory, and update the
631    /// image in WebRender when complete.
632    pub(crate) fn present(
633        &self,
634        context_id: WebGPUContextId,
635        pending_texture: Option<PendingTexture>,
636        size: Size2D<u32>,
637        canvas_epoch: Epoch,
638    ) {
639        let mut webgpu_contexts = self.wgpu_image_map.lock().unwrap();
640        let context_data = webgpu_contexts.get_mut(&context_id).unwrap();
641
642        let Some(image_key) = context_data.image_key else {
643            return;
644        };
645
646        let Some(PendingTexture {
647            texture_id,
648            encoder_id,
649            configuration,
650        }) = pending_texture
651        else {
652            context_data.clear_presentation();
653            self.paint_api.update_image(
654                image_key,
655                ImageDescriptor {
656                    format: ImageFormat::BGRA8,
657                    size: size.cast_unit().cast(),
658                    stride: None,
659                    offset: 0,
660                    flags: ImageDescriptorFlags::empty(),
661                },
662                SerializableImageData::External(image_data(context_id)),
663                Some(canvas_epoch),
664            );
665            return;
666        };
667        let Some(staging_buffer) = context_data.get_or_make_available_buffer(&configuration) else {
668            warn!("Failure obtaining available staging buffer");
669            context_data.clear_presentation();
670            self.paint_api.update_image(
671                image_key,
672                configuration.into(),
673                SerializableImageData::External(image_data(context_id)),
674                Some(canvas_epoch),
675            );
676            return;
677        };
678        let epoch = context_data.next_epoch();
679        let wgpu_image_map = self.wgpu_image_map.clone();
680        let paint_api = self.paint_api.clone();
681        drop(webgpu_contexts);
682        self.texture_download(
683            texture_id,
684            encoder_id,
685            staging_buffer,
686            configuration,
687            move |staging_buffer| {
688                let mut webgpu_contexts = wgpu_image_map.lock().unwrap();
689                let context_data = webgpu_contexts.get_mut(&context_id).unwrap();
690                if staging_buffer.is_mapped() {
691                    context_data.replace_presentation(PresentationStagingBuffer::new(
692                        epoch,
693                        staging_buffer,
694                    ));
695                } else {
696                    context_data.return_staging_buffer(staging_buffer);
697                    context_data.clear_presentation();
698                }
699                // update image in WR
700                paint_api.update_image(
701                    image_key,
702                    configuration.into(),
703                    SerializableImageData::External(image_data(context_id)),
704                    Some(canvas_epoch),
705                );
706            },
707        );
708    }

710    /// Copies data from provided texture using `encoder_id` to the provided [`StagingBuffer`].
711    ///
712    /// `callback` is guaranteed to be called.
713    ///
714    /// Returns a [`StagingBuffer`] with the [`StagingBufferState::Mapped`] state
715    /// on success or [`StagingBufferState::Available`] on failure.
716    fn texture_download(
717        &self,
718        texture_id: TextureId,
719        encoder_id: CommandEncoderId,
720        mut staging_buffer: StagingBuffer,
721        config: ContextConfiguration,
722        callback: impl FnOnce(StagingBuffer) + Send + 'static,
723    ) {
724        let Ok(command_buffer_id) =
725            staging_buffer.prepare_load_texture_command_buffer(texture_id, encoder_id, &config)
726        else {
727            return callback(staging_buffer);
728        };
729        let StagingBufferState::Available(buffer) = &staging_buffer.state else {
730            unreachable!("`prepare_load_texture_command_buffer` should make buffer available")
731        };
732        let buffer_id = staging_buffer.buffer_id;
733        let buffer_size = buffer.size;
734        {
735            let _guard = self.poller.lock();
736            let result = self
737                .global
738                .queue_submit(buffer.queue_id, &[command_buffer_id]);
739            if result.is_err() {
740                return callback(staging_buffer);
741            }
742        }
743        staging_buffer.state = match staging_buffer.state {
744            StagingBufferState::Available(buffer) => StagingBufferState::Mapping(buffer),
745            _ => unreachable!("`prepare_load_texture_command_buffer` should make buffer available"),
746        };
747        let map_callback = {
748            let token = self.poller.token();
749            Box::new(move |result: Result<(), BufferAccessError>| {
750                drop(token);
751                staging_buffer.state = match staging_buffer.state {
752                    StagingBufferState::Mapping(buffer) => {
753                        if let Ok((data, len)) = result.and_then(|_| {
754                            staging_buffer.global.buffer_get_mapped_range(
755                                staging_buffer.buffer_id,
756                                0,
757                                Some(buffer.size),
758                            )
759                        }) {
760                            StagingBufferState::Mapped(MappedBuffer {
761                                buffer,
762                                data,
763                                len,
764                                image_size: config.size,
765                                image_format: config.format,
766                                is_opaque: config.is_opaque,
767                            })
768                        } else {
769                            StagingBufferState::Available(buffer)
770                        }
771                    },
772                    _ => {
773                        unreachable!("Mapping buffer should have StagingBufferState::Mapping state")
774                    },
775                };
776                callback(staging_buffer);
777            })
778        };
779        let map_op = BufferMapOperation {
780            host: HostMap::Read,
781            callback: Some(map_callback),
782        };
783        // error is handled by map_callback
784        let _ = self
785            .global
786            .buffer_map_async(buffer_id, 0, Some(buffer_size), map_op);
787        self.poller.wake();
788    }

790    pub(crate) fn destroy_context(&mut self, context_id: WebGPUContextId) {
791        self.wgpu_image_map
792            .lock()
793            .unwrap()
794            .remove(&context_id)
795            .unwrap()
796            .destroy(&self.script_sender, &self.paint_api);
797    }
